।𑌵𑌸𑌿𑌷𑍍𑌠𑌸𑍍𑌯𑌾𑌪𑌾𑌮𑍀𑌵𑍇𑌚𑍇 𑌵𑌾𑌯𑍋𑌰𑍍𑌵𑌾𑌭𑌿𑌕𑍃𑌨𑍍𑌦𑍗 ।

15 ।𑌵𑌸𑌿𑌷𑍍𑌠𑌸𑍍𑌯𑌾𑌪𑌾𑌮𑍀𑌵𑍇𑌚𑍇 𑌵𑌾𑌯𑍋𑌰𑍍𑌵𑌾𑌭𑌿𑌕𑍃𑌨𑍍𑌦𑍗 ।

𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌸𑍋𑌮𑌸𑍁𑌷𑍁𑌤𑌾𑌃।𑌪𑌾𑌰𑍀𑌸𑍍𑌰𑌾𑌵𑌾।𑌆𑌪𑌾
𑌕𑍀𑌚𑍀𑌚𑍍𑌯𑌕𑌫

𑌕𑌿𑌚𑍁𑌚𑍍𑌯𑌕𑌫𑌕𑍍𑌯
𑌮𑌤𑍍𑌸𑌤𑌾𑌚𑌾𑌯𑌾𑌵𑍀𑌨𑌾𑌃।𑌨𑍍𑌦𑌰𑌾𑌵𑌾𑌇𑌣𑌸𑍍𑌵𑌨𑍍𑌤𑌇𑌹𑌸

𑌚𑍢𑌤𑍍𑌰
