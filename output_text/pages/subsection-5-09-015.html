।𑌮𑌰𑍂𑌤𑌾𑌨𑍍𑌧𑍇𑌨𑍁𑌨𑍀𑌚𑍇 ।

14 ।𑌮𑌰𑍂𑌤𑌾𑌨𑍍𑌧𑍇𑌨𑍁𑌨𑍀𑌚𑍇 ।

𑌪𑍍𑌰𑌾𑌦𑍇।𑌵𑌾𑌮𑌚𑍍𑌛𑌾𑌮𑌧𑍁𑌮𑌨𑍍𑌤𑌾𑌆𑌇𑌨𑍍𑌦𑌾𑌵𑌾𑌃।𑌆𑌸𑌿𑌷𑍍𑌯𑌾
𑌤𑍍𑌯𑌕𑍂𑌚𑌕𑌟𑌂𑌟𑍍𑌯𑌚𑌚𑍍𑌯
𑌦𑌂𑌤𑌾𑌗𑌾𑌵𑌆𑌨𑌾𑌧𑌾𑌇𑌨𑌾𑌵𑌾𑌃।𑌬𑌰𑍍𑌹𑌿𑌷𑌾𑌦𑌾𑌃।𑌵𑌾
𑌕𑌥𑌟𑌿𑌚𑌕𑌟𑌚𑌚𑌿𑌚𑌿𑌣
𑌚𑌾𑌨𑌾𑌵𑌾𑌨𑍍𑌤𑌾𑌊𑌧𑌾𑌭𑌾𑌇𑌃𑌪𑌾𑌰𑌿𑌸𑍍𑌸𑌰𑍂𑌤𑌂।𑌉𑌸𑍍𑌰𑌿𑌯𑌾
𑌚𑍍𑌯𑌟𑍀𑌚𑌕𑌕𑌿𑌚𑍍𑌯𑌣
𑌨𑌿𑌰𑍍𑌣𑌿𑌜𑌨𑍍𑌧𑌾𑌇𑌰𑌾𑌏𑌧𑌾𑌇𑌰𑌾𑌔𑌹𑍋𑌵𑌾
𑌟𑍇𑌚𑌿𑌶𑌿
𑌧𑌾𑌇𑌰𑌾𑌇।।27।।
𑌤𑌟𑌚𑌶
𑌤𑍍𑌰𑌾𑌇𑌰𑌸𑍍𑌮𑍈𑌸𑌪𑍍𑌤𑌧𑍇𑌨𑌵𑍋𑌦𑍁𑌦𑍗𑌹𑍋𑌹𑍍𑌰𑌾𑌇𑌰𑌾𑌇।
𑌪𑌪𑍍𑌲𑌨𑍍𑌦𑌶𑌿
𑌸𑌤𑍍𑌯𑌾𑌮𑌾𑌶𑍀𑌰𑌂𑌪𑌰𑌮𑍇𑌵𑍇𑌯𑌾𑌮𑌾𑌨𑍀।𑌚𑌤𑍍𑌵𑌾𑌰𑍍𑌯𑌨𑍍𑌯𑌾𑌭𑍁𑌵
𑌕𑍢𑌟𑍍𑌯
𑌨𑌾𑌨𑌿𑌨𑌿𑌰𑍍𑌣𑌾𑌇𑌜𑌾𑌇।𑌚𑌾𑌰𑍂𑌣𑌾𑌇𑌚𑌾𑌕𑍇𑌰𑌅𑌦𑍍𑌰𑍁
𑌕𑍢𑌟𑌿𑌶𑌚𑌿𑌣𑍍𑌯𑌕
