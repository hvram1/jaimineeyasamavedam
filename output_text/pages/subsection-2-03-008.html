8 ।𑌭𑌾𑌰𑌚𑌾𑌜𑌸𑍍𑌯𑌮𑍗𑌕𑍍𑌷𑍇𑌚𑍇 ।

𑌷𑌤𑍃𑌤𑌶
𑌪𑍍𑌰𑌜𑌾𑌵𑌾𑌤𑍍𑌸𑌾।𑌵𑍀𑌸𑍍𑌸𑍗𑌭𑌗𑌾𑌂𑌪𑌾𑌰𑌾𑌦𑍂𑌷𑍍𑌵𑌾।
𑌟𑍃𑌤𑌟𑍍𑌯𑌚𑍍𑌯𑌯𑌟𑍍𑌯𑌤
𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌪𑍍𑌨𑌿𑌯𑌂𑌸𑍂𑌵𑌾।𑌦𑌕𑍍𑌷𑌾𑌯𑌾।।10।।

𑌚𑍍𑌯𑌶𑍍𑌰𑍂𑌕𑍍𑌯𑌕𑍀
