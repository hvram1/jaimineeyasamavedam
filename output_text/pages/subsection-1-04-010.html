।𑌮𑌣𑍍𑌡𑍋𑌰𑍍𑌜𑌾𑌮𑌦𑌗𑍍𑌨𑍍𑌯𑌸𑍍𑌯𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।।

10 ।𑌮𑌣𑍍𑌡𑍋𑌰𑍍𑌜𑌾𑌮𑌦𑌗𑍍𑌨𑍍𑌯𑌸𑍍𑌯𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।।

𑌅𑌗𑍍𑌨𑍇𑌵𑍀𑌵𑌾𑌹𑌾𑌬𑍁।𑌸𑍍𑌵𑌾𑌦𑍂𑌷𑌾𑌸𑌾
𑌤𑍀𑌤𑌶𑌕𑌟𑍍𑌯𑌤
𑌶𑍍𑌪𑌾𑌇𑌤𑍍𑌰𑌾𑌂𑌹𑌾𑌇𑌰𑌾𑌧𑍋𑌹𑌾𑌅𑌮𑌾𑌕𑍍𑌤𑌰𑌾𑌯𑌂।𑌆
𑌟𑌿𑌤𑌟𑌿𑌤𑌟𑍍𑌯𑌚𑌣𑌚
𑌦𑌾𑌶𑍁𑌷𑍇𑌜𑌾𑌤𑌾𑌵𑍇𑌦𑍋𑌵𑌾𑌹𑌾𑌤𑍂𑌵𑌾𑌂।
𑌯𑌟𑍍𑌯𑌚𑍍𑌯𑌥𑍍𑌯𑌚𑌕𑍍𑌯𑌯𑌟
𑌅𑌦𑍍𑌯𑌾𑌹𑍋𑌇𑌦𑌾𑌇𑌵𑌂।𑌊𑌷𑌾𑌰𑍍𑌬𑍁𑌧𑌾
𑌟𑍁𑌤𑍍𑌯𑌕𑌟𑍍𑌯𑌟𑌚
𑌔𑌹𑍋𑌵𑌾।𑌹𑍁𑌵𑍇𑌵𑌾𑌸𑍂:
𑌶𑌿𑌤𑍍𑌯𑌟𑍍𑌯𑌚
𑌅𑌗𑍍𑌨𑍇𑌵𑌿𑌵𑌸𑍍𑌵𑌦𑍁𑌷𑌾𑌸𑌾:।𑌚𑌿𑌤𑍍𑌰𑌂𑌰𑌾𑌧𑍋
𑌤𑍃𑌤𑌕𑍀
𑌅𑌮𑌾𑌕𑍍𑌤𑍍𑌯𑌰𑌾𑌮𑌾𑌦𑌾𑌶𑍂𑌷𑍇।𑌜𑌾𑌤𑌾𑌵𑍇𑌦𑍋
𑌕𑍍𑌯𑌕𑍍𑌯𑌯𑌟𑍍𑌯𑌚𑍍𑌯𑌥𑍍𑌯𑌚
𑌵𑌾𑌹𑌾𑌤𑍂𑌵𑌾𑌂।𑌅𑌦𑍍𑌯𑌾𑌦𑌾𑌇𑌵𑌂।𑌊𑌷𑌾𑌰𑍍𑌬𑍂
𑌕𑍍𑌯𑌯𑌟𑌟𑌿𑌤𑍍𑌯𑌚𑌕𑌟𑍍𑌯𑌟
𑌧𑌾𑌔𑌹𑍋𑌵𑌾।𑌵𑌿𑌦𑌾𑌵𑌾𑌸𑍂𑌃
𑌚𑌶𑌿𑌤𑍍𑌯𑌟𑍍𑌯𑌚
