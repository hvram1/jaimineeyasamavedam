।𑌸𑍋𑌮𑌸𑌾𑌮𑌾𑌨𑌿𑌷 𑌟𑍍 ।

30 ।𑌸𑍋𑌮𑌸𑌾𑌮𑌾𑌨𑌿𑌷 𑌟𑍍 ।

𑌹𑌾𑌬𑍁𑌸𑍋𑌮𑌾𑌃।𑌉𑌷𑍍𑌵𑌾𑌣𑌸𑍍𑌸𑍋𑌤𑍃𑌭𑍀𑌃।𑌆𑌧𑌿𑌷𑍍𑌣𑍁𑌭𑍀
𑌤𑍀𑌥𑍍𑌯𑌚𑌯𑌿𑌟𑌕𑍀
𑌰𑌾𑌵𑌾𑌇𑌨𑌾𑌂।𑌆𑌶𑍍𑌪𑌾𑌔𑌹𑍋।𑌯𑍇𑌵𑌹𑌰𑌿𑌤𑌾𑌯𑌾𑌤𑌾𑌇
𑌚𑌯𑌟𑍍𑌯𑌟𑍍𑌯𑌟𑌤
𑌧𑌾𑌰𑌾𑌯𑌾।𑌮𑌙𑍍𑌰𑌾𑌯𑌾𑌯𑌾𑌹𑌾𑌓𑌵𑌾।𑌤𑍀𑌧𑌾𑌰𑌾𑌯𑌾।।44।।
𑌯𑍢𑌟𑍍𑌯𑌕𑌿𑌚𑌶𑌿𑌟𑌿𑌚
𑌸𑍋𑌮𑌉𑌷𑍍𑌵𑌾𑌣𑌸𑍍𑌸𑍋𑌤𑍃𑌭𑌾𑌇𑌃।𑌆𑌧𑍀𑌷𑍍𑌣𑍁𑌭𑍀𑌰𑌾𑌵𑍀𑌨𑌾
𑌚𑍀𑌪𑍍𑌲𑌶𑌕𑍀𑌕𑌥𑍍𑌯
𑌮𑌾𑌶𑍍𑌪𑌯𑍇𑌵𑌾।𑌹𑌾𑌰𑌿𑌤𑌾𑌯𑌾𑌤𑌿𑌧𑌾𑌰𑌾𑌯𑌾।𑌮𑌙𑍍𑌰𑌯𑌾𑌯𑌾।
𑌪𑌿𑌣𑌯𑍂𑌪𑌶𑌚𑌿𑌶
𑌤𑌾𑌇𑌧𑌾।𑌰𑌾𑌯𑌾𑌔𑌹𑍋𑌬𑌾।𑌹𑍋𑌇ழா𑌹।।45।।
𑌚𑌣𑍍𑌯𑌟𑌿𑌚𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌸𑍋𑌮𑌉𑌷𑍍𑌵𑌾𑌣𑌸𑍍𑌸𑍋𑌤𑍃𑌭𑌿𑌰𑍇𑌅𑌧𑍀।𑌷𑍍𑌣𑍁𑌭𑌿𑌰𑌾𑌆𑌉𑌵𑌾
𑌤𑍢𑌤𑍍𑌯𑌤𑌿𑌟𑌿
।𑌪𑍍𑌰।𑌵𑍀𑌨𑌾𑌂।𑌆𑌶𑍍𑌪𑌾𑌯𑍇𑌵𑌹𑌰𑌿𑌤𑌾𑌯𑌾𑌤𑌿𑌧𑌾।𑌪𑍍𑌰।
𑌚𑌶𑌟𑌤𑌕𑍂𑌤𑍍𑌯
𑌰𑌾𑌯𑌾।𑌮𑌾𑌙𑍍𑌰𑌾𑌯𑌾𑌯𑌾𑌤𑌿𑌧𑌾।𑌪𑍍𑌰।𑌰𑌾𑌯𑌾।।46।।
𑌚𑌶𑌟𑌤𑌕𑍍𑌯𑌤𑍍𑌯𑌚𑌶
𑌸𑍋𑌮𑌉𑌷𑍍𑌵𑌾𑌣𑌸𑍍𑌸𑍋𑌤𑍃𑌭𑌿𑌰𑍇𑌅𑌧𑌿।𑌷𑍍𑌣𑍁𑌭𑌿𑌰𑌵𑍀
𑌤𑍢𑌤𑍍𑌯𑌕𑍀
𑌨𑌾𑌮𑌾𑌶𑍍𑌪𑌯𑍇𑌵𑌾𑌹𑌰𑌿𑌤𑌾𑌯𑌾𑌤𑌾𑌇𑌧𑌾𑌰𑌯𑌾𑌮𑌙𑍍𑌰𑌯𑌾𑌯𑍗
𑌕𑍀𑌕𑍁𑌕𑍁
𑌪𑌾।𑌤𑌿𑌧𑌾।𑌰𑌾𑌯𑌾𑌔𑌹𑍋𑌬𑌾𑌹𑍋𑌇ழா𑌹।।47।।
𑌶𑍁𑌚𑌶𑌟𑌿𑌚𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌸𑍋𑌮𑌉𑌷𑍍𑌵𑌾𑌣𑌸𑍍𑌸𑍋𑌰𑍃𑌭𑌾𑌇𑌃𑌰𑌧𑌿𑌷𑍍𑌣𑍁𑌭𑌿𑌰𑌵𑍀𑌨𑌾𑌂।
𑌤𑍍𑌰
𑌆𑌧𑌿𑌷𑍍𑌣𑍁𑌭𑌿𑌰𑌵𑍀𑌨𑌾𑌮𑌾𑌶𑍍𑌪𑌯𑍇𑌵𑌾𑌹𑌾𑌰𑍀𑌤𑌾𑌯𑌾𑌤𑌾𑌇𑌧𑌾।
𑌕𑍋𑌪𑍀𑌶𑍍𑌯
𑌰𑌯𑌾𑌆𑌉𑌵𑌾।𑌮𑌾𑌙4𑌾।𑌯𑌾𑌯𑌾𑌤𑍀𑌧𑌾।𑌰𑌾𑌯𑌾𑌔𑌹𑍋
𑌤𑍍𑌯𑌟𑌿𑌚𑌶𑌚𑌿𑌣𑌣𑍍𑌯𑌚𑌕
𑌵𑌾।𑌹𑍋𑌇ழா𑌹।।48।।
𑌫𑌪𑍍𑌲𑌶𑍍𑌯
𑌸𑍋𑌮𑌉𑌷𑍍𑌵𑌾𑌣𑌸𑍍𑌸𑍋𑌤𑍃𑌭𑌿𑌰𑍗𑌹𑍋𑌆𑌧𑌿𑌷𑍍𑌣𑍁𑌭𑍀𑌰𑌵𑍀
𑌚𑍃𑌟𑍍𑌯𑌤𑌪
𑌨𑌾𑌂।𑌆𑌶𑍍𑌪𑌯𑍇𑌵𑌾𑌹𑌾𑌰𑌿𑌤𑌾𑌯𑌾𑌤𑍀𑌧𑌾𑌰𑌾𑌯𑍗𑌵𑌾𑌉𑌵𑍋
𑌶𑍂𑌕𑌕𑍀𑌥𑌿𑌚𑌕𑌕𑌿𑌪𑍍𑌲
𑌵𑌾।𑌮𑌾𑌙𑍍𑌰𑌯𑌾𑌯𑌾𑌤𑌿𑌧𑌾𑌹𑌿𑌂।𑌰𑌾𑌯𑍋𑌹𑌾𑌇।।49।।
𑌶𑌕𑌪𑍀𑌶𑌚𑌚𑌪𑍍𑌲𑌪𑍍𑌲𑌶
