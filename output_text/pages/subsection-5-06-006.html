6 ।𑌗𑍗𑌤𑌮𑌸𑍍𑌯𑌤𑌭|3𑌤1𑌭|𑌚𑍇…

𑌈𑌹𑍋𑌈𑌹𑌾𑌇𑌹𑍋𑌇𑌹𑌾।𑌹𑍋𑌵𑌾𑌹𑍋𑌇𑌹𑌾।
𑌚𑍍𑌯𑌫𑍍𑌯𑌤𑍀𑌕𑍍𑌯𑌚𑌫𑍍𑌯𑌣
𑌤𑌿𑌸𑍇𑌰𑌾𑌵𑌾𑌚𑌾𑌈𑌰𑌾𑌯𑌾𑌤𑌿𑌪𑍍𑌰𑌵𑌾𑌨𑍍𑌹𑍀:।𑌓𑌹𑌾𑌹𑍋
𑌟𑍁𑌕𑍍𑌯𑌚𑌿𑌪𑍍𑌲𑌣𑍍𑌯𑌫𑌚
𑌹𑌾𑌓𑌹𑌾।𑌰𑌤𑌸𑍍𑌯𑌧𑌾𑌇𑌤𑍀𑌂𑌬𑍍𑌰𑌾𑌹𑍍𑌮𑌾𑌣𑍋𑌮𑌾𑌨𑍀𑌷𑌾𑌂।
𑌶𑌿𑌟𑍂𑌕𑌿𑌚𑍍𑌯𑌶
𑌆𑌹𑌾𑌹𑍋𑌇𑌆𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।𑌗𑌾𑌵𑍋𑌯
𑌕𑍍𑌯𑌕𑌿𑌕𑍍𑌯𑌚𑌕𑌚𑌶𑌕
𑌭3𑌇𑌗𑍋𑌪𑌾𑌤𑌿𑌂𑌪𑌾𑌚𑍍𑌛𑌰𑌾𑌮𑌾𑌨𑌾:।𑌇𑌹𑌾𑌹𑍋𑌇𑌈
𑌟𑍁𑌕𑌿𑌚𑍍𑌯𑌪𑍍𑌲𑌕𑍍𑌯𑌕𑌿
𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।𑌸𑍋𑌮𑌯𑍍𑌯𑌭3𑌇𑌮𑌾𑌤𑌯𑍋𑌵𑌾𑌵𑌾
𑌕𑍍𑌯𑌚𑌕𑌚𑌶𑌕𑌕𑍁𑌪𑌿
𑌶𑌾𑌨𑌾:।।15।।
𑌚𑍍𑌯𑌤𑍍𑌰
𑌇𑌹𑌾𑌉𑌵𑌾𑌈𑌹𑌾𑌇𑌹𑌾।𑌤𑌿𑌸𑍇𑌰𑌾𑌵𑌾𑌚𑌾𑌈𑌰𑌾𑌯𑌾𑌤𑌿𑌪𑍍𑌰
𑌟𑍀𑌚𑌪𑍍𑌲𑌨𑍍𑌦𑌟𑍁𑌕𑍍𑌯
𑌵𑌾𑌨𑍍𑌹𑍀:।𑌓𑌹𑌾𑌹𑍋𑌹𑌾𑌓𑌹𑌾।𑌰𑌤𑌸𑍍𑌯𑌧𑌾𑌇𑌤𑍀𑌂𑌬𑍍𑌰𑌾𑌹𑍍𑌮𑌾
𑌪𑍍𑌰𑌣𑍍𑌯𑌫𑌚𑌶𑌿𑌟𑍂
𑌣𑍋𑌮𑌾𑌨𑍀𑌷𑌾𑌂।𑌆𑌹𑌾𑌹𑍋𑌇𑌆𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋
𑌕𑌿𑌚𑍍𑌯𑌶𑌕𑍍𑌯𑌕𑌿𑌕𑍍𑌯𑌚𑌕𑌚
𑌇।𑌗𑌾𑌵𑍋𑌯𑌭3𑌇𑌗𑍋𑌪𑌾𑌤𑌿𑌂𑌪𑌾𑌚𑍍𑌛𑌰𑌾𑌮𑌾𑌨𑌾:।
𑌶𑌕𑌟𑍁𑌕𑌿𑌚𑍍𑌯𑌪𑍍𑌲
𑌇𑌹𑌾𑌹𑍋𑌇𑌈𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।𑌸𑍋𑌮𑌯𑍍𑌯
𑌕𑍍𑌯𑌕𑌿𑌕𑍍𑌯𑌚𑌕𑌚𑌶𑌕
𑌭3𑌇𑌮𑌾𑌤𑌯𑍋𑌵𑌾𑌵𑌾𑌶𑌾𑌨𑌾:।।16।।
𑌤𑍁𑌪𑌿𑌚𑍍𑌯𑌤𑍍𑌰
