𑌅𑌸𑍍𑌯𑌾𑌔𑌹𑍋𑌵𑌾𑌪𑍇𑌰𑌷𑌾𑌹𑍇𑌮𑌾𑌨𑌾𑌪𑍂𑌯
𑌚𑍍𑌯𑌕𑌥𑍍𑌯𑌟𑌿𑌕𑌿
𑌮𑌾𑌨𑌾:।𑌦𑍇𑌵𑍋𑌔𑌹𑍋𑌵𑌾𑌦𑍇𑌵𑌾𑌇𑌭𑍀𑌸𑍍𑌸𑌾𑌮𑌾𑌪𑍃
𑌚𑍍𑌯𑌪𑍍𑌲𑌚𑍍𑌯𑌕𑌥𑍍𑌯𑌟𑍀
𑌕𑍍𑌤𑌰𑌾𑌸𑌂।𑌸𑍁𑌤𑌾𑌔𑌹𑍋𑌵𑌾।𑌪𑌵𑌾𑌇𑌤𑍍𑌰𑌾𑌂𑌪𑌾𑌰𑍀
𑌚𑍁𑌚𑍍𑌯𑌕𑌥𑍍𑌯𑌟𑍀
𑌯𑍇𑌤𑌿𑌰𑍇𑌭𑌨𑍍।𑌮𑌿𑌤𑌾𑌔𑌹𑍋𑌵𑌾𑌵𑌸𑌾𑌦𑍍𑌮𑌾𑌪𑌾𑌶𑍁

𑌚𑍁𑌕𑌥𑍍𑌯𑌟𑌿𑌕𑌿

𑌕𑌿𑌚𑍍𑌯𑌶𑌚𑍁𑌕𑌥𑍍𑌯

𑌟𑍀𑌕𑌿𑌚𑍍𑌯𑌶𑌚𑍁𑌕
