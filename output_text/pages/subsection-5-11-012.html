12 ।𑌸𑌨𑍍𑌤𑌨𑌿𑌚 ।

𑌏𑌷𑌾𑌹𑌾𑌬𑍁।𑌸𑍍𑌯𑌧𑌾𑌰𑌯𑌾𑌆𑌉𑌵𑌾।𑌸𑍂𑌤𑌾𑌃।
𑌤𑍍𑌯𑌤𑌶𑌕𑍍𑌯𑌤𑍍𑌯𑌟𑌿𑌚𑌪𑍍𑌲
𑌅𑌵𑍍𑌯𑌾𑌵𑌾𑌰𑍇𑌭𑌿ǐ𑌪𑌵𑌤𑍇𑌮𑌾𑌦𑌿𑌨𑍍𑌤𑌾𑌮𑌾𑌃।𑌕𑍍𑌰𑍀ழா𑌹𑌨𑍍𑌹𑌾𑌬𑍁।
𑌕𑍄𑌚𑌕𑌭𑌿𑌤𑍍𑌯𑌤𑌶
𑌊𑌰𑍍𑌮𑍍𑌮𑍀𑌰𑌵𑌾𑌆𑌉𑌵𑌾।𑌮𑍀𑌵𑌾।𑌯𑌾𑌉𑌸𑍍𑌰𑌿𑌯𑌾𑌅𑌪𑌿𑌯𑌾𑌆
𑌕𑍍𑌯𑌤𑍍𑌯𑌟𑌿𑌚𑌶𑌕𑍃𑌚
𑌨𑍍𑌤𑌾𑌰𑌶𑍍𑌮𑌾𑌨𑍀।𑌨𑌿𑌰𑍍𑌗𑌾𑌹𑌾𑌬𑍁।𑌆𑌕𑍍𑌰𑌨𑍍𑌤𑌦𑍋𑌆𑌉𑌵𑌾𑌜𑌾
𑌚𑍀𑌤𑍍𑌯𑌤𑌶𑌕𑍍𑌯𑌤𑍍𑌯𑌟𑌿𑌚
𑌸𑌾।𑌆𑌭𑌿𑌵𑍃𑌜𑌨𑍍𑌤𑌨𑍍𑌤𑍀𑌷𑍇𑌗𑌵𑍍𑌯𑌮𑌶𑍍𑌪𑌿𑌯𑌾𑌂।𑌵𑌰𑍍𑌮𑍍𑌮𑍀𑌹𑌾𑌬𑍁।
𑌶𑌕𑍃𑌨𑍁𑌤𑍍𑌯𑌤𑌶
𑌵𑌾𑌧𑍃𑌷𑍍𑌣𑌵𑌾𑌆𑌉𑌵𑌾।𑌰𑍂𑌜𑌾।।35।।
𑌕𑍍𑌯𑌤𑍍𑌯𑌟𑌿𑌚𑌶
