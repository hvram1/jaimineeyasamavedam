14 ।𑌗𑍋𑌶𑍍𑌰𑌂𑌗𑍇𑌤𑍍𑌵𑍇 ।

𑌤𑌰𑌣𑌿𑌰𑌿𑌤𑍍𑌸𑍀𑌷𑌾𑌸𑌤𑌿𑌵𑌾𑌜𑌂𑌪𑍁𑌰𑌨𑍍𑌧𑌯𑌾𑌯𑍁𑌜𑌾।
𑌫𑍂𑌤𑍍𑌯𑌪𑌪𑍍𑌲𑌨𑍍𑌦𑌶
𑌵𑌾𑌜𑌂𑌪𑍁𑌰𑌨𑍍𑌧𑌯𑌾𑌯𑍁𑌜𑌾𑌵𑌾𑌇𑌨𑍍𑌦𑌰𑌾𑌂𑌪𑍁𑌰𑍁𑌹𑍂𑌤𑌨𑍍𑌨𑌮𑍇
𑌚𑍇𑌫𑌪𑍍𑌲
𑌗𑌾𑌇𑌰𑌾𑌃।𑌨𑍇𑌮𑌾𑌇𑌨𑍍𑌤𑌾𑌷𑍍𑌟𑍇𑌵𑌸𑍂𑌦𑍍𑌰𑍂𑌵𑌾𑌮𑌿ழா𑌹𑌭𑌾।
𑌤𑌤𑍍𑌯𑌕𑌟𑌿𑌤𑌚𑌿𑌟𑌿𑌚𑌣
𑌓𑌇ழா𑌹।।22।।
𑌪𑌪𑍍𑌲
𑌤𑌰𑌣𑌿𑌰𑌿𑌤𑍍𑌸𑍀𑌷𑌾𑌸𑌤𑍀𑌵𑌾𑌜𑌂𑌪𑍁𑌨𑍍𑌧𑌯𑌾𑌯𑍂𑌜𑌾।
𑌫𑍂𑌤𑍍𑌯𑌪𑌪𑍍𑌲𑌨𑍍𑌦𑌶
𑌵𑌾𑌜𑌂𑌪𑍁𑌰𑌨𑍍𑌧𑌯𑌾𑌯𑍁𑌜𑌾𑌵𑌇𑌨𑍍𑌦𑌰𑌂𑌪𑍁𑌰𑍁𑌹𑍂𑌤𑌨𑍍𑌨𑌮𑌾𑌇𑌗𑌾
𑌟𑍇
𑌇𑌰𑌾𑌹𑍋𑌵𑌾𑌉𑌉𑌵𑍋𑌇।।𑌨𑍇𑌮𑌿𑌨𑍍𑌤𑌷𑍍𑌟𑍇𑌵𑌸𑌾𑌉𑌵𑌾
𑌕𑍁𑌤𑌚𑌿𑌶𑌚𑍂𑌕𑍍𑌯
𑌓𑌬𑌾𑌦𑍍𑌰𑍂𑌵𑌾𑌂𑌹𑌾𑌇।।23।।
