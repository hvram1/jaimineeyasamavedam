।𑌶𑍍𑌨𑍗𑌷𑍍𑌟𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿…

8 ।𑌶𑍍𑌨𑍗𑌷𑍍𑌟𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿…

𑌔𑌹𑍋𑌹𑌾𑌅𑌯𑍋𑌹𑌾𑌇।𑌪𑌾𑌵𑌾𑌪𑌵𑌸𑍍𑌵𑍈𑌨𑌾𑌵𑌾
𑌤𑍁𑌤𑌶𑌕𑍁
𑌸𑍂𑌨𑍀।𑌔𑌹𑍋𑌇𑌹𑌾𑌈𑌯𑌾।𑌪𑍍𑌰।𑌮𑌾𑌂𑌶𑍍𑌪𑌤𑍍𑌵𑍈
𑌟𑍍𑌯𑌤𑌟𑌚𑌕𑍍𑌯𑌟𑌤𑌂
𑌨𑍍𑌦𑍋𑌸𑌰𑌸𑍀𑌪𑍍𑌰𑌧𑌾𑌨𑍍𑌵𑌾।𑌪𑍍𑌰।𑌬𑍃𑌧𑍍𑌨𑌶𑍍𑌪𑌿𑌦𑍍𑌯𑌸𑍍𑌯𑌵𑌾𑌤𑍋
𑌕𑍃𑌟𑍍𑌯𑌤𑌶𑍂𑌕
𑌨𑌾𑌜𑍂𑌤𑍀𑌂।𑌪𑍍𑌰।𑌪𑍁𑌰𑍂𑌮𑍇𑌧𑌶𑍍𑌪𑌿𑌕𑍍𑌤𑌕𑌵𑍇𑌨𑌾𑌰𑌾
𑌟𑍍𑌯𑌤𑌷𑍍𑌯𑌕𑍂𑌟𑍍𑌯
𑌨𑍍𑌧𑌾𑌤𑍍।𑌔𑌹𑍋𑌇𑌹𑌾𑌈𑌯𑌾𑌔𑌹𑍋𑌵𑌾।
𑌤𑌟𑌚𑌕𑍍𑌯𑌟𑌟𑌚𑌶𑌿
𑌏𑌦𑍀𑌦𑌿𑌹𑍀।।18।।
𑌤𑌤𑌿𑌚
𑌅𑌯𑍋𑌹𑌾𑌇𑌪𑌵𑍋𑌹𑌾𑌇।𑌪𑌾𑌵𑌸𑍍𑌵𑍈𑌨𑌾𑌵𑌾𑌸𑍂𑌨𑍀।
𑌤𑍍𑌯𑌤𑍀𑌤𑌶𑌤𑍀𑌟𑍍𑌯𑌤
𑌇𑌹𑍋𑌇𑌹𑌾𑌈𑌯𑌾।𑌪𑍍𑌰।𑌮𑌾𑌂𑌶𑍍𑌪𑌤𑍍𑌵𑌾𑌇𑌨𑍍𑌦𑍋𑌸𑌰𑌸𑍀𑌪𑍍𑌰
𑌕𑍍𑌯𑌕𑍍𑌯𑌟𑌤𑌂𑌕𑍃
𑌧𑌾𑌨𑍍𑌵𑌾।𑌪𑍍𑌰।𑌬𑍃𑌧𑍍𑌨𑌶𑍍𑌪𑌿𑌦𑍍𑌯𑌸𑍍𑌯𑌵𑌾𑌤𑍋𑌨𑌜𑍂𑌤𑍀𑌂।𑌪𑍍𑌰।
𑌟𑍍𑌯𑌤𑌕𑍃𑌟𑍍𑌯𑌤
𑌪𑍁𑌰𑍂𑌮𑍇𑌧𑌶𑍍𑌪𑌿𑌕𑍍𑌤𑌕𑌵𑍇𑌨𑌾𑌰𑌾𑌨𑍍𑌧𑌾𑌤𑍍।𑌇𑌹𑍋
𑌷𑌿𑌕𑍁𑌟𑍍𑌯𑌤𑌕𑍍𑌯
𑌇𑌹𑌾𑌈𑌯𑌾𑌔𑌹𑍋𑌵𑌾।𑌏𑌦𑍀𑌦𑌿𑌯𑌾।।19।।
𑌕𑍍𑌯𑌟𑌟𑌚𑌶𑌿𑌤𑌨𑌿
𑌹𑌾𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌓𑌵𑌾𑌹𑌾𑌉𑌵𑍋।𑌆
𑌚𑌿𑌶𑌚𑌿𑌶𑌤𑌪𑌪𑍍𑌲𑌤𑌿
𑌯𑌾𑌪𑌾𑌵𑌾𑌪𑌵𑌸𑍍𑌵𑍈𑌨𑌾𑌵𑌸𑍂𑌨𑍀𑌹𑍋𑌇𑌹𑌿𑌯𑌾𑌈
𑌚𑍍𑌯𑌚𑍍𑌯𑌕𑌿𑌕𑌿𑌕𑍍𑌯𑌚𑌕𑌟𑍍𑌯
𑌹𑍋𑌇𑌹𑌾।𑌪𑍍𑌰।𑌮𑌾𑌂𑌶𑍍𑌪𑌤𑍍𑌵𑌈𑌨𑍍𑌦𑍋𑌸𑌰𑌸𑌿𑌪𑍍𑌰𑌧𑌨𑍍𑌵𑌾𑌇
𑌟𑍍𑌯𑌨𑍍𑌯𑌂𑌕𑍀𑌕𑍂
𑌹𑍋।𑌪𑍍𑌰।𑌬𑍃𑌧𑍍𑌨𑌾𑌶𑍍𑌪𑌿𑌦𑍍𑌯𑌾𑌸𑍍𑌯𑌵𑌾𑌤𑍋𑌨𑌜𑍂𑌤𑌿𑌮𑌿𑌹𑍋
𑌚𑍍𑌯𑌚𑍍𑌯𑌚𑍍𑌯𑌕𑌿𑌕𑍁𑌚
।𑌪𑍍𑌰।𑌹𑌾𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌓𑌵𑌾𑌹𑌾
𑌚𑌿𑌶𑌚𑌿𑌶𑌤𑌪𑌪𑍍𑌲
𑌉𑌵𑌾।𑌪𑍂𑌰𑍂𑌮𑍇𑌧𑌶𑍍𑌪𑌿𑌕𑍍𑌤𑌕𑌵𑍇𑌨𑌰𑌾𑌨𑍍𑌧𑌾𑌦𑌿𑌹𑍋।।20।।
𑌤𑌿𑌕𑌿𑌕𑍁𑌕𑌿𑌕𑍍𑌯𑌚
