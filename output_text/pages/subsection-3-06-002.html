2 ।𑌆𑌤𑍇𑌰𑌚𑍇 ।

𑌮𑍋𑌷𑍁𑌤𑍍𑌵𑌾𑌵𑌾।𑌘𑌾𑌤𑌾𑌶𑍍𑌪𑌾𑌨𑌾।𑌆𑌰𑍇𑌅𑌸𑍍𑌮𑌿
𑌤𑍀𑌪𑌣𑌫𑌪𑍍𑌲𑌚𑌶𑌕
𑌨𑍍𑌨𑌿𑌰𑍀𑌰𑌮𑌨𑍍𑌨𑌾𑌰𑌾𑌕𑍍𑌤𑌾𑌚𑌾।𑌸𑌾𑌧𑌮𑌾𑌦𑌾𑌨𑍍𑌨𑌾𑌯𑌗𑌹𑍍𑌯𑌾𑌇𑌹𑌵𑌾
𑌕𑍃𑌚𑌯𑌟𑍍𑌯𑌟𑍀𑌕𑌿
𑌸𑌾𑌨𑍍।𑌊𑌪𑌾𑌶𑍍𑌰𑍂𑌧𑍀𑌇ழா𑌹𑌭𑌾।𑌓𑌇ழா𑌹।।3।।
𑌟𑍁𑌕𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯𑌚𑌣𑌪𑌶𑍍𑌯
𑌮𑍋𑌷𑍁𑌤𑍍𑌵𑌾𑌵𑌾𑌘𑌾𑌤𑌾𑌚𑌨𑌾𑌏।𑌆𑌰𑍇𑌅𑌸𑍍𑌮𑌿𑌨𑍍𑌨𑌿𑌰𑍀𑌰𑌮𑌨𑍍
𑌤𑍢𑌟𑍃
𑌹𑍋𑌵𑌾𑌉𑌉𑌵𑌾𑌊।𑌆𑌰𑌾𑌕𑍍𑌤𑌾𑌚𑌾।𑌸𑌾𑌧𑌮𑌾𑌦𑌾𑌂।𑌪𑍍𑌰।
𑌕𑌥𑌟𑌿𑌟𑌚𑌯𑌟𑍍𑌯𑌟𑍀
𑌨𑌾𑌯𑌾𑌗𑌾𑌹𑍀।𑌆𑌇𑌹𑌵𑌾𑌸𑌨𑍍𑌨𑌾𑌉𑌵𑍋𑌵𑌾।
𑌚𑌿𑌣𑌕𑍁𑌚𑌿𑌣
𑌊𑌪𑌶𑍍𑌰𑍂𑌧𑍀।𑌓𑌇ழா𑌹।।4।।
𑌟𑌿𑌚𑌣𑌪𑌶𑍍𑌯
