।𑌦𑌾𑌶𑌶𑍀𑌷𑍇𑌰𑌚𑍇 ।

11 ।𑌦𑌾𑌶𑌶𑍀𑌷𑍇𑌰𑌚𑍇 ।

𑌧𑌕𑍍𑌤𑌰𑌾𑌦𑌾𑌇𑌵𑌾।𑌅𑌵𑌾𑌏।𑌪𑍍𑌰।𑌪𑌵𑌤𑍇𑌕𑍃𑌤𑍍𑌵𑌿𑌯𑍋
𑌕𑌿𑌨𑍍𑌦𑌿𑌚𑌿𑌕
𑌰𑌾𑌯𑌾।𑌪𑍍𑌰।𑌦𑌕𑍍𑌷𑍋𑌦𑌾𑌇𑌵𑌾।𑌪𑍍𑌰।𑌨𑌾𑌮𑌨𑍁𑌮𑌾𑌦𑌿𑌯𑍋
𑌕𑌿𑌚𑌕𑌿𑌚𑍍𑌯𑌚𑌕𑌿𑌚
𑌨𑍃𑌭𑌾𑌇।𑌪𑍍𑌰।𑌹𑌾𑌰𑌾𑌇𑌸𑍍𑌸𑌾𑌰𑍍𑌜𑌾।𑌪𑍍𑌰।𑌨𑍋𑌅𑌤𑍇𑌯𑌾
𑌕𑌿𑌚𑌶𑌕𑍀𑌚𑌕𑌿𑌚
𑌨𑌾𑌸𑌾𑌕𑍍𑌵𑌾𑌭𑌾।𑌪𑍍𑌰।𑌵𑍃𑌥𑌾𑌪𑌾𑌜𑌾।𑌪𑍍𑌰।𑌸𑌿𑌕𑍃𑌣𑍁𑌷𑌾
𑌕𑌿𑌚𑌕𑌿𑌚𑌤𑍀
𑌇𑌨𑌾𑌦𑌾𑌇𑌷𑍂𑌵𑌾।𑌪𑍍𑌰।𑌹𑍋𑌇ழா𑌹।।19।।
𑌕𑍍𑌯𑌕𑌿𑌚𑌪𑍍𑌲𑌶𑍍𑌯
𑌧𑌕𑍍𑌤𑌰𑌾𑌭𑍁𑌹𑍋𑌹𑍋𑌹𑌾𑌇।𑌦𑍀𑌵ǐ𑌪𑌵𑌤𑌾𑌇𑌕𑍃𑌔
𑌤𑍀𑌤𑌤𑌶𑌚𑍁𑌟𑌿
𑌹𑍋𑌹𑌾𑌹𑌾𑌇।𑌪𑍍𑌰।𑌤𑍍𑌵𑌿𑌯𑍋𑌰𑌸𑍋𑌦𑌕𑍍𑌷𑍋𑌦𑌾
𑌤𑌤𑌤𑌶𑌚𑍍𑌯𑌚𑍍𑌯𑌚𑍍𑌯𑌥
𑌇𑌵𑌾𑌔।𑌪𑍍𑌰।𑌨𑌾𑌮𑌾𑌨𑍁𑌮𑌾𑌦𑍀𑌯𑍋𑌨𑍃𑌭𑌾𑌇𑌰𑍍𑌹𑌾𑌰𑌾
𑌟𑌿𑌕𑍀𑌕𑍀𑌚𑌿
𑌇𑌸𑍍𑌸𑌾𑌜𑌔।𑌪𑍍𑌰।𑌨𑍋𑌅𑌤𑍇𑌯𑌾𑌨𑌾𑌸𑌕𑍍𑌵𑌾𑌭𑌾𑌇𑌵𑍃𑌧𑌾
𑌟𑍀𑌕𑌿𑌕𑍀𑌚𑌿
𑌪𑌾𑌜𑌾𑌔।𑌪𑍍𑌰।𑌸𑌿𑌕𑍃𑌣𑍂𑌷𑌾𑌔।𑌪𑍍𑌰।𑌨𑌦𑍀𑌷𑍂
𑌥𑌟𑍍𑌯𑌚𑍍𑌯𑌥𑌟𑍍𑌯𑌕𑌿
𑌵𑌾𑌔𑌹𑍋𑌹𑌾𑌹𑌾𑌔𑌹𑍋𑌵𑌾।𑌏𑌨𑌾𑌦𑍀𑌷𑍂𑌵𑌾
𑌕𑌟𑌤𑌤𑌚𑌶𑌿𑌤𑌚𑌕𑍍𑌯𑌟𑌚
