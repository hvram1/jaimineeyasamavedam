।𑌵𑍇𑌨𑍋𑌵𑌿𑌶𑌾𑌲𑍇𑌚𑍇…

5 ।𑌵𑍇𑌨𑍋𑌵𑌿𑌶𑌾𑌲𑍇𑌚𑍇…

𑌹𑍋𑌇𑌯𑌾𑌈𑌹𑍋𑌇𑌯𑌾𑌔𑌹𑍋𑌇𑌯𑌾𑌤𑌿𑌸𑍇𑌰𑌾𑌵𑌾𑌚𑌾
𑌕𑍀𑌟𑍀𑌕𑌿𑌚
𑌈𑌰𑌾𑌯𑌾𑌤𑌿𑌪𑍍𑌰𑌵𑌾𑌨𑍍𑌹𑍀:।𑌓𑌹𑌾𑌹𑍋𑌹𑌾𑌓𑌹𑌾।𑌰𑌤𑌸𑍍𑌯
𑌟𑍁𑌕𑍍𑌯𑌚𑌿𑌪𑍍𑌲𑌣𑍍𑌯𑌫𑌚𑌶𑌿
𑌧𑌾𑌇𑌤𑍀𑌂𑌬𑍍𑌰𑌾𑌹𑍍𑌮𑌾𑌣𑍋𑌮𑌾𑌨𑍀𑌷𑌾𑌂।𑌆𑌹𑌾𑌹𑍋𑌇𑌆
𑌟𑍂𑌕𑌿𑌚𑍍𑌯𑌶𑌕𑍍𑌯𑌕𑌿
𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।𑌗𑌾𑌵𑍋𑌯𑌭3𑌇𑌗𑍋𑌪𑌾𑌤𑌿𑌂𑌪𑌾
𑌕𑍍𑌯𑌚𑌕𑌚𑌶𑌕𑌟𑍁𑌕𑌿
𑌚𑍍𑌛𑌰𑌾𑌮𑌾𑌨𑌾:।𑌇𑌹𑌾𑌹𑍋𑌇𑌈𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।
𑌚𑍍𑌯𑌪𑍍𑌲𑌕𑍍𑌯𑌕𑌿𑌕𑍍𑌯𑌚𑌕𑌚𑌶
𑌸𑍋𑌮𑌯𑍍𑌯𑌭3𑌇𑌮𑌾𑌤𑌯𑍋𑌵𑌾𑌵𑌾𑌶𑌾𑌨𑌾:।।13।।
𑌕𑌕𑍂𑌪𑌿𑌚𑍍𑌯𑌤𑍍𑌰
𑌹𑍋𑌇𑌹𑌾𑌈𑌹𑍋𑌇𑌹𑌾𑌔𑌹𑍋𑌇𑌹𑌾।𑌤𑌿𑌸𑍇𑌰𑌾𑌵𑌾
𑌕𑍀𑌟𑍀𑌕𑌿𑌚
𑌚𑌾𑌈𑌰𑌾𑌯𑌾𑌤𑌿𑌪𑍍𑌰𑌵𑌾𑌨𑍍𑌹𑍀:।𑌓𑌹𑌾𑌹𑍋𑌹𑌾𑌓𑌹𑌾।
𑌟𑍁𑌕𑍍𑌯𑌚𑌿𑌪𑍍𑌲𑌣𑍍𑌯𑌫𑌚𑌶𑌿
𑌰𑌤𑌸𑍍𑌯𑌧𑌾𑌇𑌤𑍀𑌂𑌬𑍍𑌰𑌾𑌹𑍍𑌮𑌾𑌣𑍋𑌮𑌾𑌨𑍀𑌷𑌾𑌂।𑌆𑌹𑌾
𑌟𑍂𑌕𑌿𑌚𑍍𑌯𑌶𑌕𑍍𑌯
𑌹𑍋𑌇𑌆𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।𑌗𑌾𑌵𑍋𑌯𑌭3𑌇𑌗𑍋
𑌕𑌿𑌕𑍍𑌯𑌚𑌕𑌚𑌶𑌕𑌟𑍁
𑌪𑌾𑌤𑌿𑌂𑌪𑌾𑌚𑍍𑌛𑌰𑌾𑌮𑌾𑌨𑌾:।𑌇𑌹𑌾𑌹𑍋𑌇𑌈𑌹𑌾𑌹𑌾𑌹𑌾
𑌕𑌿𑌚𑍍𑌯𑌪𑍍𑌲𑌕𑍍𑌯𑌕𑌿𑌕𑍍𑌯𑌚𑌕
𑌹𑍋𑌇।𑌸𑍋𑌮𑌯𑍍𑌯𑌭3𑌇𑌮𑌾𑌤𑌯𑍋𑌵𑌾𑌵𑌾𑌶𑌾𑌨𑌾:
𑌚𑌶𑌕𑌕𑍁𑌪𑌿𑌚𑍍𑌯𑌤𑍍𑌰
