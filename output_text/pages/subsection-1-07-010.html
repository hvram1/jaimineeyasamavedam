।𑌪𑍍𑌰𑌜𑌾𑌪𑌤𑍇𑌰𑌾𑌶𑌿𑌮𑌰𑌾𑌯𑍇𑌚𑍇𑌮𑌰𑌾𑌯𑌰𑌾𑌶𑌿𑌨𑌿𑌵𑌾।।

10 ।𑌪𑍍𑌰𑌜𑌾𑌪𑌤𑍇𑌰𑌾𑌶𑌿𑌮𑌰𑌾𑌯𑍇𑌚𑍇𑌮𑌰𑌾𑌯𑌰𑌾𑌶𑌿𑌨𑌿𑌵𑌾।।

𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌵𑌾𑌗𑍍𑌨𑍀𑌂।𑌨𑌰𑍋𑌨𑌰𑍋𑌨𑌰𑍋𑌦𑍀
𑌤𑍃𑌕𑍃
𑌧𑌿𑌤𑌿𑌭𑍀𑌰𑌰𑌾𑌣𑍇𑌯𑌾:।𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁
𑌕𑌿𑌤𑍍𑌯𑌚
𑌹𑌾𑌸𑍍𑌤𑌾𑌤𑍍।𑌚𑍍𑌯𑍁𑌤𑌾𑌨𑍍𑌚𑍁𑌤𑌾𑌨𑍍𑌚𑍁𑌤𑌾𑌨𑍍𑌜𑌨𑌾𑌯𑌾𑌤𑌪𑍍𑌰𑌾
𑌤𑍃𑌤𑌕𑍃𑌕𑍀
𑌶𑌾𑌸𑍍𑌤𑌂।𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌦𑍂𑌰𑌾𑌇।
𑌤𑌚𑌤𑍃𑌤𑌶
𑌦𑍃𑌶𑌾𑌨𑍍𑌦𑍃𑌶𑌾𑌨𑍍𑌦𑍃𑌶𑌾𑌨𑍍𑌗𑍃𑌹𑌾𑌪𑌤𑍀𑌮𑌾𑌥𑌾
𑌕𑍇𑌕𑍀𑌤
𑌵𑍍𑌯𑍂𑌂।𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌉𑌵𑌾।𑌈
𑌚𑌤𑍃𑌚
𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌵𑌾𑌗𑍍𑌨𑍀𑌂।𑌨𑌰𑍋𑌦𑍀𑌧𑍀𑌤𑌿𑌭𑍀𑌰𑌰𑌾
𑌤𑍃𑌤𑌿𑌕𑌿𑌤𑍍𑌯
𑌣𑍇𑌯𑌾𑌣𑍇𑌯𑌾𑌣𑍇𑌯𑌾:।𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁
𑌟𑌟𑌚
𑌹𑌾𑌸𑍍𑌤𑌾𑌤𑍍।𑌚𑍍𑌯𑍂𑌤𑌾𑌨𑍍𑌜𑌨𑌾𑌯𑌾𑌤𑌪𑍍𑌰𑌾𑌶𑌾𑌸𑍍𑌤𑌂𑌸𑍍𑌤𑌂
𑌤𑍃𑌤𑌚𑍍𑌯𑌕𑌶𑍀𑌤𑌟𑌟
𑌸𑍍𑌤𑌂।𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌦𑍂𑌰𑌾𑌇।𑌦𑍃𑌶𑌾𑌨𑍍
𑌚𑌤𑍃𑌤𑍍𑌰𑌶
𑌗𑍃𑌹𑌾𑌪𑌤𑍀𑌮𑌾𑌥𑌾𑌵𑍍𑌯𑍂𑌂𑌵𑍍𑌯𑍂𑌂𑌵𑍍𑌯𑍂𑌂।
𑌕𑍀𑌕𑍀𑌤𑌟𑌟𑌚
