।𑌗𑍗𑌤𑌮𑌸𑍍𑌯𑌮𑌨𑌾𑌰𑍍𑌯𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

9 ।𑌗𑍗𑌤𑌮𑌸𑍍𑌯𑌮𑌨𑌾𑌰𑍍𑌯𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌕𑍍𑌵𑍇𑌯𑌥𑌾।𑌕𑍍𑌵𑍇𑌦𑌸𑌾𑌔𑌹𑍋𑌵𑌾𑌔𑌹𑍋𑌔
𑌤𑌿𑌕𑌟𑍍𑌯𑌕𑌿𑌕𑌿
𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌪𑍁𑌰𑍂𑌤𑍍𑌰𑌾𑌚𑌾𑌇𑌡𑌿𑌤𑍇𑌮𑌨𑌾।𑌪𑍍𑌰।
𑌚𑌶𑌕𑍍𑌯𑌕𑍍𑌯𑌕𑌿𑌟𑍍𑌯
𑌆𑌲𑌾𑌰𑍍𑌷𑍀𑌯𑍂𑌧𑍍𑌮𑌾𑌚𑌜𑌕𑍍𑌰𑌤𑍍।𑌪𑍍𑌰।𑌪𑍁𑌰𑌨𑍍𑌦𑌰𑌾𑌪𑍍𑌰𑌗𑌾
𑌕𑍍𑌯𑌕𑌿𑌶𑌿𑌕𑍍𑌯𑌕𑍀
𑌯𑌾𑌤𑍍𑌰𑌾।𑌪𑍍𑌰।𑌅𑌗𑍍𑌨𑌾𑌸𑌾𑌇𑌷𑍂𑌔𑌹𑍋𑌵𑌾।
𑌟𑍍𑌯𑌤𑍍𑌯𑌟𑌚𑍍𑌯𑌶𑌿
𑌪𑍍𑌸𑍂𑌶𑌂𑌸𑌾𑌃।।12।।
𑌟𑍍𑌯𑌚
𑌕𑍂𑌵𑌾𑌕𑍂𑌵𑌾।𑌈𑌯𑌾𑌥𑌾𑌕𑍍𑌵𑍇𑌦𑌸𑌾𑌉𑌵𑌾𑌔𑌹𑍋
𑌤𑍀𑌕𑍀𑌶𑍍𑌯𑌟𑌿𑌤
।𑌪𑍍𑌰।𑌪𑍁𑌰𑍂𑌤𑍍𑌰𑌾𑌚𑌾𑌇𑌡𑌿𑌤𑍇𑌮𑌨𑌾।𑌪𑍍𑌰।𑌆𑌲𑌾
𑌕𑍀𑌕𑌿𑌕𑍍𑌯𑌕𑍍𑌯
𑌰𑍍𑌷𑌿𑌯𑍂𑌧𑍍𑌮𑌾𑌚𑌜𑌕𑍍𑌰𑌤𑍍।𑌪𑍍𑌰।𑌪𑍁𑌰𑌨𑍍𑌦𑌰𑌾𑌪𑍍𑌰𑌗𑌾𑌯𑌾𑌤𑍍𑌰𑌾
𑌕𑌿𑌶𑌿𑌕𑍍𑌯𑌕𑍀𑌟𑍍𑌯
।𑌪𑍍𑌰।𑌅𑌗𑌾𑌸𑌾𑌇𑌷𑍂𑌔𑌹𑍋𑌵𑌾।𑌪𑍍𑌸𑍂𑌶𑌂𑌸𑌾𑌃
𑌤𑍍𑌯𑌟𑌚𑍍𑌯𑌶𑌿𑌤𑌟𑌚
𑌕𑍍𑌵𑍇𑌯𑌥𑌾।𑌕𑍂𑌵𑌾𑌇𑌦𑌾𑌸𑍀।𑌪𑍁𑌰𑍂𑌤𑍍𑌰𑌾𑌚𑍀𑌡𑌿𑌤𑌾𑌇𑌮𑌾
𑌤𑌿𑌪𑍀𑌣𑌤𑍀𑌚𑍍𑌯𑌪
𑌨𑌾𑌃।𑌆𑌲𑌾𑌰𑍍𑌷𑍀𑌯𑍂𑌧𑍍𑌮𑌾𑌚𑌜𑌕𑍍𑌰𑌡𑌾𑌉𑌵𑌾।𑌪𑍂𑌰𑌨𑍍𑌦𑌾𑌰𑌾।
𑌣𑌕𑍁𑌟𑌿𑌤𑍍𑌯𑌪𑌿𑌣
𑌪𑍍𑌰𑌗𑌾𑌯𑌾𑌤𑍍𑌰𑌾𑌅𑌗𑌾𑌸𑌾𑌇𑌷𑍂𑌔𑌹𑍋𑌵𑌾।
𑌕𑍍𑌯𑌟𑍍𑌯𑌚𑌕𑍍𑌯𑌟𑌚𑍍𑌯𑌶𑌿
