।𑌵𑌾𑌮𑍍𑌮𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

5 ।𑌵𑌾𑌮𑍍𑌮𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌆𑌮𑌨𑍍𑌦𑍈𑌰𑌰𑌾𑌇𑌨𑍍𑌦𑌰𑌹𑌰𑌿𑌭𑌾𑌇𑌃।𑌯𑌾𑌹𑍀𑌮𑌯𑍂𑌰𑌾𑌰𑍋𑌮𑌾
𑌤𑍀𑌤𑍁𑌶𑌕𑌕𑌿𑌚𑌕𑌕
𑌭𑌾𑌇𑌰𑍍𑌮𑍍𑌮𑌾𑌤𑍍𑌵𑌾𑌕𑌾𑌇𑌚𑍀𑌤𑍍।𑌨𑍀𑌯𑍇𑌮𑍂𑌰𑍍𑌵𑍀𑌂।𑌨𑌾𑌪𑌾𑌶𑌿
𑌪𑌶𑌕𑌿𑌶𑍍𑌰𑌯𑌚𑌿𑌶𑌕
𑌨𑍋𑌤𑍀𑌧𑌨𑍍𑌵𑍇𑌵𑌾𑌤𑌂𑌆𑌇𑌹𑌾𑌔𑌹𑍋𑌵𑌾।𑌵𑌾𑌯𑌾𑌃
𑌚𑌿𑌟𑌿𑌚𑌕𑌟𑌟𑌚𑍍𑌯𑌶𑌿𑌚𑌪𑍍𑌲
𑌆𑌮𑌨𑍍𑌦𑍈𑌰𑌰𑌿𑌨𑍍𑌦𑌰𑌹𑌾𑌰𑌿𑌭𑌾𑌇𑌃।𑌯𑌾𑌹𑍀𑌮𑌯𑍂𑌰𑌾𑌰𑍋𑌮𑌭𑌾
𑌤𑍂𑌤𑍍𑌯𑌶𑌕𑌚𑌿𑌕
𑌰𑍍𑌮𑍍𑌮𑌾𑌤𑍍𑌵𑌾𑌕𑌾𑌇𑌚𑍀𑌤𑍍।𑌨𑍀𑌯𑍇𑌮𑍁𑌰𑍍𑌵𑍀𑌨𑍍𑌨𑌾𑌪𑌾𑌶𑌾𑌇𑌨𑌾𑌃।
𑌟𑍃𑌤𑍍𑌯𑌕𑍍𑌯𑌕𑍍𑌯𑌕𑌟𑍍𑌯𑌤𑍍𑌯
𑌆𑌤𑌾𑌇𑌧𑌾𑌨𑍍𑌵𑍇।𑌵𑌾𑌤𑌂𑌆𑌇𑌹𑌾𑌔𑌹𑍋𑌵𑌾।
𑌟𑍀𑌤𑌚𑌕𑌟𑌟𑌚𑍍𑌯𑌶𑌿
𑌵𑌯𑍋𑌭𑍀𑌃।।9।।
𑌤𑍍𑌯𑌟𑌚
𑌆𑌮𑌨𑍍𑌦𑍈𑌰𑌰𑌿𑌨𑍍𑌦𑌰𑌹𑌾𑌰𑍀𑌭𑍀𑌃।𑌯𑌾𑌹𑌿𑌮𑌯𑍂𑌰𑌾𑌰𑍋𑌮𑌭𑍀𑌰𑌾
𑌤𑍂𑌤𑍍𑌯𑌕𑌚𑌿𑌕
𑌉𑌵𑌾।𑌮𑌾𑌤𑍍𑌵𑌾𑌕𑍇𑌚𑌿𑌨𑍍𑌨𑌿𑌯𑍇𑌮𑍁𑌰𑍍𑌵𑌿𑌨𑍍𑌨𑌾𑌪𑌾𑌶𑌿𑌨𑍋
𑌶𑍍𑌰𑍂𑌟𑍍𑌯𑌚𑌕𑌕𑍁
𑌉𑌵𑌾।𑌆𑌤𑍀𑌧𑌨𑍍𑌵𑍇𑌵𑌤𑌂𑌆𑌉𑌵𑌾।𑌈𑌹𑌿।।10।।
𑌟𑍂𑌟𑍍𑌯𑌚𑍍𑌯𑌤𑍍𑌯𑌟𑌿𑌚𑌶
