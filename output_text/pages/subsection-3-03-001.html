।𑌹𑌰𑌾𑌯𑌣𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

1 ।𑌹𑌰𑌾𑌯𑌣𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌶𑌗𑍍𑌧𑍂𑌷𑍂।𑌶𑌾𑌚𑌾𑌇𑌪𑌾𑌤𑌾।𑌈𑌨𑍍𑌦𑌰𑌾।𑌵𑌿𑌶𑍍𑌪𑌾𑌭𑍀𑌰𑍂𑌤𑍀
𑌤𑌿𑌚𑍀𑌕𑌚𑌶𑌥𑍍𑌯𑌚
𑌭𑌾𑌇𑌰𑍍𑌭𑌾𑌗𑌂।𑌨𑌹𑌿𑌤𑍍𑌵𑌾𑌯𑌾𑌶𑌾𑌸𑌂𑌵𑌾𑌸𑍂𑌵𑌾𑌇𑌦𑌂।
𑌕𑍀𑌚𑍍𑌯𑌶𑌚𑍀𑌚𑍍𑌯𑌕𑍍𑌯𑌚𑌶𑍍𑌯
𑌆𑌨𑍂𑌶𑍂𑌰𑌾𑌔𑌹𑍋𑌵𑌾।𑌚𑌰𑌾𑌮𑌸𑍀।।1।।
𑌟𑌿𑌟𑌚𑌶𑌿𑌟𑍍𑌯𑌚𑍍𑌯
𑌶𑌗𑍍𑌧𑍂𑌷𑍍𑌵𑍗𑌹𑍋𑌶𑌚𑍀𑌪𑌤𑌾𑌇।𑌆𑌇𑌨𑍍𑌦𑌰𑌵𑌿𑌶𑍍𑌪𑌾𑌭𑍀𑌰𑍂𑌤𑍀𑌭𑍀
𑌫𑍀𑌕𑍀𑌶
𑌰𑍍𑌭𑌗𑌾𑌨𑍍𑌨𑌾𑌹𑍀।𑌤𑍍𑌵𑌾𑌯𑌾𑌶𑌾𑌸𑌂𑌵𑌾𑌸𑍂𑌹𑌾𑌇𑌵𑌾𑌇𑌦𑌾𑌂।
𑌟𑍋𑌤𑌚𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯𑌤𑌟𑍀
𑌆𑌨𑍂𑌶𑍂𑌰𑌚𑌰𑌾𑌉𑌵𑌾𑌓𑌬𑌾𑌮𑌾𑌸𑍋𑌹𑌾𑌇।।2।।
𑌷𑌕𑍁𑌪𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌶𑌗𑍍𑌧𑍂𑌷𑍁𑌶𑌾𑌚𑍀𑌪𑌤𑌾𑌇𑌶𑌗𑍍𑌧𑍁𑌷𑍁।𑌶𑌾𑌚𑌾𑌇𑌪𑌾𑌤𑌾𑌇
𑌫𑍁𑌚𑍍𑌯𑌶𑍀𑌟𑍁
𑌇𑌹𑌿𑌮𑌾𑌹𑌾𑌇।𑌪𑍍𑌰।𑌆𑌇𑌨𑍍𑌦𑌰𑌵𑌿𑌶𑍍𑌪𑌾𑌭𑍀𑌰𑍂𑌤𑍀𑌭𑍀𑌰𑌾।𑌪𑍍𑌰।
𑌚𑌿𑌣𑌶𑌕𑍁𑌚𑌕𑍍𑌯𑌟𑌿
𑌭𑌾𑌗𑌨𑍍𑌨𑌹𑍀𑌤𑍍𑌵𑌾𑌯𑌾𑌶𑌾𑌸𑌂𑌵𑌾𑌸𑍂𑌵𑍀𑌦𑌾𑌮𑌾।𑌆𑌨𑍂𑌶𑍂
𑌕𑍃𑌚𑌕𑍍𑌯𑌟𑌿𑌚𑍍𑌯𑌯
𑌰𑌾।𑌪𑍍𑌰।𑌚𑌾𑌰𑌾𑌮𑌾𑌸𑌾𑌇।𑌓𑌇ழா𑌹।।3।।
𑌟𑌟𑌿𑌚𑌣𑌶𑌪𑌶𑍍𑌯
