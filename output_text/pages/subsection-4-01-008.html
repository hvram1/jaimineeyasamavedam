।𑌵𑌾𑌤𑍍𑌸𑌪𑍍𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

7 ।𑌵𑌾𑌤𑍍𑌸𑌪𑍍𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌹𑍋𑌯𑍇𑌹𑍋𑌯𑍇𑌹𑍋𑌯𑍇।𑌜𑌗𑍃𑌹𑍍𑌮𑌾𑌤𑍇𑌦𑌕𑍍𑌷𑌿𑌣𑌾𑌮𑌿
𑌟𑍍𑌯𑌟𑍍𑌯𑌟𑍍𑌯𑌥𑍀𑌕𑍀𑌚
𑌨𑍍𑌦𑌰𑌾𑌹𑌾𑌸𑍍𑌤𑌾𑌂𑌹𑌾𑌸𑍍𑌤𑌾𑌂𑌹𑌾𑌸𑍍𑌤𑌾𑌂।𑌵𑌸𑍂𑌯𑌵𑍋𑌵𑌸𑍁𑌪
𑌕𑌟𑍍𑌯𑌟𑍍𑌯𑌟𑍍𑌯
𑌤𑍇𑌵𑌾𑌸𑍂𑌨𑌾𑌂।𑌤𑍍𑌰𑌿𑌃।𑌵𑌿𑌦𑍍𑌮𑌾𑌹𑌿𑌤𑍍𑌵𑌾𑌗𑍋𑌪𑌤𑌿𑌂𑌶𑍂
𑌕𑍄𑌚𑌕𑌟𑍍𑌯𑌕𑍀𑌚𑌕𑍀𑌚
𑌰𑌾𑌗𑍋𑌨𑌾𑌂।𑌤𑍍𑌰𑌿𑌃।𑌅𑌸𑍍𑌮𑌭𑍍𑌯𑌞𑍍𑌚𑌭𑍍𑌯𑌤𑍍𑌰𑌂𑌵𑍍𑌰𑌷𑌣𑌂𑌰𑌾𑌯𑌾𑌇
𑌕𑌟𑍍𑌯𑌕𑌿𑌕𑍁𑌚𑌕
𑌨𑍍𑌦𑌾𑌆𑌇𑌨𑍍𑌦𑌾𑌆𑌇𑌨𑍍𑌦𑌾𑌃।𑌹𑍋𑌯𑍇𑌹𑍋𑌯𑍇𑌹𑍋𑌯𑌾𑌵𑌾
𑌟𑌿𑌟𑌿𑌟𑌿𑌟𑍍𑌯𑌟𑍍𑌯𑌟𑍍𑌯𑌟𑌚
𑌔𑌹𑍋𑌵𑌾।𑌈।।11।।
𑌶𑌿𑌚
𑌹𑌾𑌬𑍁𑌹𑌬𑍁𑌹𑌾𑌬𑍁।𑌓𑌹𑍋𑌹𑍋𑌵𑌾।𑌤𑍍𑌰𑌿𑌃।𑌜𑌗𑍃
𑌤𑍁𑌶𑌕𑍀
𑌹𑍍𑌮𑌾𑌤𑌾𑌇𑌦𑌾𑌕𑍍𑌷𑍀𑌣𑌾𑌮𑌿𑌨𑍍𑌦𑍃𑌹𑌸𑍍𑌤𑌨𑍍𑌦𑌰𑌹𑌸𑍍𑌤𑌨𑍍𑌦𑌰𑌹𑌸𑍍𑌤𑌂।𑌵𑌸𑍂
𑌕𑌿𑌕𑌿𑌚𑌕𑌚𑌶𑍇𑌕𑍍𑌯
𑌯𑌵𑍋𑌵𑌾𑌸𑍂𑌪𑌾𑌤𑌾𑌇𑌵𑌸𑍂𑌨𑌾𑌂𑌵𑌸𑍂𑌨𑌾𑌂𑌵𑌸𑍂
𑌟𑌿𑌚𑌕𑌚
𑌨𑌾𑌂।𑌵𑌿𑌦𑍍𑌮𑌾𑌹𑌿𑌤𑍍𑌵𑌾𑌗𑍋𑌪𑌾𑌤𑍀𑌂𑌶𑍂𑌰𑌗𑍋𑌨𑌾𑌂𑌰𑌗𑍋𑌨𑌾𑌂
𑌶𑍈𑌕𑍁𑌚𑌕𑌚
𑌰𑌗𑍋𑌨𑌾𑌂।𑌅𑌸𑍍𑌮𑌭𑍍𑌯𑌨𑍍𑌚𑌾𑌇𑌤𑍍𑌰𑌾𑌂𑌵𑌾𑌰𑍍𑌷𑌾𑌣𑌂𑌰𑌯𑌿𑌨𑍍𑌦𑌾
𑌶𑍇𑌟𑍂𑌚𑌚𑍍𑌯𑌚
𑌰𑌯𑌿𑌨𑍍𑌦𑌾𑌰𑌯𑌿𑌨𑍍𑌦𑌾𑌃।𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁।𑌓𑌹𑍋𑌹𑍋
𑌶𑍢𑌤𑍁𑌶
𑌵𑌾।𑌚𑌿𑌃।𑌓𑌹𑍋𑌹𑍋𑌵𑌾𑌔𑌹𑍋𑌵𑌾।𑌈।।12।।
𑌕𑌿𑌟𑌿𑌟𑌚𑌶𑌿𑌚
𑌆𑌬𑍁𑌹𑍋𑌆𑌬𑍁𑌹𑍋𑌆𑌬𑍁𑌹𑍋।𑌵𑌾𑌔𑌹𑍋𑌔
𑌤𑍢
𑌹𑍋𑌵𑌾।𑌜𑌗𑍍𑌰𑌹𑍍𑌮𑌾𑌤𑌾𑌇𑌦𑌾𑌕𑍍𑌷𑍀𑌣𑌾𑌮𑌿𑌨𑍍𑌦𑌰𑌹𑌸𑍍𑌤𑌨𑍍𑌦𑌰
𑌤𑍁𑌤𑌕𑌿𑌕𑌿𑌚𑌕𑌚
𑌹𑌸𑍍𑌤𑌨𑍍𑌦𑌰𑌹𑌸𑍍𑌤𑌂।𑌵𑌸𑍂𑌯𑌵𑍋𑌵𑌾𑌸𑍂𑌪𑌾𑌤𑌾𑌇𑌵𑌸𑍂
𑌶𑍍𑌰𑍇𑌕𑍍𑌯𑌕𑌿𑌚𑌕𑌚
𑌨𑌾𑌂।𑌤𑍍𑌰𑌿𑌃।𑌵𑌿𑌦𑍍𑌮𑌾𑌹𑌿𑌤𑍍𑌵𑌾𑌗𑍋𑌪𑌾𑌤𑍀𑌂𑌶𑍂𑌰𑌗𑍋𑌨𑌾𑌂
𑌶𑍍𑌰𑍁𑌕𑍁𑌚𑌕𑌚𑌶𑍍𑌰𑌿
।𑌤𑍍𑌰𑌿𑌃।𑌅𑌸𑍍𑌮𑌭𑍍𑌯𑌨𑍍𑌚𑌾𑌇𑌤𑍍𑌰𑌾𑌵𑌾𑌰𑍍𑌷𑌾𑌣𑌂𑌰𑌯𑌿𑌨𑍍𑌦𑌾𑌃।𑌤𑍍𑌰𑌿𑌃।
𑌟𑍂𑌕𑌕𑌚𑌶𑍁
𑌆𑌬𑍁𑌹𑍋𑌆𑌬𑍁𑌹𑍋𑌆𑌬𑍁𑌹𑍋।𑌵𑌾𑌔𑌹𑍋𑌔
𑌤𑍢𑌟𑍍𑌯𑌟𑍍𑌯
𑌹𑍋𑌵𑌾𑌔𑌹𑍋𑌵𑌾।𑌈।।13।।
𑌟𑌟𑌚𑌶𑌿𑌚
