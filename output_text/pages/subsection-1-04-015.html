।𑌦𑍈𑌰𑍍𑌘𑌶𑍍𑌰𑌵𑌸𑍇𑌚𑍇 ।।

15 ।𑌦𑍈𑌰𑍍𑌘𑌶𑍍𑌰𑌵𑌸𑍇𑌚𑍇 ।।

𑌯𑍋𑌵𑌿𑌶𑍍𑌪𑌾𑌦𑌯𑌤𑍇𑌵𑌸𑍍𑌵𑍋𑌹𑌾𑌓𑌹𑌾𑌏।
𑌤𑍗𑌤
𑌹𑍋𑌤𑌾𑌮𑌨𑍍𑌦𑍇𑌰𑌾𑌜𑌨𑌾𑌨𑌾𑌮𑍋𑌹𑌾𑌓𑌹𑌾𑌏।
𑌟𑍍𑌯𑌕𑌕𑌿𑌟𑍍𑌯𑌤𑌟𑌚𑍍𑌯
𑌮𑌧𑍋𑌰𑍍𑌨𑌪𑌾।𑌤𑍍𑌰𑌾𑌪𑍍𑌰𑌾𑌥𑌮𑌾𑌨𑍍𑌨𑌾𑌯𑌾𑌸𑍍𑌮𑌾𑌓
𑌚𑍍𑌯𑌤𑍍𑌯𑌕𑍍𑌯𑌕𑍍𑌯𑌚𑌕𑍍𑌯𑌟
𑌹𑌾𑌓𑌹𑌾𑌏𑌪𑍍𑌰𑌸𑍍𑌤𑍋𑌮𑌾𑌯𑌾𑌭𑍂𑌵𑍋𑌬𑌾𑌗𑍍𑌨𑌾
𑌤𑌟𑌚𑍍𑌯𑌚𑍍𑌯𑌤𑍍𑌯𑌚𑌕𑌪𑌪𑍍𑌲
𑌯𑍋𑌹𑌾𑌇
𑌫𑍍𑌲𑌶𑍍𑌯
𑌯𑍋𑌵𑌿𑌶𑍍𑌪𑌾𑌦𑌯𑌤𑍇𑌵𑌸𑍂𑌏।𑌹𑍋𑌤𑌾
𑌤𑍂𑌤𑌚𑍍𑌯
𑌮𑌨𑍍𑌦𑍇𑌰𑌾𑌜𑌨𑌾𑌨𑌾𑌮𑍍𑌮𑌾𑌧𑍋𑌰𑍍𑌨𑌾𑌪𑍗𑌵𑌾।
𑌚𑌕𑌚𑌤𑌤
𑌤𑍍𑌰𑌾𑌪𑍍𑌰𑌥𑌮𑌾𑌨𑍍𑌯𑌸𑍍𑌮𑍈।𑌪𑍍𑌰𑌾𑌸𑍍𑌤𑍋𑌮𑌾𑌯𑍗𑌵𑌾।
𑌚𑍂𑌕𑌤𑌤𑌤
𑌤𑍍𑌵𑌗𑍍𑌨𑌾𑌯𑌾𑌇ழா𑌭𑌾।𑌓𑌇ழா
𑌚𑍍𑌯𑌟𑌿𑌚𑌣𑌪𑌶𑍍𑌯
