।𑌵𑌾𑌸𑌿𑌷𑍍𑌠𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 𑌭𑌾𑌰𑍍𑌗𑌵𑌂 𑌵𑌾 𑌤𑍍𑌰𑌤𑍀𑌯𑌂 ।

7 ।𑌵𑌾𑌸𑌿𑌷𑍍𑌠𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 𑌭𑌾𑌰𑍍𑌗𑌵𑌂 𑌵𑌾 𑌤𑍍𑌰𑌤𑍀𑌯𑌂 ।

𑌇𑌨𑍍𑌦𑌰𑌾𑌔𑌹𑍋।𑌕𑍍𑌰𑌤𑍁𑌨𑍍𑌨𑌾𑌆𑌭𑌾𑌰𑌾।𑌪𑌿𑌤𑌾𑌔
𑌭𑌿𑌤𑌚𑍍𑌯𑌕𑍍𑌯𑌯𑌟𑌭𑌿
𑌹𑍋।𑌪𑍁𑌤𑍇𑌰𑌭𑍀𑌯𑍋𑌯𑌾𑌥𑌾।𑌶𑌿𑌕𑍍𑌷𑌾𑌔𑌹𑍋।
𑌤𑌥𑍍𑌯𑌚𑌕𑍍𑌯𑌯𑌟𑌭𑌿𑌤
𑌣𑍋𑌅𑌸𑍍𑌮𑌿𑌨𑍍𑌪𑍁𑌰𑍂𑌹𑍂𑌤𑌾𑌯𑌾𑌮𑌾𑌨𑍀।𑌜𑍀𑌵𑌾𑌜𑍇𑌯𑌾𑌤𑌾
𑌕𑍂𑌕𑍍𑌯𑌯𑌟𑌟𑌿𑌚
𑌔𑌹𑍋𑌵𑌾।𑌅𑌶𑍀𑌮𑌹𑍀।।।16।।
𑌶𑌿𑌚𑍀
𑌇𑌨𑍍𑌦𑌰𑌕𑍍𑌰𑌤𑍁𑌨𑍍𑌨𑌆𑌭𑌰𑌾।𑌪𑌿𑌤𑌾𑌪𑍁𑌤𑍇𑌰𑌭𑍀𑌯𑍋𑌯𑌥𑌾𑌶𑌾
𑌫𑍀𑌕𑍀𑌕𑍀
𑌇𑌕𑍍𑌷𑌾𑌣𑍋𑌆।𑌸𑍍𑌮𑌾𑌇𑌨𑍍𑌪𑍁𑌰𑍁𑌹𑍂𑌤𑌾𑌯𑌾𑌮𑌾𑌨𑍀।
𑌟𑍃𑌤𑌕𑌕𑍀𑌚𑌯𑌟𑍍𑌯
𑌔𑌹𑍗𑌹𑍂𑌵𑌾𑌔𑌹𑍋𑌵𑌾।𑌜𑍀𑌵𑌾𑌜𑍇𑌯𑌾𑌤𑍀𑌃।𑌅𑌶𑍀
𑌟𑌟𑌕𑌿𑌤𑌤𑌟𑌿𑌤
𑌮𑌾𑌹𑌾𑌇।𑌓𑌇ழா𑌹।।।17।।
𑌟𑌿𑌚𑌣𑌪𑌪𑍍𑌲
𑌇𑌨𑍍𑌤𑍍𑌰𑌕𑍍𑌰𑌤𑍁𑌨𑍍𑌨𑌆𑌭𑌰𑌾𑌉𑌵𑍋𑌵𑌾।𑌪𑌿𑌤𑌾𑌪𑍁𑌤𑍇𑌰
𑌫𑍂𑌚𑍀𑌶𑌚𑍀
𑌭𑍀𑌯𑍋𑌯𑌾𑌥𑌾।𑌹𑌾𑌪𑌿𑌤𑌾𑌪𑍁𑌤𑍇𑌰𑌭𑌿𑌯𑍋𑌯𑌾𑌥𑌾।
𑌚𑍍𑌯𑌪𑍍𑌲𑌤𑌚𑍂𑌕𑌪𑍍𑌲
𑌹𑌾𑌬𑍁𑌶𑌾𑌇𑌕𑍍𑌷𑌾𑌣𑍋𑌆𑌸𑍍𑌮𑌾𑌇𑌨𑍍𑌪𑍁𑌰𑍂𑌹𑍂𑌤𑌯𑌾
𑌶𑍃𑌕𑍍𑌯𑌕𑌿𑌚𑍍𑌯
𑌮𑌾𑌨𑍀𑌹𑌾𑌬𑍁𑌜𑍀𑌵𑌾𑌜𑍇𑌯𑌾𑌤𑍀𑌃।𑌆𑌶𑍋𑌵𑌾
𑌪𑍍𑌲𑌶𑍂𑌕𑌪
𑌮𑌾𑌹𑍋𑌹𑌾𑌇।।18।।
