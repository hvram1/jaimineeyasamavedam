।𑌸𑌨𑍍𑌨𑌤𑍇𑌤𑍍𑌵𑍇 ।।

3 ।𑌸𑌨𑍍𑌨𑌤𑍇𑌤𑍍𑌵𑍇 ।।

𑌅𑌭𑌿𑌪𑍍𑌰𑌵𑌾𑌃।𑌸𑍂𑌰𑌾𑌧𑌾𑌸𑌾𑌂।𑌇𑌨𑍍𑌦𑌰𑌮𑌰𑍍𑌚𑌯𑌾𑌥𑌾𑌵𑌿𑌦𑌾
𑌤𑍀𑌟𑌿𑌤𑌯𑍂𑌪𑍍𑌲
𑌇𑌯𑍋𑌜𑌾𑌰𑌿𑌤𑍃𑌭𑍇𑌯𑌾𑌮𑌘𑌵𑌾𑌪𑍂𑌰𑍋𑌵𑌾𑌸𑍂𑌃।
𑌚𑌿𑌤𑍍𑌯𑌯𑍂𑌟𑍍𑌯
𑌸𑌾𑌹𑌸𑍍𑌰𑌾𑌇𑌣𑌾𑌔𑌹𑍋𑌵𑌾।𑌵𑌾𑌶𑍀𑌕𑍍𑌷𑌾𑌤𑍀।।5।।
𑌟𑌿𑌟𑌚𑍍𑌯𑌶𑌿𑌟𑌿𑌚
𑌆𑌭𑌾𑌇𑌪𑍍𑌰𑌵𑌾𑌸𑍍𑌸𑍁𑌰𑌾𑌧𑌾𑌸𑌂।𑌇𑌨𑍍𑌦𑌰𑌾𑌮𑌰𑍍𑌚𑌾𑌯𑌾
𑌟𑍁𑌚𑌿𑌣𑌟𑍁𑌟
𑌥𑌾𑌔𑌹𑍋𑌵𑌾।𑌵𑍀𑌦𑍇𑌯𑍋𑌜𑌰𑌿𑌤𑍃𑌭𑍇𑌯𑌾𑌮𑌾𑌘𑌵𑌾
𑌚𑌶𑌿𑌚𑌶𑌚𑌕𑍍𑌯𑌕𑍍𑌯𑌚𑌟𑌿
𑌪𑍁𑌰𑍋𑌵𑌾𑌸𑍂𑌃।𑌸𑌹𑌾𑌶𑍍𑌰𑍇𑌣𑍇𑌵𑌾𑌶𑌾𑌇𑌕𑍍𑌷𑌾
𑌤𑍍𑌯𑌟𑍍𑌯𑌚𑌤𑍍𑌯𑌕𑌕𑍍𑌯𑌟𑌿𑌟
𑌤𑌾𑌔𑌹𑍋𑌵𑌾।𑌸𑍁𑌭𑍂𑌤𑌯𑍇।।6।।
𑌚𑌶𑌿𑌕𑍍𑌯𑌟𑍍𑌯𑌚
