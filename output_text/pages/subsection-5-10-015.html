।𑌸𑍋𑌮𑌸𑌾𑌮𑌾𑌨𑌿 𑌚𑌤𑍍𑌵𑌾𑌰𑌿 ।

15 ।𑌸𑍋𑌮𑌸𑌾𑌮𑌾𑌨𑌿 𑌚𑌤𑍍𑌵𑌾𑌰𑌿 ।

𑌪𑍍𑌰𑌾𑌪𑍁𑌨𑌾𑌨𑌾𑌯𑌵𑍇𑌧𑌸𑌾𑌇।𑌹𑍋𑌇𑌹𑍋𑌇𑌸𑍋𑌮𑌾
𑌪𑌿𑌶𑍂
𑌯𑌾𑌵𑌾𑌚𑌾𑌉𑌚𑍍𑌯𑌾𑌤𑌾𑌇।𑌭𑍃𑌤𑌿𑌨𑍍𑌨𑌭𑌰𑌾𑌮𑌾𑌤𑌿𑌭𑌾𑌇
𑌟𑍢𑌭𑌿𑌶𑌕𑍁𑌕𑌿𑌶
𑌰𑍍𑌜𑍂𑌜𑍋𑌷𑌾𑌤𑌾𑌇।𑌓𑌇ழா𑌹।।32।।
𑌟𑌿𑌚𑌣𑌶𑌪𑌶𑍍𑌯
𑌪𑍍𑌰𑌪𑍁𑌨𑌾𑌨𑍗𑌹𑍋𑌯𑌵𑍇𑌧𑌸𑌾𑌇।𑌸𑍋𑌮𑌾𑌔
𑌫𑍁𑌕𑍀𑌶𑌕𑌕𑍍𑌯
𑌹𑍋𑌯𑌵𑌚𑌉𑌚𑍍𑌯𑌤𑌾𑌇।𑌭𑍃𑌤𑌾𑌔𑌹𑍋𑌇𑌨𑍍𑌨𑌭𑌰𑌾
𑌚𑌶𑍃𑌕𑌿𑌚𑌪𑍍𑌲
𑌮𑌾𑌤𑌿𑌭𑌾𑌇𑌃।𑌜𑍁𑌜𑍋𑌆𑌉𑌵𑌾।𑌷𑌾𑌤𑍇।।33।।
𑌨𑍍𑌦𑌶𑌿𑌕𑍍𑌯𑌟𑌿𑌚𑌶
𑌗𑍋𑌮𑌨𑍍𑌨𑍋𑌹𑍋𑌇𑌨𑍍𑌦𑍋𑌅𑌶𑍍𑌪𑌵𑌾𑌦𑍇।𑌸𑍂𑌤𑌃𑌸𑍁𑌦𑌕𑍍𑌷
𑌫𑌿𑌪𑍍𑌲𑌚𑌿𑌪𑍍𑌲𑌶
𑌯𑌾𑌨𑍀𑌵𑌾𑌃।𑌶𑍁𑌚𑌿𑌞𑍍𑌚𑌵𑌾𑌰𑍍𑌣𑌾𑌮𑌧𑌿𑌗𑍋𑌔𑌹𑍋।
𑌯𑍂𑌶𑍍𑌯𑌚𑍀𑌣𑌫𑌪𑍍𑌲𑌚𑍀𑌤𑍍𑌯
𑌷𑍁𑌧𑌾𑌰𑌾𑌯𑌾𑌔𑌹𑍋𑌬𑌾।𑌹𑍋𑌇ழா𑌹।।14।।
𑌕𑍍𑌯𑌚𑌕𑌟𑍍𑌯𑌚𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌹𑌾𑌵𑌾𑌸𑍍𑌮𑌾।𑌭𑍍𑌯𑌾𑌨𑍍𑌤𑍍𑌵𑌾।𑌵𑌾𑌸𑍁𑌵𑌾𑌇𑌦𑌂।𑌅𑌭𑌾𑌇𑌵𑌾𑌣𑍀
𑌤𑌿𑌚𑌶𑌚𑌿𑌣𑍍𑌯𑌟𑍍𑌯𑌚𑍍𑌯𑌶
𑌰𑌨𑌾𑌉𑌵𑍋𑌵𑌾।𑌷𑌾𑌤𑌾।𑌗𑍋𑌭𑌿𑌷𑍍𑌟𑍇𑌵𑌰𑍍𑌣𑌾𑌮𑌾𑌭𑌿
𑌚𑍀𑌶𑌚𑌶𑌚𑌿𑌥𑍍𑌯𑌚
𑌵𑌾𑌸𑌯𑌾𑌮𑌸𑌿𑌹𑍋𑌇ழா𑌹।।35।।
𑌶𑍃𑌚𑌪𑍍𑌲
