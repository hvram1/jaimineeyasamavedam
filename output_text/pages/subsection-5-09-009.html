।𑌵𑌿𑌰𑍂𑌪𑌸𑍍𑌯𑌚𑌤𑌨𑍍𑌤𑍇𑌰 ।

9 ।𑌵𑌿𑌰𑍂𑌪𑌸𑍍𑌯𑌚𑌤𑌨𑍍𑌤𑍇𑌰 ।

𑌪𑍇𑌰𑌾𑌅𑌯𑌾𑌸𑌾𑌇𑌦𑌿𑌨𑍍𑌦𑍂𑌰𑌿𑌨𑍍𑌦𑌰𑌾𑌸𑍍𑌯𑌾𑌨𑌿𑌷𑍍𑌕𑍃𑌤𑌾𑌂।𑌸𑌾𑌚𑌾
𑌤𑌚𑌿𑌥𑍍𑌯𑌟𑍀𑌪𑌿𑌚𑍍𑌯
𑌸𑌾𑌚𑍍𑌯𑍁𑌨𑍍𑌨𑌰𑌪𑍍𑌰𑌾𑌮𑌿𑌨𑌾𑌤𑌿𑌸𑌂𑌗𑌾𑌇𑌰𑌂।𑌮𑌰𑍍𑌯𑌾𑌇𑌵𑌾
𑌕𑌫𑌚𑍍𑌯𑌕𑌫𑌚𑌿𑌶𑍍𑌯𑌥𑍍𑌯𑌕𑍍𑌯
𑌯𑍁𑌵𑌾𑌤𑌿𑌭𑌾𑌇𑌸𑍍𑌸𑌾𑌮𑌰𑍍𑌷𑌤𑌾𑌏𑌸𑍋𑌮𑌾𑌃𑌕𑌾𑌲𑌾𑌶𑍇
𑌶𑍍𑌯𑌭𑍀𑌪𑍀𑌚𑍍𑌯𑌕𑌫
𑌶𑌾𑌤𑌾𑌯𑌾𑌮𑌨𑌾𑌪𑌾𑌥𑌾।।15।।
𑌚𑍍𑌯𑌕𑌫𑌚𑌿𑌤𑍍𑌰
𑌪𑍇𑌰𑌾𑌅𑌯𑌾𑌸𑍀𑌤𑍍।𑌇𑌨𑍍𑌦𑍁𑌰𑌿𑌨𑍍𑌦𑌰𑌾𑌸𑍍𑌯𑌨𑌿𑌷𑍍𑌕𑍃𑌤𑌂।𑌸𑌾𑌚𑌾
𑌚𑌿𑌣𑌪𑍀𑌶𑌫𑌿𑌚𑍍𑌯
𑌸𑌾𑌚𑍍𑌯𑍁𑌨𑍍𑌨𑌰𑌪𑍍𑌰𑌮𑌿𑌨𑌾𑌤𑍀𑌸𑌂𑌗𑍀𑌰𑌂।𑌮𑌰𑍍𑌯𑌾𑌈𑌵𑌾𑌯𑍁𑌵𑌾
𑌕𑌪𑍁𑌶𑌕𑍍𑌯𑌫𑌚𑍍𑌯𑌕𑌫
𑌤𑌿𑌭𑌾𑌏𑌸𑌾।𑌮𑌰𑍍𑌷𑌾𑌤𑌾𑌇।𑌸𑍋𑌮𑌾𑌃𑌕𑌾𑌲𑌾𑌶𑍇𑌶𑌤
𑌪𑍁𑌶𑌕𑍍𑌯𑌫𑌶𑌚𑍍𑌯𑌕𑌶
𑌯𑌾𑌮𑌨𑌾𑌪𑌾।𑌥𑌾।।16।।
𑌟𑍀𑌚𑌿𑌤𑍍𑌰
