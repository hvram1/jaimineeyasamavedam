। 𑌕𑍗𑌲𑍍𑌮𑌲 𑌬𑌰𑍍𑌹𑌿𑌷𑍇𑌚𑍇 ।

1 । 𑌕𑍗𑌲𑍍𑌮𑌲 𑌬𑌰𑍍𑌹𑌿𑌷𑍇𑌚𑍇 ।

𑌪𑍍𑌰𑌤𑍍𑌯𑌸𑍍𑌮𑍈𑌪𑍀𑌬𑌾𑌹𑌾𑌬𑍁।𑌓𑌇𑌷𑌾𑌤𑌾𑌇।𑌵𑌾𑌇𑌶𑍍𑌪𑌾𑌨𑍀
𑌤𑍁𑌤𑌶𑌤𑌿𑌤𑌶𑌕𑌿𑌚𑌕
𑌵𑌾𑌇𑌦𑍂𑌷𑍇𑌹𑌾𑌇𑌭𑌾𑌰𑌾।𑌆𑌰𑌾𑌂𑌗𑌮𑌾𑌯𑌾𑌜𑌾𑌹𑌾।
𑌟𑍀𑌤𑌤𑍍𑌯𑌤𑌟𑍍𑌯𑌚𑌕𑌟𑍍𑌯𑌤
𑌗𑍍𑌮𑌾𑌯𑌾𑌯𑌪𑌾𑌶𑍍𑌪𑌾𑌦𑍍𑌦𑌾𑌔𑌹𑍋𑌵𑌾।𑌘𑍍𑌵𑌨𑍇𑌨𑌰𑌾𑌃।।1।।
𑌥𑍍𑌯𑌟𑌿𑌟𑌚𑌶𑌿𑌟𑍍𑌯𑌚𑍍𑌯
𑌪𑍍𑌰𑌤𑍍𑌯𑌸𑍍𑌮𑍈𑌪𑍀𑌪𑍀𑌷𑌤𑌾𑌇।𑌵𑌾𑌇𑌶𑍍𑌪𑌾𑌨𑍀𑌵𑌾𑌇𑌦𑍂
𑌤𑍀𑌤𑌿𑌶𑌕𑌿𑌚𑌕
𑌷𑍇𑌭𑌾𑌰𑌾।𑌆𑌰𑌾𑌂𑌗𑌾𑌮𑌾𑌯𑌜𑌗𑍍𑌮𑌾𑌯𑌾𑌯𑌪𑌶𑍍𑌪𑌾𑌦𑍍𑌦𑌾𑌘𑍍𑌵𑌨𑌾
𑌟𑍂𑌟𑍍𑌯𑌚𑌕𑌟𑍈
𑌹𑍋𑌇।𑌨𑌾𑌰𑌾𑌔𑌹𑍋𑌬𑌾।𑌹𑍋𑌇ழா𑌹।।2।।
