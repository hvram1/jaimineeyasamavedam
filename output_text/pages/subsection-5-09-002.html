।𑌪𑍍𑌰𑌜𑌾𑌪𑌤𑍇𑌰𑍍𑌵𑌾𑌜𑌸𑌨𑌿𑌨𑍀𑌚𑍇 ।

2 ।𑌪𑍍𑌰𑌜𑌾𑌪𑌤𑍇𑌰𑍍𑌵𑌾𑌜𑌸𑌨𑌿𑌨𑍀𑌚𑍇 ।

𑌏𑌆𑌭𑍀𑌪𑍍𑌰𑍀𑌯𑌾𑌓𑌣𑍀𑌪𑌾𑌵𑌾𑌤𑌾𑌏।𑌚𑌨𑍋
𑌤𑌚𑍍𑌯𑌕𑌫𑌤𑌚𑌿𑌕𑍍𑌯
𑌹𑍀𑌤𑌏𑌨𑌾𑌮𑌾𑌨𑍀𑌯𑌾।𑌓𑌹𑍍𑌵𑍋𑌅𑌧𑍀𑌯𑌾𑌏।𑌷𑍁𑌵𑌰𑍍𑌧
𑌚𑌿𑌤𑍍𑌯𑌚𑍍𑌯𑌕𑌫𑌤𑌚𑌿𑌤𑍍𑌯𑌚𑌿
𑌤𑌾𑌏।𑌆𑌸𑍂𑌰𑍀𑌯𑌾𑌓𑌸𑍍𑌯𑌬𑍃𑌹𑌾𑌤𑌾𑌏।𑌬𑍃𑌹𑌾𑌨𑍍
𑌤𑍍𑌯𑌚𑍍𑌯𑌕𑌫𑌤𑌚𑌿𑌤𑍍𑌯𑌚𑍍𑌯
𑌨𑌾𑌧𑍀।𑌏𑌰𑌾𑌥𑌾𑌂𑌵𑍀𑌷𑍍𑌵𑌾।𑌓𑌞𑍍𑌚𑌮𑌾𑌰𑍂𑌹𑌾𑌦𑍇।
𑌕𑌫𑌤𑌚𑍍𑌯𑌕𑌫𑌤𑌚𑌿𑌤𑍍𑌯
𑌵𑌿𑌚𑌕𑍍𑌷𑌣𑌾𑌹𑍋𑌇ழா𑌹।।2।।
𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌅𑌭𑍍𑌯𑍗𑌹𑍋𑌵𑌾𑌹𑌾𑌇𑌪𑍍𑌰𑌿𑌯𑌾𑌣𑍀।𑌵𑌾𑌵𑌾𑌤𑌾𑌇𑌚𑌾𑌨𑍋।
𑌤𑍢𑌕𑌿𑌚𑍍𑌯𑌣
𑌹𑌿𑌂𑌤𑍋𑌨𑌾𑌮𑌾𑌨𑌿𑌯𑌹𑍍𑌵𑍋𑌅𑌧𑌿𑌯𑌾𑌇𑌷𑍂𑌵𑌾𑌰𑍍𑌧𑌾𑌤𑌾𑌇।
𑌟𑌿𑌕𑍀𑌤𑌿𑌶𑌚𑍍𑌯𑌚𑍍𑌯𑌶
𑌆𑌸𑍂𑌰𑌿𑌯𑌾𑌸𑍍𑌯𑌬𑍃𑌹𑌾𑌤𑍋𑌬𑍃𑌹𑌨𑍍𑌨𑌾𑌧𑌾𑌇।𑌰𑌾𑌥𑌂
𑌟𑌕𑌿𑌚𑍀𑌚𑍍𑌯𑌕𑌚𑌶
𑌵𑌾𑌇𑌷𑍍𑌵𑌾𑌞𑍍𑌚𑌾𑌮𑌰𑍂𑌹𑌾𑌚𑌿𑌚𑌾𑌆𑌉𑌵𑌾।𑌕𑍍𑌷𑌾𑌣𑌾𑌃।।3।।
𑌕𑍂𑌕𑌿𑌤𑍍𑌯𑌟𑌿𑌤𑌟𑌚
