।𑌐𑌡𑍍𑌮𑌵𑌾𑌹𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

10 ।𑌐𑌡𑍍𑌮𑌵𑌾𑌹𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌅𑌘𑌾𑌯𑍇𑌆𑌗𑍍𑌨𑌿𑌮𑌿𑌨𑍍𑌧𑌾𑌤𑌾𑌇।𑌸𑍍𑌤𑍃𑌣𑌨𑍍𑌤𑌿𑌬
𑌤𑌿𑌤𑌶𑌕𑌟𑌿
𑌰𑍍𑌹𑌿𑌰𑌾𑌨𑍁𑌷𑌾𑌗𑍇𑌯𑌷𑌾𑌮𑌿𑌨𑍍𑌦𑍇𑌰𑌾𑌯𑍁𑌵𑌾।
𑌕𑍁𑌚𑌿𑌶𑍍𑌯
𑌇𑌹𑌾𑌉𑌵𑌾𑌉𑌵𑍋𑌬𑌾𑌸𑌾𑌚𑍋𑌹𑌾𑌇।।19।।
𑌪𑍁𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌆𑌘𑌾𑌯𑍇𑌇𑌹𑌾।𑌗𑍍𑌨𑍀𑌮𑌾𑌇𑌨𑍍𑌧𑌾𑌤𑌾𑌉𑌵𑍋𑌵𑌾।
𑌤𑍁𑌟𑍁𑌚𑍍𑌯𑌶
𑌈𑌹𑌾।𑌸𑍍𑌤𑍃𑌣𑌨𑍍𑌤𑍀𑌬𑌰𑍍𑌹𑍀𑌰𑌾𑌨𑍁𑌷𑌾𑌉𑌵𑍋𑌵𑌾।
𑌚𑌶𑌕𑌿𑌤𑌟𑍀𑌚𑍍𑌯𑌶
𑌈𑌹𑌾।𑌏𑌷𑌾𑌮𑌾𑌇𑌨𑍍𑌦𑌰𑌾𑌉𑌵𑍋𑌵𑌾।𑌈𑌹𑌾।
𑌚𑌶𑌕𑌟𑍀𑌚𑍍𑌯𑌶𑌚𑌶
𑌯𑍂𑌵𑌾𑌯𑍁𑌵𑌾𑌸𑌾𑌚𑌾।𑌈𑌹𑌾।।20।।
𑌟𑌿𑌚𑌤𑍍𑌰𑌚𑌶
𑌔𑌹𑍋𑌆𑌘𑌾𑌯𑌾𑌏।𑌗𑍍𑌨𑍀𑌮𑌾𑌇𑌨𑍍𑌧𑌾𑌤𑌾𑌔𑌹𑍋𑌵𑌾।
𑌤𑍁𑌤𑌟𑍁𑌚𑍍𑌯𑌶
𑌸𑍍𑌤𑍃𑌣𑌨𑍍𑌤𑍀𑌬𑌰𑍍𑌹𑍀𑌰𑌾𑌨𑍁𑌷𑌾𑌔𑌹𑍋𑌵𑌾।𑌏𑌷𑌾𑌮𑌾𑌇
𑌕𑌿𑌥𑌟𑍀𑌚𑍍𑌯𑌶𑌕
𑌨𑍍𑌦𑌰𑌾𑌔𑌹𑍋𑌵𑌾।𑌯𑍂𑌵𑌾𑌸𑌾𑌚𑌾𑌉𑌵𑌾𑌹𑌾𑌬𑍁।𑌬𑌾
𑌟𑍀𑌚𑍍𑌯𑌶𑌕𑍍𑌯𑌪𑌪𑍍𑌲𑌶𑌶
