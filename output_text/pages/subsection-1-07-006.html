6 ।𑌐𑌟𑌤𑍇𑌚𑍇।।

𑌵𑌿𑌤𑍍𑌵𑌦𑍋𑌹𑌾𑌇।𑌆𑌪𑍋𑌨𑌪𑌰𑍍𑌵𑌤𑌾𑌸𑍍𑌯𑌾𑌪𑌾
𑌤𑌿𑌤𑌶𑌕𑍂𑌟𑌿𑌰𑌿
𑌷𑍍𑌠𑌾𑌤𑍍।𑌉𑌤𑍍𑌥𑍇𑌭𑍀𑌰𑌗𑍍𑌨𑍇𑌜𑌨𑌾𑌯𑌨𑍍𑌤𑌾𑌦𑌾𑌇𑌵𑌾:।
𑌤𑍍𑌯𑌕𑍢𑌟𑌤𑍍𑌯
𑌤𑌨𑍍𑌤𑍍𑌵𑌾𑌗𑍀𑌰𑌸𑍍𑌸𑍁𑌷𑍍𑌟𑍁𑌤𑌯𑍋𑌵𑌾𑌜𑌯𑌾𑌨𑍍𑌤𑍀।𑌆𑌜𑌿𑌨𑍍𑌨𑌗𑌾
𑌕𑍢𑌟𑍍𑌯𑌤𑌕
𑌇𑌰𑍍𑌵𑌵𑌾𑌹𑍋।𑌜𑌾𑌇𑌗𑍍𑌯𑍂𑌰𑌾𑌔𑌹𑍋𑌵𑌾।
𑌟𑍂𑌤𑌟𑌿𑌟𑌚𑌶𑌿
𑌅𑌶𑍍𑌪𑌾:
𑌟𑍍𑌯𑌚
𑌹𑌾𑌯𑌾𑌯𑌾𑌇𑌦𑍀𑌵𑍋𑌹𑌾𑌇𑌵𑌿𑌤𑍍𑌵𑌾𑌤𑍍।𑌆𑌪𑍋𑌨𑌾
𑌫𑌚𑍁𑌶𑍀𑌕𑍍𑌯
𑌪𑌾𑌰𑍍𑌵𑌾𑌤𑌸𑍍𑌯𑌪𑌾𑌰𑌿𑌷𑍍𑌟𑌾𑌤𑍍।𑌹𑌾𑌯𑌾𑌯𑌾𑌇𑌦𑌿
𑌕𑍍𑌯𑌚𑍀𑌶𑍍𑌯𑌫
𑌵𑍋𑌹𑌾𑌇𑌉𑌕𑍍𑌥𑌾𑌇।𑌭𑍀𑌰𑌾𑌗𑍍𑌨𑍇𑌜𑌾𑌨𑌾𑌯𑌨𑍍𑌤𑌦𑍇
𑌚𑍁𑌶𑍁𑌟𑌿𑌕𑌿𑌚𑍍𑌯
𑌵𑌾:।𑌹𑌾𑌯𑌾𑌯𑌾𑌇𑌦𑍀𑌵𑍋𑌹𑌾𑌇𑌤𑌨𑍍𑌤𑍍𑌵𑌾।𑌗𑍀𑌰𑌾
𑌪𑍍𑌲𑌫𑌚𑍁𑌶𑍀𑌕𑍍𑌯
𑌸𑍍𑌸𑍁𑌷𑍍𑌟𑍂𑌤𑌯𑍋𑌵𑌾𑌜𑌯𑌾𑌨𑍍𑌤𑌿।𑌹𑌾𑌯𑌾𑌯𑌾𑌇𑌦𑍀𑌵𑍋
𑌕𑍁𑌚𑍍𑌯𑌶𑌫𑌚𑍁
𑌹𑌾𑌯𑌾𑌜𑌿𑌂।𑌨𑌾𑌗𑌾𑌇𑌰𑍍𑌵𑌵𑌾𑌹𑍋𑌜𑌾𑌇𑌗𑍍𑌯𑍂𑌰𑌾
𑌶𑌿𑌟𑍁𑌤𑌟𑌿𑌟𑌚
𑌔𑌹𑍋𑌵𑌾।𑌅𑌶𑍍𑌪𑌾:
𑌶𑌿𑌨𑍍𑌯
