।𑌆𑌶𑍍𑌪𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿 ।

25 ।𑌆𑌶𑍍𑌪𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿 ।

𑌆𑌸𑍋।𑌮𑌾𑌸𑍍𑌵𑌾𑌨𑍋𑌅𑌦𑍍𑌰𑌾𑌇𑌭𑌾𑌉𑌵𑍋𑌵𑌾।𑌪𑍍𑌰।
𑌚𑌣𑌕𑌥𑍍𑌯𑌟𑍀𑌚𑍍𑌯𑌶
𑌤𑌿𑌰𑍋𑌵𑌾𑌰𑌾𑌣𑌿।𑌅𑌵𑍍𑌯𑌯𑌾𑌜𑌾𑌨𑌾।𑌪𑍍𑌰।𑌨𑌾𑌪𑌾
𑌟𑍀𑌚𑌟𑌿𑌟𑍍𑌯𑌟𑍍𑌯
।𑌪𑍍𑌰।𑌰𑌿𑌚𑌾𑌮𑍁𑌵𑍋𑌰𑍍𑌵𑍀𑌶𑌾𑌡𑌰𑍀𑌸𑍍𑌸𑌾𑌦𑌾।𑌪𑍍𑌰।𑌵𑌾𑌨𑌾।
𑌟𑍀𑌚𑌟𑌿𑌟𑍍𑌯𑌟𑍍𑌯
𑌷𑍁𑌦𑌾𑌡𑍍𑌰𑍀𑌷𑌾𑌇।𑌹𑍋𑌇ழா𑌹।।35।।
𑌪𑍍𑌲𑌶𑍍𑌯𑌶𑌿
𑌹𑌾𑌵𑌾𑌸𑍋𑌮𑌸𑍍𑌵𑌾।𑌨𑍋𑌅𑌦𑍍𑌰𑌾𑌇𑌭𑌿𑌸𑍍𑌤𑍀𑌰𑍋𑌵𑌾𑌰𑌾।
𑌤𑍁𑌚𑌿𑌶𑍍𑌯𑌚𑌿𑌣
𑌣𑌿𑌯𑌾।𑌆𑌉𑌵𑌾।𑌪𑍍𑌰।𑌵𑍍𑌯𑌾𑌯𑌾।𑌜𑌨𑍋𑌨𑌾𑌪𑍂𑌰𑍀𑌚𑌾
𑌤𑍍𑌯𑌟𑌿𑌚𑌶𑌚𑌿𑌶
𑌮𑍂𑌵𑍋𑌃।𑌵𑌿𑌶𑌾।𑌪𑍍𑌰।𑌰𑍍𑌡𑌾𑌰𑌿𑌸𑍍𑌸𑌾𑌦𑌾𑌵𑌾𑌨𑍇।𑌷𑍁𑌦𑌾
𑌚𑌿𑌣𑌤𑍍𑌯𑌚𑌶𑌚𑌿𑌣𑌤𑍍𑌯
𑌆𑌉𑌵𑌾𑌡𑍍𑌰𑍀𑌷𑍇।।36।।
𑌟𑌿𑌚𑌶
𑌆𑌸𑍋𑌮𑌸𑍍𑌵𑌾𑌨𑍋𑌅𑌦𑍍𑌰𑌿𑌭𑌾𑌇𑌃।𑌤𑍀𑌰𑍋𑌵𑌾𑌰𑌾𑌣𑌿𑌯𑌾
𑌤𑍁𑌤𑌿𑌶
𑌵𑍍𑌯𑌾𑌯𑌾।𑌜𑌨𑍋𑌨𑌪𑍂𑌰𑌿𑌚𑌾𑌮𑍂𑌵𑍋𑌰𑍍𑌵𑍀𑌶𑌾𑌰𑍍𑌡𑌾𑌰𑌿𑌃।
𑌯𑍃𑌟𑌯𑍃𑌟𑌕𑍍𑌯𑌯𑌟
𑌸𑌦𑌾𑌵𑌾𑌨𑌾𑌇𑌷𑍂𑌦𑌡𑍍𑌰𑍀𑌷𑍋𑌇ழா𑌹।।37।।
𑌟𑍍𑌯𑌚𑌟𑍀𑌚𑌿𑌪𑍍𑌲𑌶𑍍𑌯
𑌆𑌸𑍋𑌮𑌸𑍍𑌵𑌾𑌨𑍋𑌅𑌦𑍍𑌰𑌿𑌭𑌿𑌸𑍍𑌤𑌿𑌰𑍋𑌵𑌾𑌰𑌾𑌣𑍀𑌯𑌵𑍍𑌯𑌯𑌾।
𑌫𑍇𑌚𑍍𑌯𑌪𑍍𑌲
𑌜𑌨𑍋𑌨𑌪𑍁𑌰𑌿𑌚𑌮𑍍𑌵𑍋𑌰𑍍𑌵𑌿𑌶𑌾𑌰𑍍𑌡𑌾𑌰𑍀𑌰𑍗।𑌹𑍂𑌵𑌾𑌏
𑌕𑍢𑌟𑌿𑌚𑌟𑌿
𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌸𑌾𑌦𑌾𑌵𑌨𑍇𑌷𑍂𑌦𑍗।𑌪𑍍𑌰।𑌡𑌿𑌷𑌾𑌔
𑌚𑍍𑌯𑌕𑍀𑌟𑍍𑌯𑌚𑌟𑌿
𑌹𑍋𑌬𑌾।𑌹𑍋𑌇ழா𑌹।।38।।
𑌚𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
