।𑌅𑌂𑌗𑌿𑌰𑌸𑌾𑌮𑍁𑌤𑍍 𑌸𑍇𑌧𑌨𑌿𑌷𑍇𑌧𑍗 𑌚𑍗।

29 ।𑌅𑌂𑌗𑌿𑌰𑌸𑌾𑌮𑍁𑌤𑍍 𑌸𑍇𑌧𑌨𑌿𑌷𑍇𑌧𑍗 𑌚𑍗।

𑌪𑍍𑌰𑌸𑍋𑌮𑌦𑍇𑌵𑌵𑍀𑌤𑌯𑍇𑌸𑌿𑌨𑍍𑌧𑍂𑌨𑍍𑌨𑌰𑌪𑌾𑌔𑌹𑍋𑌵𑌾।
𑌫𑍇𑌚𑍍𑌯𑌣𑌫𑌪𑍍𑌲
𑌪𑍇𑌯𑌅𑌰𑍍𑌣𑌾𑌸𑌾𑌹𑌾𑌉𑌵𑌾।𑌊𑌪𑌾।𑌅𑌂𑌶𑍋ǐ𑌪𑌯𑌾
𑌕𑌟𑌿𑌚𑌯𑌟𑍍𑌯𑌚𑌶
𑌔𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌸𑌾𑌮𑌦𑌿𑌰𑍋𑌨𑌜𑌾𑌗𑍃𑌵𑌿𑌰𑍍𑌹𑌾
𑌶𑍍𑌰𑍢𑌕𑌕𑌿𑌕𑌿𑌯𑍍𑌯
𑌉𑌵𑌾।𑌊𑌪𑌾।𑌅𑌚𑍍𑌛𑌾𑌕𑍋𑌶𑌂𑌔𑌹𑍋𑌵𑌾𑌹𑌾𑌇।
𑌟𑍍𑌯𑌚𑌶𑌶𑍢
𑌮𑌧𑍂𑌶𑍂𑌤𑌾𑌂।𑌊।।42।।
𑌚𑌿𑌤𑍍𑌰𑌚
𑌪𑍍𑌰𑌸𑍋𑌮𑌦𑌾𑌇𑌵𑌾𑌵𑍀𑌤𑌯𑌾𑌇।𑌸𑌿𑌨𑍍𑌧𑍂𑌨𑍍𑌨𑌰𑌪𑌾𑌇𑌪𑍇𑌯
𑌤𑍂𑌤𑌿𑌶𑌚𑍂
𑌅𑌰𑍍𑌣𑌾𑌸𑌾𑌇𑌹𑌾।𑌆𑌂𑌶𑍋ǐ𑌪𑌾𑌯𑌾।𑌹𑌾𑌹𑍋𑌹𑌾𑌇
𑌥𑌟𑍍𑌯𑌤𑍍𑌯𑌟𑍍𑌯𑌚𑌶𑌚𑍍𑌯𑌣𑌶
।𑌪𑍍𑌰।𑌸𑌾𑌮𑌦𑌿𑌰𑍋𑌨𑌜𑌾𑌗𑍄𑌵𑍀𑌰𑌿𑌹𑌾।𑌆𑌚𑍍𑌛𑌾𑌕𑍋𑌶𑌾𑌂
𑌕𑌕𑌿𑌟𑍍𑌯𑌤𑍀𑌟𑍍𑌯𑌚𑌶
।𑌪𑍍𑌰।𑌮𑌧𑍂𑌶𑍍𑌪𑍂𑌤𑌾𑌂।𑌹𑍇।।43।।
