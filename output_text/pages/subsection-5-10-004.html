।𑌐𑌷𑌿𑌰𑌾𑌣𑌿𑌪 𑌞𑍍𑌚।

4 ।𑌐𑌷𑌿𑌰𑌾𑌣𑌿𑌪 𑌞𑍍𑌚।

𑌪𑍍𑌰𑌧𑌨𑍍𑌵𑌾𑌸𑍗।𑌹𑍋𑌔𑌹𑍋𑌵𑌾।𑌮𑌜𑌾𑌗𑍍𑌰𑌵𑍀𑌃𑌇𑌨𑍍𑌦𑌰𑌾
𑌕𑍍𑌯𑌶𑍍𑌯𑌕𑍍𑌯𑌚𑌣𑌕𑍍𑌯𑌟𑍍𑌯𑌥𑍍𑌯
𑌯𑍇𑌨𑍍𑌦𑍗।𑌹𑍋𑌔𑌹𑍋𑌵𑌾।𑌪𑌰𑍀𑌸𑍍𑌰𑌾𑌵𑌾𑌦𑍍𑌯𑍁𑌮𑌨𑍍𑌤𑌾𑌂
𑌶𑍍𑌯𑌕𑍍𑌯𑌚𑌣𑌚𑍍𑌯𑌟𑍍𑌯𑌕𑍍𑌯
𑌶𑍗।𑌹𑍋𑌔𑌹𑍋𑌵𑌾।𑌷𑍍𑌮𑌮𑌾𑌭𑌾𑌰𑌾𑌸𑍁𑌵𑌰𑍍𑌵𑌿𑌦𑍗
𑌶𑍍𑌯𑌕𑍍𑌯𑌚𑌣𑌕𑍍𑌯𑌟𑍍𑌯𑌕𑍍𑌯
𑌹𑍋𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾।।6।।
𑌕𑍀𑌚𑌤𑍍𑌰𑌚𑌶
𑌪𑍍𑌰𑌧𑌨𑍍𑌵𑌾𑌸𑍗𑌹𑍋𑌵𑌾𑌹𑍋𑌮𑌜𑌾𑌗𑍍𑌰𑌵𑍀𑌃।𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑍇𑌨𑍍𑌦𑍗
𑌕𑍍𑌯𑌕𑍀𑌕𑌿𑌟𑍍𑌯𑌕𑍍𑌯
𑌹𑍋𑌵𑌾𑌹𑍋𑌇𑌪𑌰𑍀𑌸𑍍𑌰𑌾𑌵𑌾।𑌦𑍍𑌯𑍁𑌮𑌨𑍍𑌤𑌾𑌂𑌶𑍗𑌹𑍋𑌵𑌾
𑌕𑍀𑌕𑍀𑌟𑍍𑌯𑌕𑍍𑌯𑌕𑍀
𑌹𑍋𑌷𑍍𑌮𑌮𑌾𑌭𑌾𑌰𑌾।𑌸𑍁𑌵𑌰𑍍𑌵𑌿𑌦𑍗𑌹𑍋𑌵𑌾𑌹𑍋𑌵𑌾
𑌕𑌿𑌟𑍍𑌯𑌕𑍍𑌯𑌕𑌿𑌟𑍍𑌯𑌟𑌚
𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾।।7।।
𑌶𑌿𑌚𑌶
𑌔𑌹𑍋𑌔𑌹𑍋𑌇।𑌪𑍍𑌰।𑌪𑍍𑌰𑌧𑌨𑍍𑌵𑌾𑌸𑍋।𑌔𑌹𑍋
𑌕𑌚𑌿𑌶𑌕𑍍𑌯𑌟𑍍𑌯𑌕
𑌔𑌹𑍋𑌮𑌜𑌾𑌗𑍍𑌰𑌵𑍀𑌃।𑌪𑍍𑌰।𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌾𑌇𑌨𑍍𑌦𑍋𑌔𑌹𑍋
𑌕𑍁𑌟𑍍𑌯𑌥𑍍𑌯𑌟𑌿𑌚𑌕
𑌔𑌹𑍋𑌇𑌪𑌰𑍀𑌸𑍍𑌰𑌾𑌵𑌾।𑌪𑍍𑌰।𑌦𑍍𑌯𑍁𑌮𑌨𑍍𑌤𑌾𑌂𑌶𑍂𑌔
𑌕𑍂𑌟𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯𑌚𑌕
𑌹𑍋𑌔𑌹𑍋𑌷𑍍𑌮𑌮𑌾𑌭𑌾𑌰𑌾।𑌪𑍍𑌰।𑌸𑍂𑌵𑌾𑌰𑍍𑌵𑌾𑌇𑌦𑌾
𑌕𑍁𑌟𑍍𑌯𑌕𑍍𑌯𑌤𑌟𑌚𑍍𑌯
𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾।।8।।
𑌶𑌿𑌚𑌶
𑌹𑍁𑌵𑌾𑌇𑌹𑍁𑌵𑌾𑌹𑍋𑌇।𑌪𑍍𑌰।𑌪𑍍𑌰𑌧𑌨𑍍𑌵𑌾𑌸𑍋𑌮𑌜𑌾
𑌟𑍍𑌯𑌟𑌿𑌶𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯𑌕𑍍𑌯
𑌗𑍍𑌰𑌵𑍀𑌃।𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌾𑌇𑌨𑍍𑌦𑍋𑌪𑌾𑌰𑍀𑌸𑍍𑌰𑌾𑌵𑌾।𑌦𑍍𑌯𑍁𑌮𑌨𑍍𑌤𑌾𑌂𑌶𑍂
𑌟𑍍𑌯𑌥𑍍𑌯𑌟𑌿𑌚𑍍𑌯𑌟𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯
𑌷𑍍𑌮𑌾𑌮𑌾𑌭𑌾𑌰𑌾।𑌪𑍍𑌰।𑌸𑍂𑌵𑌾𑌰𑍍𑌵𑌾𑌇𑌦𑌾𑌔𑌹𑍋𑌵𑌾।
𑌕𑍍𑌯𑌟𑍍𑌯𑌕𑍍𑌯𑌤𑌟𑌚𑍍𑌯𑌶𑌿
𑌊𑌪𑌾।।9।।
𑌚𑌶
𑌪𑍍𑌰𑌾𑌧𑌨𑍍𑌵𑌾𑌸𑍋𑌮𑌜𑌾𑌗𑍍𑌰𑌵𑍀𑌃।𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌾𑌇𑌨𑍍𑌦𑍋।𑌓𑌇
𑌪𑌿𑌶𑍁𑌟𑍍𑌯𑌚𑌣𑍍𑌯
𑌪𑌾𑌰𑌾𑌇𑌸𑍍𑌰𑌾𑌵𑌾।𑌦𑍍𑌯𑍁𑌮𑌾𑌨𑍍𑌤𑌾𑌂𑌶𑍂।𑌓𑌷𑍍𑌮𑌮𑌾𑌭𑌾𑌰𑌾।
𑌟𑌿𑌪𑌿𑌶𑌟𑍍𑌯𑌚𑌣𑌟𑌿𑌚𑌤𑍍𑌰
𑌸𑍂𑌵𑌾𑌰𑍍𑌵𑍀𑌦𑌾𑌮𑍍।।10।।
𑌤𑍍𑌯𑌟𑌚
