।𑌵𑌸𑌿𑌷𑍍𑌠𑌸𑍍𑌯𑌪𑍍𑌰𑌮𑌂𑌹𑌿𑌷𑍍𑌠𑍀𑌯𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿।।

1 ।𑌵𑌸𑌿𑌷𑍍𑌠𑌸𑍍𑌯𑌪𑍍𑌰𑌮𑌂𑌹𑌿𑌷𑍍𑌠𑍀𑌯𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿।।

𑌪𑍍𑌰𑌮𑌂𑌹𑌾𑌇𑌷𑍍𑌠𑌾𑌯𑌗𑌾𑌯𑌤𑌾।𑌋𑌤𑌾𑌵𑍍𑌨𑍇𑌬𑍃𑌹𑌤𑍇
𑌚𑌿𑌶𑍂𑌟𑌿𑌚𑌕
𑌶𑍂𑌕𑍍𑌰𑌾𑌶𑍋𑌚𑌾𑌇𑌷𑌾𑌇।𑌉𑌪𑌾𑌔𑌹𑍋
𑌪𑍍𑌲𑌶𑌤𑌤𑍍𑌯𑌶𑌪𑌿𑌶
𑌸𑍍𑌤𑍁𑌤𑌾𑌸𑍋𑌆।𑌗𑍍𑌨𑌾𑌯𑍋𑌹𑌾𑌇।।1।।
𑌪𑌿𑌶𑌚𑌪𑍍𑌲𑌶𑍍𑌯
𑌪𑍍𑌰𑌮𑌂𑌹𑌿𑌷𑍍𑌠𑌾𑌯𑌗𑌾।𑌯𑌾𑌤𑌾𑌋𑌤𑌾𑌵𑍍𑌨𑍇𑌬𑍃𑌹𑌾𑌤𑌾
𑌤𑍂𑌚𑌟𑌿𑌟𑌚𑌕𑍍𑌯𑌯
𑌇𑌶𑍂।𑌕𑍍𑌰𑌾𑌶𑍋𑌚𑌾𑌇𑌷𑌾𑌇।𑌊𑌪𑌾𑌸𑍍𑌤𑍁𑌤𑍗𑌹𑍁
𑌟𑍍𑌯𑌕𑌟𑍍𑌯𑌤𑍍𑌯𑌶𑌚𑍍𑌯𑌟𑍍𑌯𑌚𑌕
𑌵𑌾𑌏𑌹𑍋𑌵𑌾।𑌸𑌾𑌓𑌬𑌾𑌗𑍍𑌨𑌾𑌯𑍋𑌹𑌾𑌇।।2।।
𑌟𑍍𑌯𑌕𑍍𑌯𑌕𑌪𑌪𑍍𑌯𑌪𑍍𑌯𑌶
𑌪𑍍𑌰𑌮𑌂𑌹𑌿𑌷𑍍𑌠𑌾𑌯𑌾𑌗𑌾𑌇𑌯𑍍𑌯𑌾𑌈𑌯𑌾।𑌯𑌾𑌤𑌾𑌋𑌤𑌾𑌵𑍍𑌨𑍇
𑌫𑍁𑌚𑍍𑌯𑌚𑌶𑌥𑌕𑍀
𑌬𑍃𑌹𑌤𑍇𑌶𑍂𑌕𑍍𑌰𑌾𑌶𑍗𑌹𑍗𑌹𑍁𑌵𑌾𑌇𑌚𑌾𑌇𑌷𑌾𑌇।
𑌟𑍀𑌟𑍍𑌯𑌟𑌟𑍍𑌯𑌟𑍀𑌶
𑌉𑌪𑌾𑌹𑍋𑌇𑌸𑍍𑌤𑍁𑌤𑌾𑌹𑍋𑌸𑍋𑌅𑌗𑍍𑌨𑌾𑌯𑌾𑌇।
𑌕𑍍𑌯𑌕𑍀𑌕𑌥𑌟𑍍𑌯𑌚𑌣𑌶
𑌓𑌇।।3।।
𑌪𑌶𑍍𑌯
𑌪𑍍𑌰𑌮𑌂𑌹𑌿𑌷𑍍𑌠𑌾𑌯𑌗𑌾𑌯𑌤𑌪𑍍𑌰𑌮𑌂𑌹𑌿𑌷𑍋𑍍𑌠𑌵𑌾।𑌯𑌾𑌗𑌾
𑌤𑍋𑌤𑌕𑍍𑌯
𑌯𑌤𑌾𑌋𑌤𑌾𑌵𑍍𑌨𑍇𑌈𑌬𑍃𑌹𑌤𑍇𑌶𑍁𑌕𑍍𑌰𑌾𑌶𑍋𑌚𑌾
𑌟𑍀𑌚𑍍𑌯𑌣𑍀𑌫𑌿𑌤
𑌇𑌷𑌾𑌇।𑌉𑌪𑍗𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌸𑍍𑌤𑍁𑌤𑍗
𑌤𑍍𑌯𑌶𑌫𑍍𑌯𑌤𑍍𑌯𑌤𑌶𑌫𑍍𑌯
𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌸𑌓𑌬𑌾𑌗𑍍𑌨𑌾𑌯𑍋𑌹𑌾𑌇।।4।।
