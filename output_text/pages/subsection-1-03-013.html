।𑌗𑍗𑌰𑌾𑌂𑌗𑌿𑌰𑌸𑌸𑍍𑌯𑌸𑌾𑌮𑌨𑍀𑌚𑍇।।

13 ।𑌗𑍗𑌰𑌾𑌂𑌗𑌿𑌰𑌸𑌸𑍍𑌯𑌸𑌾𑌮𑌨𑍀𑌚𑍇।।

𑌕𑌸𑍍𑌯𑌾𑌨𑍂𑌨𑌾𑌂𑌪𑌾𑌰𑍀𑌣𑌾𑌸𑍀।𑌧𑍀𑌯𑍋𑌜𑌿𑌨𑍍𑌵𑌾
𑌟𑍀𑌚𑌿𑌣𑌟𑌿
𑌸𑍀𑌸𑌾𑌤𑍍𑌪𑌾𑌤𑌾𑌇।𑌗𑍋𑌷𑌾𑌤𑌾𑌯𑌾𑌸𑍍𑌯𑌾𑌤𑌾
𑌚𑌿𑌣𑌚𑌥𑍍𑌯𑌟𑍍𑌯𑌟𑌚
𑌔𑌹𑍋𑌵𑌾।𑌉𑌪𑍍𑌗𑍀𑌰𑌾:
𑌶𑌿𑌚𑍍𑌯𑌶
𑌓𑌹𑍋𑌵𑌾𑌇𑌹𑍁𑌵𑌾𑌇𑌹𑍁𑌵𑌾𑌏।𑌕𑌸𑍍𑌯𑌨𑍂
𑌕𑍍𑌯𑌥𑌟𑌿𑌚𑌿𑌤
𑌨𑌾𑌂𑌪𑌾𑌰𑍀𑌣𑌾𑌸𑍀।𑌓𑌹𑍋𑌵𑌾𑌇𑌹𑍁𑌵𑌾𑌇𑌹𑍁
𑌚𑍀𑌚𑍍𑌯𑌫𑍍𑌯𑌕𑍍𑌯𑌥𑌟𑌿
𑌵𑌾𑌏।𑌧𑍀𑌯𑍋𑌜𑌿𑌨𑍍𑌵𑌾𑌸𑍀𑌸𑌾𑌤𑍍𑌪𑌾𑌤𑍀𑌂।𑌓
𑌚𑌿𑌤𑌟𑌿𑌚𑍍𑌯𑌫𑍍𑌯
𑌹𑍋𑌵𑌾𑌇𑌹𑍁𑌵𑌾𑌇𑌹𑍁𑌵𑌾𑌏।𑌗𑍋𑌷𑌾𑌤𑌾
𑌕𑍍𑌯𑌥𑌟𑌿𑌚𑌿𑌤
𑌯𑌸𑍍𑌯𑌤𑌾𑌗𑌾𑌇𑌰𑌾:
