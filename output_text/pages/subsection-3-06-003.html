3 ।𑌗𑍗𑌤𑌮𑍇𑌚𑍇 ।

𑌸𑍁𑌨𑍋𑌤𑌸𑍋𑌮𑌪𑌾𑌵𑍍𑌨𑌾𑌏।𑌸𑍋𑌮𑌮𑌿𑌨𑍍𑌦𑌰𑌾𑌹𑍋𑌵𑌾
𑌷𑌤𑍁𑌤𑌕𑍀𑌟𑍍𑌯
𑌹𑌾𑌯𑌾𑌵𑌜𑍍𑌰𑌾𑌇𑌣𑌾𑌇।𑌪𑌚𑌤𑌾𑌪𑌕𑍍𑌤𑌾𑌇𑌰𑌵𑌸𑍇
𑌤𑌚𑌕𑌟𑍍𑌯𑌤𑍍𑌯𑌶𑌕𑌿𑌚𑌕𑍂𑌚
𑌕𑍃𑌣𑍂𑌧𑍍𑌵𑌾𑌮𑍀𑌡𑌾𑌇।𑌪𑍃𑌣𑌾𑌨𑍍𑌨𑌾𑌇𑌤𑍍𑌪𑍃𑌹𑌾𑌇।
𑌕𑍍𑌯𑌯𑌟𑌤𑌶𑌚𑍍𑌯𑌯𑌟𑍍𑌯𑌤𑌶
𑌣𑌾𑌤𑌾𑌇𑌮𑌾𑌯𑌾𑌃।𑌓𑌇ழா𑌹।।5।।
𑌟𑍀𑌚𑌣𑌪𑌶𑍍𑌯
𑌸𑍁𑌨𑍋𑌤𑌸𑍋𑌮𑌪𑌾𑌵𑍍𑌨𑌾𑌉𑌵𑍋𑌵𑌾𑌇𑌯𑌾𑌹𑌾𑌇।
𑌫𑍂𑌚𑌿𑌶𑍁
𑌸𑍋𑌮𑌮𑌿𑌨𑍍𑌦𑌰𑌾𑌹𑍁𑌵𑍇𑌹𑍁𑌵𑍇𑌹𑍋𑌯𑌾𑌵𑌾𑌜𑍍𑌰𑍀𑌣𑌾𑌇।
𑌟𑍀𑌟𑍀𑌯𑌿𑌟𑍍𑌯𑌶
𑌪𑌚𑌤𑌾𑌪𑌕𑍍𑌤𑌾𑌇𑌰𑌵𑌸𑍇।𑌕𑍃𑌣𑍂𑌧𑍍𑌵𑌾𑌮𑍀𑌡𑌾𑌇।𑌪𑍃
𑌚𑌿𑌚𑍂𑌟𑌕𑍍𑌯𑌯𑌟𑌤𑌶
𑌣𑌾𑌨𑍍𑌨𑌾𑌇𑌤𑍍𑌪𑍃𑌹𑌾𑌇।𑌣𑌾𑌤𑌾𑌇𑌮𑌾𑌯𑌾𑌃।
𑌚𑍍𑌯𑌯𑌟𑍍𑌯𑌤𑌶𑌟𑍀𑌚𑌣
𑌓𑌇ழா𑌹।।6।।
𑌪𑌶𑍍𑌯
