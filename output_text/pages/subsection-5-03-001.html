।𑌯𑌾𑌮𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿…

1 ।𑌯𑌾𑌮𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿…

𑌇𑌹𑌾𑌇𑌹𑌾।𑌪𑍍𑌰।𑌉𑌪𑍋𑌷𑍁𑌜𑌾𑌤𑌮𑌾𑌪𑍍𑌤𑍂𑌰𑌾𑌮𑌿𑌹𑌾
𑌟𑍍𑌯𑌚𑌕𑌶𑌟𑍂𑌚𑌕𑌚𑌶𑍍𑌯
।𑌪𑍍𑌰।𑌗𑍋𑌭𑌿𑌰𑍍𑌭𑌂𑌗𑌂𑌪𑌰𑌾𑌇𑌷𑍍𑌕𑍃𑌤𑌾𑌮𑌿𑌹𑌾।𑌪𑍍𑌰।
𑌟𑍂𑌚𑌕𑍍𑌯𑌚𑌶𑍍𑌯
𑌇𑌨𑍍𑌦𑍁𑌨𑍍𑌦𑍇𑌵𑌾𑌅𑌯𑌾𑌸𑍀𑌷𑍂:।𑌇𑌹𑌾।।1।।
𑌟𑍂𑌚𑌕𑌚𑌕𑌶
𑌆𑌉𑌪𑍋𑌷𑍁𑌜𑌾𑌤𑌮𑌾𑌪𑍍𑌤𑍁𑌰𑌾𑌮𑍁𑌪𑌾।𑌗𑍋𑌭𑌾
𑌟𑍃𑌚𑌕𑌚𑌶𑍍𑌯𑌟𑍍𑌯
𑌇𑌹𑍋𑌰𑌇।𑌭𑌂𑌗𑌂𑌪𑌰𑌾𑌇𑌷𑍍𑌕𑍃𑌤𑌾𑌮𑍁𑌪𑌾।𑌇𑌨𑍍𑌦𑍁𑌂
𑌕𑍍𑌯𑌚𑌶𑌟𑍀𑌚𑌿𑌶𑍍𑌯𑌟𑍍𑌯𑌚
𑌹𑍋𑌇𑌦𑍇𑌵𑌾𑌅𑌯𑌾𑌸𑌿𑌷𑍁𑌰𑌾𑌆𑌉𑌵𑌾।𑌉𑌊𑌪𑌾।।2।।
𑌕𑌟𑍂𑌤𑌿𑌟𑌿𑌚𑍍𑌯𑌶
𑌉𑌪𑍋𑌷𑍍𑌵𑍗𑌹𑍋𑌇𑌜𑌾𑌤𑌾𑌂।𑌆𑌪𑍍𑌤𑍂𑌰𑌾𑌮𑍗𑌹𑍋𑌵𑌾
𑌤𑍃𑌟𑍍𑌯𑌤𑌟𑍍𑌯𑌤
𑌇𑌗𑍋𑌭𑌿𑌰𑍍𑌭𑌾𑌂𑌗𑌂।𑌓𑌇𑌪𑌾𑌰𑍀𑌷𑍍𑌕𑍃𑌤𑌾𑌂।𑌇𑌨𑍍𑌦𑍂𑌨𑍍𑌦𑍇𑌵𑌾
𑌚𑍍𑌯𑌚𑍍𑌯𑌣𑌯𑍀𑌟𑍍𑌯𑌟𑍍𑌯𑌚𑍍𑌯
𑌔𑌹𑍋𑌵𑌾।𑌅𑌯𑌾𑌸𑌿𑌷𑍂:।।3।।
