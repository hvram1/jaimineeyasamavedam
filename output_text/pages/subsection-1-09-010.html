10 ।𑌕𑌶𑍍𑌯𑌵𑌸𑍍𑌯𑌸𑌯𑍋𑌨𑌿:।।

𑌜𑌾𑌤𑌾:𑌪𑌰𑍇𑌣𑌾𑌧𑌾𑌈𑌹𑌾𑌰𑍍𑌮𑌾𑌣𑌾𑌇𑌹𑌾
𑌚𑍍𑌯𑌟𑍍𑌯𑌚𑌕𑌟𑍍𑌯𑌂𑌟𑌿
𑌯𑍋𑌨𑌿𑌂।𑌯𑍋𑌨𑌿𑌮𑌿𑌨𑍍𑌦𑌰𑌾𑌶𑍍𑌪𑌗𑌚𑍍𑌛𑌥𑍋𑌯𑌤𑍍𑌸𑌾𑌵𑍃
𑌚𑍍𑌯𑌶𑌕𑍀𑌕𑍀𑌟𑌿
𑌦𑍍𑌭𑌿𑌸𑌾𑌹𑌾𑌈𑌹𑌾𑌭𑍂𑌵𑌾𑌈𑌹𑌾𑌯𑍋𑌨𑌿𑌂।
𑌟𑍍𑌯𑌕𑌟𑍍𑌯𑌂𑌟𑌿𑌚𑍍𑌯𑌶
𑌯𑍋𑌨𑌿𑌮𑌿𑌨𑍍𑌦𑌰𑌾𑌶𑍍𑌪𑌗𑌚𑍍𑌛𑌥𑌾:।𑌪𑌿𑌤𑌾𑌯𑌾𑌤𑍍𑌕𑌾
𑌕𑍀𑌂𑌕𑌿𑌟𑌿𑌕
𑌶𑍍𑌯𑌾𑌪𑌾𑌈𑌹𑌾𑌸𑍍𑌯𑌾𑌗𑍍𑌨𑌾𑌈𑌹𑌾𑌯𑍋𑌨𑌿𑌂।
𑌟𑌿𑌂𑌟𑌿𑌚𑍍𑌯𑌶
𑌯𑍋𑌨𑌿𑌮𑌿𑌨𑍍𑌦𑌰𑌾𑌶𑍍𑌪𑌗𑌚𑍍𑌛𑌥𑌾𑌶𑍍𑌰𑌡𑌾𑌮𑌾𑌤𑌾𑌮𑌾𑌨𑌾𑌈
𑌕𑍀𑌕𑍂𑌥𑍍𑌯𑌚𑌕𑌟𑍍𑌯
