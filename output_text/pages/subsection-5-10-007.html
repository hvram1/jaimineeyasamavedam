।𑌵𑍈𑌶𑍍𑌪𑌦𑍇𑌵𑍇𑌚𑍇 ।

7 ।𑌵𑍈𑌶𑍍𑌪𑌦𑍇𑌵𑍇𑌚𑍇 ।

𑌪𑍍𑌰𑌾𑌣𑌾𑌶𑌾𑌇𑌶𑍂𑌰𑍍𑌮𑍍𑌮𑌾𑌹𑌾𑌇𑌨𑌾𑌂।𑌹𑌿𑌨𑍍𑌵𑌾𑌨𑍍𑌨𑌾𑌕𑍍𑌤𑌰𑌾।
𑌤𑍍𑌯𑌚𑌶𑍍𑌯𑌪𑍍𑌲𑌚𑍍𑌯𑌣𑍍𑌯𑌟𑍍𑌯𑌚𑌣
𑌸𑍍𑌯𑌾𑌦𑌾𑌇𑌧𑍀𑌤𑌿𑌂।𑌵𑌾𑌇𑌶𑍍𑌪𑌾𑌪𑌰𑌿𑌪𑍍𑌰𑌿𑌯𑌾𑌭𑍁𑌵𑌾𑌦𑌧𑌚𑌿
𑌤𑌪𑌿𑌶𑌕𑌿𑌕𑍀𑌟𑍍𑌯
𑌤𑌾।𑌓𑌇ழா𑌹।।19।।
𑌚𑍀𑌣𑌪𑌶𑍍𑌯
𑌪𑍍𑌰𑌾𑌣𑌾𑌪𑍍𑌰𑌾𑌣𑌾।𑌶𑌾𑌇𑌶𑍂𑌶𑍍𑌶𑌾𑌇𑌶𑍂𑌮𑍍𑌮𑌰𑌹𑌾।𑌆𑌉𑌵𑌾
𑌤𑍀𑌟𑌿𑌟𑌿𑌤𑍍𑌯
𑌏।𑌪𑍍𑌰।𑌨𑌾𑌂𑌹𑌿𑌨𑍍𑌵𑌨𑍍𑌨𑌾𑌕𑍍𑌤𑌰𑌾𑌸𑍍𑌯𑌦𑌾।𑌪𑍍𑌰।𑌧𑍀𑌤𑌿𑌂।
𑌭𑍀𑌚𑌕𑌕𑌿𑌤𑍍𑌯𑌚𑌶
𑌵𑌿𑌶𑍍𑌪𑌾𑌪𑌰𑌿𑌪𑍍𑌰𑌿𑌯𑌾𑌆𑌉𑌵𑌾।𑌭𑍂𑌵𑌾𑌤𑍍।𑌅𑌧𑍗𑌹𑍗
𑌕𑍍𑌯𑌤𑍀𑌟𑌿𑌚𑌶𑌟𑍍𑌯𑌟
𑌹𑍁𑌵𑌾𑌇𑌚𑌾𑌇𑌤𑌾𑌔𑌹𑍋𑌵𑌾।𑌉𑌊𑌪𑌾।।20।।
𑌟𑍀𑌟𑌚𑍍𑌯𑌶𑌿𑌚𑍍𑌯𑌶
