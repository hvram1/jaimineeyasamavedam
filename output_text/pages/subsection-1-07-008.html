।𑌵𑍈𑌶𑍍𑌯𑌜𑍇𑌯𑌾𑌤𑌿𑌷𑍇𑌚।।

8 ।𑌵𑍈𑌶𑍍𑌯𑌜𑍇𑌯𑌾𑌤𑌿𑌷𑍇𑌚।।

𑌇𑌨𑍍𑌧𑌾𑌓𑌇𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁।𑌰𑌾𑌜𑌾𑌸𑌮
𑌤𑍍𑌯𑌪𑌣𑍍𑌯𑌶𑍁
𑌯𑍋𑌰𑌨𑌮𑍋𑌭𑍀𑌯𑍋𑌭𑍀𑌯𑍋𑌭𑍀:।𑌯𑌸𑍍𑌯𑌾𑌓𑌹𑌾
𑌤𑍋𑌤𑍍𑌯𑌪𑌣
𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁।𑌪𑍍𑌰𑌤𑍀𑌕𑌮𑌾𑌹𑍁𑌤𑌂𑌘𑍃𑌤𑌾
𑌶𑍍𑌰𑍁
𑌇𑌨𑌾𑌇𑌨𑌾𑌇𑌨𑌾।𑌨𑌰𑌾𑌓𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾
𑌤𑍋𑌤𑍍𑌯𑌪𑌣
𑌬𑍁।𑌹𑌵𑍇𑌯𑌭𑍀𑌰𑌿𑌡𑌤𑍇𑌸𑌬𑌾𑌧𑌾𑌬𑌾𑌧𑌾𑌬𑌾𑌧𑌾:।
𑌶𑍍𑌰𑍁𑌤𑍋
𑌅𑌗𑍍𑌨𑌾𑌓𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌉𑌵𑌾।𑌅𑌗𑍍𑌰𑌾𑌮𑍂𑌷
𑌤𑍍𑌯𑌪𑌣𑌤𑍂𑌚𑍍𑌯
𑌸𑌾𑌮𑌶𑍋𑌉𑌵𑌾।𑌚𑍀
𑌟𑌿𑌕𑍍𑌯𑌤𑍍𑌯𑌚
𑌹𑌾𑌬𑍁𑌹𑍋𑌵𑌾𑌹𑌾𑌬𑍁𑌹𑍋𑌵𑌾𑌹𑌾𑌬𑍁𑌹𑍋
𑌕𑍀𑌕𑍀
𑌵𑌾।𑌪𑍍𑌰।𑌇𑌨𑍍𑌡𑍇𑌰𑌾𑌜𑌾𑌸𑌮𑌯𑍋𑌰𑌨𑌾𑌮𑍋𑌭𑍀𑌯𑍋
𑌕𑍀𑌚𑍃𑌕𑌕𑌟𑌟𑍍𑌯
𑌭𑍀𑌯𑍋𑌭𑍀:।𑌯𑌸𑍍𑌯𑌪𑍍𑌰𑌤𑌿𑌕𑌮𑌾𑌹𑍁𑌤𑌂𑌘𑍃𑌤𑌾𑌇
𑌟𑍍𑌯𑌕𑍢𑌕
𑌨𑌾𑌆𑌇𑌨𑌾𑌆𑌇𑌨𑌾।𑌨𑌰𑍋𑌹𑌵𑍇𑌯𑌭𑍀𑌰𑍀𑌡𑌤𑍇
𑌟𑍍𑌯𑌟𑌿𑌟𑌿𑌚𑍄
𑌸𑌾𑌬𑌾𑌧𑌾𑌬𑌾𑌧𑌾𑌬𑌾𑌧𑌾:।𑌹𑌾𑌬𑍁𑌹𑍋
𑌕𑌕𑌕𑌟𑍍𑌯𑌟𑍍𑌯
𑌵𑌾𑌹𑌾𑌬𑍁𑌹𑍋𑌵𑌾𑌹𑌾𑌬𑍁𑌹𑍋𑌵𑌾।𑌅𑌗𑍍𑌨𑌿𑌰𑌾
𑌕𑍀𑌕𑍀𑌕𑍀
𑌗𑍍𑌰𑌾𑌮𑍂𑌷𑌸𑌾𑌮𑌶𑍋𑌉𑌵𑌾।𑌚𑍀
𑌚𑍀𑌟𑌿𑌕𑍍𑌯𑌤𑍍𑌯𑌚
