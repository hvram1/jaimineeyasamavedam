।𑌵𑌾𑌸𑌿𑌷𑍍𑌠𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

11 ।𑌵𑌾𑌸𑌿𑌷𑍍𑌠𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌉𑌦𑍁𑌤𑍇𑌯𑌮𑌾।𑌧𑍂𑌮𑌕𑍍𑌤𑌮𑌾𑌗𑌾𑌇𑌰𑌾𑌸𑍍𑌤𑍋𑌮𑌾।𑌸𑌾𑌈
𑌤𑍀𑌕𑌟𑌿𑌚𑍀𑌣𑌕
𑌰𑌤𑌾𑌏।𑌸𑌾𑌤𑍍𑌰𑌾𑌜𑌾𑌇𑌤𑍋।𑌧𑌾𑌨𑌾𑌸𑌾𑌯𑌾।𑌕𑍍𑌷𑍀
𑌭𑍀𑌚𑌿𑌶𑍍𑌯𑌚𑌿𑌣𑌕
𑌤𑍋𑌤𑌯𑍋𑌵𑌾𑌜𑌾𑌯𑌾𑌨𑍍𑌤𑌾𑌃।𑌰𑌥𑌾𑌆𑌇𑌵𑌾।।21।।
𑌟𑌿𑌚𑌿𑌣𑌚𑌿𑌤𑍍𑌰
𑌉𑌦𑍁𑌤𑍇𑌯𑌮𑌾𑌧𑍁𑌮𑌾𑌕𑍍𑌤𑌮𑌾𑌃।𑌗𑍀𑌰𑌸𑍍𑌤𑍋𑌮𑌾𑌸𑌯𑌾
𑌫𑍀𑌕𑍀𑌟𑍂
𑌈𑌰𑌾𑌤𑌾𑌇।𑌸𑌤𑍍𑌰𑌾𑌜𑌿𑌤𑍋𑌧𑌨𑌾𑌸𑌾𑌯𑌾𑌕𑍍𑌷𑌿𑌤𑍋
𑌟𑍍𑌯𑌶𑍍𑌯𑌟𑍂𑌕𑍀
𑌤𑌾𑌯𑍋।𑌵𑌾𑌜𑌯𑌨𑍍𑌤𑍋𑌰𑌥𑌾𑌆𑌉𑌵𑌾।𑌈𑌵𑌾।।22।।
𑌟𑍍𑌯𑌤𑍂𑌤𑌿𑌚𑌶
𑌹𑌾𑌉𑌦𑍁𑌤𑍇𑌯𑌮𑌧𑍂𑌮𑌾𑌕𑍍𑌤𑌮𑍋𑌹𑌾𑌇।𑌗𑌾𑌇𑌰𑌸𑍍𑌤𑍋𑌮𑌾
𑌤𑌪𑍍𑌲𑌚𑍍𑌯𑌶𑍍𑌯𑌚𑌟𑍀
𑌸𑌯𑌾𑌇।𑌰𑌾𑌤𑍇।𑌸𑌤𑍍𑌰𑌾𑌜𑌿𑌤𑍋𑌧𑌨𑌾𑌸𑌾𑌯𑌾𑌕𑍍𑌷𑌿𑌤𑍋
𑌚𑌣𑌫𑌪𑍍𑌲𑌶𑌚𑌶𑌕𑍂𑌕𑍀
𑌤𑌾𑌯𑌾।𑌹𑌾𑌵𑌾𑌜𑌯𑌨𑍍𑌤𑍋𑌰𑌥𑌾𑌇𑌵𑍋𑌹𑌾𑌇।𑌵𑌾𑌜𑌯𑌨𑍍𑌤𑍋𑌰
𑌚𑍍𑌯𑌤𑌫𑌚𑍍𑌯𑌶𑍍𑌯𑌕𑍀
𑌥𑌾𑌈𑌵𑌾𑌔𑌹𑍋𑌬𑌾𑌹𑍋𑌇ழா𑌹।।23।।
𑌤𑍍𑌯𑌚𑌕𑌟𑍍𑌯𑌚𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
