2 ।𑌗𑍗𑌤𑌮𑌸𑍍𑌯𑌮𑌨𑌾𑌯𑍇𑌰𑌚𑍇 ।।

𑌏𑌨𑌾𑌵𑍋𑌅𑌗𑍍𑌨𑌿𑌮𑍍𑌮𑍇𑌨𑌮𑌸𑌾।𑌊𑌜𑍋𑌰𑌨𑌪𑌾
𑌤𑍂𑌤𑌿𑌕𑌟𑌿𑌚
𑌤𑌾𑌮𑌾𑌹𑍁𑌵𑍇𑌪𑍍𑌰𑌾𑌯𑌾𑌂।𑌚𑌾𑌇𑌤𑍀𑌷𑍍𑌠𑌾𑌮𑌾𑌰𑌤𑌿𑌂
𑌚𑍍𑌯𑌕𑍍𑌯𑌟𑌤𑌚𑌿𑌕𑍍𑌯𑌟𑍍𑌯𑌚
𑌸𑍂𑌆𑌧𑍍𑌵𑌾𑌰𑌾𑌂।𑌵𑌿𑌶𑍍𑌪𑌾𑌸𑍍𑌯𑌾𑌦𑍂।𑌤𑌾𑌮𑌾𑌮𑍃𑌤𑌾
𑌕𑍍𑌯𑌯𑌟𑌕𑍍𑌯𑌯𑌟𑌿𑌚𑌚𑌕
𑌮𑌿ழா𑌭𑌾।𑌓𑌇ழா
𑌟𑍍𑌯𑌚𑌣𑌪𑌶𑍍𑌯
𑌏𑌨𑌾𑌵𑍋𑌅𑌗𑍍𑌨𑌿𑌨𑍍𑌨𑌮𑌸𑌊𑌜𑍋𑌰𑌨𑌪𑍋𑌵𑌾।𑌤𑌾
𑌤𑍇𑌤
𑌮𑌾𑌹𑍁𑌵𑍇𑌪𑍍𑌰𑌾𑌯𑌾𑌂।𑌚𑌾𑌇𑌤𑍀𑌷𑍍𑌠𑌾𑌮𑌾𑌰𑌾𑌤𑍀𑌂।
𑌚𑍍𑌯𑌕𑍍𑌯𑌟𑌤𑌚𑌿𑌕𑍍𑌯𑌟𑌤
