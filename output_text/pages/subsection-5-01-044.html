43 ।𑌔𑌰𑍍𑌣𑌾𑌯𑌵𑍇𑌚𑍇…

𑌪𑌰𑌿𑌪𑍍𑌰𑌿𑌯𑌾𑌦𑌿𑌵:𑌕𑌾𑌵𑍀:।𑌵𑌾𑌯𑌾𑌂𑌸𑌿𑌨𑌪𑍍𑌤𑍇𑌯𑌾𑌰𑍍𑌹𑌿𑌤𑌾𑌸𑍍𑌵𑌾
𑌷𑌿𑌤𑍂𑌷𑌿𑌕𑍁
𑌨𑍈𑌰𑍍𑌯𑌾𑌤𑍀।𑌆𑌯𑌾𑌉𑌵𑌾𑌇𑌯𑌾𑌈𑌯𑌾।𑌕𑌾𑌵𑍀:𑌕𑍍𑌰𑌾𑌤𑍋
𑌟𑍍𑌯𑌤𑌕𑍍𑌯𑌶𑍀𑌟𑌤𑌚𑍍𑌯𑌟𑍍𑌯𑌟
𑌯𑌾𑌔𑌹𑍋𑌵𑌾।𑌈।।68।।
𑌚𑌶𑌿𑌚
𑌪𑌰𑌿𑌪𑍍𑌰𑌿𑌯𑌾𑌦𑌿𑌵;𑌕𑌵𑌾𑌇:।𑌵𑌯𑌾𑌹𑌾𑌉𑌵𑌾𑌂𑌸𑌿
𑌫𑍀𑌶𑍍𑌯𑌕𑍍𑌯𑌶𑌟𑍍𑌯𑌚𑌕𑌕𑌿
𑌨𑌪𑍍𑌤𑍇𑌯𑌾𑌵𑌾𑌰𑍍𑌹𑍀𑌤𑌾:।𑌸𑍍𑌵𑌾𑌨𑍈𑌰𑍍𑌯𑌾𑌤𑍀।𑌹𑍁𑌵𑌾
𑌕𑍀𑌚𑌕𑌟𑍍𑌯𑌤𑌕𑍍𑌯
𑌹𑍋𑌵𑌾𑌹𑍋𑌵𑌾𑌹𑍁𑌵𑌾𑌈𑌯𑌾।𑌕𑌾𑌵𑍀:𑌕𑍍𑌰𑌾𑌤𑍋
𑌕𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯𑌟𑌤𑌚𑍍𑌯𑌟𑍍𑌯𑌟
𑌯𑌾𑌔𑌹𑍋𑌵𑌾।𑌈𑌯𑌾𑌂।।69।।
𑌚𑌶𑌿𑌤𑌟𑌚
