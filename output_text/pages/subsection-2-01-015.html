14 ।𑌗𑍗𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌇𑌦𑌂𑌵𑌸𑌾𑌬𑍁।𑌸𑍂𑌤𑌮𑌾𑌨𑍍𑌧𑌾𑌃।𑌪𑍀𑌬𑌾𑌸𑍁𑌪𑍂
𑌤𑍀𑌶𑌟𑌿𑌤𑌚𑍍𑌯𑌕𑍍𑌯
𑌰𑍍𑌣𑌾𑌮𑍁𑌦𑌾𑌰𑌾𑌮𑌨𑌾𑌭𑌯𑌾𑌈𑌨𑍍।𑌰𑌰𑍋𑌬𑌾
𑌟𑌿𑌚𑌚𑍍𑌯𑌤𑍍𑌯𑌶𑌪𑍍𑌲
𑌮𑌾𑌤𑍋𑌹𑌾𑌇।।21।।
𑌫𑌪𑍍𑌲𑌶𑍍𑌯
𑌈𑌦𑌾𑌂𑌵𑌸𑍋𑌸𑍂𑌤𑌮𑌨𑍍𑌧𑌾𑌃।𑌪𑍀𑌬𑌾𑌸𑍂𑌪𑍁
𑌣𑍍𑌯𑌫𑌚𑍍𑌯𑌶𑍀𑌟𑌿𑌤
𑌰𑍍𑌣𑌾𑌮𑍂𑌦𑌾𑌰𑌂।𑌆𑌨𑌾𑌭𑌾𑌯𑌾𑌇𑌨𑍍।𑌰𑌰𑌾
𑌟𑍍𑌯𑌚𑌣𑌟𑌤𑌟𑌚𑌟𑌟𑍍𑌯
𑌹𑌾𑌉𑌵𑌾।𑌮𑌾𑌤𑍇।।22।।
𑌤𑌿𑌟𑍍𑌯𑌚
𑌇𑌦𑌂𑌵𑌸𑍋𑌸𑍁𑌤𑌮𑌨𑍍𑌧𑌾𑌏।𑌪𑍀𑌬𑌾𑌸𑍁𑌪𑍂𑌰𑍍𑌣𑌾
𑌤𑍄𑌤𑌕𑍍𑌯𑌕𑍍𑌯𑌕
𑌮𑍂𑌦𑌰𑍗𑌹𑍋𑌵𑌾।𑌪𑍀𑌬𑌾𑌸𑍁𑌪𑍂𑌰𑍍𑌣𑌾
𑌕𑌟𑍍𑌯𑌤𑌤𑌚𑍁
𑌮𑍂𑌦𑌰𑍗𑌹𑍋𑌵𑌾।𑌆𑌨𑌾𑌭𑌾𑌯𑍀𑌨𑍍।𑌰𑌰𑌿𑌮𑌾
𑌥𑌿𑌤𑌤𑌭𑌿𑌤𑌚𑌿
