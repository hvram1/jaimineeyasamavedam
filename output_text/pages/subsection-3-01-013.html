।𑌕𑌾𑌲𑍇𑌯𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

12 ।𑌕𑌾𑌲𑍇𑌯𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌤𑌰𑍋𑌭𑌿𑌵𑍋𑌰𑌵𑌿𑌦𑌾𑌚𑌸𑍂𑌂।𑌇𑌨𑍍𑌤𑍍𑌰𑌂𑌸𑌬𑌾𑌧𑌾𑌊𑌤𑌾
𑌟𑍍𑌯𑌚𑍀𑌶𑍍𑌯𑌚𑍀𑌟𑌚𑍍𑌯
𑌯𑌾𑌓𑌵𑌾𑌓𑌵𑌾।𑌪𑍍𑌰।𑌬𑍍𑌰𑌹𑌤𑍍𑌗𑌾𑌯𑌨𑍍𑌤𑌃𑌸𑍂𑌤𑌸𑍋𑌮𑌾
𑌕𑌚𑌶𑌪𑍍𑌲𑌚𑌣𑌚𑍂𑌕𑍍𑌯𑌚
𑌯𑌾𑌧𑍍𑌵𑌾𑌰𑌾।𑌪𑍍𑌰।𑌹𑍂𑌵𑌾𑌇𑌭𑌰𑌨𑍍𑌨𑌾𑌕𑌾𑌰𑌾𑌇𑌣𑌂।𑌓𑌵𑌾
𑌯𑌟𑍍𑌯𑌚𑍂𑌕𑌚𑌣𑍍𑌯𑌚𑌶𑌪𑍍𑌲
𑌓𑌇ழா𑌹।।17।।
𑌚𑌶𑍍𑌯
𑌤𑌾𑌰𑍋𑌭𑌿𑌵𑍋𑌰𑌵𑌿𑌦𑌾𑌚𑌸𑍂𑌂।𑌇𑌨𑍍𑌦𑌰𑌂𑌸𑌬𑌾𑌧𑌾𑌊
𑌟𑍀𑌚𑍍𑌯𑌶𑍍𑌯𑌚𑍍𑌯𑌕𑍍𑌯𑌚𑌕𑍍𑌯
𑌤𑌾𑌯𑌾।𑌔𑌹𑍋𑌵𑌾𑌔𑌹𑍋𑌵𑌾।𑌬𑍍𑌰𑌹𑌤𑍍𑌗𑌾𑌯𑌨𑍍𑌤𑌸𑍍𑌸𑍂
𑌯𑌟𑌟𑌤𑌟𑍍𑌯𑌤𑌤
𑌤𑌸𑍋𑌮𑌾𑌯𑌾𑌧𑍍𑌵𑌾𑌰𑌾।𑌪𑍍𑌰।𑌹𑍂𑌵𑌾𑌇𑌭𑌾𑌰𑌾।𑌪𑍍𑌰।𑌨𑌾
𑌚𑍃𑌕𑌯𑌟𑍍𑌯𑌚𑍍𑌯𑌯𑍍𑌯𑌟
𑌕𑌾𑌰𑌿𑌣𑌾𑌮𑌿ழா𑌹𑌭𑌾।𑌓𑌇ழா𑌹।।18।।
𑌚𑍍𑌯𑌟𑍀𑌚𑌣𑌪𑌶𑍍𑌯
𑌤𑌰𑍋𑌭𑌾𑌇𑌵𑍋𑌰𑌵𑍀𑌦𑌚𑌸𑍂𑌂।𑌇𑌨𑍍𑌦𑌰𑌂𑌸𑌬𑌾𑌧𑌊
𑌚𑍁𑌪𑍍𑌲𑌕𑌚𑌿𑌕𑍍𑌯𑌚
𑌤𑌾𑌯𑌾𑌇𑌬𑍍𑌰𑌹𑌾𑌤𑍍𑌗𑌾𑌯𑌾𑌨𑍍𑌤𑌾𑌸𑍂𑌤𑌾𑌸𑍋𑌮𑍇𑌆𑌧𑍍𑌵𑌾
𑌕𑌪𑌪𑍍𑌲𑌤𑍍𑌯𑌪𑌣𑌿𑌫𑍍𑌯𑌤
𑌰𑌾𑌇।𑌹𑍂𑌵𑌾𑌇𑌭𑌰𑍗𑌵𑌾𑌓𑌵𑌾।𑌨𑌕𑌾𑌰𑌿𑌣𑌾
𑌤𑌶𑌚𑍍𑌯𑌕𑌿𑌚𑌚𑌣𑌪𑍍𑌲
𑌹𑍋𑌇ழா𑌹।।19।।
𑌪𑍍𑌲𑌶𑍍𑌯
