।𑌵𑌿𑌰𑍂𑌪𑌸𑍍𑌯 𑌸𑌮𑌿𑌚𑍀𑌨 𑌪𑍍𑌰𑌾𑌚𑍀𑌨𑍇𑌚𑍇 ।

8 ।𑌵𑌿𑌰𑍂𑌪𑌸𑍍𑌯 𑌸𑌮𑌿𑌚𑍀𑌨 𑌪𑍍𑌰𑌾𑌚𑍀𑌨𑍇𑌚𑍇 ।

𑌹𑌾𑌬𑍁𑌤𑍍𑌵𑌮𑌿𑌨𑍍𑌦𑌰𑌾𑌯𑌾𑌶𑌾𑌯𑌸𑍀𑌹𑍋𑌇𑌹𑍋𑌇
𑌤𑍁𑌚𑍍𑌯𑌕𑍍𑌯
𑌹𑍋𑌯𑍇𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁।𑌪𑍍𑌰।𑍠𑌜𑍀𑌷𑍀
𑌪𑍂𑌶𑍂𑌕𑍍𑌯
𑌶𑌵𑌾𑌸𑌸𑍍𑌖𑌾𑌤𑌿𑌹𑍋𑌰।𑌪𑍍𑌰।𑌤𑍍𑌵𑌂𑌵𑍃𑌤𑍍𑌰𑌾𑌣𑌿
𑌚𑌿𑌚𑌿𑌕𑍀
𑌹𑌂𑌸𑍍𑌯𑌾𑌪𑍍𑌰𑌾𑌤𑍀𑌨𑍇𑌯𑌕𑌾𑌈𑌤𑍍𑌪𑍁𑌰𑍂𑌃।𑌪𑍍𑌰।
𑌚𑍍𑌯𑌚𑍍𑌯𑌶𑌚𑍍𑌯𑌶𑌕𑍍𑌯
𑌆𑌨𑍁𑌕𑍍𑌤𑌶𑍍𑌪𑌰𑍍𑌷𑌾𑌣𑌿𑌧𑍃𑌤𑍀𑌹𑍋𑌰𑌇𑌹𑍋𑌇𑌹𑍋𑌯𑍇
𑌕𑍂𑌚𑌕𑌪𑍂
𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌉𑌵𑌾।𑌸𑍍𑌵𑌰𑍍𑌮𑌹𑌾𑌃।।15।।
𑌪𑍍𑌲𑌕𑌟𑍍𑌯𑌚
𑌹𑍋𑌤𑍍𑌵𑌮𑌿𑌨𑍍𑌦𑌰𑌾।𑌯𑌾𑌶𑌾𑌯𑌸𑍀𑌹𑍋𑌯𑍇𑌹𑍋।𑍠𑌜𑍀
𑌤𑍀𑌚𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯𑌚𑌕𑍍𑌯
𑌷𑍀𑌶𑌵𑌾𑌸𑌸𑍍𑌖𑌾𑌤𑌿𑌹𑍋𑌰𑌯𑍇𑌹𑍋।𑌤𑍍𑌵𑌂𑌵𑍃𑌤𑍍𑌰𑌾𑌣𑍀
𑌚𑌿𑌚𑍍𑌯𑌕𑌟𑍍𑌯𑌚𑌕𑍀
𑌹𑌂𑌸𑍍𑌯𑌾𑌪𑍍𑌰𑌾𑌤𑍀𑌨𑍇𑌯𑌕𑌾𑌈𑌤𑍍।𑌪𑍁𑌰𑍂𑌹𑍋𑌰𑌯𑍇
𑌚𑍍𑌯𑌚𑌿𑌚𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯
𑌹𑍋𑌆𑌨𑍁𑌕𑍍𑌤𑌶𑍍𑌪𑌰𑍍𑌷𑌾𑌣𑌿𑌧𑍃𑌤𑍀𑌹𑍋𑌰𑌯𑍇𑌹𑍋
𑌚𑌟𑍃𑌟𑍍𑌯𑌚
𑌹𑌾𑌉𑌵𑌾।𑌸𑍍𑌵𑌰𑍍𑌮𑌯𑌾𑌃।।16।।
𑌤𑌿𑌕𑌟𑍍𑌯𑌚
