।𑌦𑍍𑌯𑍁𑌤𑌾𑌨𑌸𑍍𑌯 𑌮𑌾𑌰𑍂𑌤𑌸𑍍𑌯 𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।

4 ।𑌦𑍍𑌯𑍁𑌤𑌾𑌨𑌸𑍍𑌯 𑌮𑌾𑌰𑍂𑌤𑌸𑍍𑌯 𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।

𑌸𑌾𑌤𑍍𑌰𑌾𑌹𑌣𑌾𑌔𑌹𑍋𑌵𑌾।𑌦𑌾𑌧𑍃𑌷𑍀𑌂𑌤𑍂𑌮𑍍𑌰𑌮𑍀𑌨𑍍𑌦𑌰𑌾
𑌫𑌫𑌣𑌫𑌪𑍍𑌲𑌟𑍃
𑌉𑌵𑍋𑌵𑌾।𑌮𑌾𑌹𑌾𑌮𑌪𑌾𑌰𑌂𑌵𑍃𑌷𑌾𑌭𑌂𑌸𑍂𑌵𑌜𑍍𑌰𑌾𑌂𑌹
𑌚𑍍𑌯𑌶𑌕𑌕𑌿𑌕𑍀𑌟𑌿
𑌨𑍍𑌤𑌾𑌯𑍋𑌵𑍃।𑌤𑍍𑌰𑌂𑌸𑌨𑌿𑌤𑍋𑌤𑌾𑌵𑌾𑌜𑌾𑌂।𑌦𑌾𑌤𑌾𑌮𑌾
𑌟𑍍𑌯𑌚𑌣𑌪𑍀𑌚𑍍𑌯𑌤𑍍𑌰𑌥𑍍𑌯𑌚𑌕
𑌘𑌾𑌨𑌿𑌮𑌾𑌘𑌵𑌾𑌸𑍂𑌰𑌾𑌧𑌾𑌃।।5।।
𑌕𑌿𑌟𑌿𑌚𑌕𑍍𑌯𑌨
𑌸𑌾𑌤𑍍𑌰𑌾𑌹𑌣𑌨𑍍𑌦𑌾𑌧𑍃𑌷𑍀𑌇𑌨𑍍𑌤𑍂𑌮𑍃𑌮𑍀𑌨𑍍𑌦𑌰𑌾𑌂।𑌮𑌾𑌹𑌾𑌮𑌪𑌾
𑌫𑌫𑌫𑌚𑌿𑌶𑌕𑌕𑌿
𑌰𑌂𑌵𑍃𑌷𑌾𑌭𑌂𑌸𑍂𑌵𑌜𑍍𑌰𑌾𑌂𑌹𑌨𑍍𑌤𑌾𑌯𑍋𑌵𑍃।𑌤𑍍𑌰𑌂𑌸𑌨𑌿𑌤𑍋
𑌕𑌚𑌿𑌟𑌿𑌚𑌿𑌣𑌪𑍀
𑌤𑌾𑌵𑌾𑌜𑌾𑌂।𑌦𑌾𑌤𑌾𑌮𑌾𑌘𑌾𑌨𑌿𑌮𑌾𑌘𑌵𑌾𑌸𑍂𑌰𑌾
𑌚𑍍𑌯𑌤𑍍𑌰𑌥𑍍𑌯𑌚𑌕𑌕𑌿𑌟𑍍𑌯𑌚𑌕𑍍𑌯
