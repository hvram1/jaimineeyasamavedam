।𑌕𑌾𑌣𑍍𑌵𑍇𑌚𑍇 ।।

10 ।𑌕𑌾𑌣𑍍𑌵𑍇𑌚𑍇 ।।

𑌕𑌾𑌯𑌮𑌾𑌨𑍋𑌵𑌨𑌾𑌤𑍁𑌵𑌾𑌂।𑌯𑌨𑍍𑌮𑌾𑌤𑍃𑌰𑌾𑌜𑌾𑌗
 𑌫𑍀 𑌕𑍀 𑌕 𑌟𑍍𑌯 𑌕𑌿
𑌨𑍍𑌨𑌾𑌪𑌾:।𑌨𑌤𑌤𑍍𑌤𑌯𑌾𑌗𑍍𑌨𑌾𑌇।𑌪𑍍𑌰𑌮𑍃𑌷𑍇𑌹𑌾𑌇।
 𑌚 𑌣 𑌤𑍀 𑌤 𑌶 𑌟𑌿 𑌤 𑌶
𑌨𑍀𑌵𑌾𑌰𑍍𑌤𑌾𑌨𑌂।𑌯𑌦𑍍𑌦𑍂𑌰𑌾𑌇𑌸𑌾𑌨𑍍।𑌇𑌹𑌾𑌭𑍂
 𑌟𑍍𑌯 𑌚 𑌣 𑌟𑌿 𑌤𑍍𑌯 𑌕𑍍𑌯𑌚 𑌕
𑌵𑌾𑌔𑌹𑍋𑌬𑌾𑌹𑍋𑌇ழா
 𑌟𑍍𑌯 𑌚 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌪𑍍𑌲
𑌏𑌕𑌾𑌯𑌾।𑌮𑌾𑌨𑍋𑌵𑌨𑌾𑌤𑍂𑌵𑌾𑌮𑍋𑌇𑌤𑍂
 𑌤𑌿 𑌕𑍍𑌯 𑌕𑍍𑌯 𑌚 𑌪𑍍𑌲 𑌚𑍀
𑌵𑌾𑌮𑍁𑌹𑍁𑌵𑌾𑌹𑌾𑌇।𑌔𑌹𑍋𑌇𑌯𑌨𑍍𑌮𑌾𑌤𑍃𑌰𑌾𑌜𑌾
 𑌶𑍂 𑌟 𑌤 𑌚𑍍𑌯 𑌟𑍍𑌯
𑌗𑌨𑍍𑌨𑌾𑌪𑌾𑌆𑌪𑌾𑌉𑌹𑍁𑌵𑌾𑌹𑌾𑌇।𑌔𑌹𑍋
 𑌕𑌿 𑌚 𑌪𑍍𑌲 𑌚 𑌶𑍂 𑌟 𑌤𑌚
𑌇𑌨𑌾𑌤𑌤𑍍𑌤𑌯𑌾𑌗𑍍𑌨𑌾𑌇𑌪𑍍𑌰𑌮𑍃𑌷𑌾𑌇𑌨𑍀𑌵𑌾𑌰𑍍𑌤𑌾𑌨𑌾
 𑌕𑍍𑌯 𑌟𑌿   𑌟𑍀 𑌟𑍀 𑌚 𑌪𑍍𑌲
𑌆𑌭𑌾𑌨𑌾𑌮𑍁𑌹𑍁𑌵𑌾𑌹𑌾𑌇।𑌔𑌹𑍋𑌇𑌯𑌦𑍍𑌦𑍂𑌰𑍇
 𑌪𑍍𑌲 𑌚 𑌶𑍂 𑌟 𑌤𑌚 𑌕𑍀
𑌸𑌨𑍍𑌨𑌿𑌹𑌾𑌭𑍂𑌵𑌾𑌆𑌭𑍂𑌵𑌾𑌉𑌹𑍁𑌵𑌾𑌹𑌾𑌇।
 𑌟𑌿 𑌚 𑌪𑍍𑌲 𑌚𑍍𑌯 𑌶𑍍𑌰
𑌔𑌹𑍋𑌯𑌾𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾
 𑌟 𑌤 𑌚 𑌶𑌿 𑌚 𑌶
