।𑌮𑌹𑌾𑌵𑌾𑌮𑌦𑍇𑌵𑍍𑌯𑌂𑌤𑍃𑌤𑍀𑌯𑌂 ।

8 ।𑌮𑌹𑌾𑌵𑌾𑌮𑌦𑍇𑌵𑍍𑌯𑌂𑌤𑍃𑌤𑍀𑌯𑌂 ।

𑌕𑌾𑌯𑌾𑌨𑌶𑍍𑌪𑌾𑌇𑌤𑍍𑌰𑌾𑌯𑌾𑌭𑍁𑌵𑌾𑌤𑍍।𑌊𑌤𑍀
𑌣𑌫𑌫𑌚𑍍𑌯𑌶𑌿𑌕𑍍𑌯
𑌸𑍍𑌸𑌾𑌦𑌾𑌵𑍃𑌧𑌸𑍍𑌸𑌚𑌾𑌔𑌹𑍋𑌹𑌾𑌇𑌕𑌯𑌾𑌶
𑌕𑍍𑌯𑌕𑍁𑌤𑍍𑌯𑌟𑌿
𑌚𑌾𑌇।𑌷𑍍𑌠𑌯𑍗𑌹𑍋𑌹𑌿𑌮𑍍𑌮𑌾𑌵𑌾𑌕𑍍𑌤𑍇𑌰𑍍𑌾𑌹𑌾𑌇।।16।।
𑌤𑍍𑌯𑌶𑌤𑌿𑌟𑍍𑌯𑌟𑌚𑌤𑍍𑌰𑌶
𑌕𑌾𑌸𑍍𑌤𑍍𑌵𑌾𑌸𑌤𑍇𑌯𑌾𑌮𑌾𑌦𑌾𑌨𑌾𑌂।𑌮𑌂𑌹𑌿𑌷𑍋𑍍𑌠𑌮𑌤𑍍𑌸
𑌣𑌫𑌫𑍍𑌯𑌚𑌶𑍍𑌯𑌕𑌿
𑌦𑌨𑍍𑌧𑌸𑌾𑌔𑌹𑍋𑌹𑌾𑌇𑌦𑍃𑌢𑌾𑌚𑌿𑌦𑌾𑌰𑍂𑌜𑍗
𑌕𑍀𑌕𑌿𑌤𑍍𑌯𑌟𑌿𑌤𑍍𑌯
𑌹𑍋𑌹𑌿𑌮𑍍𑌮𑌾।𑌵𑌾𑌸𑍋𑌹𑌾𑌇।।17।।
𑌤𑌿𑌟𑍍𑌯𑌟𑌚𑌤𑍍𑌰𑌶
𑌆𑌭𑍀𑌷𑍂𑌣𑌾𑌸𑍍𑌸𑌾𑌚𑍀𑌨𑌾𑌂।𑌅𑌵𑌿𑌤𑌾𑌜𑌰𑌾𑌇𑌤𑍃𑌣𑌾𑌮𑌾
𑌣𑌫𑌫𑌚𑌶𑍍𑌯𑌕𑌿𑌕𑍀
𑌔𑌹𑍋𑌹𑌾𑌈𑌶𑌾𑌤𑌾𑌂𑌭𑌵𑌾।𑌸𑌿𑌯𑍗𑌹𑍋𑌹𑌿𑌮𑍍𑌮𑌾।𑌤𑌾
𑌤𑍁𑌟𑌿𑌤𑍍𑌯𑌤𑌿𑌟𑍍𑌯𑌟
