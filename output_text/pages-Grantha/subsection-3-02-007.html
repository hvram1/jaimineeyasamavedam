।𑌸𑌾𑌧𑍍𑌰𑌾𑌣𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

7 ।𑌸𑌾𑌧𑍍𑌰𑌾𑌣𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌤𑍍𑌵𑌮𑌿𑌨𑍍𑌦𑌰𑌾।𑌯𑌾𑌶𑌾𑌯𑌸𑍀𑌋𑌜𑍀𑌷𑍀𑌶𑌵𑌾𑌸𑌸𑍍𑌖𑌾𑌤𑌾𑌇।
𑌤𑌿𑌚𑍀𑌕𑍍𑌯𑌚𑌿𑌚𑌿𑌶
𑌤𑍍𑌵𑌂𑌵𑍍𑌰𑌤𑍍𑌰𑌾𑌣𑍀𑌹𑌂𑌸𑍍𑌯𑌾।𑌪𑍍𑌰𑌾𑌤𑍀𑌨𑌾𑌯𑍇𑌕𑌾𑌈𑌤𑍍𑌪𑍂
𑌤𑍀𑌤𑍍𑌯𑌕𑍀𑌚𑌕𑌿
𑌰𑍂𑌰𑌾𑌨𑍂।𑌹𑍋𑌕𑍍𑌤𑌾𑌶𑍍𑌪𑌾𑌰𑍍𑌷𑌾𑌣𑌾𑌔𑌹𑍋𑌵𑌾।
𑌭𑌟𑍍𑌯𑌟𑌿𑌟𑌟𑌚𑌶𑌿
𑌧𑌾𑌕𑍍𑌤𑌰𑌿𑌃।।12।।
𑌚𑌪𑍍𑌲
𑌤𑍍𑌵𑌮𑌾𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌾𑌶𑌾𑌯𑌸𑌾𑌇।𑍠𑌜𑌾𑌇𑌷𑍀𑌶𑌾𑌵𑌾𑌸𑌾।
𑌚𑍀𑌪𑍍𑌲𑌶𑌕𑌕𑌿𑌟𑌚𑍍𑌯𑌣𑌫𑌪𑍍𑌲
𑌸𑍍𑌖𑌾𑌤𑌿𑌃।𑌤𑍍𑌵𑌂𑌵𑍍𑌰𑌤𑍍𑌰𑌾𑌣𑍀𑌹𑌂𑌸𑍍𑌯𑌾𑌪𑍍𑌰𑌾𑌤𑌿𑌨𑍇𑌯
𑌚𑌶𑌕𑌕𑌿𑌚𑍍𑌯𑌚𑍍𑌯𑌶
𑌕𑌾𑌈𑌤𑍍।𑌪𑍁𑌰𑍂𑌰𑌾𑌨𑌾𑌉𑌵𑍋𑌵𑌾।𑌪𑍍𑌰।𑌕𑍍𑌤𑌾𑌶𑍍𑌪𑌾
𑌚𑍍𑌯𑌭𑍀𑌚𑍍𑌯𑌶𑌟𑍍𑌯
।𑌪𑍍𑌰।𑌰𑍍𑌷𑌣𑌾𑌇𑌧𑍍𑌰𑌤𑌿𑌃।𑌹𑍋𑌇ழா𑌹।।13।।𑍀
𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌹𑌾𑌬𑍁𑌤𑍍𑌵𑌮𑌿𑌨𑍍𑌦𑌰𑌾।𑌯𑌾𑌶𑌾𑌯𑌾𑌸𑍀𑌹𑌾𑌬𑍁।𑍠𑌜𑍀𑌷𑍀𑌶𑌾
𑌤𑍁𑌚𑌿𑌶𑌿𑌚𑌿𑌶
𑌵𑌾𑌸𑌸𑍍𑌖𑌾𑌤𑍀𑌰𑍍𑌹𑌾𑌬𑍁।𑌤𑍍𑌵𑌂𑌵𑍃𑌤𑍍𑌰𑌾𑌣𑍀𑌹𑌂𑌸𑍍𑌯𑌾𑌹𑌾𑌬𑍁।
𑌚𑌿𑌶𑌿𑌕𑌕𑍍𑌯𑌭𑍀𑌶
𑌪𑍍𑌰𑌾𑌤𑍀𑌨𑌾𑌯𑍇।𑌕𑌾𑌈𑌤𑍍𑌪𑍂𑌰𑍂𑌰𑍍𑌹𑌾𑌬𑍁।𑌆𑌨𑍁𑌕𑍍𑌤𑌾
𑌚𑌿𑌣𑌪𑌿𑌪𑍍𑌲𑌨𑍍𑌦𑌶𑌪𑌿
𑌶𑍍𑌪𑌾𑌰𑍍𑌹𑌾𑌬𑍁।𑌷𑌾𑌣𑌾𑌔𑌹𑍋𑌵𑌾।𑌧𑌾𑌕𑍍𑌤𑌰𑍀𑌃।।14।।
𑌫𑌨𑍍𑌦𑌶𑌟𑌚𑌶𑌿𑌚𑌪𑍍𑌲
