7 ।𑌸𑍗𑌬𑌭𑍍𑌰𑌵𑍇𑌚𑍇 ।

𑌪𑌾𑌹𑍀𑌗𑌾𑌯𑌾।𑌨𑍍𑌧𑌾𑌸𑌾𑌮𑌾𑌦𑌾𑌇।𑌆𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌮𑍇𑌧𑍀𑌯𑌾
𑌤𑍀𑌟𑌿𑌤𑌶𑌕𑍁𑌚𑌕
𑌤𑌾𑌇𑌥𑌾𑌇।𑌯𑌸𑍍𑌸𑌮𑍍𑌮𑌿𑌶𑍍𑌲𑍇𑌯𑌾𑌹𑌯𑍋𑌰𑌯𑍋𑌰𑌹𑌾𑌇𑌰𑌾𑌣𑍍𑌯𑌾
𑌟𑍍𑌯𑌤𑍍𑌯𑌶𑌟𑍀𑌯𑍂
𑌯𑌾𑌃।𑌇𑌨𑍍𑌦𑍇𑌰𑌾𑌵𑌾𑌜𑍍𑌰𑌿।𑌹𑍀𑌰𑍋𑌬𑌾𑌣𑍍𑌯𑌾𑌯𑍋𑌹𑌾𑌇।।12।।
𑌟𑍍𑌯𑌭𑌿𑌤𑌚𑌕𑌪𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌪𑌾𑌹𑍇𑌯𑌪𑌾𑌹𑍀।𑌗𑌾𑌯𑌾𑌨𑍍𑌧𑌾𑌸𑌾𑌮𑌾𑌦𑌾𑌇।𑌇𑌨𑍍𑌦𑌰𑌾𑌯
𑌤𑍀𑌥𑍍𑌯𑌤𑍍𑌯𑌟𑌤𑌶
𑌮𑍇𑌧𑍀𑌯𑌾𑌤𑌾𑌇𑌥𑌾𑌇।𑌯𑌸𑍍𑌸𑌮𑍍𑌮𑌿𑌶𑍍𑌲𑍋𑌹𑌯𑍋𑌰𑌯𑍋𑌰
𑌕𑍀𑌚𑌕𑌟𑍍𑌯𑌤𑍍𑌯𑌶𑌟𑍀
𑌹𑌾𑌇𑌰𑌾𑌣𑍍𑌯𑌾𑌯𑌾𑌃।𑌆𑌇𑌨𑍍𑌦𑍇𑌰𑌾𑌵𑌾𑌜𑍍𑌰𑍀।𑌹𑍀𑌰𑍋𑌬𑌾
𑌯𑍂𑌟𑍍𑌯𑌭𑍀𑌤𑌚𑌕𑌪𑌪𑍍𑌲
𑌣𑍍𑌯𑌾𑌯𑍋𑌹𑌾𑌇।।13।।
𑌪𑍍𑌲𑌶𑍍𑌯
