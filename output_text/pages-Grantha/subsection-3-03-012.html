11 ।𑌶𑍍𑌰𑍁𑌷𑍍𑌟𑌿𑌗𑌵𑌞𑍍𑌚 ।

𑌓𑌹𑌾𑌇𑌯𑌦𑌿𑌨𑍍𑌦𑌰𑌨𑌾𑌹𑍁𑌷𑍀𑌷𑍂𑌵𑌾।𑌓𑌜𑍋𑌨𑍍𑌰𑌮𑍍𑌣𑌾
 𑌤 𑌤𑍢 𑌤 𑌟𑍍𑌯 𑌟𑍍𑌯
𑌞𑍍𑌚𑌾𑌕𑍍𑌰𑌷𑍍𑌟𑌿𑌷𑍁।𑌐𑌹𑍀𑌆𑌇𑌹𑍀𑌹𑍋𑌵𑌾𑌉𑌉𑌵𑍋
 𑌕 𑌚𑌿 𑌟 𑌶𑍀 𑌕 𑌥 𑌚𑌿
𑌇।𑌪𑍍𑌰।𑌯𑌚𑌾𑌪𑌞𑍍𑌚𑌾𑌕𑍍𑌷𑌿𑌤𑍀𑌨𑌾𑌮𑍈𑌹𑍀।𑌪𑍍𑌰।
 𑌶 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌶 𑌟𑍍𑌯
𑌦𑍍𑌯𑍁𑌮𑍍𑌨𑌮𑌾𑌭𑌰𑌾𑌐𑌹𑍀।𑌪𑍍𑌰।𑌸𑌾𑌤𑍍𑌰𑌾𑌵𑌾𑌇𑌶𑍍𑌪𑌾
 𑌕𑍁 𑌟 𑌟𑍍𑌯 𑌟𑌿𑌚
𑌨𑍀𑌪𑍗𑌂𑌸𑍍𑌯𑌾𑌐𑌹𑍀𑌆𑌇𑌹𑍀𑌹𑍋𑌵𑌾𑌉𑌉𑌵𑍋
 𑌕𑍍𑌯 𑌟𑌿 𑌕𑌿 𑌕 𑌥 𑌟𑌿𑌟
𑌯𑌾𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾।।25।।
 𑌚 𑌶𑌿 𑌚 𑌶
