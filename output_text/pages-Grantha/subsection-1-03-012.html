।𑌵𑌸𑍁𑌰𑍋𑌚𑌿𑌷𑌸𑍍𑌸𑍗𑌰𑍍𑌯𑌵𑌚𑌰𑍍𑍍𑌚𑌸𑌸𑍍𑌯𑌸𑌾𑌮𑌨𑍀𑌚𑍇।।

12 ।𑌵𑌸𑍁𑌰𑍋𑌚𑌿𑌷𑌸𑍍𑌸𑍗𑌰𑍍𑌯𑌵𑌚𑌰𑍍𑍍𑌚𑌸𑌸𑍍𑌯𑌸𑌾𑌮𑌨𑍀𑌚𑍇।।

𑌶𑌨𑍍𑌨𑍋𑌦𑍇𑌵𑍀:।𑌆𑌭𑍀𑌷𑍍𑌟𑌾𑌯𑌾𑌇𑌶𑌨𑍍𑌨𑍋𑌭𑍁𑌵𑌾𑌂।
 𑌤𑍀 𑌟𑌿 𑌚 𑌶𑍁
𑌤𑍂𑌪𑍀𑌤𑌾𑌯𑌾𑌇𑌶𑌯𑍇𑌯𑌾𑌰𑌭𑍀:।𑌶𑍍𑌰𑌾𑌵𑌾𑌨𑍍𑌤𑍂𑌨𑌾
 𑌟𑌿 𑌚 𑌶𑍁 𑌟𑌿𑌟 𑌚
𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾
 𑌶𑌿 𑌚 𑌶
𑌹𑍁𑌵𑌾𑌹𑍋𑌇𑌶𑌨𑍍𑌨𑍋𑌦𑍇𑌵𑍀𑌰𑌭𑌿𑌷𑍍𑌟𑌯𑌾𑌇।
 𑌤𑍍𑌯 𑌪 𑌪𑍍𑌲 𑌨𑍍𑌦 𑌶
𑌹𑍁𑌵𑌾𑌹𑍋𑌇𑌶𑌨𑍍𑌨𑍋𑌭𑍁𑌵𑌾𑌂𑌤𑍂𑌪𑍀𑌤𑌾𑌯𑌾𑌇।
 𑌤𑍍𑌯 𑌪 𑌪𑍍𑌲 𑌨𑍍𑌦 𑌶
𑌹𑍁𑌵𑌾𑌹𑍋𑌇𑌶𑌯𑍇𑌯𑌾𑌰𑌭𑍀𑌸𑍍𑌰𑌵𑌨𑍍𑌤𑍁𑌨𑌾:।
