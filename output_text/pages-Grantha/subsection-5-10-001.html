।𑌵𑌸𑌿𑌷𑍍𑌠𑌸𑍍𑌯𑌪𑌦𑍇𑌚𑍇।

1 ।𑌵𑌸𑌿𑌷𑍍𑌠𑌸𑍍𑌯𑌪𑌦𑍇𑌚𑍇।

𑌇𑌨𑍍𑌦𑌰𑌮𑌚𑍍𑌛𑌾।𑌸𑍂𑌤𑌾𑌈𑌮𑌾𑌉𑌵𑍋𑌵𑌾।𑌵𑍃𑌷𑌾𑌣𑌯𑍍𑌯
𑌤𑍀𑌟𑍀𑌚𑍍𑌯𑌶𑌕𑍍𑌯𑌥𑍍𑌯
𑌨𑍍𑌤𑍂𑌹𑌾𑌰𑌯𑌾𑌉𑌵𑍋𑌵𑌾।𑌶𑍍𑌰𑍂𑌷𑍍𑌟𑌾𑌇।𑌜𑌾𑌤𑌾𑌸𑌇
𑌟𑌚𑌕𑌟𑍍𑌯𑌚𑍍𑌯𑌶𑌚𑍍𑌯𑌶
𑌨𑍍𑌦𑌾𑌵𑌾𑌃।𑌸𑍂𑌵𑌰𑍍𑌵𑌾𑌇𑌦𑌾𑌃।𑌓𑌇ழா𑌹।।1।।
𑌚𑍂𑌟𑌿𑌚𑍍𑌯𑌣𑌪𑌶𑍍𑌯
𑌇𑌨𑍍𑌦𑌰𑌮𑌚𑍍𑌛𑌾।𑌸𑍂𑌤𑌾𑌈𑌮𑌾𑌇𑌤𑌾𑌈𑌮𑌾𑌇।𑌵𑍃𑌷𑌾𑌣𑌯𑍍𑌯
𑌤𑍀𑌟𑍀𑌟𑌿𑌤𑌶𑌕𑍍𑌯𑌥𑍍𑌯
𑌨𑍍𑌤𑍂𑌹𑌾𑌰𑌯𑌾𑌹𑌾𑌰𑌾𑌯𑌾𑌃।𑌶𑍍𑌰𑍁𑌷𑍍𑌟𑍇𑌜𑌾𑌤𑌾𑌸𑌾𑌇𑌨𑍍𑌦𑌵𑌾
𑌚𑌕𑌟𑍍𑌯𑌕𑌟𑌤𑌕𑍍𑌯𑌥𑍍𑌯𑌟𑌟𑌿
𑌆𑌇𑌨𑍍𑌦𑌾𑌵𑌾𑌃।𑌸𑍁𑌵𑌰𑌾𑌔𑌹𑍋𑌵𑌾𑌵𑌿𑌦𑍋𑌵𑍀𑌦𑌾𑌃।।2।।
