।𑌆𑌂𑌗𑍀𑌰𑌸𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

6 ।𑌆𑌂𑌗𑍀𑌰𑌸𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌅𑌚𑍋𑌹𑌾𑌇।𑌦𑌾𑌸𑍋𑌨𑍋𑌧𑌨𑍁𑌵𑌾𑌨𑍍𑌤𑍁𑌵𑌿𑌨𑍍𑌦𑌾𑌵𑌾𑌃।
𑌤𑍍𑌯𑌤𑌶𑌷𑌿𑌕𑍁𑌟𑍍𑌯
𑌪𑍍𑌰𑌸𑍍𑌵𑌾𑌨𑌾𑌸𑍋𑌬𑍃𑌹𑌦𑍍𑌦𑍇𑌵𑌾𑌇𑌷𑍂𑌹𑌰𑌯𑌾𑌃।𑌵𑌿𑌚𑍋
𑌕𑍃𑌕𑌿𑌟𑌿𑌚𑍍𑌯
𑌹𑌾𑌇𑌦𑌶𑍍𑌨𑍋𑌹𑌾𑌇।𑌨𑌾𑌇𑌷𑌯𑍋𑌅𑌰𑍋𑌹𑌾𑌇।
𑌶𑍍𑌯𑌚𑍍𑌯𑌶𑍍𑌯𑌚𑍂𑌣𑌶
𑌤𑌾𑌯𑌾𑌅𑌯𑍋𑌰𑌹𑌾𑌇𑌨𑌸𑍍𑌸𑍋𑌹𑌾𑌨𑍍𑌤𑍁𑌸𑌾𑌨𑍀𑌷𑌾𑌂।
𑌕𑍍𑌯𑌚𑍍𑌯𑌶𑍍𑌯𑌚𑍍𑌯𑌶𑌚𑌿𑌣
𑌤𑍁𑌨𑌾𑌉𑌵𑌾।𑌧𑍀𑌯𑌾𑌃।।10।।
𑌟𑍍𑌯𑌤𑍍𑌯𑌚𑌪𑍍𑌲
𑌅𑌚𑍗𑌹𑍋𑌵𑌾।𑌦𑌾𑌸𑍋𑌨𑍋𑌧𑌨𑍁𑌵𑌾𑌨𑍍𑌤𑍁𑌵𑌿𑌨𑍍𑌦𑌾𑌵𑌾𑌃।
𑌤𑌿𑌤𑌷𑌿𑌕𑍂𑌟
𑌪𑍍𑌰𑌸𑍍𑌵𑌨𑌾𑌸𑍋𑌬𑍃𑌹𑌦𑍍𑌦𑍇𑌵𑌾𑌇𑌷𑍁𑌹𑌾𑌰𑌾𑌯𑌾𑌃।𑌵𑌾𑌇𑌚𑍀𑌦𑌾
𑌷𑍁𑌕𑍁𑌟𑌿𑌟𑌿𑌚
𑌶𑍍𑌨𑌾𑌨𑌾𑌇𑌷𑌾𑌯𑍋।𑌹𑍋𑌅𑌰𑌾𑌤𑌾𑌯𑌾𑌃।𑌹𑍋𑌆𑌯𑍇𑌰𑍍𑌯𑌾
𑌚𑌚𑌿𑌶𑌚𑍀𑌣𑌤𑌟𑍍𑌯
𑌨𑌾𑌸𑍍𑌸𑌾𑌨𑍍𑌤𑍁𑌸𑌾𑌨𑍀𑌷𑌾।𑌹𑍋𑌨𑍍𑌤𑍁𑌨𑍋𑌧𑌾𑌯𑌾𑌃।।11।।
𑌚𑌶𑌚𑌿𑌣𑌚𑍀𑌤𑍍𑌰
𑌅𑌚𑍗𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌦𑌾𑌸𑍋𑌨𑍋𑌧𑌨𑍁𑌵𑌾𑌨𑍍𑌤𑍁𑌵𑌿𑌨𑍍𑌦𑌾
𑌤𑍀𑌤𑌶𑌷𑌿𑌕𑍁
𑌵𑌾𑌃।𑌪𑍍𑌰𑌸𑍍𑌵𑌾𑌨𑌾𑌸𑍋𑌬𑍃𑌹𑌦𑍍𑌦𑍇𑌵𑌾𑌇𑌷𑍂𑌹𑌾𑌰𑌾𑌯𑌾𑌃।𑌵𑌾
𑌟𑍍𑌯𑌷𑌿𑌕𑍃𑌟𑌿𑌟
𑌇𑌚𑍀𑌦𑌾𑌶𑍍𑌨𑌾𑌨𑌾𑌇𑌷𑌯𑍋𑌅𑌰𑌾𑌤𑌾𑌯𑌾𑌃।𑌆𑌯𑍋𑌰𑌨𑌾
𑌤𑍍𑌯𑌟𑌚𑌣𑌫𑍁𑌤𑌤𑌟𑌤𑌟
𑌸𑍍𑌸𑌾𑌨𑍍𑌤𑍂𑌸𑌨𑍀𑌷𑌾𑌨𑍍𑌤𑍁𑌨𑍋𑌧𑌾𑌯𑌾𑌃।।12।।
𑌚𑌣𑌫𑌿𑌚𑌿𑌤𑍍𑌰
