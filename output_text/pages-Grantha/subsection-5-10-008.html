8 ।𑌇𑌨𑍍𑌦𑌰𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।

𑌪𑍍𑌰𑌾𑌣𑌾𑌶𑌿𑌶𑍂𑌃।𑌮𑌹𑌾𑌇𑌨𑌮𑍗𑌹𑍋𑌵𑌾।𑌹𑌿𑌨𑍍𑌵𑌨𑍍𑌵𑌰𑌤𑌸𑍍𑌯
𑌤𑍀𑌟𑍍𑌯𑌚𑍀𑌤𑍍𑌰𑌕
𑌦𑍀𑌧𑌿𑌤𑌿𑌂।𑌵𑌾𑌇𑌶𑍍𑌪𑌾𑌉𑌵𑌾𑌪𑌾𑌰𑌾𑌉𑌵𑌾।𑌪𑍍𑌰𑌿𑌯𑌾𑌭𑍁𑌵
𑌕𑍃𑌕𑌿𑌕𑍍𑌯𑌕𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯
𑌦𑌧𑌾𑌉𑌵𑌾।𑌏𑌚𑌿𑌤𑌾।।21।।
𑌕𑍀𑌤𑍍𑌯𑌤𑌤𑍍𑌯𑌚
𑌪𑍍𑌰𑌾𑌣𑌾𑌹𑍋𑌇𑌯𑌾𑌹𑌾𑌇।𑌅𑌶𑌾𑌇𑌶𑍂𑌮𑍍𑌮𑌰𑌹𑌾𑌇𑌨𑌾𑌮𑌾।
𑌚𑌿𑌶𑍀𑌟𑌿𑌚𑌟𑍀𑌚
𑌬𑍁𑌹𑍋𑌔𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌹𑌿𑌨𑍍𑌵𑌨𑍍𑌨𑍃𑌤𑌸𑍍𑌯𑌦𑍀𑌧𑌿𑌤𑌾𑌇𑌮𑌾
𑌕𑍍𑌯𑌕𑌕𑍍𑌯𑌚𑍇
।𑌪𑍍𑌰।𑌵𑌿𑌶𑌵𑌾𑌪𑌾𑌰𑌾।𑌪𑍍𑌰।𑌪𑍍𑌰𑍀𑌯𑌾𑌭𑍂𑌵𑌾𑌔𑌹𑍋
𑌟𑍀𑌟𑌿𑌟𑌚
𑌵𑌾।𑌆𑌧𑌾𑌚𑌿𑌤𑌾।।22।।
𑌶𑌿𑌚𑌤𑌿𑌚
