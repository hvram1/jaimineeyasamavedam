।𑌶𑍈𑌤𑍇𑌷𑍍𑌯𑌾𑌣𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿 ।

9 ।𑌶𑍈𑌤𑍇𑌷𑍍𑌯𑌾𑌣𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿 ।

𑌤𑍍𑌵𑌂𑌹𑌿𑌯𑌾𑌂।𑌗𑌦𑌾𑌇𑌵𑍍𑌯𑌪𑌵𑌮𑌾𑌨𑌜𑌨𑌿𑌮𑌾𑌨𑌿𑌦𑍍𑌯𑍁𑌮𑌾𑌕𑍍𑌤
𑌤𑌿𑌚𑍇𑌕𑌚𑌕𑍍𑌯𑌟
𑌮𑌾𑌃।𑌆𑌮𑌾𑌕𑍍𑌤𑌰𑌾𑌤𑍍𑌵𑌾𑌯𑌾𑌘𑍋𑌬𑌾𑌷𑌾𑌯𑌾𑌨𑍍𑌹𑌾𑌇।।27।।
𑌟𑍍𑌯𑌟𑍍𑌯𑌟𑍍𑌯𑌕𑌪𑌪𑍍𑌲𑌶𑍀
𑌤𑍍𑌵𑌂𑌹𑌿𑌯𑌾𑌂।𑌗𑌦𑌾𑌇𑌵𑍍𑌯𑌪𑌵𑌮𑌾𑌨𑌾𑌹𑍋𑌵𑌾𑌹𑌾𑌇।
𑌤𑌿𑌕𑍃𑌚𑌕𑌟𑍍𑌯𑌤𑌶
𑌜𑌨𑌿𑌮𑌾𑌨𑍀𑌰𑍍𑌦𑍂𑌮𑌾𑌕𑍍𑌤𑌰𑌾𑌮𑌾𑌃।𑌆𑌮𑌾𑌕𑍍𑌤𑌰𑌾𑌤𑍍𑌵𑌾𑌔𑌹𑍋
𑌟𑌿𑌚𑌪𑍍𑌲𑌤𑌤𑌟𑌿𑌟𑌚
𑌵𑌾।𑌯𑌘𑍋𑌷𑌾𑌯𑌾𑌨𑍍।।28।।
𑌶𑌿𑌕𑍍𑌯𑌨𑍍𑌯
𑌤𑍍𑌵𑌂𑌹𑍍𑌯𑌂𑌗𑌦𑌾।𑌇𑌵𑍍𑌯𑌪𑌵𑌮𑌾𑌨𑌜𑌨𑌿𑌮𑌾𑌨𑌿𑌦𑍍𑌯𑍂𑌮𑌾𑌕𑍍𑌤𑌾𑌮𑌾𑌃।
𑌤𑍀𑌕𑍢𑌟𑌿𑌤𑍍𑌯
𑌆𑌮𑌾𑌤𑍍𑌤𑌰𑌾𑌤𑍍𑌵𑌾𑌯𑌾𑌘𑍋𑌬𑌾𑌷𑌾𑌯𑌾𑌨𑍍𑌹𑌾𑌇।।29।।
𑌟𑌿𑌤𑌚𑌕𑌪𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌤𑍍𑌵𑌂𑌹𑌾𑌯𑌾𑌂𑌗𑌾𑌦𑌾𑌇𑌵𑍍𑌯𑌾𑌂।𑌪𑌾𑌵𑌮𑌾𑌨𑌾𑌜𑌨𑌿𑌮𑌾𑌨𑌾𑌯𑍇
𑌚𑌿𑌪𑍍𑌲𑌟𑌿𑌂𑌟𑌿𑌚𑌕
𑌦𑍍𑌯𑍂𑌮𑌾𑌕𑍍𑌤𑌰𑌾𑌮𑌾𑌃।𑌆𑌮𑌾𑌕𑍍𑌤𑌰𑌾𑌤𑍍𑌵𑌾।𑌯𑌾𑌘𑍋𑌬𑌾𑌷𑌾
𑌪𑌪𑍍𑌲𑌤𑌤𑌭𑌿𑌤𑌚𑌕𑌪𑌪𑍍𑌲
𑌯𑌾𑌨𑍍𑌹𑌾𑌇।।30।।
𑌪𑍍𑌲𑌶𑍍𑌯
