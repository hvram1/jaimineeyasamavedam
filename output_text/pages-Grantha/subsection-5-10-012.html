।𑌸𑍁 𑌜𑍍𑌞𑌾𑌨𑍇𑌚𑍇 ।

12 ।𑌸𑍁 𑌜𑍍𑌞𑌾𑌨𑍇𑌚𑍇 ।

𑌸𑍋𑌮ǐ𑌪𑍁𑌨𑌾।𑌨𑌊𑌰𑍍𑌮𑍍𑌮𑍀𑌣𑌾𑌵𑍍𑌯𑌂𑌵𑌾𑌰𑌾𑌂𑌵𑌿𑌧𑌾𑌵
 𑌤𑍀 𑌕𑍍𑌯 𑌚𑍍𑌯 𑌥 𑌟𑍍𑌯 𑌕𑍍𑌯
𑌤𑌾𑌇।𑌅𑌗𑍇𑌰𑌵𑌾𑌚𑌾𑌃𑌪𑌾𑌵𑌮𑌾𑌨𑌾𑌔𑌹𑍋𑌵𑌾।
 𑌚𑍍𑌯 𑌶 𑌥𑍍𑌯 𑌟𑍍𑌯𑌚 𑌕 𑌟𑍍𑌯𑌟 𑌚 𑌶𑌿
𑌕𑌨𑍀𑌕𑍃𑌦𑌦𑍇𑌊𑌪𑌾।।26।।
 𑌕𑍍𑌯 𑌤𑌿 𑌟 𑌚
𑌸𑍋𑌮ǐ𑌪𑍂𑌨𑌾।𑌨𑌊𑌰𑍍𑌮𑍍𑌮𑌿𑌣𑌾𑌉𑌵𑌾𑌓𑌵𑌾𑌔𑌹𑍋
 𑌤𑍀 𑌕𑍍𑌯 𑌕𑍍𑌯 𑌕𑍍𑌯 𑌟𑌟 𑌚
𑌵𑌾।𑌪𑍍𑌰।𑌅𑌵𑍍𑌯𑌂𑌵𑌾𑌰𑌂𑌵𑌿𑌧𑌾𑌵𑌤𑍍𑌯𑌗𑍍𑌰𑌾𑌉𑌵𑌾।𑌪𑍍𑌰।
 𑌶𑌿 𑌟𑍂 𑌤𑌿 𑌕𑍍𑌯
𑌵𑌾𑌚ǐ𑌪𑌵𑌮𑌾𑌨Ǐ𑌕𑌨𑌾𑌉𑌵𑌾।𑌪𑍍𑌰।𑌕𑍍𑌰𑌾𑌦𑌾𑌤𑍍।।27।।
 𑌕 𑌚𑍃 𑌕𑍍𑌯 𑌚 𑌶
