2 ।𑌰𑍗𑌦𑍇𑌰𑌚𑍇 ।।

𑌤𑌚𑍋𑌗𑌾𑌯𑌾।𑌸𑍁𑌤𑌾𑌇𑌸𑌚𑌾𑌪𑍁𑌰𑍂𑌹𑍁𑌤𑌾𑌹𑌾
 𑌤𑍀 𑌤𑍍𑌯 𑌤𑌿 𑌟𑍍𑌯 𑌚𑍍𑌯 𑌶𑌪𑍍𑌲
𑌹𑌾𑌇𑌯𑌾𑌸𑌤𑍍𑌵𑌾𑌨𑌾𑌇।𑌶𑌂𑌯𑌤𑍍𑌗𑌾𑌵𑍇।𑌨𑌾𑌶𑍗𑌹𑍗
 𑌤 𑌚𑍍𑌯 𑌚𑍍𑌯 𑌣 𑌶 𑌕 𑌟𑌿 𑌤 𑌟𑍍𑌯 𑌟𑌚
𑌹𑍂𑌵𑍋𑌬𑌾𑌕𑌾𑌇𑌨𑍋𑌹𑌾𑌇।।2।।
 𑌕 𑌪 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶
𑌤𑌚𑍋𑌗𑌾𑌯𑌸𑍁𑌤𑍇𑌸𑌾𑌚𑌾𑌏।𑌪𑍁𑌰𑍁𑌹𑍂𑌤𑌾𑌯𑌸𑌤𑍍𑌵𑌾
 𑌤𑍃 𑌤 𑌕𑌿
𑌨𑍇।𑌪𑍁𑌰𑍂𑌹𑍂𑌤𑌾𑌯𑌾𑌸𑌾𑌤𑍍𑌵𑌾𑌨𑌾𑌇𑌶𑌂𑌯𑌾𑌤𑍍
 𑌕𑍁 𑌚𑍍𑌯 𑌥𑍍𑌯 𑌚 𑌯 𑌟 𑌚 𑌫𑌿
𑌗𑍗𑌵𑌾𑌉𑌵𑍋𑌵𑌾।𑌨𑌾𑌶𑌾𑌕𑌾𑌇𑌨𑍋𑌹𑌾𑌇।।3।।
 𑌚𑍀 𑌣 𑌪 𑌶 𑌚 𑌪𑍍𑌲 𑌶𑍍𑌯
