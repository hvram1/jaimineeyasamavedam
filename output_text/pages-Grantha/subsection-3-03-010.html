।𑌵𑍈𑌷𑍍𑌟𑌂 𑌭𑍇𑌚𑍇 ।

9 ।𑌵𑍈𑌷𑍍𑌟𑌂 𑌭𑍇𑌚𑍇 ।

𑌔𑌹𑍋𑌵𑌾𑌇𑌵𑌯𑌂𑌘𑌤𑍍𑌵𑌾𑌸𑍁𑌤𑌾𑌵𑌨𑍍𑌤𑌔𑌹𑍋𑌵𑌾।
𑌷𑌤𑍢𑌤
𑌔𑌹𑍋𑌵𑌾𑌯𑌾𑌪𑍋𑌨𑌵𑍃𑌕𑍍𑌤𑌬𑌰𑍍𑌹𑌿𑌷𑌃𑌪𑌵𑌾𑌇𑌤𑍍𑌰𑌾𑌸𑍍𑌯𑌾।
𑌷𑌪𑌶
𑌪𑍍𑌰𑌸𑍍𑌰𑌵𑌣𑍇𑌷𑍁𑌵𑌾𑌤𑍍𑌰𑌾𑌹𑌾𑌨𑍍।𑌔𑌹𑍋𑌵𑌾𑌔
𑌯𑍂𑌪𑌶𑌤𑍍𑌯𑌤𑌷
𑌹𑍋𑌵𑌾𑌇𑌪𑌰𑌿𑌸𑍍𑌤𑍋𑌤𑌾।𑌰𑌆𑌸𑌾𑌤𑌾𑌔𑌹𑍋𑌬𑌾
𑌟𑍂𑌤𑌕𑍍𑌯𑌕𑌟𑍍𑌯𑌚𑌪𑍍𑌲
𑌹𑍋𑌇ழா𑌹।।22।।
𑌪𑍍𑌲𑌶𑍍𑌯
𑌵𑌯𑌂𑌘𑌤𑍍𑌵𑍋𑌹𑌾𑌇𑌸𑍁𑌤𑌾𑌵𑌨𑍍𑌤𑍋𑌵𑌾।𑌆𑌪𑍋𑌨𑌵𑍃𑌕𑍍𑌤𑌾
𑌤𑍀𑌤𑍂𑌤𑌕𑍁
𑌬𑌾𑌰𑍍𑌹𑌾𑌇𑌷𑌾𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌪𑍍𑌰।𑌪𑌾𑌵𑌿𑌤𑍍𑌰𑌸𑍍𑌯
𑌚𑌯𑌟𑍍𑌯𑌟𑍍𑌯𑌤𑌶𑌕𑍀
𑌪𑍍𑌰𑌸𑍍𑌰𑌵𑌣𑍇𑌷𑍂𑌵𑌾𑌤𑍍𑌰𑌾𑌹𑌾𑌨𑍍।𑌪𑍍𑌰।𑌪𑌰𑌾𑌇
𑌕𑍀𑌚𑌕𑍍𑌯𑌯𑌟𑌚𑍍𑌯
𑌸𑍍𑌤𑍋𑌤𑌾।𑌪𑍍𑌰।𑌰𑌆𑌸𑌾𑌤𑌾𑌔𑌹𑍋𑌵𑌾।𑌦𑍀𑌶𑌾𑌃।।23।।
𑌯𑍍𑌯𑌟𑌟𑌿𑌟𑌚𑌶𑌿𑌚𑌶
