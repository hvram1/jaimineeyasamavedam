38 ।𑌆𑌶𑍍𑌪𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿…

𑌆𑌸𑌾।𑌵𑍍𑌯𑌂𑌶𑍁𑌰𑍍𑌮𑍍𑌮𑌾𑌆𑌉𑌵𑌾।𑌦𑌾𑌯𑌾𑌪𑍍𑌸𑍁𑌦𑌾𑌕𑍍𑌷𑌾:।
𑌤𑍍𑌯𑌕𑌤𑍍𑌯𑌟𑌿𑌚𑌶𑌿
𑌗𑌿𑌰𑌾𑌆𑌉𑌵𑌾𑌏।𑌷𑍍𑌠𑌾:।𑌶𑍇𑌯𑌨𑍋𑌨𑌾𑌯𑍋।𑌨𑌿𑌮𑌾
𑌤𑍍𑌯𑌭𑍀𑌚𑌕𑍍𑌯𑌟𑌤𑌤𑍍𑌯
𑌆𑌉𑌵𑌾𑌸𑌾𑌦𑌾𑌤𑍍।।52।।
𑌟𑌿𑌚𑌶
𑌅𑌸𑌾𑌵𑌿𑌯𑌾।𑌶𑍁𑌰𑍍𑌮𑌦𑌾𑌯𑌾𑌪𑍍𑌸𑍁𑌦𑌕𑍍𑌷𑍋𑌗𑌾𑌇𑌰𑌿𑌷𑍍𑌠𑌾:।𑌶𑍍𑌯𑌾
𑌤𑍀𑌟𑌿𑌚𑌕𑌕𑍃
𑌇𑌨𑍋𑌨𑌾𑌯𑍋।𑌨𑍀𑌮𑍋𑌬𑌾𑌸𑌾𑌦𑌾𑌤𑍍𑌹𑌾𑌇
𑌯𑌿𑌟𑍍𑌯𑌚𑌕𑌪𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌅𑌸𑌾𑌵𑍍𑌯𑌾𑌉𑌵𑍋𑌵𑌾।𑌪𑍍𑌰।𑌶𑍁𑌮𑍍𑌮𑌰𑌦𑌾𑌯𑌾।𑌪𑍍𑌰।𑌆𑌪𑍍𑌸𑌾
𑌫𑍍𑌯𑌚𑌿𑌶𑌕𑌟𑌿𑌟𑍍𑌯
।𑌪𑍍𑌰।𑌦𑌕𑍍𑌷𑍋𑌗𑌿𑌰𑌾𑌇𑌷𑍍𑌠𑌾।𑌪𑍍𑌰।𑌶𑍇𑌯𑌨𑍋𑌨𑌯𑍋𑌨𑌿𑌮𑌾
𑌟𑍂𑌕
𑌸𑌾𑌦𑌾𑌤𑍍।।54।।
𑌟𑍂𑌚
𑌅𑌸𑌾𑌵𑍍𑌯𑍗𑌵𑌾𑌉𑌵𑍋𑌵𑌾।𑌶𑍁𑌰𑍍𑌮𑌦𑌾𑌯𑍗𑌵𑌾𑌉𑌵𑍋𑌵𑌾
𑌫𑍍𑌯𑌚𑍀𑌶𑌕𑌕𑌿𑌚𑌿𑌶
।𑌪𑍍𑌰।𑌆𑌪𑍍𑌸𑍂𑌦𑌕𑍍𑌷𑍗।𑌪𑍍𑌰।𑌗𑍀𑌰𑌾𑌇𑌷𑍍𑌠𑍗।𑌪𑍍𑌰।𑌶𑍇𑌯𑌨𑍋
𑌫𑍍𑌯𑌕𑍀𑌫𑍍𑌯
𑌨𑌯𑍗।𑌪𑍍𑌰।𑌨𑍀𑌮𑌾𑌸𑌾𑌦𑍗𑌵𑌾𑌉𑌵𑍋𑌬𑌾𑌹𑍋𑌇ழா𑌹।।55।।
