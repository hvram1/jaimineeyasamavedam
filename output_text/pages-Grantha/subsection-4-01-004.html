। 𑌗𑍃 𑌤𑍍 𑌸𑌮𑌦𑌸𑍍𑌯𑌯𑍋𑌨𑌿𑌨𑍀𑌚𑍇 ।

3 । 𑌗𑍃 𑌤𑍍 𑌸𑌮𑌦𑌸𑍍𑌯𑌯𑍋𑌨𑌿𑌨𑍀𑌚𑍇 ।

𑌯𑍋𑌨𑍀𑌃।𑌤𑌆𑌇𑌨𑍍𑌦𑌰𑌾𑌸𑌾𑌦𑌾𑌨𑌾𑌯𑌕𑌾𑌰𑍀𑌤𑌾𑌮𑌾।
𑌤𑍍𑌯𑌟𑍀𑌕𑌿𑌚𑍍𑌯𑌶𑌿
𑌨𑍃𑌭𑌾𑌇𑌃𑌪𑍂𑌰𑍂𑌹𑍂𑌤𑌾𑌪𑍍𑌰𑌯𑌾𑌹𑍀𑌆𑌸𑌾𑌃।𑌯𑌥𑌾
𑌕𑍂𑌚𑌿𑌶𑌿
𑌨𑍋𑌆𑌵𑍀𑌤𑌾𑌵𑍃𑌧𑌾𑌶𑍍𑌪𑌿𑌤𑍍𑌦𑌾𑌦𑌾𑌃।𑌵𑌸𑍂𑌨𑌿𑌮𑌾
𑌭𑌿𑌕𑌿𑌚𑍍𑌯𑌶𑌿𑌭𑍀
𑌮𑌦𑌾𑌶𑍍𑌪𑌾𑌸𑍋।𑌮𑌾𑌇𑌃।।3।।
𑌪𑍍𑌲𑌚𑍍𑌯𑌤𑍍𑌰𑌶
𑌯𑍋𑌨𑌿𑌷𑍍𑌟𑌆𑌇𑌨𑍍𑌦𑌰𑌸𑌦𑌨𑌾𑌇𑌹𑍋𑌵𑌾।𑌆𑌕𑌾𑌰𑌾
𑌚𑌚𑌿𑌶𑍄𑌕𑍍𑌯𑌚
𑌇𑌤𑌮𑌾𑌨𑍃𑌭𑌿𑌹𑍋𑌰𑌵𑌾।𑌪𑍂𑌰𑍂𑌹𑍂𑌤𑌾𑌪𑍍𑌰𑌯𑌾
𑌶𑍃𑌕𑌚𑌚𑌿
𑌹𑍀𑌹𑍋𑌵𑌾।𑌆𑌸𑍋𑌯𑌥𑌾𑌨𑍋𑌵𑌿𑌤𑌾𑌹𑍋𑌵𑌾।𑌵𑌾
𑌶𑌿𑌕𑍍𑌯𑌚𑍍𑌯𑌶𑍁𑌕
𑌰𑍍𑌧𑌶𑍍𑌪𑌾𑌇𑌤𑍍𑌦𑌦𑍋𑌵𑌸𑍂𑌹𑍋𑌵𑌾।𑌨𑌾𑌇𑌮𑌾𑌮𑌾𑌦𑌶𑍍𑌪𑌸𑍋
𑌚𑍍𑌯𑌶𑍃𑌭𑍁𑌚𑍍𑌯
𑌮𑍈𑌹𑍋𑌰𑌵𑌾।𑌹𑍋𑌇ழா𑌹।।4।।
𑌶𑌿𑌪𑍍𑌲𑌶𑍍𑌯
