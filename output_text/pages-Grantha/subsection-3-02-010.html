10 ।𑌆𑌤𑍍𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌇𑌮𑌾𑌉𑌕𑍍𑌤𑍍𑌵𑌾𑌪𑍁𑌰𑍋𑌵𑌸𑍋𑌗𑌿𑌰𑌏𑌗𑌿𑌰𑌾𑌃।𑌵𑌾𑌰𑍍𑌧𑌨𑍍𑌤𑍁
 𑌤𑍈 𑌤𑍍𑌯
𑌯𑌾𑌮𑌮𑌾।𑌪𑌾𑌵𑌕𑌵𑌰𑍍𑌣𑌾𑌶𑍍𑌶𑍁𑌚𑌯𑍋𑌵𑍀𑌪𑌾।𑌹𑌿𑌂𑌹𑌿𑌂
 𑌕𑍀 𑌟𑍍𑌯 𑌤𑍁 𑌪𑌿 𑌶 𑌟
𑌶𑍍𑌪𑌾𑌇𑌤𑌾𑌃।𑌆𑌭𑌿𑌸𑍍𑌤𑍋𑌮𑍈𑌰𑌨𑍂𑌹𑍂𑌵𑌾𑌏𑌹𑍋
 𑌚𑍍𑌯 𑌣𑍍𑌯 𑌕𑍀 𑌟𑍍𑌯𑌚 𑌕𑍍𑌯 𑌕
𑌵𑌾।𑌷𑌾𑌤𑌾𑌔𑌹𑍋𑌬𑌾।𑌹𑍋𑌇ழா𑌹।।18।।
 𑌟𑍍𑌯 𑌕 𑌟𑍍𑌯 𑌚
𑌇𑌮𑌾𑌉𑌕𑍍𑌤𑍍𑌵𑌾𑌪𑍁𑌰𑍋𑌵𑌸𑍋𑌹𑌾𑌬𑍁।𑌗𑍀𑌰𑍋𑌵𑌰𑍍𑌧
 𑌤𑍢 𑌶 𑌚𑍍𑌯 𑌕𑍍𑌯
𑌨𑍍𑌤𑍁𑌯𑌾𑌮𑌮𑌾।𑌇𑌹𑌾𑌹𑌾𑌹𑍋𑌇𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।
𑌪𑌾𑌵𑌾𑌕𑌵𑌰𑍍𑌣𑌾𑌶𑍍𑌶𑍁𑌚𑌯𑍋।𑌪𑍍𑌰।𑌵𑍀𑌪𑌶𑍍𑌪𑍀𑌤𑍋।
 𑌥 𑌕𑍀 𑌕𑌿 𑌚𑍀
𑌆𑌭𑌿𑌸𑍍𑌤𑍋𑌮𑍈𑌰𑌿𑌹𑌾।𑌪𑍍𑌰।𑌆𑌨𑍂𑌷𑌾𑌤𑌾
 𑌤𑍂 𑌟𑌿𑌟 𑌚
𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾।।19।।
 𑌶𑌿 𑌚 𑌶
𑌇𑌮𑌾𑌉𑌕𑍍𑌤𑍍𑌵𑌾𑌪𑍁𑌰𑍋𑌵𑌸𑌾𑌬𑍁𑌗𑌾𑌇𑌰𑍋𑌵𑌰𑍍𑌧𑌨𑍍𑌤𑍁𑌯𑌾𑌮𑌾
 𑌫𑍂 𑌤𑍍𑌯 𑌪𑍍𑌲 𑌫𑍂 𑌨𑍍𑌦
𑌮𑌾।𑌪𑌾𑌵𑌾𑌕𑌵𑌰𑍍𑌣𑌾𑌶𑍍𑌶𑍁𑌚𑌯𑍋𑌵𑍀𑌪𑌶𑍍𑌪𑌿𑌤𑌾𑌔𑌹𑍋।
 𑌶 𑌥 𑌕𑍀 𑌟𑌿𑌚 𑌭𑍂 𑌤
𑌆𑌔𑌹𑍋।𑌆𑌭𑌿𑌸𑍍𑌤𑍋𑌮𑍈𑌰𑌨𑍂𑌹𑍂𑌵𑌾𑌏𑌹𑍋
 𑌟𑍍𑌯 𑌤 𑌚𑍀 𑌟𑍍𑌯𑌚 𑌕𑍍𑌯 𑌟
𑌵𑌾।𑌷𑌾𑌤𑌾𑌔𑌹𑍋𑌬𑌾𑌹𑍋𑌇ழா𑌹।।20।।
 𑌕𑍍𑌯 𑌟𑌿 𑌚 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
