9 ।𑌕𑍗𑌤𑍍𑌸𑍇𑌚𑍇 𑌪𑌾𑌞𑍍𑌚𑌵𑌾𑌜𑍇𑌵𑌾 ।

𑌯𑌾𑌇𑌨𑍍𑌦𑌰𑌾𑌚𑌮𑌸𑍇𑌷𑍍𑌵𑌾𑌇𑌯𑌾।𑌸𑍋𑌮𑌾𑌶𑍍𑌪𑌾𑌮𑍂𑌷𑍁
 𑌥𑍍𑌯 𑌪 𑌶𑍍𑌰𑍂 𑌕𑍍𑌯
𑌤𑍇𑌸𑍂𑌤𑌸𑍍𑌸𑍋𑌮𑌶𑍍𑌪𑌮𑍂𑌷𑍁𑌤𑌾𑌇𑌸𑍂𑌤𑌾𑌃।
 𑌕𑍀𑌚 𑌕 𑌕𑍁𑌚 𑌕 𑌟𑌿 𑌤
𑌪𑌾𑌇𑌬𑌾𑌹𑌾𑌇𑌦𑌾𑌸𑍇𑌯𑌾𑌹𑌾𑌇।𑌤𑍍𑌵𑌾𑌮𑍀𑌶𑌾
 𑌟𑌿 𑌤 𑌟𑌿 𑌤 𑌶 𑌟𑌿
𑌇𑌷𑌾𑌇।𑌓𑌇।।25।।
 𑌚𑍍𑌯𑌣 𑌶 𑌪 𑌶𑍍𑌯
𑌯𑌇𑌨𑍍𑌦𑌰𑌚𑌾𑌮𑌾𑌸𑍇𑌷𑍍𑌵𑌾।𑌸𑍋𑌮𑌶𑍍𑌪𑌮𑍂𑌷𑍁𑌤𑌾𑌇
 𑌤𑍁 𑌤𑍍𑌯 𑌥𑍂
𑌸𑍁𑌤𑌾𑌃।𑌸𑍋𑌮𑌶𑍍𑌪𑌮𑍂𑌷𑍂।𑌤𑌾𑌇𑌸𑍂𑌤𑌾𑌃।
 𑌪 𑌶 𑌤𑍀 𑌤 𑌪𑌿 𑌶
