5 ।𑌕𑌾𑌶𑍍𑌯𑌪𑍇𑌚𑍇।।

𑌪𑍍𑌰𑌹𑍋𑌤𑌾𑌜𑌾𑌤𑌾:।𑌮𑌾𑌹𑌾𑌨𑍍𑌨𑌭𑍋𑌵𑌾𑌇𑌨𑍃𑌷𑌦𑍍𑌮𑌾
 𑌤𑍁  𑌟𑍢
𑌸𑍀।𑌦𑌾𑌦𑌪𑌾𑌂𑌵𑌿𑌵𑌰𑍍𑌤𑌾𑌇𑌦𑌧𑌦𑍇𑌯𑌾𑌧𑌾।𑌯𑌾𑌇
 𑌤 𑌤𑌚 𑌕 𑌟𑍃 𑌤
𑌸𑍂𑌤𑍇𑌵𑌯𑌾𑌂𑌸𑌿𑌯𑌨𑍍𑌤𑌾𑌉𑌵𑌾।𑌵𑌾𑌸𑍂𑌨𑍀𑌵𑍀𑌧𑌾
 𑌕𑍢 𑌤𑍍𑌯 𑌕𑍀 𑌚
𑌤𑍋𑌯𑌾𑌔𑌹𑍋𑌵𑌾।𑌤𑌨𑍂𑌪𑌾
 𑌟𑌟 𑌚 𑌶𑌿 𑌤𑍍𑌯 𑌚
𑌪𑍍𑌰𑌹𑍋𑌤𑌾𑌜𑌾𑌤𑌉𑌹𑍁𑌵𑌾𑌹𑌾𑌇।𑌮𑌾𑌹𑌾𑌨𑍍𑌨
 𑌤𑍄 𑌤 𑌶 𑌟𑍍𑌯
𑌭𑍋𑌵𑌾𑌇𑌨𑍃𑌷𑌦𑍍𑌮𑌾𑌸𑍀𑌦𑌦𑌪𑌾𑌂𑌵𑍀𑌵𑌾𑌕𑍍𑌤𑌰𑌾𑌇।𑌓
 𑌟𑍍𑌯 𑌟𑍁 𑌕𑍀 𑌟𑍍𑌯 𑌤 𑌶
𑌔𑌹𑍋𑌇𑌹𑌾।𑌦𑌾𑌧𑌾𑌦𑍇𑌯𑌾𑌧𑌾𑌯𑌾𑌇𑌸𑍁𑌤𑍇
 𑌕𑍍𑌯 𑌤 𑌤𑍍𑌯 𑌟𑍍𑌯 𑌟𑍍𑌯
𑌵𑌯𑌾𑌂𑌸𑌿𑌯𑌨𑍍𑌤𑌾𑌵𑌾𑌸𑍂𑌨𑍀।𑌓𑌔𑌹𑍋𑌇𑌹𑌾।
 𑌕𑍢 𑌟𑍍𑌯 𑌤 𑌕𑍍𑌯 𑌤 𑌤𑍍𑌯
𑌵𑍀𑌧𑌾𑌤𑌾𑌇𑌤𑌨𑍂𑌪𑌾𑌔𑌹𑍋𑌵𑌾।𑌹𑌾𑌵𑌿𑌷𑍍𑌮𑌾
 𑌕𑍍𑌯 𑌟𑍀𑌟 𑌚 𑌶𑌿 𑌟𑌿
