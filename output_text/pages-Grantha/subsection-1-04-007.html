7 ।𑌭𑌾𑌰𑌚𑌾𑌜𑌸𑍍𑌯𑌪𑍃𑌶𑍍𑌨𑍀𑌚𑍇 ।।

𑌬𑍃𑌹𑌾𑌤𑍍𑌭𑍀𑌰𑌗𑍍𑌨𑍇𑌰𑌚𑍍𑌚𑌰𑌿𑌭𑍀𑌰𑍍𑌹𑌾𑌬𑍁।𑌶𑍁𑌕𑍍𑌰𑌾𑌇𑌣𑌾
𑌪𑌿𑌶𑍍𑌰𑍇
𑌦𑍇𑌵𑌾𑌶𑍋𑌚𑌿𑌷𑌾𑌭𑌾𑌰𑌾𑌚𑌾𑌜𑍇𑌹𑍋𑌵𑌾𑌹𑌾𑌇।
𑌕𑍁𑌚𑍀𑌕𑍍𑌯𑌯𑌟𑌟𑍍𑌯𑌤𑌶
𑌸𑌮𑍀𑌧𑌾𑌨𑍋𑌯𑌾𑌵𑍀𑌷𑍍𑌠𑍀𑌯𑌾𑌹𑍋𑌵𑌾𑌹𑌾𑌇।
𑌚𑍍𑌯𑌥𑍍𑌯𑌚𑌯𑌟𑌿𑌟𑍍𑌯𑌤𑌶
𑌰𑍇𑌵𑌾𑌤𑍍𑌖𑌾𑌵𑌾𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌕𑌾𑌦𑍀𑌦𑍀𑌹𑍀𑌇ழா
𑌚𑍍𑌯𑌯𑌟𑌟𑍍𑌯𑌤𑌶𑌕𑌿

𑌤𑍂𑌤
