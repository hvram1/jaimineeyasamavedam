4 ।𑌕𑌾𑌣𑍍𑌵𑍇𑌚𑍇 ।

𑌵𑌯𑌂𑌵𑌾𑌯𑌾𑌂𑌊𑌤𑍍𑌵𑌾।𑌤𑌾𑌦𑍀𑌦𑌾𑌰𑍍𑌤𑍍𑌥𑌾𑌃।𑌇𑌨𑍍𑌦𑌰𑌤𑍍𑌵𑌾𑌯
 𑌤𑍀 𑌚 𑌶 𑌚𑌿 𑌣
𑌨𑍍𑌤𑌸𑍍𑌸𑌚𑌾𑌯𑌾𑌃।𑌕𑌣𑍍𑌵𑌾𑌊𑌕𑍍𑌥𑍇𑌭𑌿𑌰𑍍𑌜𑌰𑌨𑍍𑌤𑍇𑌏𑌹𑌿𑌯𑌾
 𑌚𑍃 𑌪𑍍𑌲 𑌤𑍍𑌯 𑌪 𑌪𑍍𑌲
𑌹𑌾𑌹𑍋𑌇।।10।।
 𑌶𑌿 𑌪𑍍𑌲 𑌶𑍍𑌯
𑌵𑌯𑌮𑍂𑌤𑍍𑌵𑌾𑌤𑌦𑌿𑌦𑌾𑌰𑍍𑌤𑍍𑌥𑌾𑌃।𑌐𑌹𑍀𑌹𑍋𑌇।
 𑌚𑌿 𑌶𑍁 𑌟𑌿 𑌶
𑌵𑌾𑌯𑌮𑍁𑌤𑍍𑌵𑌾𑌤𑌦𑌿𑌦𑌰𑍍𑌤𑍍𑌥𑌾𑌇𑌨𑍍𑌦𑌰𑌤𑍍𑌵𑌾𑌯𑌨𑍍𑌤𑌸𑍍𑌸𑌚𑌾𑌯𑌾𑌃।
 𑌷 𑌕𑍄 𑌟𑌿 𑌤
𑌕𑌾𑌣𑍍𑌵𑌾𑌃।𑌉𑌕𑍍𑌥𑌾𑌇𑌭𑍀𑌜𑍋𑌰𑌬𑌾।𑌰𑌨𑍍𑌤𑌾𑌯𑌾।।11।।
 𑌚 𑌣 𑌕𑍀 𑌪 𑌪𑍍𑌲 𑌤𑍍𑌯 𑌨
