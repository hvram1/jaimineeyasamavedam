।𑌭𑌾𑌰𑌞𑍍𑌚 𑌜𑍇𑌚𑍇 ।

16 ।𑌭𑌾𑌰𑌞𑍍𑌚 𑌜𑍇𑌚𑍇 ।

𑌔𑌹𑍋𑌇𑌪𑌰𑌿𑌤𑍋𑌷𑍀।𑌣𑌤𑌾𑌸𑍂𑌤𑌮𑍗।𑌹𑍋𑌔
𑌤𑍃𑌤𑍍𑌯𑌚𑌟𑌿
𑌹𑍋𑌵𑌾𑌔𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌸𑍋𑌮𑍋𑌯𑌊𑌕𑍍𑌤𑌾𑌮𑌂
𑌤𑌿𑌚𑌕𑌕𑌚𑌶𑌟𑍍𑌯𑌚𑍍𑌯𑌕𑍍𑌯
𑌹𑌾𑌵𑌿𑌰𑍗।𑌪𑍍𑌰।𑌦𑌧𑌨𑍍𑌵𑌾𑌯𑍋𑌨𑌯𑍋𑌰𑌅𑌪𑍍𑌸𑌭𑌾𑌰𑌾𑌔
𑌟𑍀𑌕𑍍𑌯𑌟𑍀𑌚𑌕𑌿𑌟𑍍𑌯
।𑌪𑍍𑌰।𑌸𑍂𑌷𑌾𑌵𑌸𑍋𑌮𑌾𑌮𑌾𑌦𑍍𑌰𑌾𑌭𑌿𑌰𑍗𑌹𑍋𑌔𑌹𑍋
𑌚𑍍𑌯𑌚𑍀𑌤𑍂𑌚
𑌵𑌾𑌔𑌹𑍋𑌵𑌾𑌉𑌹𑍁𑌵𑌾𑌹𑌾𑌬𑍁।𑌬𑌾।।19।।
𑌕𑌕𑌪𑌪𑍍𑌲𑌶𑌶
𑌪𑌯𑍇𑌰𑌪𑌰𑍀।𑌤𑍋𑌷𑌿𑌞𑍍𑌚𑌤𑌾𑌸𑍂𑌤𑌮𑌾𑌉𑌵𑌾𑌹𑌾𑌬𑍁
𑌤𑍀𑌕𑌟𑌿𑌚𑌪𑍍𑌲
𑌹𑌾𑌬𑍁𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌸𑍋𑌮𑍋𑌯𑌊𑌕𑍍𑌤𑌾𑌮𑌂𑌹𑌾𑌵𑌿
𑌤𑌿𑌕𑌿𑌟𑍍𑌯𑌚𑍍𑌯𑌚𑍍𑌯
𑌰𑌾𑌉𑌵𑌾।𑌪𑍍𑌰।𑌦𑌧𑌨𑍍𑌵𑌾𑌯𑍋𑌨𑌯𑍋𑌰𑌅𑌪𑍍𑌸𑌭𑌾𑌰𑌾
𑌫𑍁𑌕𑍍𑌯𑌟𑍀𑌚𑌕𑌿
𑌉𑌵𑌾।𑌸𑍂𑌷𑌾𑌵𑌾𑌸𑍋।𑌪𑍍𑌰।𑌮𑌾𑌮𑌦𑍍𑌰𑌾𑌇𑌭𑌿𑌰𑍂𑌹𑍁𑌵𑌾
𑌟𑌿𑌟𑌿𑌤𑌚𑌪𑍍𑌲𑌪𑍍𑌲
𑌹𑌾𑌬𑍁।𑌬।।20।।
