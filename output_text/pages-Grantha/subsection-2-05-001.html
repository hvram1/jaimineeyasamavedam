।𑌵𑍈𑌤𑌹𑌵𑍍𑌯𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

1 ।𑌵𑍈𑌤𑌹𑌵𑍍𑌯𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌪𑌾𑌨𑍍𑌤𑌮𑌾𑌵𑍋𑌅𑌨𑍍𑌧𑌸𑌾𑌃।𑌇𑌨𑍍𑌦𑌰𑌾𑌮𑌾𑌭𑌿𑌪𑍍𑌰𑌾𑌗𑌾𑌯𑌾
𑌤𑍃𑌕𑌕𑍍𑌯
𑌤𑌾𑌹𑌾𑌹𑌾𑌇।𑌪𑍍𑌰।𑌵𑌿𑌶𑍍𑌪𑌾𑌸𑌾𑌹𑌂𑌶𑌾𑌤𑌾𑌕𑍍𑌰𑌾𑌤𑍍𑌵𑌂
𑌟𑍁𑌤𑌤𑌶𑌕𑍀𑌚𑌕𑍍𑌯𑌟𑍍𑌯
।𑌪𑍍𑌰।𑌮𑌂𑌹𑌾𑌇𑌷𑍍𑌠𑌾𑌞𑍍𑌚।𑌪𑍍𑌰।𑌰𑍍𑌷𑌣𑌾𑌇𑌨𑌾
𑌕𑌟𑍀𑌟𑍍𑌯𑌚𑍍𑌯
𑌮𑍗𑌹𑍋𑌵𑌾।𑌉𑌊𑌪𑌾।।1।।
𑌶𑌿𑌚𑍍𑌯𑌶
𑌪𑌾𑌨𑍍𑌤𑌮𑌾𑌓𑌅𑌨𑍍𑌧𑌸𑌈𑌹𑌾।𑌇𑌨𑍍𑌦𑌰𑌾𑌮𑌭𑌾𑌇𑌪𑍍𑌰𑌗𑌾𑌯
𑌷𑌿𑌤𑍂𑌥𑌕𑍂
𑌤𑌾𑌇𑌹𑌾।𑌵𑌿𑌶𑍍𑌪𑌾𑌸𑌾𑌹𑌂𑌶𑌤𑌾𑌕𑍍𑌰𑌾𑌤𑍂𑌮𑌿𑌹𑌾।
𑌟𑍍𑌯𑌤𑍍𑌯𑌕𑌿𑌟𑌿𑌚𑌟𑍀
𑌮𑌂𑌹𑌾𑌇𑌷𑍍𑌠𑌾𑌞𑍍𑌚𑌾𑌇𑌹𑌾𑌰𑍍𑌷𑌣𑌾𑌈𑌨𑌾𑌮𑌿𑌹𑌾।।2।।
𑌚𑍍𑌯𑌶𑌥𑌟𑌿𑌟𑍍𑌯𑌚𑌕𑍍𑌯𑌨𑍍𑌯
𑌪𑌾𑌨𑍍𑌤𑌾𑌮𑌾𑌵𑍋𑌅𑌨𑍍𑌧𑌸𑌾𑌃।𑌆𑌇𑌨𑍍𑌦𑌰𑌾𑌮𑌭𑌾𑌇𑌪𑍍𑌰
𑌭𑍍𑌯𑌫𑌚𑌶𑌿
𑌗𑌾𑌯𑌾𑌤𑌾।𑌵𑌿𑌶𑍍𑌪𑌾𑌸𑌾𑌹𑌂।𑌶𑌾𑌤𑌾𑌕𑍍𑌰𑌾𑌤𑍁𑌂।
𑌟𑍄𑌚𑌣𑌟𑍍𑌯𑌚𑌣𑌤𑌪𑍍𑌲𑌶
𑌮𑌂𑌹𑌿𑌷𑍍𑌠𑌞𑍍𑌚𑌰𑍍𑌷𑌾𑌣𑌾𑌇।𑌨𑌾𑌮𑌾𑌔𑌹𑍋𑌵𑌾।
