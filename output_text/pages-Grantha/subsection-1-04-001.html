।𑌭𑌾𑌰𑌚𑌾𑌜𑌸𑍍𑌯𑍗𑌪𑌹𑌵𑍗𑌚𑍗।।

1 ।𑌭𑌾𑌰𑌚𑌾𑌜𑌸𑍍𑌯𑍗𑌪𑌹𑌵𑍗𑌚𑍗।।

𑌯𑌜𑍍𑌞𑌾𑌯𑌜𑍍𑌞𑌾।𑌵𑍋𑌗𑍍𑌨𑌯𑌾𑌈𑌗𑍀𑌰𑌾𑌗𑌿𑌰𑌾𑌹𑌾
  𑌤𑍀 𑌤 𑌤𑍍𑌯 𑌟𑌿 𑌚𑍍𑌯 𑌶𑌪𑍍𑌲
𑌹𑌾𑌇𑌚𑌾𑌦𑌕𑍍𑌷𑌾𑌸𑌾𑌇।𑌪𑍍𑌰𑌪𑍍𑌰𑌾𑌵𑌾𑌯𑌮𑌮𑍃𑌤
 𑌤 𑌚𑍍𑌯 𑌚𑍍𑌯 𑌣 𑌶 𑌟𑍍𑌯 𑌚 𑌕
𑌨𑍍𑌜𑌾𑌤𑌾𑌵𑍇𑌦𑌾𑌸𑌾𑌂।𑌪𑍍𑌰𑍀𑌯𑌾𑌮𑍍𑌮𑌿𑌤𑍍𑌰𑌾𑌨𑍍𑌨𑌾𑌶𑌂𑌸𑌿𑌷𑌾
 𑌥𑍃 𑌟𑍍𑌯 𑌚𑍍𑌯 𑌕𑌿
𑌮𑍇𑌹𑌿𑌯𑌾𑌔𑌹𑍋𑌔𑌹𑍋𑌇ழா
 𑌕𑍀 𑌚𑍂 𑌪𑍍𑌲
𑌯𑌜𑍍𑌞𑌾𑌯𑌜𑍍𑌞𑌾।𑌹𑍋𑌇𑌵𑍋𑌗𑍍𑌨𑌾𑌯𑌾𑌏𑌹𑌿𑌯𑌾।
  𑌤𑍀 𑌤 𑌤𑍍𑌯 𑌕𑍍𑌯 𑌚 𑌣𑍍𑌯
𑌗𑍀𑌰𑌾𑌗𑍀𑌰𑌾𑌚𑌾𑌦𑌾𑌕𑍍𑌷𑌾𑌸𑌾𑌇।𑌪𑍍𑌰𑌪𑍍𑌰𑌾𑌵𑌾
 𑌕𑍍𑌯𑌚 𑌚 𑌟𑍍𑌯 𑌚 𑌚𑍍𑌯 𑌶 𑌥𑍍𑌯𑌚
𑌯𑌾𑌮𑌾𑌮𑍃𑌤𑌨𑍍𑌜𑌾𑌤𑌾𑌵𑍇𑌦𑌾𑌸𑌂।𑌪𑍍𑌰𑍀𑌯𑌮𑍍𑌮𑌿𑌤𑍍𑌰𑌾𑌨𑍍𑌨𑌾
 𑌚𑍍𑌯 𑌕 𑌟𑍁 𑌚 𑌣 𑌚𑍍𑌯 𑌕𑌿
𑌶𑌂𑌸𑌿𑌷𑌾𑌮𑍇𑌹𑌿𑌯𑌾𑌔𑌹𑍋𑌇ழா
 𑌕𑍀 𑌚𑍀 𑌪𑍍𑌲
