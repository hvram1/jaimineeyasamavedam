2 ।𑌪𑍍𑌰𑌶𑌸𑍍𑌯𑌜𑌾𑌨𑌸𑍍𑌯𑌾𑌭𑍀𑌵𑌰𑍍𑌤𑍗𑌚𑍗…

𑌪𑍍𑌰𑌤𑍍𑌵𑍇𑌪𑍍𑌰𑌤𑍁।𑌦𑍍𑌰𑌾𑌵𑌾𑌦𑍍𑌰𑌵𑌾𑌪𑌾𑌰𑍀𑌕𑍋𑌶𑌾𑌨𑍍𑌨𑍀𑌷𑍀𑌦𑌾।
𑌕𑍍𑌯𑌤𑍍𑌯𑌕𑍃𑌚𑌿𑌶
𑌨𑍃𑌭𑌿𑌰𑍇𑌨𑍃𑌭𑍀:𑌪𑍂𑌨𑌾𑌪𑍁𑌨𑌾𑌨𑍋𑌆𑌭𑍀𑌵𑌾𑌜𑌮𑌾
𑌤𑌿𑌤𑍍𑌯𑌟𑍁𑌕𑌿𑌚𑍍𑌯
𑌰𑍍𑌷𑌾।𑌅𑌶𑍍𑌪𑌮𑍇𑌅𑌶𑍍𑌪𑌾𑌂।𑌨𑌤𑍍𑌵𑌾𑌨𑌤𑍍𑌵𑌾𑌵𑌾𑌜𑍀𑌨𑌮𑍍𑌮𑌾𑌰𑍍𑌜𑌯𑌾
𑌶𑌤𑌿𑌤𑍍𑌯𑌟𑍁𑌕𑌿𑌚𑍍𑌯
𑌭3𑌇।𑌅𑌚𑍍𑌛𑌯𑍇𑌅𑌚𑍍𑌛𑌾𑌬𑌰𑍍𑌹𑌾𑌬𑌰𑍍𑌹𑌾𑌇𑌰𑌾𑌶𑌨𑌾𑌭𑌾
𑌶𑍍𑌯𑌤𑌿𑌤𑍍𑌯𑌕𑌿𑌕𑌿𑌪𑌿
𑌇𑌰𑍍𑌨𑍍𑌨𑌾𑌯𑌾𑌭3𑌇।।4।।
𑌚𑌿𑌤𑍍𑌰𑌶
𑌪𑍍𑌰𑌤𑍂𑌦𑍍𑌰𑌾।𑌵𑌾𑌪𑌰𑌿𑌕𑍋𑌶𑌾𑌨𑍍𑌨𑍀𑌷𑍀𑌦𑌾।𑌸𑌾𑌇𑌦𑌾।𑌬𑍁
𑌤𑌿𑌟𑍃𑌤𑌕𑌿
𑌹𑍋𑌔𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌨𑍃𑌭𑌿:𑌪𑍁𑌨𑌾𑌨𑍋𑌅𑌭𑍀𑌵𑌾𑌜𑌮𑌾
𑌕𑍍𑌯𑌚𑌕𑌕𑍍𑌯𑌟𑍇
𑌰𑍍𑌷𑌾𑌆𑌰𑍍𑌷𑌾।𑌪𑍍𑌰।𑌅𑌶𑍍𑌪𑌨𑍍𑌨𑌤𑍍𑌵𑌾𑌵𑌾𑌜𑌿𑌨𑌮𑍍𑌮𑌾𑌰𑍍𑌜𑌯𑌾𑌭3𑌇।
𑌤𑌕𑍍𑌯𑌟𑍇𑌤𑌶
𑌯𑌭3।𑌪𑍍𑌰।𑌅𑌚𑍍𑌛𑌾𑌬𑌰𑍍𑌹𑌾𑌇𑌰𑌾𑌶𑌨𑌾𑌭𑌿𑌰𑍍𑌨𑍍𑌨𑌾𑌯𑌾𑌏
𑌕𑍍𑌯𑌕𑌿𑌕𑌿𑌟𑌿𑌕𑍍𑌯𑌚
𑌹𑌿𑌯𑌾𑌹𑌾𑌉𑌵𑌾।𑌭3𑌇।।5।।
