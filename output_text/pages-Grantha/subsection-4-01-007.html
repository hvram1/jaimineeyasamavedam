।𑌸𑍗𑌮𑌿𑌤𑍇𑌰𑌚𑍇 ।

6 ।𑌸𑍗𑌮𑌿𑌤𑍇𑌰𑌚𑍇 ।

𑌜𑌗𑍃𑌹𑍍𑌮𑌾𑌤𑍇𑌦𑌕𑍍𑌷𑌿𑌣𑌮𑍋𑌹𑌾𑌓𑌹𑌾𑌏।𑌇𑌨𑍍𑌦𑌰𑌹𑌾𑌸𑍍𑌤𑌾𑌂।
𑌤𑍇𑌤𑌟𑌿𑌤
𑌵𑌸𑍂𑌯𑌵𑍋𑌵𑌾𑌸𑍂𑌪𑌾𑌤𑌾𑌇𑌵𑌸𑍂𑌨𑌾𑌓𑌹𑌾𑌏।𑌪𑍍𑌰।
𑌕𑍍𑌯𑌯𑌿𑌪𑌶𑍁𑌤𑌤𑌤
𑌵𑌿𑌦𑍍𑌮𑌾𑌹𑌿𑌤𑍍𑌵𑌾𑌗𑍋𑌪𑌾𑌤𑌿𑌂𑌶𑍂𑌰𑌗𑍋𑌨𑌾𑌓।𑌪𑍍𑌰।
𑌯𑍁𑌪𑌶𑌿𑌤
𑌅𑌸𑍍𑌮𑌾𑌭𑍍𑌯𑌨𑍍𑌚𑌾𑌇𑌤𑍍𑌰𑌾𑌂।𑌵𑌾𑌰𑍍𑌷𑌾𑌣𑌂𑌰𑌾𑌯𑌿𑌨𑍍𑌦𑌾।𑌪𑍍𑌰।
𑌟𑍂𑌚𑍍𑌯𑌕𑍍𑌯𑌫𑌶
𑌰𑌾𑌯𑌾𑌇𑌨𑍍𑌦𑌾𑌉𑌵𑌾।𑌊𑌪𑌾।𑌔𑌹𑍋𑌔𑌹𑍋𑌵𑌾
𑌟𑌿𑌤𑌚𑌶𑌕𑌕𑍍𑌯𑌪
𑌹𑌾𑌉𑌵𑌾।𑌈।।9।।
𑌶𑌿𑌚
𑌜𑌗𑍍𑌰𑌹𑍍𑌮𑌾𑌤𑍇𑌦𑌕𑍍𑌷𑌿𑌣𑌮𑍗𑌹𑍋𑌔𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌇𑌨𑍍𑌦𑌰𑌾
𑌤𑍋𑌤𑌶
𑌹𑌾𑌸𑍍𑌤𑌂।𑌵𑌸𑍂𑌯𑌵𑍋𑌵𑌾𑌸𑍂𑌪𑌾𑌤𑌾𑌇𑌵𑌸𑍂𑌨𑍗
𑌚𑌿𑌶𑌕𑍍𑌯𑌯𑌿𑌪𑌶𑍁
𑌵𑌾𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌇।𑌪𑍍𑌰।𑌵𑌿𑌦𑍍𑌮𑌾𑌹𑌿𑌤𑍍𑌵𑌾𑌗𑍋𑌪𑌾𑌤𑌿𑌂
𑌚𑌿𑌪𑍍𑌲𑌤𑌶𑌯𑍁𑌪
𑌶𑍂𑌰𑌗𑍋𑌨𑍗।𑌪𑍍𑌰।𑌅𑌸𑍍𑌮𑌭𑍍𑌯𑌞𑍍𑌚𑌇𑌤𑍍𑌰𑌾𑌂𑌵𑌾𑌰𑍍𑌷𑌾𑌣𑌂
𑌶𑌿𑌟𑍂𑌚𑍍𑌯𑌕
𑌰𑌾𑌯𑌿𑌂𑌦𑍗𑌵𑌾𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌔𑌹𑍋𑌵𑌾।𑌈।।10।।
𑌫𑍍𑌯𑌚𑍀𑌪𑍍𑌲𑌚𑌶𑌿𑌚
