।𑌚𑍍𑌯𑌾𑌵𑌨𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿…

39 ।𑌚𑍍𑌯𑌾𑌵𑌨𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿…

𑌅𑌸𑌾𑌵𑍍𑌯𑍁𑌹𑍁𑌵𑌾𑌹𑌾𑌇।𑌅𑌂𑌶𑍁𑌰𑍍𑌮𑌦𑌾𑌯𑌾𑌪𑍍𑌸𑍁𑌦𑌕𑍍𑌷𑍋
 𑌤𑍁 𑌤 𑌶
𑌗𑌿𑌰𑌿𑌷𑍍𑌠𑌾𑌉𑌹𑍁𑌵𑌾𑌹𑍋𑌇।𑌶𑍍𑌯𑌾𑌇𑌨𑍋𑌨𑌾𑌯𑍋
 𑌟𑌂 𑌚 𑌶 𑌕 𑌤𑍍𑌯 𑌟 𑌚
𑌔𑌹𑍋𑌵𑌾।𑌨𑍀𑌮𑌾𑌸𑌾𑌦𑌾𑌤𑍍।।56।।
 𑌶𑌿 𑌟𑌿 𑌚
𑌅𑌸𑌾𑌵𑍍𑌯𑌂𑌶𑍁𑌰𑍗𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌮𑌦𑌾𑌯𑌾𑌪𑍍𑌸𑍁𑌦𑌕𑍍𑌷𑍋
 𑌤𑍃 𑌤 𑌶 𑌟𑍍𑌯𑌚
𑌗𑌿𑌰𑌿𑌷𑍍𑌠𑌾:।𑌶𑍍𑌯𑌾𑌇𑌨𑍋𑌨𑌾𑌯𑍋𑌨𑌾𑌇𑌮𑌾𑌔𑌹𑍋𑌵𑌾।
 𑌕𑍃 𑌯𑌿 𑌟𑍍𑌯 𑌟 𑌚𑍍𑌯 𑌶𑌿
𑌉𑌪𑍍𑌸𑌾𑌦𑌾𑌤𑍍।।57।।
  𑌶𑌿
𑌇𑌹𑌾𑌓𑌵𑌾𑌇𑌹𑌾𑌅𑌸𑌾𑌵𑍍𑌯𑌂𑌶𑍁𑌰𑍍𑌮𑌦𑌯𑍇𑌹𑌾।𑌆𑌪𑍍𑌸𑍁𑌦
 𑌤𑍍𑌯 𑌪
𑌕𑍍𑌷𑍋𑌗𑌿𑌰𑌿𑌷𑍍𑌠𑌾:।𑌈𑌹𑌾।𑌶𑍇𑌯𑌨𑍋𑌨𑌯𑍋𑌨𑌿
 𑌚𑍃𑌣𑌫𑌪𑍍𑌲 𑌚 𑌶 𑌕
𑌮𑌾।𑌈𑌹𑌾।𑌆𑌸𑌦𑌾𑌤𑍍।𑌈𑌹𑌾।।58।।
 𑌚𑍍𑌯𑌣𑌫𑌪𑍍𑌲 𑌚 𑌶 𑌟 𑌚𑍍𑌯𑌣𑌫𑌪𑍍𑌲
𑌅𑌸𑌾𑌵𑍍𑌯𑌂𑌶𑍁𑌰𑍍𑌮𑌦𑌾𑌯𑌾𑌹𑌾𑌬𑍁।𑌆𑌪𑍍𑌸𑍁𑌦𑌾𑌕𑍍𑌷𑌾:।
 𑌚𑍍𑌯 𑌪𑍍𑌲 𑌚𑍍𑌯 𑌶𑌿 𑌚𑌿 𑌣
𑌗𑌿𑌰𑌾𑌇𑌷𑍋𑍍𑌠𑌹𑌾𑌇।𑌶𑍍𑌯𑌾𑌇𑌨𑍋𑌨𑌾𑌯𑍋।𑌨𑌾𑌇𑌮𑌾
  𑌪𑍀 𑌣 𑌶 𑌭𑍀 𑌤 𑌟𑌿
𑌹𑌾𑌇।𑌸𑌾𑌦𑌾𑌤𑍍𑌹𑌾𑌇।।59।।
 𑌚𑌣 𑌶 𑌪 𑌪𑍍𑌲 𑌶𑍍𑌯
