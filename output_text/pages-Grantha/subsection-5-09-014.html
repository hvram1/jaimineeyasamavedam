।𑌯𑌾𑌮𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

13 ।𑌯𑌾𑌮𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌅𑌸𑌾𑌵𑌿𑌸𑍋𑌮𑍋𑌆𑌰𑍂𑌷𑍋𑌵𑍃𑌷𑌾𑌹𑌰𑌾𑌇।𑌰𑌾
 𑌕𑍍𑌯 𑌕𑌿𑌚 𑌕 𑌪𑍍𑌲 𑌶𑍍𑌯 𑌤𑍍𑌯 𑌶
𑌜𑍇𑌵𑌦𑌾𑌸𑍍𑌮𑍋𑌆𑌭𑍀𑌗𑌾𑌚𑍀𑌕𑍃𑌦𑌾𑌤𑍍।𑌪𑍁𑌨𑌾𑌨𑍋𑌵𑌾𑌰𑌾
 𑌥𑍍𑌯  𑌕𑌿𑌚 𑌕 𑌪 𑌪𑍍𑌲 𑌤𑍍𑌯 𑌕𑍍𑌯 𑌕𑌿𑌚
𑌮𑌾𑌤𑌿𑌧𑌾𑌏𑌷𑍀𑌯𑌾𑌵𑍍𑌯𑌯𑌾𑌶𑍇𑌯𑌾𑌨𑍋𑌨𑌯𑍋𑌨𑍀𑌂𑌘𑍃𑌤
 𑌕 𑌪𑌿 𑌪𑍍𑌲 𑌤 𑌥𑍍𑌯 𑌕𑌿𑌚 𑌕
𑌵𑌾𑌤𑍇𑌮𑌾𑌸𑌦𑌾𑌤𑍍।𑌹𑍋𑌇ழா𑌹।।24।।
 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
𑌆𑌸𑌾𑌵𑍀𑌸𑍋𑌮𑍋𑌆𑌰𑍂𑌷𑌾𑌃।𑌵𑌾𑌰𑍍𑌷𑌾𑌹𑌰𑌾𑌏।𑌰𑌾
 𑌚𑌿 𑌶 𑌚𑌿 𑌣 𑌕 𑌭𑍀
𑌜𑍇𑌵𑌾𑌦𑌾𑌸𑍍𑌮𑍋𑌆𑌭𑍀𑌗𑌾𑌃।𑌆𑌚𑌿𑌕𑍍𑌰𑌦𑌾𑌤𑍍।𑌪𑍂𑌨𑌾𑌨𑍋
 𑌚𑌿 𑌶 𑌚𑌿 𑌣 𑌕 𑌟𑌿 𑌚𑌿
𑌵𑌾𑌰𑌾𑌮𑌾𑌤𑍀𑌯𑍇।𑌷𑍀𑌅𑌵𑍍𑌯𑌯𑌾𑌂𑌶𑍇𑌯𑌨𑍋𑌨𑌾𑌯𑍋𑌨𑍀𑌂𑌘𑌾
 𑌶 𑌚𑌿 𑌣 𑌕 𑌟𑌿 𑌚𑌿 𑌶
𑌕𑍍𑌤𑌰𑌾𑌵𑌾𑌂।𑌤𑌮𑌾𑌸𑌾𑌦𑌾𑌔𑌹𑍋𑌵𑌾।𑌦𑍇𑌦𑍀𑌵𑍀।।25।।
 𑌚𑌿 𑌣 𑌤𑍍𑌯 𑌟𑌟 𑌚 𑌶𑌿 𑌤 𑌟
𑌅𑌸𑌾𑌵𑌿𑌸𑍋𑌮𑍋𑌅𑌰𑍁𑌷𑍋𑌵𑍃𑌷𑌾𑌹𑌰𑌾𑌇𑌃।𑌰𑌾𑌜𑍇𑌵
 𑌷𑍁 𑌤𑍃 𑌶
𑌦𑌸𑍍𑌮𑍋𑌅𑌭𑌿𑌗𑌾𑌅𑌚𑌿𑌕𑍍𑌰𑌦𑌾𑌤𑍍।𑌪𑍁𑌨𑌾𑌨𑍋𑌵𑌾𑌰𑌮𑌤𑍇𑌯
 𑌷𑍁 𑌤𑍃
𑌷𑍍𑌯𑌵𑍍𑌯𑌯𑌾𑌂।𑌶𑍇𑌯𑌨𑍋𑌨𑌯𑍋𑌨𑌿𑌂𑌘𑍃𑌤𑌵𑌾𑌨𑍍𑌤𑌮𑌾।
 𑌤𑍇 𑌤𑍃 𑌤𑍍𑌯
𑌸𑌾𑌦𑌾𑌔𑌹𑍋𑌵𑌾।𑌏𑌦𑍀𑌵𑍀।।26।।
 𑌟𑌟 𑌚 𑌶𑌿 𑌤
