।𑌋𑌤𑍁𑌸𑌾𑌮𑌨𑍀𑌚।।

2 ।𑌋𑌤𑍁𑌸𑌾𑌮𑌨𑍀𑌚।।

𑌓𑌇𑌚𑌿𑌤𑍍𑌰𑌇𑌚𑍍𑌛𑌾𑌇𑌶𑍋𑌸𑍍𑌤𑌰𑍁𑌣𑌾𑌸𑍍𑌯𑌾𑌵𑌾𑌕𑍍𑌷𑌾
 𑌯 𑌪𑌿 𑌕𑌿
𑌥𑌾:।𑌓𑌇𑌨𑌯𑍋𑌮𑌾𑌤𑌾𑌰𑌾𑌵𑌨𑍁𑌵𑌾𑌇𑌤𑍀𑌧𑌾𑌤𑌾
 𑌫𑍍𑌯 𑌯𑍃 𑌪𑌿 𑌕𑍀
𑌵𑍇।𑌓𑌅𑌨𑍁𑌧𑌾𑌯𑌾𑌦𑌜𑍀𑌜𑌨𑌾𑌦𑌧𑌾𑌚𑍀𑌦𑌾।
 𑌫 𑌯𑍁 𑌪𑍀 𑌚𑍍𑌯 𑌫𑍍𑌯
𑌓𑌇𑌵𑌵𑌕𑍍𑌷𑌸𑌾𑌦𑍇𑌯𑌾𑌮𑌾𑌹𑍀𑌦𑍍𑌯𑍂𑌤𑍍𑌯𑌾𑌨𑍍𑌚𑌾𑌰𑌾𑌨𑍍।
 𑌯𑍃 𑌟𑍍𑌯 𑌚𑌿 𑌤𑍍𑌰
𑌦𑍁𑌤𑍍𑌯𑌨𑍍𑌚𑌰𑌨𑍍𑌮𑌾𑌹𑍀
 𑌟𑍁 𑌚
𑌚𑌿𑌤𑍍𑌰𑌾𑌏।𑌏𑌏𑌇𑌚𑍍𑌛𑌿𑌶𑍋𑌸𑍍𑌤𑌰𑍂𑌣𑌸𑍍𑌯
 𑌤𑍍𑌯 𑌤 𑌤 𑌪 𑌷𑍁
𑌵𑌕𑍍𑌷𑌥𑌕𑍍𑌷𑌥𑍋𑌹𑍀𑌹𑍀𑌹𑍀𑌯𑌾𑌹𑌾𑌬𑍁।𑌏𑌏
 𑌤𑍈 𑌤 𑌶 𑌝𑍍𑌯
𑌨𑌯𑍋𑌮𑌾𑌤𑌰𑌾𑌵𑌨𑍍𑌵𑍇𑌤𑌿𑌧𑌾𑌤𑌵𑍇𑌤𑌵𑍇𑌹𑍀𑌹𑍀𑌹𑍀𑌯𑌾
 𑌷𑍁 𑌤𑍈
𑌹𑌾𑌬𑍁।𑌏𑌏𑌅𑌨𑍁𑌧𑌾𑌯𑌦𑌜𑍀𑌜𑌨𑌦𑌧𑌾𑌚𑌿𑌦𑌾
 𑌤 𑌶 𑌝𑍍𑌯 𑌷
𑌚𑌿𑌦𑌾𑌹𑍀𑌹𑍀𑌹𑍀𑌯𑌾𑌹𑌾𑌬𑍁।𑌏𑌏𑌵𑌵𑌕𑍍𑌷𑌸𑍍𑌸𑌦𑍇𑌯𑌾
 𑌤𑍈 𑌤 𑌶 𑌝𑍍𑌯 𑌷𑍁
𑌮𑌹𑌿𑌦𑍍𑌯𑍁𑌤𑍍𑌯𑌨𑍍𑌨𑍍𑌚𑌰𑌨𑍍𑌨𑍍𑌚𑌰𑌂𑌹𑍀𑌹𑍀𑌹𑍀𑌯𑌾𑌹𑌾𑌉𑌵𑌾।
      𑌤𑍈 𑌤𑌿
𑌏𑌋𑌤𑍂
 𑌤 𑌨𑍍𑌯
