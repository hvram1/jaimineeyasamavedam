5 ।𑌪𑌾𑌥𑍇𑌰𑌚𑍇 ।

𑌸𑍁𑌷𑍍𑌵𑌾𑌣𑌾𑌸𑌾𑌃।𑌇𑌨𑍍𑌦𑌰𑌾𑌸𑍍𑌤𑍁𑌮𑌸𑍀𑌤𑍍𑌵𑌾।𑌸𑌨𑌿𑌷𑍍𑌯𑌨𑍍𑌤𑌶𑍍𑌪𑌿𑌕𑍍𑌤𑍂𑌵𑌿
 𑌤𑍀 𑌚𑍁 𑌶       𑌷𑌿
𑌨𑍍𑌰𑌮𑍍𑌣𑌾𑌵𑌾𑌜𑌂।𑌆𑌨𑍋𑌭𑌰𑌾𑌉𑌵𑍋𑌵𑌾।𑌸𑍁𑌵𑌿𑌤𑌯𑍍𑌯𑌸𑍍𑌯
 𑌚𑍂 𑌶 𑌚 𑌶 𑌚𑍀 𑌣
𑌕𑍋𑌨𑌾𑌤𑌾𑌨𑌾𑌤𑍍𑌮𑌨𑌾।𑌸𑌹𑍍𑌯𑌾𑌮𑌾𑌤𑍂𑌵𑍋।𑌤𑌾𑌃।।7।।
 𑌤𑍈 𑌚𑌿 𑌚𑍍𑌯 𑌤𑍍𑌰
𑌓𑌹𑍋𑌹𑍋𑌇𑌸𑍂𑌷𑍍𑌵𑌾।𑌣𑌾𑌸𑌾𑌇𑌨𑍍𑌦𑌰𑌾𑌸𑍍𑌤𑍁𑌮𑌸𑍀𑌤𑍍𑌵𑌾।
 𑌤 𑌤 𑌚𑌿 𑌣 𑌕𑍍𑌯 𑌚𑍁 𑌶
𑌓𑌹𑍋𑌹𑍋𑌇𑌸𑌾𑌨𑍀।𑌷𑍍𑌯𑌨𑍍𑌤𑌾𑌶𑍍𑌪𑍀𑌤𑍂𑌵𑍀𑌨𑍍𑌰𑌮𑍍𑌣𑌾𑌵𑌾𑌜𑌂।
 𑌤 𑌤 𑌚𑌿 𑌣 𑌭𑌿 𑌚𑍁 𑌶
𑌓𑌹𑍋𑌹𑍋𑌆𑌨𑍋।𑌭𑌰𑌾𑌸𑍁𑌵𑌿𑌤𑌯𑍍𑌯𑌸𑍍𑌯𑌕𑍋𑌨𑌾।
 𑌤 𑌤 𑌚𑍍𑌯 𑌣 𑌭𑍍𑌯 𑌚𑍂 𑌶
𑌓𑌹𑍋𑌹𑍋𑌇𑌤𑌾𑌨𑌾।𑌤𑍍𑌮𑌨𑌾𑌸𑌹𑍍𑌯𑌮𑌾𑌤𑍂𑌵𑍋।𑌤𑌾𑌃
 𑌤 𑌤 𑌚𑌿 𑌣 𑌕𑍍𑌯 𑌪𑌿 𑌚𑍍𑌯 𑌤𑍍𑌰
