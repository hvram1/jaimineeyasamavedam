2 ।𑌸𑌭𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌪𑌵𑌸𑍍𑌵𑌮𑌧𑍁𑌮𑌾𑌇𑌹𑌾।𑌕𑍍𑌤𑌾𑌮𑌾𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌸𑍋𑌮Ǐ𑌕𑍃𑌤𑍁𑌵𑍀
𑌤𑍃𑌕𑍇
𑌕𑍍𑌤𑌮𑍋𑌮𑌾𑌦𑌾𑌇𑌹𑌾।𑌮𑌹𑌾𑌇𑌦𑍍𑌯𑍂𑌕𑍍𑌷𑌾𑌇𑌹𑌾।𑌤𑌾𑌮𑍋
𑌟𑌿𑌤𑌿𑌟𑍀𑌤𑌿
𑌮𑌾𑌦𑌾𑌃।𑌓𑌇ழா𑌹।।3।।
𑌟𑌿𑌚𑌣𑌪𑌶𑍍𑌯
𑌪𑌾𑌵𑌸𑍍𑌵𑌮𑌧𑍂𑌮𑌾𑌕𑍍𑌤𑌾𑌮𑌾𑌃।𑌆𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌸𑍋𑌮𑌾𑌕𑍍𑌰𑌾
𑌚𑌪𑍍𑌲𑌨𑍍𑌦𑌶𑌟𑍂𑌟𑌕
𑌤𑍁𑌵𑌾𑌇𑌤𑍍𑌤𑌾𑌮𑍋𑌮𑌾𑌦𑌾𑌃𑌮𑌾𑌹𑌿𑌦𑍍𑌯𑍁𑌕𑍍𑌷𑌾𑌤𑌾𑌮𑍋𑌮𑌾
𑌪𑍀𑌶𑌤𑌤𑌪𑍁𑌶𑌚
𑌦𑍋𑌹𑌾𑌇।।4।।
𑌫𑍍𑌲𑌶𑍍𑌯
𑌪𑌾𑌵𑌾𑌸𑍍𑌵𑌾𑌮𑌧𑍂𑌮𑌕𑍍𑌤𑌾𑌮𑌾𑌃।𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑌸𑍋𑌮𑌾𑌕𑍍𑌰𑌾
𑌪𑌿𑌪𑍍𑌲𑌚𑍍𑌯𑌣𑌟𑍁𑌚𑌕
𑌤𑍁𑌵𑌾𑌇𑌕𑍍𑌤𑌾𑌮𑍋𑌮𑌾𑌦𑌾𑌃।𑌮𑌹𑌾𑌇𑌦𑍍𑌯𑍁𑌕𑍍𑌷𑌾𑌤𑌾𑌮𑍋𑌮𑌾
𑌪𑍀𑌶𑌤𑌤𑌪𑍂𑌶𑌚
𑌦𑍋𑌹𑌾𑌇।।5।।
𑌪𑍍𑌲𑌶𑍍𑌯
