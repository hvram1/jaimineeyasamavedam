4 ।𑌦𑌾𑌶𑌸𑍍𑌖𑌤𑍍𑌯𑌾𑌨𑌿𑌷𑌟𑍍…

𑌸𑌾𑌚𑌾𑌂।𑌉𑌕𑍍𑌷𑍋𑌮𑌜𑌰𑍍𑍍𑌜𑌯𑌭𑌸𑍍𑌵𑌾𑌸𑌾𑌰𑌾:।𑌦𑌾𑌶𑌾𑌧𑍀𑌰
𑌤𑍍𑌯𑌷𑍍𑌯𑌕𑍁𑌟𑌤𑌟𑍍𑌯
𑌸𑍍𑌯𑌧𑍀𑌤𑌯𑍋𑌧𑌾𑌨𑍂𑌤𑍍𑌰𑍀:।𑌹𑌾𑌰𑍀:𑌪𑌰𑍍𑌯𑌦𑍍𑌰𑌵𑌜𑍍𑌜𑌾𑌸𑍍𑌸𑍂𑌯𑌾
𑌕𑍂𑌟𑍍𑌯𑌤𑌟𑍍𑌯𑌟𑍃
𑌸𑍍𑌯𑌾।𑌦𑍇𑌰𑌾𑌣𑌾𑌨𑍍𑌨𑌾𑌨𑌕𑍍𑌷𑍇𑌅𑌤𑍇𑌯𑌾𑌨𑌵𑌾𑌹𑌾𑌉𑌵𑌾।
𑌤𑌟𑍍𑌯𑌚𑌕𑌟𑍂𑌤𑌿
𑌜𑍀।।9।।
𑌚
𑌸𑌾𑌕𑌮𑍁𑌕𑍍𑌷𑌾𑌏।𑌏𑌏𑌮𑌜𑌰𑍍𑍍𑌜𑌯𑌭𑌸𑍍𑌵𑌾𑌸𑌾𑌰𑍋𑌦𑌶𑌧𑍀𑌰𑌾
𑌤𑍀𑌤𑌤𑌪𑌤𑍈
𑌏।𑌏𑌏𑌸𑍍𑌯𑌧𑍀𑌤𑌯𑍋𑌧𑌨𑍁𑌤𑍍𑌰𑌿𑌰𑍍𑌹𑌰𑌿:𑌪𑌰𑍍𑌯𑌾𑌏।𑌏𑌏
𑌤𑌤𑌪𑌤𑍈𑌤𑌤𑌪
𑌦𑍍𑌰𑌵𑌜𑍍𑌜𑌸𑍍𑌸𑍂𑌰𑍍𑌯𑌸𑍍𑌯𑌦𑍇𑌰𑌾𑌣𑌨𑍍𑌨𑌨𑌾𑌏।𑌏𑌏𑌕𑍍𑌷𑍇𑌅
𑌤𑍇𑌤𑌤𑌪
𑌤𑍇𑌯𑌾𑌨𑌵𑌾𑌹𑌾𑌉𑌵𑌾।𑌜𑍀।।10।।
𑌪𑍍𑌲𑌶𑍍𑌯𑌚
𑌇𑌨𑍍𑌨𑍍𑌦𑍂𑌰𑍍𑌵𑌾𑌜𑍀𑌪𑌾𑌵𑌤𑍇𑌗𑍋𑌨𑍇𑌯𑌾𑌘𑌾𑌇𑌨𑍍𑌦𑌰𑌾𑌇
𑌥𑍍𑌯𑌟𑍍𑌯𑌚𑌕𑍂𑌚𑌕
𑌸𑍍𑌸𑍋𑌮𑌾:।𑌸𑌹𑌇𑌨𑍍𑌵𑌨𑍍𑌮𑌦𑌾𑌯𑌾𑌹𑌭3𑌇𑌰𑌾𑌕𑍍𑌷𑌾:।𑌬𑌾𑌧𑌤𑍇
𑌟𑌿𑌤𑌕𑍂𑌚𑌕𑌟𑌿𑌤
𑌪𑌰𑌿𑌯𑌰𑌾𑌤𑌿𑌂𑌵𑌾𑌰𑌾𑌇𑌪𑌾:𑌕𑍃।𑌣𑍍𑌵𑌾𑌨𑍍𑌵𑍃𑌜𑌨𑌾𑌸𑍍𑌯𑌾𑌰𑌾
𑌕𑍄𑌚𑌕𑌤𑍀𑌥𑌟𑌿𑌪𑍍𑌲
𑌜𑍗𑌵𑌾𑌉𑌵𑍋𑌵𑌾।𑌹𑍋𑌇ழா𑌹।।11।।
𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
𑌇𑌨𑍍𑌦𑍁𑌰𑍍𑌵𑌾𑌜𑍀𑌪𑌵𑌤𑍗𑌹𑍋𑌔𑌹𑍋𑌵𑌾𑌹𑌾𑌇𑌗𑍋𑌨𑍇𑌯𑌾
𑌤𑍋𑌤𑌶
𑌘𑍗𑌵𑌾𑌉𑌵𑍋𑌵𑌾।𑌪𑍍𑌰।𑌈𑌨𑍍𑌦𑍇𑌰𑌸𑍍𑌸𑍋𑌮𑌸𑍍𑌸𑌾𑌹𑌾𑌇𑌨𑍍𑌵𑌨𑍍𑌮𑌦𑌾
𑌕𑌿𑌚𑌿𑌶𑌕𑍍𑌯𑌕𑍀
𑌯𑍗।𑌪𑍍𑌰।𑌹𑌭[𑌰𑌕𑍍𑌷𑍋𑌬𑌾𑌧𑌤𑍇𑌪𑌰𑌿𑌯𑌰𑌾𑌤𑍗।𑌪𑍍𑌰।
𑌕𑍁𑌕𑌟𑍂𑌚𑌕𑍁
𑌵𑌰𑌿𑌵:𑌕𑍃𑌣𑍍𑌵𑌨𑍍𑌵𑍃𑌜𑌨𑌾𑌸𑍍𑌯𑌾𑌰𑌾𑌜𑍗𑌵𑌾𑌉𑌵𑍋𑌹𑍋
𑌕𑍀𑌟𑍀𑌪𑍍𑌲𑌪𑍍𑌲𑌪𑍍𑌲
𑌇ழா𑌹।।12।।
𑌶𑍍𑌯
𑌇𑌨𑍍𑌦𑍁𑌰𑍍𑌵𑌾𑌜𑍀𑌪𑌵𑌤𑍇𑌗𑍋𑌨𑍇𑌯𑌾𑌘𑌾:।𑌇𑌨𑍍𑌦𑍇𑌰𑌸𑍍𑌸𑍋𑌮𑌸𑍍𑌸𑌹
𑌷𑍀𑌤𑍁𑌤
𑌇𑌨𑍍𑌵𑌨𑍍𑌮𑌦𑌾𑌯𑌾।𑌹𑌭𑌭𑍍𑌯𑌰𑌕𑍍𑌷𑍋𑌬𑌾𑌧𑌤𑍇𑌪𑌰𑍍𑌯𑌰𑌾𑌤𑌿𑌂।𑌵
𑌤𑍇𑌶𑌤𑍇𑌶
𑌰𑌿𑌵:𑌕𑍃𑌣𑍍𑌵𑌨𑍍𑌵𑍃𑌜𑌨𑌸𑍍𑌯𑌰𑌾𑌜𑌾𑌰𑌾𑌜𑌾𑌔𑌹𑍋𑌵𑌾।𑌏𑌸𑍍𑌯𑌾
𑌤𑍇𑌚𑌿𑌶𑌿𑌤𑌚
𑌰𑌾𑌜𑌾।।13।।
𑌟𑍍𑌯𑌚
𑌇𑌨𑍍𑌦𑍁𑌰𑍗𑌹𑍋𑌵𑌾𑌹𑌾𑌈𑌯𑌾।𑌵𑌾𑌜𑌾𑌉𑌵𑌾𑌹𑌾𑌉𑌵𑌾।𑌪𑍍𑌰।
𑌟𑍃𑌤𑌤𑍀𑌤𑌿
𑌪𑌵𑌾𑌤𑍇𑌗𑍋𑌨𑍇𑌯𑌾𑌘𑌾।𑌪𑍍𑌰।𑌇𑌨𑍍𑌦𑌰𑌾𑌇𑌸𑍋𑌮𑌾𑌸𑍍𑌸𑌹
𑌕𑍍𑌯𑌥𑍍𑌯𑌕𑍍𑌯𑌕𑌕𑍀
𑌇𑌨𑍍𑌵𑌨𑍍।𑌪𑍍𑌰।𑌮𑌾𑌦𑌾𑌔𑌹𑍋𑌵𑌾।𑌯𑌾।𑌹𑌭3𑌇।𑌰
𑌕𑍀𑌟𑌟𑌚𑌶𑌿𑌚𑌕𑌚𑌶
𑌕𑍍𑌷𑍋𑌬𑌾𑌧𑌾𑌤𑌾𑌇𑌪𑌾।𑌪𑍍𑌰।𑌰𑍀𑌯𑌾𑌰𑌾𑌔𑌹𑍋𑌵𑌾।𑌤𑍀𑌂।
