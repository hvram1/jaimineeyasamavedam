।𑌯𑌾𑌮𑌞𑍍𑌚।

10 ।𑌯𑌾𑌮𑌞𑍍𑌚।

𑌆𑌯𑌾𑌮𑌯𑌾𑌯𑌮𑍗𑌹𑍋𑌵𑌾𑌊।𑌪𑍍𑌰।𑌨𑌾𑌕𑍇𑌸𑍁𑌪𑌰𑍍𑌣𑌾
 𑌕𑍀 𑌟𑍍𑌯 𑌥𑍍𑌯 𑌟
𑌮𑍁𑌪𑌯𑌾𑌤𑍍𑌪𑌤𑌨𑍍𑌤𑌂𑌪𑌤𑌨𑍍𑌤𑌮𑍗𑌹𑍋𑌵𑌾𑌊।𑌪𑍍𑌰।
 𑌕𑍇 𑌕𑌿 𑌟𑍍𑌯 𑌥𑍍𑌯 𑌟
𑌹𑍃𑌦𑌾𑌵𑍇𑌨𑌨𑍍𑌤𑍋𑌅𑌭𑍍𑌯𑌾𑌚𑌾𑌕𑍍𑌷𑌤𑌤𑍍𑌵𑌾𑌕𑍍𑌷𑌤𑌤𑍍𑌵𑍗𑌹𑍋
 𑌕𑍃 𑌟𑌿 𑌟𑌿
𑌵𑌾𑌊।𑌪𑍍𑌰।𑌹𑌿𑌰𑌣𑍍𑌯𑌪𑌕𑍍𑌷𑌂𑌵𑌰𑍂𑌣𑌾𑌸𑍍𑌯𑌦𑍂𑌤𑌂𑌸𑍍𑌯𑌦𑍂
 𑌥𑍍𑌯 𑌟 𑌕𑍇 𑌕𑌿
𑌤𑌮𑍗𑌹𑍋𑌵𑌾𑌊।𑌪𑍍𑌰।𑌯𑌾𑌮𑌸𑍍𑌯𑌯𑍋𑌨𑍗𑌶𑌕𑍁𑌨𑌾𑌂𑌭𑍁
 𑌟𑍍𑌯 𑌥𑍍𑌯 𑌟
𑌰𑌣𑍍𑌯𑍁𑌂𑌭𑍁𑌰𑌣𑍍𑌯𑍁𑌮𑍗𑌹𑍋𑌵𑌾𑌊।𑌪𑍍𑌰।𑌵𑌾𑌹𑌾
 𑌕𑍇 𑌕𑌿 𑌟𑍍𑌯 𑌥𑍍𑌯 𑌟 𑌟𑌚 𑌥
𑌉𑌵𑌾।𑌏𑌦𑌿𑌵𑌮𑍇𑌦𑌿𑌵𑌮𑍇𑌦𑌿𑌵𑌾𑌂।।16।।
 𑌟𑍍𑌯 𑌤 𑌕𑌿 𑌕𑌿 𑌟𑍍𑌯𑌚
