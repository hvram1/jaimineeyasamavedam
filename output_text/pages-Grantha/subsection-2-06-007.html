।𑌵𑌾𑌚𑌸𑍍𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।

7 ।𑌵𑌾𑌚𑌸𑍍𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।

𑌕𑌯𑌾𑌨𑌶𑍍𑌪𑍀।𑌤𑍍𑌰𑌾𑌯𑌾𑌭𑍂𑌵𑌾𑌤𑍍।𑌊𑌤𑌾𑌇
  𑌤𑍀 𑌤 𑌪 𑌶 𑌕
𑌸𑌦𑌾𑌵𑌾𑌰𑍍𑌧𑌸𑍍𑌸𑌾𑌚𑌾।𑌕𑌯𑌾𑌶𑌾𑌚𑌾𑌇।
 𑌕𑍁 𑌚𑍍𑌯 𑌣 𑌟𑌿 𑌤 𑌶
𑌷𑍍𑌠𑌾𑌯𑌾𑌵𑌾𑌤𑍋𑌰𑌹𑌾𑌇।।13।।
 𑌪 𑌶 𑌚  𑌪𑍍𑌲 𑌶𑍍𑌯
𑌹𑍋𑌵𑌾𑌇𑌹𑍋𑌵𑌾𑌇𑌕𑌯𑌾𑌨𑌶𑍍𑌪𑍀।𑌤𑍍𑌰𑌾𑌯𑌾𑌭𑍂𑌵𑌾𑌤𑍍।
 𑌤𑍇 𑌤 𑌪𑍍𑌲 𑌶
𑌹𑍋𑌵𑌾𑌇𑌹𑍋𑌵𑌾𑌇𑌊𑌤𑍀𑌸𑍍𑌸𑌾𑌦𑌾𑌵𑍃𑌧𑌾𑌸𑍍𑌸𑌾𑌚𑌾
 𑌚𑍄 𑌕𑍍𑌯𑌚 𑌕𑍍𑌯 𑌪 𑌶
𑌹𑍋𑌵𑌾𑌇𑌹𑍋𑌵𑌾𑌇𑌕𑌯𑌾𑌶𑌚𑌾𑌇।𑌷𑍍𑌠𑌾𑌯𑌾𑌵𑌾𑌕𑍍𑌤𑌰𑌾
 𑌤𑍄 𑌤𑍍𑌯 𑌶 𑌟𑌿𑌟 𑌚
𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾।।15।।
 𑌶𑌿 𑌚 𑌶
