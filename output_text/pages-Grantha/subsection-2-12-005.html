।𑌕𑍗𑌤𑍍𑌸𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿 ।

5 ।𑌕𑍗𑌤𑍍𑌸𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿 ।

𑌆𑌯𑌾𑌹𑍀।𑌊𑌪𑌾𑌨𑌾𑌸𑍍𑌸𑍂𑌤𑌂𑌹𑍋𑌵𑌾𑌹𑌾𑌇।
𑌤𑌿𑌚𑌚𑍍𑌯𑌶𑍍𑌯𑌟𑍍𑌯𑌤𑌶
𑌵𑌾𑌜𑍇𑌭𑌿𑌰𑍍𑌮𑍍𑌮𑌾𑌹𑍃𑌣𑍀𑌯𑌥𑌾।𑌪𑍍𑌰।𑌮𑌾𑌹𑌂𑌈𑌵𑌾
𑌕𑌿𑌕𑌿𑌤𑍍𑌯𑌚𑍍𑌯
𑌯𑍁𑌵𑌾।𑌪𑍍𑌰।𑌜𑌾𑌨𑌾𑌔𑌹𑍋𑌵𑌾।𑌉𑌊𑌪𑌾।।7।।
𑌟𑍀𑌟𑌟𑌚𑌶𑌿𑌚𑍍𑌯𑌶
𑌓𑌹𑍋𑌇।𑌕𑌾𑌦𑌾𑌵𑌸𑍋𑌸𑍍𑌤𑍋𑌤𑍍𑌰𑌾𑌂𑌹𑌰𑍍𑌯𑌤𑌾𑌯𑌾।
𑌟𑌚𑌚𑌶𑌕𑍍𑌯𑌫𑍍𑌯𑌟𑍍𑌯𑌚𑌿𑌶
𑌓𑌹𑍋𑌇।𑌯𑌾𑌵𑌾𑌶𑍍𑌮𑌾𑌶𑌾𑌰𑍂𑌧𑌾𑌦𑍂𑌵𑌾𑌃।𑌓𑌹𑍋𑌇।
𑌟𑌚𑌚𑌶𑌕𑍍𑌯𑌫𑍍𑌯𑌚𑌿𑌣𑌟𑌚𑌚𑌶
𑌦𑍀𑌰𑍍𑌘𑌂𑌸𑍂𑌤𑌂𑌵𑌾𑌤𑌾𑌪𑍀𑌯𑌾𑌯𑌾।𑌈।।8।।
𑌕𑍍𑌯𑌫𑍍𑌯𑌪𑍍𑌲𑌚𑍍𑌯𑌤𑍍𑌰𑌚
𑌕𑌾𑌦𑌾𑌔𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌵𑌾𑌸𑍋𑌸𑍍𑌤𑍋𑌤𑍍𑌰𑌾𑌂
𑌚𑍍𑌯𑌕𑌥𑍍𑌯𑌟𑍀
𑌹𑌰𑍍𑌯𑌤𑌾𑌯𑌾।𑌅𑌵𑌾।𑌪𑍍𑌰।𑌶𑍍𑌪𑌶𑌾𑌰𑍂𑌧𑌾𑌦𑍂𑌵𑌾𑌃।
𑌚𑌿𑌶𑌚𑍍𑌯𑌕𑍍𑌯𑌚𑌿𑌣
𑌦𑍀𑌰𑍍𑌘𑌾।𑌪𑍍𑌰।𑌸𑍂𑌤𑌾𑌂।𑌵𑌾𑌤𑌾𑌪𑍀𑌯𑌾𑌯𑌾।।9।।
𑌚𑍍𑌯𑌟𑍍𑌯𑌚𑍍𑌯𑌚𑍍𑌯𑌤𑍍𑌰
𑌔𑌹𑍋𑌵𑌾𑌇।𑌪𑍍𑌰।𑌕𑌾𑌦𑌾𑌵𑌾𑌸𑍋𑌸𑍍𑌤𑍋𑌤𑍍𑌰𑌾𑌂
𑌚𑌶𑌿𑌕𑌿𑌫𑍍𑌯𑌟𑍍𑌯
𑌹𑌰𑍍𑌯𑌾𑌤𑌾𑌯𑌾।𑌪𑍍𑌰।𑌯𑌾𑌵𑌾𑌶𑍍𑌮𑌾𑌶𑌾𑌰𑍂𑌧𑌾𑌦𑍂𑌵𑌾𑌃।
𑌚𑌿𑌶𑌕𑍍𑌯𑌫𑌚𑌿𑌣
।𑌪𑍍𑌰।𑌦𑍀𑌰𑍍𑌘𑌂𑌸𑍂𑌤𑌂𑌵𑌾𑌤𑌾𑌪𑍀𑌯𑌾𑌯𑌾।𑌏।।10।।
𑌕𑍍𑌯𑌫𑍍𑌯𑌪𑍍𑌲𑌚𑍍𑌯𑌤𑍍𑌰𑌤𑌚
