2 ।𑌭𑌾𑌰𑌚𑌾𑌜𑍇𑌚𑍇 ।

𑌤𑍍𑌵𑌾𑌮𑌿𑌡𑍀।𑌹𑌵𑌾𑌮𑌾𑌹𑌾𑌆𑌔𑌹𑍋𑌵𑌾𑌹𑌾𑌉𑌵𑌾।
𑌤𑍀𑌟𑍍𑌯𑌚𑍍𑌯𑌚𑌕𑌶𑍁
𑌊𑌪𑌾।𑌪𑍍𑌰।𑌸𑌾𑌤𑍗𑌵𑌾𑌜𑌸𑍍𑌯𑌾𑌕𑌾𑌰𑌾𑌵𑌾।𑌪𑍍𑌰।
𑌚𑌶𑌚𑌿𑌕𑌟𑍍𑌯𑌟𑍍𑌯
𑌤𑍍𑌵𑌾𑌂𑌵𑍍𑌰𑌤𑍍𑌰𑌾𑌇𑌷𑍂𑌇𑌨𑍍𑌦𑌰𑌾𑌸𑌾𑌤𑍍𑌖𑌾𑌤𑍀𑌮𑌾𑌔।𑌪𑍍𑌰।𑌣𑌰
𑌟𑍁𑌥𑌟𑍍𑌯𑌕𑍍𑌯𑌚𑍍𑌯
𑌸𑍍𑌤𑍍𑌵𑌾𑌂𑌕𑌾𑌷𑍍𑌠𑌾𑌸𑍁𑌵𑌾𑌰𑍍𑌵𑌾𑌤𑌾𑌆𑌔𑌹𑍋𑌵𑌾𑌹𑌾𑌉𑌵𑌾।
𑌕𑌿𑌟𑍍𑌯𑌟𑍍𑌯𑌕𑍍𑌯𑌚𑌕𑌶𑍁
𑌉𑌊𑌪𑌾।।3।।
𑌚𑍍𑌯𑌶
𑌤𑍍𑌵𑌾𑌮𑌿𑌢𑌿𑌹𑌵𑌾𑌮𑌹𑍇𑌸𑌾𑌤𑍗𑌵𑌾𑌜𑍋𑌵𑌾।𑌸𑍍𑌯𑌾𑌕𑌾𑌰𑌾𑌵𑌾𑌃।
𑌤𑍈𑌤𑌚𑌯𑌟𑍍𑌯
𑌤𑍍𑌵𑌾𑌂𑌵𑍍𑌰𑌤𑍍𑌰𑌾𑌇𑌷𑍂𑌇𑌨𑍍𑌦𑌰𑌾𑌸𑌾𑌤𑍍𑌪𑌾𑌤𑍀𑌨𑍍𑌨𑌾𑌰𑌾𑌃।𑌤𑍂𑌵𑌾𑌂𑌕𑌾
𑌕𑍄𑌚𑌯𑌟𑍍𑌯𑌟𑌿
𑌷𑍍𑌠𑌾।𑌸𑍂𑌵𑌰𑍍𑌵𑌾𑌤𑌾𑌃।𑌓𑌇ழா𑌹।।4।।
