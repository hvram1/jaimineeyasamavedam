। 𑌗𑍗𑌤𑌮𑌸𑍍𑌯𑌰𑌯𑌿𑌷𑍍𑌠𑍇𑌚𑍇।

13 । 𑌗𑍗𑌤𑌮𑌸𑍍𑌯𑌰𑌯𑌿𑌷𑍍𑌠𑍇𑌚𑍇।

𑌯𑍋𑌰𑌯𑌿𑌂𑌵𑍋𑌰𑌯𑌾𑌹𑌾𑌬𑍁।𑌤𑌾𑌮𑌾𑌃।𑌯𑍋𑌦𑍍𑌯𑍁𑌮𑍍𑌨𑍈𑌦𑍍𑌯𑍁
𑌕𑍂𑌤𑌶𑌚𑌪𑍍𑌲𑌷𑍍𑌯
𑌮𑍍𑌨𑌵𑌾𑌕𑍍𑌤𑌮𑌾𑌸𑍋𑌮𑌸𑍍𑌸𑍁𑌤𑌸𑍍𑌸𑌾𑌯𑌾𑌹𑍋𑌇𑌨𑍍𑌦𑌰𑌤𑌾𑌇।
𑌕𑍂𑌟𑍂𑌟𑍀𑌶
𑌆𑌸𑍍𑌤𑌿𑌸𑍍𑌵𑌧𑌾𑌪𑌾𑌤𑌾𑌇𑌹𑍋𑌯𑍇𑌮𑌦𑍋𑌇ழா𑌹।।27।।
𑌟𑍂𑌟𑌿𑌚𑍍𑌯𑌪𑍍𑌲
𑌯𑍋𑌰𑌯𑌿𑌂𑌵𑍋𑌰𑌯𑌿𑌨𑍍𑌤𑌮𑍋𑌹𑌾𑌇।𑌯𑍋𑌦𑍍𑌯𑍁𑌮𑍍𑌨𑍈𑌰𑍍𑌦𑍁𑌮𑍍𑌨𑌾
𑌫𑍂𑌚𑍍𑌯𑌶𑍍𑌯
𑌵𑌾𑌕𑍍𑌤𑌮𑍋𑌹𑌾𑌇।𑌸𑍋𑌮𑌸𑍁𑌤𑌸𑍍𑌸𑌇𑌨𑍍𑌦𑌰𑌤𑍋𑌹𑌾𑌇।
𑌫𑍂𑌚𑍍𑌯𑌶𑍍𑌯𑌫𑍂𑌚𑍍𑌯𑌶𑍍𑌯
𑌅𑌸𑍍𑌤𑌿𑌸𑍍𑌵𑌾𑌧𑌾𑌪𑌾𑌤𑍇𑌮𑌦𑍋𑌹𑌾।𑌹𑍋𑌇ழா𑌹।।28।।
𑌚𑍀𑌕𑌫𑌚𑍍𑌯𑌪𑍍𑌲𑌪𑍍𑌲𑌶𑍍𑌯
