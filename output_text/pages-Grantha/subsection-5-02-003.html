3 ।𑌬𑌭𑍇𑌰𑌾:𑌕𑍂𑌂𑌭𑌸𑍍𑌯𑌸𑌾𑌮𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿…

𑌇𑌨𑍍𑌦𑍂𑌰𑍗𑌹𑍋𑌵𑌾𑌹𑌾𑌇𑌪𑌾𑌵𑍀।𑌷𑍍𑌟𑌚𑌾𑌈𑌤𑌨𑍋𑌹𑌾𑌇।
 𑌚𑍍𑌯 𑌶𑍃 𑌟𑍍𑌯 𑌚𑌿 𑌣 𑌶
𑌹𑌾𑌹𑍋𑌏𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌪𑍍𑌰𑍀𑌯𑌾:𑌕𑌵𑍀𑌨𑌾𑌮𑍍𑌮𑌤𑌿𑌰𑍋𑌹𑌾
 𑌕 𑌟𑍍𑌯 𑌟 𑌤 𑌕𑍀 𑌪𑍀 𑌣
𑌇।𑌪𑍍𑌰।𑌸𑍃𑌜𑌾𑌦𑌶𑍍𑌪𑌾𑌂𑌹𑌾𑌇।𑌪𑍍𑌰।𑌰𑌾𑌥𑌾𑌔𑌹𑍋𑌵𑌾।
 𑌶 𑌟𑍍𑌯 𑌚𑍍𑌯 𑌣 𑌶 𑌟 𑌚 𑌶𑌿
𑌈𑌵𑌾।।6।।
 𑌚 𑌶
𑌇𑌨𑍍𑌦𑍁:𑌪𑌵𑌿𑌷𑍍𑌟𑌚𑍇𑌤𑌨:𑌪𑍍𑌰𑌿𑌯:𑌕𑌵𑌾𑌇।𑌹𑌿𑌂𑌹𑌿𑌨𑍍𑌨𑌾𑌮𑍍𑌮𑌾𑌤𑍀:।
 𑌤𑍋 𑌶 𑌟 𑌚𑌿 𑌣
𑌸𑍃𑌜𑌦𑌾𑌶𑍍𑌪𑌾𑌂।𑌹𑌿𑌂𑌹𑌿𑌂𑌰𑌾𑌥𑌾𑌔𑌹𑍋𑌵𑌾।𑌈𑌵𑌾।।7।।
 𑌪𑌿 𑌶 𑌟 𑌤 𑌟𑌟 𑌚 𑌶𑌿 𑌚 𑌶
𑌇𑌨𑍍𑌦𑍁:𑌪𑌵𑌿𑌷𑍍𑌟𑌚𑍇𑌤𑌨:𑌪𑍍𑌰𑌿𑌯:𑌕𑌵𑌿𑌨𑌾𑌮𑍍𑌮𑌤𑌿𑌂𑌸𑍃𑌜𑌾:𑌦𑌶𑍍𑌪𑌾𑌂।
 𑌷𑌿 𑌤𑍇 𑌤𑍍𑌯
𑌓𑌵𑌾𑌓𑌵𑌾𑌰𑌾𑌥𑌾𑌔𑌹𑍋𑌵𑌾।𑌈𑌵𑌾।।8।।
 𑌕𑍍𑌯 𑌕𑍍𑌯 𑌟𑌟 𑌚 𑌶𑌿 𑌚 𑌶
