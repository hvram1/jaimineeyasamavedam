8 ।𑌮𑌰𑍂𑌤𑌾𑌂𑌕𑌾𑌲𑌕𑌾𑌕𑍍𑌰𑌭2…

𑌓𑌔𑌹𑍋𑌔𑌹𑍋𑌵𑌾𑌹𑌾।𑌤𑌿𑌸𑍇𑌰𑌾𑌵𑌾𑌚𑌾𑌈𑌰𑌾𑌯𑌾
𑌕𑍍𑌯𑌚𑌫𑍍𑌯𑌤𑌤𑌚𑌟𑍁𑌕𑍍𑌯
𑌤𑌿𑌪𑍍𑌰𑌵𑌾𑌨𑍍𑌹𑍀:।𑌓𑌹𑌾𑌹𑍋𑌹𑌾𑌓𑌹𑌾।𑌰𑌤𑌸𑍍𑌯𑌧𑌾𑌇𑌤𑍀𑌂
𑌚𑌿𑌪𑍍𑌲𑌣𑍍𑌯𑌫𑌚𑌶𑌿𑌟𑍂
𑌬𑍍𑌰𑌾𑌹𑍍𑌮𑌾𑌣𑍋𑌮𑌾𑌨𑍀𑌷𑌾𑌂।𑌆𑌹𑌾𑌹𑍋𑌇𑌆𑌹𑌾𑌹𑌾
𑌕𑌿𑌚𑍍𑌯𑌶𑌕𑍍𑌯𑌕𑌿𑌕𑍍𑌯𑌚
𑌹𑌾𑌹𑍋𑌇।𑌗𑌾𑌵𑍋𑌯𑌭3𑌇𑌗𑍋𑌪𑌾𑌤𑌿𑌂𑌪𑌾𑌚𑍍𑌛𑌰𑌾𑌮𑌾
𑌕𑌚𑌶𑌕𑌟𑍁𑌕𑌿𑌚𑍍𑌯
𑌨𑌾:।𑌇𑌹𑌾𑌹𑍋𑌇𑌈𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।𑌸𑍋𑌮𑌯𑍍𑌯
𑌪𑍍𑌲𑌕𑍍𑌯𑌕𑌿𑌕𑍍𑌯𑌚𑌕𑌚𑌶𑌕
𑌭3𑌇𑌮𑌾𑌤𑌯𑍋𑌵𑌾𑌵𑌾𑌶𑌾𑌨𑌾:।।19।।
𑌕𑍁𑌪𑌿𑌚𑍍𑌯𑌤𑍍𑌰
𑌓𑌔𑌹𑍋𑌔𑌹𑍋𑌔𑌹𑍋𑌔𑌹𑍋𑌵𑌾𑌹𑌾
𑌟𑌿𑌚𑌕𑌚𑌫𑍍𑌯𑌫𑍍𑌯𑌤𑌤
𑌹𑌾𑌇।𑌤𑌿𑌸𑍇𑌰𑌾𑌵𑌾𑌚𑌾𑌈𑌰𑌾𑌯𑌾𑌤𑌿𑌪𑍍𑌰𑌵𑌾𑌨𑍍𑌹𑍀:।𑌓𑌹𑌾
𑌤𑌚𑌶𑌟𑍁𑌕𑍍𑌯𑌚𑌿𑌪𑍍𑌲𑌣𑍍𑌯𑌫
𑌹𑍋𑌹𑌾𑌓𑌹𑌾।𑌰𑌤𑌸𑍍𑌯𑌧𑌾𑌇𑌤𑍀𑌂𑌬𑍍𑌰𑌾𑌹𑍍𑌮𑌾𑌣𑍋𑌮𑌾𑌨𑍀
𑌚𑌶𑌿𑌟𑍂𑌕𑌿𑌚𑍍𑌯
𑌷𑌾𑌂।𑌆𑌹𑌾𑌹𑍋𑌇𑌆𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।𑌗𑌾𑌵𑍋
𑌶𑌕𑍍𑌯𑌕𑌿𑌚𑍍𑌯𑌚𑌕𑌚𑌶𑌕
𑌯𑌭3𑌇𑌗𑍋𑌪𑌾𑌤𑌿𑌂𑌪𑌾𑌚𑍍𑌛𑌰𑌾𑌮𑌾𑌨𑌾:।𑌇𑌹𑌾𑌹𑍋𑌇
𑌟𑍁𑌕𑌿𑌚𑍍𑌯𑌪𑍍𑌲𑌕𑍍𑌯
𑌈𑌹𑌾𑌹𑌾𑌹𑌾𑌹𑍋𑌇।𑌸𑍋𑌮𑌯𑍍𑌯𑌭3𑌇𑌮𑌾𑌤𑌯𑍋𑌵𑌾
𑌕𑌿𑌕𑍍𑌯𑌚𑌕𑌚𑌶𑌕𑌕𑍁𑌪𑌿
𑌵𑌾𑌶𑌾𑌨𑌾:।।20।।
