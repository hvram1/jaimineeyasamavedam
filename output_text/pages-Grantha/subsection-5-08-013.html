।𑌸𑍋𑌮𑌸𑌾𑌮𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

14 ।𑌸𑍋𑌮𑌸𑌾𑌮𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌅𑌭𑌿𑌨𑍋𑌵𑌾।𑌜𑌾𑌸𑌾𑌔𑌹𑍋𑌵𑌾।𑌤𑌾𑌮𑌂।𑌰𑌯𑌿𑌮𑌾
𑌤𑍀𑌟𑌟𑌚𑌶𑌿𑌚𑌶𑌟𑌿
𑌰𑍍𑌷𑌶𑌾𑌤𑌸𑍍𑌖𑍃𑌹𑌾𑌮𑌿𑌨𑍍𑌦𑍋𑌸𑌹𑌸𑍍𑌰𑌭𑌾𑌹𑍋𑌰𑍍𑌣𑌸𑌾𑌂।
𑌕𑍍𑌯𑌕𑌿𑌟𑍂𑌟𑌿
𑌤𑍁𑌵𑌿𑌦𑍍𑌯𑍁𑌮𑍍𑌨𑌂𑌵𑌿𑌭𑌾𑌹𑍋𑌏।𑌸𑌹𑌮𑍋𑌇ழா𑌹।।25।।
𑌟𑍂𑌟𑍍𑌯𑌚𑌿𑌶𑍍𑌯
𑌅𑌭𑌿𑌨𑍋𑌵𑌾।𑌜𑌾𑌸𑌾𑌤𑌮𑌂𑌰𑌾𑌯𑍀𑌮𑌰𑍍𑌷𑌾𑌉𑌵𑌾𑌓𑌹𑌾𑌇।
𑌤𑍀𑌕𑍀𑌚𑍍𑌯𑌕𑍍𑌯𑌕𑍍𑌯𑌟𑌤𑌶
𑌶𑌤𑌸𑍍𑌖𑍃𑌹𑌾𑌮𑌿𑌨𑍍𑌦𑍋𑌸𑌾𑌹𑌾𑌉𑌵𑌾𑌓𑌹𑌾𑌇।𑌸𑍍𑌰𑌭𑌰𑍍𑌣
𑌚𑌿𑌟𑌿𑌚𑌕𑌟𑌿𑌟𑌤𑌶
𑌸𑌨𑍍𑌤𑍁𑌵𑌾𑌇𑌦𑍍𑌯𑍁𑌮𑍍𑌨𑌾𑌔𑌹𑍋𑌵𑌾।𑌵𑌿𑌭𑌾𑌸𑌾𑌹𑌾𑌂।।26।।
𑌟𑍄𑌟𑌚𑌶𑌿𑌤𑍍𑌯𑌟𑌚
𑌅𑌭𑍀𑌹𑍋𑌇𑌨𑍋𑌵𑌾𑌜𑌾𑌸𑌾𑌹𑍋𑌤𑌾𑌮𑌂।𑌰𑌯𑌿𑌂𑌹𑍋
𑌟𑍍𑌯𑌕𑌿𑌶𑌿𑌚𑍍𑌯𑌶𑌟𑍍𑌯𑌕
𑌅𑌰𑍍𑌷𑌶𑌾𑌤𑌾𑌹𑍋𑌸𑍍𑌖𑍃𑌹𑌂।𑌇𑌨𑍍𑌦𑍋𑌹𑍋𑌇𑌸𑌹𑌸𑍍𑌰𑌾
𑌚𑌚𑍍𑌯𑌶𑌚𑍍𑌯𑌶𑌟𑍍𑌯𑌚𑌶𑌚𑌿
𑌭𑌾𑌹𑍋𑌰𑍍𑌣𑌾𑌸𑌂।𑌤𑍁𑌵𑌾𑌹𑍋𑌇𑌦𑍍𑌯𑍁𑌮𑍍𑌨𑌂𑌵𑌿𑌭𑌾𑌹𑍋
𑌶𑌚𑍍𑌯𑌶𑌟𑍍𑌯𑌕𑌿
𑌸𑌾𑌹𑌾𑌂।।27।।
