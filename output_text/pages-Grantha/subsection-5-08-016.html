।𑌪𑍈𑌰𑌯𑍍𑌯𑌮𑍇𑌧𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

17 ।𑌪𑍈𑌰𑌯𑍍𑌯𑌮𑍇𑌧𑌾𑌨𑌿 𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌆𑌭𑍀।𑌓𑌇𑌨𑌵𑌨𑍍𑌤𑌾𑌯𑌾𑌦𑍍𑌰𑍂𑌹𑌾𑌃।𑌓𑌇𑌪𑍍𑌰𑌿𑌯𑌮𑌿𑌨𑍍𑌦𑌰𑌸𑍍𑌯𑌕𑌾𑌮𑌾
 𑌚 𑌣 𑌯𑍃 𑌟 𑌯𑍇
𑌯𑌾𑌂।𑌓𑌇𑌵𑌤𑍍𑌸𑌨𑍍𑌨𑌪𑍂𑌰𑍍𑌵𑌆𑌯𑍂𑌨𑍀।𑌓𑌇𑌜𑌾𑌤𑌂𑌰𑌿𑌹𑌨𑍍𑌤𑌿𑌮𑍋
 𑌟 𑌯𑍇 𑌟 𑌪𑍄
𑌬𑌾।𑌤𑌾𑌰𑌾𑌃।।30।।
 𑌪𑍍𑌲 𑌚 𑌪𑍍𑌲
𑌆𑌭𑍀𑌨𑌵𑌨𑍍𑌤𑌯𑌾𑌦𑍍𑌰𑍂𑌹𑌾𑌃।𑌪𑍍𑌰𑍀𑌯𑌾𑌮𑌿𑌨𑍍𑌦𑌰𑌸𑍍𑌯𑌾𑌕𑌾𑌮𑌾𑌯𑌾𑌂।
 𑌫𑍍𑌯 𑌚𑍀 𑌶𑍍𑌯 𑌤𑍍𑌯 𑌤𑍍𑌯 𑌕𑍍𑌯 𑌚 𑌣
𑌵𑌤𑍍𑌸𑌨𑍍𑌨𑌪𑍂𑌰𑍍𑌵𑌾𑌆𑌯𑍂𑌨𑍀।𑌜𑌾𑌤𑌂𑌹𑍋𑌇𑌰𑌿𑌹𑌾𑌹𑍋
 𑌤𑍀 𑌕𑍍𑌯 𑌚 𑌣 𑌕𑍍𑌯 𑌕𑍀 𑌂
𑌨𑍍𑌤𑍀𑌮𑌾𑌤𑌾𑌰𑌾𑌃।𑌓𑌇ழா𑌹।।31।।
 𑌷𑌿 𑌚𑌣 𑌪 𑌶𑍍𑌯
𑌅𑌭𑌿𑌨𑌵𑌨𑍍𑌤𑍇𑌅𑌦𑍍𑌰𑍁𑌹𑌃𑌓𑌹𑌾𑌇𑌪𑍍𑌰𑌿𑌯𑌮𑌿𑌨𑍍𑌦𑌰𑌸𑍍𑌯𑌕𑌾𑌮𑌿𑌯𑌮𑍋
 𑌷𑍍𑌯 𑌤𑍃 𑌤𑍈
𑌹𑌾𑌇𑌵𑌤𑍍𑌸𑌨𑍍𑌨𑌪𑍂𑌰𑍍𑌵𑌾𑌯𑍁𑌨𑍍𑌯𑍗𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌜𑌾𑌤𑌂𑌰𑌿
 𑌤𑍂 𑌤𑍁 𑌤
𑌹𑌾𑌉𑌵𑌾।𑌨𑍍𑌤𑍀𑌮𑌾।𑌤𑌾𑌰𑌾𑌔𑌹𑍋𑌬𑌾𑌹𑍋𑌇ழா𑌹।।32।।
 𑌕𑍀 𑌶𑍍𑌯 𑌚 𑌶 𑌟𑌿 𑌚 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
