7 ।𑌮𑌰𑍂𑌤𑌾𑌂𑌪𑍍𑌰𑌕𑍀ழா𑌹𑌸𑍍𑌤𑌰𑌯:𑌸𑌂𑌕𑍍𑌰𑍀ழா𑌹𑌵𑌾𑌨𑌿𑌕𑍍𑌰𑌿ழா𑌹𑌵𑌾…

𑌪𑌾𑌰𑍀।𑌸𑍍𑌵𑌾𑌨𑌾𑌸𑌇𑌨𑍍𑌦𑌵𑍋𑌮𑌦𑌾𑌯𑌾𑌬𑌰𑍍𑌹𑌣𑌾𑌗𑌿𑌰𑌾
 𑌤𑍍𑌯 𑌷𑍀 𑌚𑍁 𑌷𑌿 𑌚𑍍𑌯
𑌮𑌧𑍋𑌅𑌰𑍍𑌷𑌾𑌂।𑌤𑍀𑌧𑍋𑌬𑌾𑌰𑌾𑌯𑍋𑌹𑌾𑌇।।16।।
 𑌚𑍍𑌯 𑌤𑍍𑌯 𑌕 𑌪 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
𑌪𑌯𑍇𑌰𑌪𑌾𑌰𑍀।𑌸𑍍𑌵𑌾𑌨𑌾𑌸𑌇𑌨𑍍𑌦𑌵𑌾𑌉𑌵𑌾𑌹𑌾𑌉𑌵𑌾𑌹𑍋𑌵𑌾
 𑌤𑍀 𑌟𑍄 𑌟𑌿 𑌕 𑌥
𑌇𑌯𑌾।𑌪𑍍𑌰।𑌮𑌦𑌾𑌯𑌬𑌰𑍍𑌹𑌣𑌾𑌗𑌿𑌰𑌾𑌉𑌵𑌾।𑌪𑍍𑌰।𑌮𑌾𑌧𑍋𑌅𑌰𑍍𑌷
 𑌚𑍍𑌯  𑌷𑌿 𑌟𑍃 𑌷𑌿
𑌧𑌾𑌰𑌯𑌾𑌉𑌵𑌾।𑌹𑌾𑌉𑌵𑌾𑌹𑍋𑌵𑌾𑌇𑌯𑍋𑌯𑌾𑌔𑌹𑍋
 𑌟𑍃 𑌟𑌿 𑌕 𑌥 𑌟𑍍𑌯𑌟 𑌚
𑌵𑌾।𑌊𑌪𑌾।।17।।
 𑌶𑌿 𑌚  𑌶
𑌪𑌾𑌰𑍀𑌸𑍍𑌵𑌾𑌨𑌾𑌸𑌾𑌇𑌨𑍍𑌦𑌵𑌾:।𑌮𑌦𑌾𑌯𑌾𑌬𑌰𑍍𑌹𑌣𑌾
 𑌫𑍍𑌯 𑌣𑍍𑌯𑌫 𑌚 𑌶𑌿 𑌟𑍍𑌯 𑌚 𑌚𑌿
𑌗𑍀𑌰𑌾𑌮𑌧𑍋𑌆𑌰𑍍𑌷𑌾𑌨𑍍।𑌊𑌰𑍍𑌮𑍍𑌮𑌿𑌰𑌿𑌵𑌾𑌈𑌯𑌾।𑌤𑌿𑌧𑌾𑌰𑌾
 𑌕 𑌟𑍀 𑌤 𑌟𑍀 𑌟 𑌤 𑌕𑍍𑌯𑌚 𑌕
𑌯𑌾𑌔𑌹𑍋𑌬𑌾𑌹𑍋𑌇ழா𑌹।।18।।
 𑌟𑍍𑌯 𑌚 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
