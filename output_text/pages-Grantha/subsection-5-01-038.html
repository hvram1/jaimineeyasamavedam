।𑌆𑌗𑍍𑌨𑍇𑌯𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿…

37 ।𑌆𑌗𑍍𑌨𑍇𑌯𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿…

𑌇𑌨𑍍𑌦𑌰𑌯𑍇𑌨𑍍𑌦𑍋𑌹𑌾𑌬𑍁।𑌮𑌾𑌰𑍂𑌤𑍍𑌵𑌤𑍇𑌹𑌾𑌬𑍁।𑌪𑌾𑌵𑌾
  𑌤𑍀 𑌤 𑌶 𑌟𑍀 𑌤 𑌶
𑌸𑍍𑌵𑌮𑌾𑌹𑌾𑌬𑍁।𑌧𑍂𑌮𑌾𑌕𑍍𑌤𑌮𑌾𑌹𑌾𑌬𑍁।𑌆𑌰𑍍𑌕𑌾𑌸𑍍𑌯
 𑌕𑍀 𑌤 𑌶 𑌟𑍀 𑌤 𑌶
𑌯𑍋𑌹𑌾𑌬𑍁।𑌨𑌿𑌮𑌾𑌹𑍋𑌬𑌾𑌸𑌾𑌦𑍋𑌹𑌾𑌇।।49।।
 𑌟𑍀 𑌤 𑌶 𑌟𑍍𑌯 𑌚 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
𑌇𑌨𑍍𑌦𑌰𑌯𑍇𑌨𑍍𑌦𑍋𑌵𑌾𑌓𑌵𑌾।𑌮𑌾𑌰𑍂𑌤𑍍𑌵𑌤𑌾𑌉𑌵𑌾𑌆𑌉𑌵𑌾।
  𑌤𑍀 𑌤𑍍𑌯 𑌤 𑌕𑍍𑌯 𑌕𑍀 𑌕𑌿
𑌪𑌾𑌵𑌾𑌸𑍍𑌵𑌮𑌾𑌉𑌵𑌾𑌆𑌉𑌵𑌾।𑌧𑍂𑌮𑌾𑌕𑍍𑌤𑌮𑌾𑌉𑌵𑌾
 𑌕𑍍𑌯 𑌕𑍀 𑌕𑌿 𑌕𑍍𑌯 𑌕𑍀
𑌆𑌉𑌵𑌾।𑌅𑌰𑍍𑌕𑌾𑌹𑌾𑌬𑍁𑌸𑍍𑌯𑌯𑍋𑌹𑌾𑌬𑍁।𑌨𑍀𑌮𑌾𑌉𑌵𑌾
 𑌚𑌿 𑌶𑍃 𑌕𑍀
𑌹𑍋𑌬𑌾𑌸𑌾𑌦𑍋𑌹𑌾𑌇।।50।।
 𑌪 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
𑌇𑌨𑍍𑌦𑌰𑌾𑌯𑍇𑌨𑍍𑌦𑍋𑌮𑌰𑍂𑌤𑍍𑌵𑌾𑌤𑌾𑌇।𑌪𑌵𑌾𑌸𑍍𑌵𑌮𑌧𑍂𑌮𑌾𑌕𑍍𑌤3
 𑌤𑍃 𑌤 𑌶 𑌕𑍀 𑌶 𑌚 𑌯
𑌮𑌾𑌅𑌰𑍍𑌕𑌸𑍍𑌯𑌯𑍋𑌨𑌿𑌮𑌾𑌹𑌾𑌬𑍁।𑌸𑌾𑌦𑍋𑌹𑌾𑌇।।51।।
 𑌪 𑌶𑍄 𑌪𑍍𑌲 𑌶𑍍𑌯
