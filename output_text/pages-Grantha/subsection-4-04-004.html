। 𑌆𑌷𑍍𑌟𑌾𑌦𑌂𑌷𑍍𑌟𑍇𑌰𑌚𑍇।

4 । 𑌆𑌷𑍍𑌟𑌾𑌦𑌂𑌷𑍍𑌟𑍇𑌰𑌚𑍇।

𑌇𑌨𑍍𑌦𑌂𑌵𑌿𑌶𑍍𑌪𑌾𑌅𑌵𑍀𑌵𑍃𑌧𑌨𑍍𑌨𑌯𑍍𑌯𑌾𑌹𑌾𑌇।𑌸𑌾𑌮𑍂𑌦𑍍𑌰𑌾𑌵𑍍𑌯𑌾𑌚𑌾
 𑌤𑍢 𑌤 𑌶 𑌚𑍍𑌯 𑌯 𑌟𑌚 𑌕
𑌸𑌾𑌂𑌗𑌾𑌇𑌰𑌾𑌅𑌯𑍍𑌯𑌾𑌹𑌾𑌇।𑌰𑌾𑌥𑌾𑌇𑌤𑌾𑌮𑌾𑌂𑌰𑌾𑌥𑌾𑌇
 𑌕 𑌯 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌤 𑌶 𑌚𑍍𑌯 𑌯𑍍𑌯 𑌟𑌚
𑌨𑌾𑌮𑌯𑍍𑌯𑌾𑌹𑌾𑌇।𑌵𑌾𑌜𑌾𑌨𑌾𑌂𑌸𑌾𑌤𑍍𑌪𑌾𑌤𑌾𑌇𑌂𑌪𑌾𑌤𑍀
 𑌕𑍀 𑌟𑍍𑌯 𑌤 𑌶 𑌚𑍍𑌯 𑌯 𑌟𑌚 𑌕𑍍𑌯 𑌟𑍍𑌯 𑌟
𑌮𑌯𑍍𑌯𑌾𑌹𑌾𑌇।𑌓𑌇ழா𑌹।।7।।
 𑌟𑍍𑌯 𑌚𑌣 𑌪 𑌶𑍍𑌯
𑌇𑌨𑍍𑌦𑌰𑌂𑌵𑌿𑌶𑍍𑌪𑌾𑌅𑌵𑍀𑌵𑍃𑌧𑌨𑍍𑌨𑌯𑍍𑌯𑌾𑌦𑍗𑌹𑍋𑌵𑌾।𑌸𑌮𑍁𑌦𑍍𑌰𑌵𑍍𑌯𑌾𑌚
 𑌚𑍋 𑌤𑍍𑌰
𑌸𑌂𑌗𑌾𑌇𑌰𑌾𑌅𑌯𑍍𑌯𑌾𑌦𑍗𑌹𑍋𑌵𑌾।𑌰𑌥𑍀𑌤𑌮𑌾𑌂𑌰𑌾𑌥𑌾𑌇
 𑌕𑍂 𑌟𑌿 𑌟𑍍𑌯 𑌟 𑌤 𑌤 𑌚𑍀 𑌕 𑌕
𑌨𑌾𑌂।𑌪𑍍𑌰।𑌵𑌾𑌜𑌾𑌨𑌾𑌂𑌸𑌾𑌤𑍍𑌪𑌤𑍀𑌂𑌪𑌾𑌤𑍀𑌮𑌯𑍍𑌯𑌦𑍗𑌹𑍋
 𑌟𑍍𑌯 𑌚 𑌥𑍍𑌯 𑌕𑌿 𑌟𑍍𑌯 𑌟𑌿 𑌚
𑌵𑌾𑌃।𑌓𑌇ழா𑌹।।8।।
 𑌚𑌣 𑌪 𑌶𑍍𑌯
