।𑌶𑍍𑌰𑍗𑌤𑌰𑍍𑌷𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿।।

6 ।𑌶𑍍𑌰𑍗𑌤𑌰𑍍𑌷𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿।।

𑌅𑌗𑍍𑌨𑌿𑌰𑍍𑌵𑍃𑌤𑍍𑌰𑌾।𑌣𑌾𑌇𑌜𑌾𑌔𑌹𑍋𑌵𑌾।𑌘𑌾
 𑌤𑍀 𑌟 𑌚𑍍𑌯 𑌶𑌿 𑌚
𑌨𑌾𑌤𑍍।𑌦𑍍𑌰𑌵𑌿𑌣𑌾𑌸𑍍𑌯𑍁𑌰𑍍𑌵𑍀𑌪𑌨𑍍𑌯𑌾𑌯𑌾𑌓𑌇𑌸𑌮𑌿
 𑌶𑍍𑌯 𑌚 𑌶 𑌕 𑌕𑌿 𑌟𑍍𑌯
𑌡𑌾𑌶𑍍𑌶𑍂।𑌕𑍍𑌰𑌾𑌯𑌾𑌹𑍁𑌤𑌾𑌇ழா𑌭𑌾।𑌓𑌇ழா
 𑌟𑍁 𑌤 𑌚𑍍𑌯 𑌟𑍀 𑌚𑌣 𑌪 𑌪𑍍𑌲
𑌅𑌗𑍍𑌨𑍀𑌰𑍗𑌹𑍋𑌵𑌾𑌹𑌾𑌇𑌵𑍃𑌤𑍍𑌰𑌾𑌣𑍀।𑌜𑌂𑌘𑌾
 𑌚𑍍𑌯 𑌶𑌿 𑌶𑍍𑌰𑍁 𑌟𑍍𑌯
𑌨𑌾𑌦𑍗𑌹𑍋𑌵𑌾𑌇।𑌦𑍍𑌰𑌵𑌿𑌣𑌾𑌸𑍍𑌯𑍂:।𑌓𑌇𑌵𑌾𑌇
 𑌤 𑌟𑍍𑌯 𑌤 𑌶 𑌪𑌿 𑌣
𑌪𑌨𑍍𑌨𑌯𑌾𑌯𑌾𑌸𑌾𑌮𑌾𑌇𑌡𑌾𑌶𑍍𑌶𑍂𑌔𑌹𑍋𑌵𑌾।𑌕𑍍𑌰𑌾𑌯𑌾
 𑌟𑍃 𑌟𑍀 𑌚 𑌶𑌿
𑌹𑍂𑌤𑌾:
 𑌟𑌿 𑌚
𑌓𑌗𑍍𑌨𑍀:।𑌵𑍃𑌤𑍍𑌰𑌾𑌣𑍀𑌜𑌂𑌘𑌨𑌾𑌦𑍗𑌹𑍋𑌔𑌹𑍋
 𑌤 𑌤 𑌕𑍃 𑌚𑌿
𑌵𑌾।𑌦𑍍𑌰𑌵𑍀𑌣𑌾𑌸𑍍𑌯𑍁𑌰𑍍𑌵𑍀𑌪𑌨𑍍𑌨𑌯𑌯𑌾𑌔𑌹𑍋𑌔
 𑌶 𑌚𑍍𑌯 𑌕 𑌕𑌿 𑌚𑍍𑌯 𑌕
𑌹𑍋𑌵𑌾।𑌸𑌮𑌿𑌡𑌾𑌶𑍍𑌶𑍁𑌕𑍍𑌰𑌯𑌾𑌔𑌹𑍋𑌔𑌹𑍋
 𑌚𑌿 𑌶 𑌚𑌿 𑌚𑌿 𑌕 𑌚𑌿
𑌬𑌾𑌹𑍂𑌤𑍋𑌹𑌾𑌇
 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶
