।𑌔𑌕𑍍𑌷𑍍𑌣𑍋𑌨𑌿𑌧𑌨𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

34 ।𑌔𑌕𑍍𑌷𑍍𑌣𑍋𑌨𑌿𑌧𑌨𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌮𑍃𑌜𑍍𑌯𑌾𑌏।𑌮𑌾𑌨𑌸𑍍𑌸𑍁𑌹𑌸𑍍𑌤𑌿𑌯𑌾𑌔𑌹𑍋।𑌸𑌾𑌮𑍁𑌦𑍇𑌰𑌵𑌾
 𑌤𑍍𑌯 𑌤 𑌟𑍃 𑌤
𑌚𑌮𑌿𑌨𑍍𑌵𑌸𑌾𑌔𑌹𑍋।𑌰𑌾𑌯𑌿𑌂𑌪𑌿𑌶𑌂𑌗𑌂𑌬𑌹𑍁𑌲𑌂।
 𑌕𑍂 𑌭𑌿 𑌤 𑌕𑍀 𑌕 𑌕𑌿
𑌪𑍂𑌰𑍂𑌸𑍍𑌖𑍃𑌹𑌾𑌔𑌹𑍋।𑌪𑌾𑌵𑌮𑌾𑌨𑌾।𑌭𑍀𑌯𑍋𑌬𑌾
 𑌭𑍁 𑌤 𑌟𑌿 𑌤 𑌕 𑌪 𑌪𑍍𑌲
𑌰𑍍𑌷𑌾𑌸𑍋𑌹𑌾𑌇।।58।।
 𑌪𑍍𑌲 𑌶𑍍𑌯
𑌮𑍃𑌜𑍍𑌯𑌮𑌾𑌨𑌸𑍍𑌸𑍁𑌹𑌸𑍍𑌤𑍇𑌯𑌾𑌵𑌾𑌹𑌾𑌇।𑌸𑌾𑌮𑍁𑌦𑍍𑌰𑌾𑌇𑌵𑌾𑌚𑌮𑌿
 𑌤𑍄 𑌤 𑌶 𑌟𑍃
𑌨𑍍𑌵𑌾𑌸𑍀।𑌰𑌾𑌯𑌾𑌇𑌂𑌪𑌿𑌶𑌾𑌂𑌗𑌂𑌬𑌹𑍁𑌲𑌂।𑌪𑍂𑌰𑍂
 𑌚 𑌣 𑌚𑍍𑌯 𑌤𑌿 𑌕 𑌕𑌿 𑌯𑍍𑌯
𑌸𑍍𑌖𑍍𑌰𑌹𑌾𑌂।𑌪𑌾𑌵𑌾𑌮𑌾𑌨𑌾।𑌭𑍀𑌯𑍋𑌬𑌾𑌰𑍍𑌷𑌾𑌸𑍋
 𑌪𑍍𑌲 𑌚𑍍𑌯 𑌤𑍍𑌯 𑌕 𑌪 𑌪𑍍𑌲 𑌪𑍍𑌲
𑌹𑌾𑌇।।59।।
 𑌶𑍍𑌯
𑌮𑍃𑌜𑍍𑌯𑌮𑌾𑌨𑌾𑌸𑍍𑌸𑍁𑌹𑌸𑍍𑌤𑌿𑌯𑌾।𑌸𑌮𑍂𑌦𑍇𑌰𑌵𑌾𑌚𑌮𑌾
 𑌫𑍀 𑌶𑍍𑌯 𑌕𑍍𑌯 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌟𑍍𑌯
𑌇𑌨𑍍𑌵𑌾𑌸𑍀।𑌰𑌾𑌯𑌿𑌂𑌪𑌿𑌶𑌾𑌂𑌗𑌂𑌬𑌹𑍁𑌲𑌂।𑌪𑍂𑌰𑍂𑌸𑍍𑌖𑍃𑌹𑌾
 𑌟𑌿 𑌕𑍀 𑌕𑍀
𑌔𑌹𑍋।𑌪𑌾𑌵𑌮𑌾𑌨𑌾।𑌭𑍀𑌯𑍋𑌬𑌾𑌰𑍍𑌷𑌾𑌸𑍋
 𑌭𑍁 𑌤 𑌟𑌿 𑌤 𑌕 𑌪 𑌪𑍍𑌲 𑌪𑍍𑌲
