।𑌨𑌕𑍂𑌲𑌸𑍍𑌯 𑌵𑌾𑌮 𑌦𑍇𑌵𑌸𑍍𑌯 𑌪𑍇𑌰𑌂 𑌚𑍗 𑌚𑍗 ।

1 ।𑌨𑌕𑍂𑌲𑌸𑍍𑌯 𑌵𑌾𑌮 𑌦𑍇𑌵𑌸𑍍𑌯 𑌪𑍇𑌰𑌂 𑌚𑍗 𑌚𑍗 ।

𑌪𑍁𑌰𑍋𑌜𑌿𑌤𑍀।𑌵𑍋𑌅𑌨𑍍𑌧𑌾𑌸𑌾𑌃।𑌸𑍂𑌤𑌾𑌯𑌮𑌾𑌦𑌯𑌾।
 𑌤𑍀 𑌚𑌿 𑌣 𑌟𑍀 𑌤𑍍𑌯
𑌆𑌉𑌵𑌾𑌏।𑌪𑍍𑌰।𑌤𑍍𑌨𑌾𑌵𑍇।𑌆𑌪𑌶𑍍𑌪𑌾𑌨𑌂𑌶𑍍𑌨𑌥𑌾।𑌪𑍍𑌰।
 𑌭𑌿 𑌚 𑌶 𑌟𑍀 𑌤𑍍𑌯
𑌷𑍍𑌟𑌾𑌨𑌾।𑌸𑌾𑌚𑌾𑌯𑍋𑌦𑍀𑌰𑍍𑌘𑌜𑌾𑌆𑌉𑌵𑌾।𑌹𑍍𑌵𑍀𑌯𑌂।।1।।
 𑌚 𑌶 𑌟𑍀 𑌤 𑌟𑌿 𑌚 𑌶
𑌪𑍂𑌰𑍋𑌜𑌿𑌤𑍀।𑌵𑍋𑌆𑌨𑍍𑌧𑌾𑌸𑌾𑌃।𑌸𑍂𑌤𑌾𑌯𑌾𑌮𑌾𑌊
 𑌤𑍀 𑌤 𑌪𑍍𑌲 𑌶 𑌕𑍍𑌯𑌚 𑌕 𑌟 𑌟
𑌹𑍋𑌵𑌾𑌊।𑌪𑍍𑌰।𑌦𑌾𑌯𑌾𑌇𑌤𑍍𑌨𑌾𑌵𑍇।𑌆𑌪𑌶𑍍𑌪𑌾𑌨𑌾
 𑌥𑍍𑌯 𑌟 𑌤 𑌪𑌿 𑌶 𑌚 𑌥𑍍𑌯 𑌟
।𑌪𑍍𑌰।𑌶𑍍𑌨𑌾𑌥𑌾𑌇𑌷𑍍𑌟𑌾𑌨𑌾।𑌸𑌾𑌚𑌾𑌯𑍋𑌦𑌾।𑌪𑍍𑌰।
 𑌤 𑌪𑌿 𑌶 𑌚 𑌥𑍍𑌯 𑌟
𑌰𑍍𑌘𑌾𑌜𑌾𑌔𑌹𑍋𑌵𑌾।𑌏𑌹𑍍𑌵𑍀𑌯𑌾𑌂।।2।।
 𑌟𑌟 𑌚 𑌶𑌿 𑌤 𑌨𑍍𑌯
