।𑌶𑍗𑌕𑍍𑌤𑌾𑌨𑌿𑌪𑌞𑍍𑌚 ।

5 ।𑌶𑍗𑌕𑍍𑌤𑌾𑌨𑌿𑌪𑌞𑍍𑌚 ।

𑌸𑌚𑌾𑌯𑌾𑌯𑌾𑌨𑍀𑌷𑍀𑌦𑌤𑌪𑍁𑌨𑌾𑌨𑌾𑌯𑌾𑌪𑍍𑌰𑌾𑌗𑌾𑌯𑌤𑌾।
 𑌕𑍍𑌯 𑌟𑍍𑌯𑌚 𑌟𑍃 𑌤𑌚 𑌚𑍍𑌯 𑌶𑍍𑌯
𑌶𑌿𑌶𑍁𑌨𑍍𑌨𑌾𑌯𑌾𑌜𑍍𑌞𑍈:𑌪𑌾𑌰𑌾𑌇𑌭𑍂𑌷𑌾𑌤𑌾𑌶𑍍𑌰𑌾𑌯𑌾𑌇।
 𑌟𑌿 𑌤 𑌟𑍁𑌚 𑌕 𑌟𑍍𑌯 𑌚𑌣 𑌶
𑌓𑌇ழா𑌹।।11।।
 𑌪 𑌶𑍍𑌯
𑌸𑌾𑌚𑌾𑌯𑌾𑌯𑌾।𑌨𑍀𑌷𑍀𑌦𑌾𑌤𑌾।𑌪𑍂𑌨𑌾𑌨𑌾𑌯𑌾𑌪𑍍𑌰𑌾𑌗𑌾
 𑌤𑍀 𑌚𑌿 𑌣 𑌟𑍍𑌯 𑌚 𑌣 𑌟𑌟 𑌚
𑌔𑌹𑍋𑌵𑌾𑌯𑌾𑌤𑌾।𑌶𑌿𑌶𑍁𑌨𑍍𑌨𑌯𑌾𑌜𑍍𑌞𑍈:𑌪𑌰𑍀𑌭𑍂𑌷𑌾
 𑌶𑌿 𑌚 𑌶 𑌚𑍁 𑌕𑍍𑌯 𑌟𑍍𑌯𑌟
𑌤𑌾𑌔𑌹𑍋𑌵𑌾।𑌶𑍍𑌰𑌿𑌯𑍇।।12।।
 𑌚 𑌶𑌿 𑌤𑍍𑌯𑌚
𑌸𑌚𑌾𑌯𑌯𑌾𑌨𑍀𑌷𑍀𑌦𑌾𑌤𑌾।𑌪𑍁𑌨𑌾𑌨𑌾𑌯𑌪𑍍𑌰𑌗𑌾𑌯𑌤𑌾𑌶𑌿𑌶𑍁𑌨𑍍𑌨𑌾
 𑌟𑍃 𑌤 𑌕𑍍𑌯
𑌯𑌾𑌜𑍍𑌞𑍈:𑌪𑌰𑌾𑌇𑌭𑍂𑌷𑌾𑌤𑌾𑌨𑌾𑌯𑌾𑌇।𑌓𑌇ழா𑌹।।13।।
 𑌟𑍗 𑌟𑌿 𑌤 𑌟 𑌚𑌣 𑌶 𑌪 𑌶𑍍𑌯
𑌓𑌹𑌾𑌇𑌸𑌚𑌾𑌯𑌯𑌾𑌨𑍀𑌷𑍀𑌦𑌾𑌤𑌾।𑌪𑍁𑌨𑌾𑌨𑍗𑌹𑍋𑌇𑌪𑍁
 𑌤 𑌤𑍢 𑌤 𑌕𑍍𑌯
𑌨𑌾𑌨𑍗𑌹𑍋𑌏।𑌯𑌾𑌪𑍍𑌰𑌾𑌯𑌾𑌪𑍍𑌰𑌾।𑌗𑌾𑌯𑌤𑍗𑌹𑍋𑌇𑌗𑌾
 𑌕𑍁 𑌪𑌿 𑌶𑍀 𑌕 𑌶𑍀 𑌕
𑌯𑌤𑍗𑌹𑍋𑌏𑌶𑌾𑌇𑌶𑍂𑌂𑌶𑌾𑌇𑌶𑍂𑌂।𑌨𑌾𑌯𑌜𑍍𑌞𑍌𑌹𑍋
 𑌪𑍀 𑌶𑍂 𑌤𑍀
𑌇𑌨𑌾𑌯𑌜𑍍𑌞𑍗𑌹𑍋𑌏।𑌪𑌾𑌰𑍀।𑌪𑌾𑌰𑌾𑌔𑌹𑍋𑌵𑌾।
 𑌶 𑌪𑍁 𑌤𑍍𑌯 𑌚𑍍𑌯 𑌶𑌿
𑌏𑌭𑍂𑌷𑌤𑌶𑍍𑌰𑌿𑌯𑍇।।14।।
 𑌤𑌚 𑌤𑍁𑌚
𑌸𑌚𑌾𑌯𑌯𑌾𑌉𑌵𑍋𑌵𑌾।𑌪𑍍𑌰।𑌨𑍀𑌷𑌾𑌇𑌦𑌾𑌤𑌾।𑌪𑍍𑌰।
 𑌫𑍍𑌯 𑌚𑍀 𑌶 𑌟𑍁
𑌪𑍁𑌨𑌾𑌨𑌾𑌯𑌪𑍍𑌰𑌗𑌾𑌯𑌤𑌾𑌶𑌾𑌇𑌶𑌾।𑌪𑍍𑌰।𑌨𑌾𑌯𑌜𑍍𑌞𑍈:
 𑌕𑍍𑌯 𑌕𑍀 𑌤𑍍𑌯 𑌟𑌿
𑌪𑌾𑌰𑌿𑌭𑍂𑌷𑌾𑌤𑌾।𑌪𑍍𑌰।𑌶𑍍𑌰𑌿𑌯𑍇।।15।।
 𑌟𑍂 𑌟𑍍𑌯 𑌤𑍍𑌯𑌚
