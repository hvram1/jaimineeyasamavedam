।𑌵𑌾𑌜𑌿𑌜𑌿𑌤𑍗𑌚𑍗 ।

3 ।𑌵𑌾𑌜𑌿𑌜𑌿𑌤𑍗𑌚𑍗 ।

𑌆𑌭𑌿𑌪𑍍𑌰𑌯𑌾𑌣𑌿𑌪𑌾𑌵𑌤𑌾𑌇।𑌹𑍋𑌇𑌹𑍋𑌵𑌾𑌹𑍋𑌏
 𑌕 𑌕𑍀𑌚 𑌶𑍀 𑌕𑍀 𑌪𑍍𑌲
।𑌪𑍍𑌰।𑌚𑌨𑍋𑌹𑌿𑌤𑌾𑌹𑌾𑌇𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌨𑌾𑌮𑌾𑌨𑌿𑌯𑌾
 𑌶𑍇 𑌥𑍍𑌯
𑌹𑍍𑌵𑍋𑌆𑌧𑌿𑌯𑌾𑌇।𑌪𑍍𑌰।𑌷𑍁𑌵𑌡𑌰𑌤𑌾𑌹𑌾𑌇𑌹𑍋𑌵𑌾𑌹𑌾𑌇।
 𑌕𑌿𑌚 𑌶𑍀
𑌆𑌸𑍂𑌰𑌿𑌯𑌾𑌸𑍍𑌯𑌬𑍃𑌹𑌤𑍋।𑌪𑍍𑌰।𑌬𑍃𑌹𑌨𑍍𑌨𑌧𑍀𑌹𑌾𑌇
 𑌥𑍍𑌯 𑌚𑌿 𑌶𑌿
𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌰𑌾𑌥𑌂𑌵𑌿𑌷𑍍𑌵𑌾𑌞𑍍𑌚𑌾𑌮𑌾𑌰𑍂𑌹𑌾𑌡𑍋𑌇
 𑌶𑍈 𑌕𑍍𑌯
𑌹𑍋𑌵𑌾𑌹𑍋𑌏।𑌵𑌿𑌚𑌕𑍍𑌷𑌣𑌾𑌃𑌹𑌾𑌇𑌹𑍋𑌵𑌾
 𑌟𑍃 𑌟𑍍𑌯 𑌪𑍍𑌲
𑌹𑌾𑌉𑌵𑌾।𑌵𑌾𑌜𑍀𑌜𑍀𑌗𑍀𑌵𑌾।।4।।
 𑌶𑌿 𑌕 𑌤𑍀𑌚
𑌆𑌭𑌿𑌪𑍍𑌰𑌿𑌯𑌾𑌣𑍀𑌪𑌵𑌤𑌾𑌇𑌚𑌾𑌨𑍋𑌹𑌿𑌤𑌾।𑌹𑍋𑌵𑌾
 𑌕 𑌕𑍀 𑌕𑍁 𑌶𑌿 𑌕𑍍𑌯
𑌹𑍋𑌇।𑌨𑌾𑌮𑌾𑌨𑌿𑌯𑌾𑌹𑍍𑌵𑍋𑌅𑌧𑌿𑌯𑌾𑌇𑌷𑍂𑌵𑌡𑌰𑌤𑌾𑌇।
 𑌚 𑌶 𑌥𑍍𑌯 𑌕𑌿 𑌕𑍁 𑌶𑍀
𑌹𑍋𑌵𑌾𑌹𑍋𑌇।𑌪𑍍𑌰।𑌆𑌸𑍂𑌰𑌿𑌯𑌾𑌸𑍍𑌯𑌾𑌬𑍃𑌹𑌤𑍋
 𑌕𑍍𑌯 𑌚 𑌶 𑌥𑍍𑌯 𑌕𑌿
𑌬𑍃𑌹𑌨𑍍𑌨𑌧𑍀।𑌪𑍍𑌰।𑌰𑌾𑌥𑌂𑌵𑌿𑌷𑍍𑌵𑌾𑌞𑍍𑌚𑌾𑌮𑌰𑍂𑌹𑌾𑌚𑍀
 𑌕𑍀 𑌶𑌿 𑌕𑍍𑌯 𑌕𑌿 𑌕𑍀
𑌚𑌕𑍍𑌷𑌣𑌾𑌹𑍋𑌵𑌾𑌹𑍋𑌵𑌾𑌔𑌹𑍋𑌵𑌾।𑌵𑌾𑌜𑍀𑌜𑍀
 𑌕𑍀 𑌟𑍍𑌯𑌟 𑌚 𑌶𑌿 𑌕
𑌗𑍀𑌵𑌾𑌵𑌿𑌶𑍍𑌪𑌾𑌧𑌨𑌾𑌨𑍀।।5।।
