। 𑌋𑌕𑍍𑌸𑌾𑌮𑌯𑍋𑌸𑍍𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।

10 । 𑌋𑌕𑍍𑌸𑌾𑌮𑌯𑍋𑌸𑍍𑌸𑌾𑌮𑌨𑍀𑌚𑍇 ।

𑍠𑌚𑌂𑌸𑌾𑌮𑌯𑌜𑌾।𑌮𑌹𑌾𑌇𑌯𑌾𑌭𑍍𑌯𑌾𑌂𑌕𑌰𑍍𑌮𑌾𑌣𑍀𑌕𑍃𑌣𑍍𑌵𑌾𑌤𑌾
𑌤𑍂𑌷𑍍𑌯𑌕𑍁𑌟𑌿𑌤
𑌇।𑌵𑍀𑌤𑍇𑌸𑌦𑌸𑌿𑌰𑌾𑌜𑌾𑌤𑌾𑌃।𑌯𑌜𑍍𑌞𑌾𑌨𑍍𑌦𑌾𑌇𑌵𑍇।𑌷𑍂
𑌶𑌕𑍀𑌟𑌿𑌤𑌟𑌿𑌤𑍍𑌯
𑌵𑌾𑌕𑍍𑌷𑌤𑌾𑌇ழா𑌹𑌭𑌾।𑌓𑌇ழா𑌹।।12।।
𑌕𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯𑌚𑌣𑌪𑌪𑍍𑌲
𑍠𑌚𑌂𑌸𑌾𑌮𑌾𑌯𑌾𑌜𑌾𑌮𑌹𑌾𑌇।𑌯𑌾𑌭𑍍𑌯𑌾𑌂𑌕𑌰𑍍𑌮𑌾𑌣𑍀𑌕𑌾
𑌚𑍀𑌪𑍍𑌲𑌶𑌚𑍍𑌯𑌥𑍍𑌯𑌚𑌕𑍍𑌯
𑌣𑍍𑌵𑌾𑌤𑌾𑌇𑌣𑍍𑌵𑌤𑌾𑌇।𑌵𑌾𑌇𑌤𑍇𑌸𑌦𑌸𑍀𑌰𑌾𑌜𑌾𑌤𑍋
𑌯𑌟𑌟𑌿𑌶𑌕𑍁𑌚𑌕𑍍𑌯𑌯𑌟
𑌜𑌾𑌤𑌾𑌃।𑌯𑌜𑍍𑌞𑌾𑌨𑍍𑌦𑌾𑌇𑌵𑍇𑌷𑍂𑌵𑌾𑌕𑍍𑌷𑌤𑌾𑌇ழா𑌹𑌭𑌾।
𑌟𑍍𑌯𑌕𑍍𑌯𑌯𑌟𑍍𑌯𑌕𑍍𑌯𑌕𑍍𑌯𑌟𑍍𑌯𑌚𑌣
