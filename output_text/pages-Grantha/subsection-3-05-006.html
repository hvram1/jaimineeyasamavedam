।𑌨𑍈𑌪𑌾𑌤𑌿𑌥𑍇𑌚𑍇 𑌵𑌾𑌶𑍇𑌵𑌾 ।

6 ।𑌨𑍈𑌪𑌾𑌤𑌿𑌥𑍇𑌚𑍇 𑌵𑌾𑌶𑍇𑌵𑌾 ।

𑌯𑌦𑌿𑌨𑍍𑌦𑌰𑌪𑍍𑌰𑌾𑌗𑌪𑌾𑌤𑍍।𑌊𑌦𑌾𑌨𑍍𑌯𑌾𑌹𑍂𑌯𑌾𑌸𑍇𑌨𑍍𑌰𑌭𑌾𑌇𑌃।
 𑌤𑍂 𑌕𑍍𑌯 𑌕𑍁 𑌟𑍍𑌯 𑌶
𑌸𑌿𑌮𑌾𑌪𑍂𑌰𑍂𑌨𑍍𑌰𑌷𑍂𑌤𑍋𑌅𑌸𑍍𑌯𑌾𑌨𑌵𑍇।𑌆𑌸𑌿𑌪𑍍𑌰𑌾
 𑌟𑍍𑌯𑌚 𑌚𑍍𑌯 𑌥𑌿 𑌟𑍀 𑌟𑍍𑌯
𑌶𑌾𑌡𑌾𑌤𑍋𑌬𑌾𑌰𑍍𑌵𑌾𑌶𑍋𑌹𑌾𑌇।।7।।
 𑌟𑍍𑌯𑌚 𑌕 𑌪 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
𑌯𑌦𑌿𑌨𑍍𑌦𑌰𑌪𑍍𑌰𑌾𑌗𑌪𑌾𑌗𑍁𑌦𑌾𑌗𑍇।𑌨𑌾𑌯𑌗𑍍𑌵𑌾𑌹𑍂𑌯𑌾𑌸𑌾𑌇𑌨𑍍𑌰
 𑌤𑍃 𑌤 𑌕𑍀𑌚 𑌕
𑌭𑌿𑌰𑍍𑌹𑌾𑌬𑍁𑌹𑍋𑌹𑌾𑌇।𑌪𑍍𑌰।𑌸𑌿𑌮𑌾𑌪𑍂𑌰𑍂𑌨𑍍𑌰𑌷𑍂
 𑌟𑍀 𑌚𑌿 𑌣 𑌶 𑌟𑍍𑌯𑌚 𑌕 𑌚
𑌤𑍋𑌅𑌸𑍍𑌯𑌾𑌨𑌵𑍇।𑌪𑍍𑌰।𑌆𑌸𑌾𑌇𑌪𑍍𑌰𑌾𑌶𑌾।𑌪𑍍𑌰।𑌡𑌾
 𑌥𑌿 𑌟𑍀 𑌟𑍁𑌚 𑌟𑌟
𑌤𑍂𑌔𑌹𑍋𑌵𑌾।𑌰𑍍𑌵𑌾𑌶𑍇।।8।।
 𑌚 𑌶𑌿 𑌚 𑌶
