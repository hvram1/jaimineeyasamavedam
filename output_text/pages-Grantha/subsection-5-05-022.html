।𑌮𑌰𑍂𑌤𑌾𑌂𑌗𑍋𑌷𑍍𑌠𑌾𑌪𑍁𑌂𑌸𑌿𑌨𑌿𑌚𑍇 ।

22 ।𑌮𑌰𑍂𑌤𑌾𑌂𑌗𑍋𑌷𑍍𑌠𑌾𑌪𑍁𑌂𑌸𑌿𑌨𑌿𑌚𑍇 ।

𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌉𑌵𑌾।𑌪𑍍𑌰।𑌪𑌾𑌰𑌿𑌤𑍋𑌷𑌿𑌞𑍍𑌚𑌤𑌾𑌸𑍂
𑌤𑍃𑌕𑍂
𑌤𑌾𑌮𑌿𑌹𑌾𑌉𑌪𑌾।𑌪𑍍𑌰।𑌸𑍋𑌮𑍋𑌯𑌉𑌕𑍍𑌤𑌮𑌂𑌹𑌵𑌿𑌰𑌿𑌹𑌾
𑌚𑍍𑌯𑌨𑍀𑌕𑍂𑌚𑍍𑌯
𑌉𑌪𑌾।𑌪𑍍𑌰।𑌦𑌧𑌾𑌯𑍋𑌨𑌯𑍋𑌰𑌅𑌪𑍍𑌸𑌭𑌰𑌾𑌇𑌹𑌾𑌉𑌪𑌾
𑌨𑍀𑌚𑍇𑌨𑍀
।𑌪𑍍𑌰।𑌸𑍁𑌷𑌾𑌵𑌸𑍋𑌮𑌮𑌾𑌦𑍍𑌰𑍀𑌭𑌿𑌰𑌿𑌹𑌾𑌉𑌪𑌾।।31।।
𑌕𑍂𑌕𑍍𑌯𑌨𑍀
𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁𑌹𑌾𑌉𑌵𑌾।𑌪𑍍𑌰।𑌪𑌾𑌰𑍀𑌤𑍋𑌷𑌿𑌞𑍍𑌚𑌤𑌾
𑌤𑍃𑌕𑌿𑌟𑌿𑌚
𑌸𑍂𑌤𑌂𑌶𑍍𑌰𑌵𑍋𑌬𑍃𑌹𑌦𑍁𑌪𑌾।𑌪𑍍𑌰।𑌸𑍋𑌮𑍋𑌯𑌾𑌉𑌕𑍍𑌤
𑌕𑍍𑌯𑌟𑍍𑌯𑌚𑌨𑍀𑌕𑌿
𑌮𑌂𑌹𑌾𑌵𑌿𑌶𑍍𑌰𑌵𑍋𑌬𑍃𑌹𑌦𑍁𑌪𑌾।𑌪𑍍𑌰।𑌦𑌧𑌨𑍍𑌵𑌾𑌯𑍋𑌨
𑌕𑌿𑌚𑌕𑌟𑌿𑌚𑌨𑍀
𑌯𑍋𑌰𑌅𑌪𑍍𑌸𑌭𑌰𑌾𑌶𑍍𑌰𑌵𑍋𑌬𑍃𑌹𑌦𑍁𑌪𑌾।𑌪𑍍𑌰।𑌸𑍂
𑌕𑍂𑌟𑍍𑌯𑌚𑌕𑌟𑌿𑌚𑌨𑍀
𑌷𑌾𑌵𑌾𑌸𑍋𑌮𑌮𑌾𑌦𑍍𑌰𑍀𑌭𑌿𑌶𑍍𑌰𑌵𑍋𑌬𑍃𑌹𑌦𑍁𑌪𑌾।।32।।
𑌕𑌿𑌟𑌿𑌚𑌕𑌟𑌿𑌚𑌨𑍀
