।𑌭𑌾𑌰𑌤𑍍𑌵𑌾𑌜𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿 𑌕𑌣𑍍𑌵𑌬𑍍𑌰𑌹𑌤𑍍𑌵𑌾𑌚𑌿𑌤𑍀𑌯𑌂 ।

4 ।𑌭𑌾𑌰𑌤𑍍𑌵𑌾𑌜𑌾𑌨𑌿𑌚𑌤𑍍𑌵𑌾𑌰𑌿 𑌕𑌣𑍍𑌵𑌬𑍍𑌰𑌹𑌤𑍍𑌵𑌾𑌚𑌿𑌤𑍀𑌯𑌂 ।

𑌆𑌤𑍍𑌵𑌾𑌸𑌹𑌾।𑌸𑍍𑌰𑌾𑌮𑌾𑌶𑌾𑌤𑌾𑌂।𑌯𑍁𑌕𑍍𑌤𑌾𑌰𑌥𑍇
𑌤𑍀𑌚𑍍𑌯𑌯𑌟𑌚𑍍𑌯𑌕𑍍𑌯𑌚
𑌹𑌿𑌰𑌣𑍍𑌯𑌯𑍇𑌬𑍍𑌰𑌹𑍍𑌮𑌾𑌯𑍂𑌜𑍋।𑌹𑌾𑌰𑌯𑌇𑌨𑍍𑌦𑌰𑌾𑌕𑌾
𑌚𑍀𑌕𑍍𑌯𑌯𑌟𑌕𑌿𑌚𑌕𑍍𑌯
𑌇𑌶𑌾𑌇𑌨𑌾𑌃।𑌵𑌾𑌹𑌾𑌨𑍍𑌤𑍂𑌸𑍋।𑌮𑌾𑌪𑌾𑌔𑌹𑍋𑌵𑌾।

𑌷𑌤𑍁𑌤𑌚𑍍𑌯𑌯𑌪𑌪𑍍𑌲𑌨𑍍𑌧𑌶
।𑌪𑍍𑌰।𑌯𑍁𑌕𑍍𑌤𑌾𑌰𑌥𑍇𑌹𑌿𑌰𑌣𑍍𑌯𑌯𑍇𑌬𑍍𑌰𑌹𑍍𑌮𑌾𑌯𑍂𑌜𑍋
𑌕𑍍𑌯𑌕𑍍𑌯𑌕𑍀𑌚𑌕𑍍𑌯𑌯𑌪
।𑌪𑍍𑌰।𑌹𑌾𑌰𑌯𑌇𑌨𑍍𑌦𑌰𑌾𑌕𑌾𑌇𑌶𑌾𑌇𑌨𑌾𑌃।𑌪𑍍𑌰।𑌵𑌾𑌹𑌾𑌨𑍍𑌤𑍂
𑌕𑍀𑌚𑌕𑍍𑌯𑌯𑍍𑌯𑌪𑌚𑍍𑌯𑌯
𑌸𑍋।𑌪𑍍𑌰।𑌮𑌾𑌪𑍀𑌤𑌾𑌯𑌾𑌉𑌹𑍁𑌵𑌾𑌹𑌾𑌬𑍁।𑌬𑌾।।5।।
𑌪𑌕𑍍𑌯𑌪𑌪𑍍𑌲𑌶𑌶
𑌆𑌤𑍍𑌵𑌾𑌸𑌹𑌸𑍍𑌰𑌮𑌾𑌶𑌤𑌮𑌾।𑌯𑍁𑌕𑍍𑌤𑌾𑌰𑌥𑍇𑌹𑌿𑌰𑌣𑍍𑌯𑌯𑍇𑌬𑍍𑌰𑌹𑍍𑌮
𑌷𑌤𑌿

𑌚𑍀𑌕𑍍𑌯𑌤𑌟𑍍𑌯𑌤𑍍𑌯𑌪𑌪𑍍𑌲𑌚
𑌆𑌤𑍍𑌵𑌾𑌸𑌹𑌸𑍍𑌰𑌾𑌮𑌾𑌶𑌤𑌾𑌮𑌾𑌤𑍍𑌵𑌾𑌸𑌾𑌹𑌾।𑌸𑍍𑌰𑌾𑌮𑌾𑌶𑌤𑌮𑌾
𑌫𑌚𑍍𑌯𑌶𑍀𑌟𑍁

𑌚𑌿𑌣𑌶𑌚𑍍𑌯𑌕𑍍𑌯𑌕𑍀𑌚𑌕𑍍𑌯
