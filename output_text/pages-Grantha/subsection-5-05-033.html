।𑌔𑌕𑍍𑌷𑍍𑌣𑍋𑌰𑌨𑍍𑌧𑍍𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

33 ।𑌔𑌕𑍍𑌷𑍍𑌣𑍋𑌰𑌨𑍍𑌧𑍍𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌮𑍃𑌜𑍍𑌯𑌮𑌾𑌨𑌾𑌃।𑌸𑍁𑌹𑌸𑍍𑌤𑌿𑌯𑌾𑌸𑌾𑌮𑍂𑌦𑍍𑌰𑌾𑌇𑌵𑌾।
 𑌤𑍀 𑌤𑍀 𑌟𑍍𑌯 𑌚 𑌶𑍍𑌯
𑌚𑌮𑌿𑌨𑍍𑌵𑌸𑌾𑌇।𑌰𑌾𑌯𑍀𑌂𑌪𑌾𑌇𑌶𑌾𑌂।𑌗𑌂𑌬𑌹𑍁𑌲𑌾𑌂।
 𑌤𑍀 𑌶 𑌟𑍍𑌯 𑌚 𑌶𑍍𑌯 𑌤𑍀
𑌪𑍂𑌰𑍂𑌸𑍍𑌖𑍃𑌹𑌂।𑌪𑌾𑌵𑌮𑌾𑌨𑌾𑌔𑌹𑍋।𑌭𑍀𑌯𑍋
 𑌟𑍍𑌯 𑌚 𑌣 𑌚𑌿 𑌪𑍍𑌲 𑌶 𑌕 𑌪
𑌬𑌾𑌰𑍍𑌷𑌾𑌸𑍋𑌹𑌾𑌇।।55।।
 𑌪𑍍𑌲 𑌪𑍍𑌲 𑌶𑍍𑌯
𑌮𑍃𑌜𑍍𑌯𑌮𑌾𑌨𑌾𑌃।𑌸𑍂𑌹𑌸𑍍𑌤𑌾𑌯𑌾।𑌸𑌮𑍁𑌦𑍇𑌰𑌵𑌾𑌚𑌾𑌮𑌿𑌨𑍍𑌵𑌾
 𑌤𑍀 𑌚𑌿 𑌟 𑌚𑍍𑌯 𑌥𑍍𑌯 𑌯𑍍𑌯
𑌸𑌾𑌇𑌰𑌯𑌾𑌇𑌂𑌪𑌿𑌶𑌾𑌂𑌗𑌂𑌬𑌹𑍁𑌲𑌂𑌪𑍂𑌰𑍂𑌸𑍍𑌖𑍃𑌹𑌾𑌂।
 𑌪𑍍𑌲 𑌚𑌿 𑌤𑌿 𑌕𑍀 𑌚 𑌯 𑌟𑍍𑌯
𑌪𑌾𑌵𑌾𑌮𑌾𑌨𑌾𑌔𑌹𑍋𑌵𑌾।𑌭𑍍𑌯𑌰𑍍𑌷𑌾𑌸𑍀।।56।।
 𑌟𑌿𑌟 𑌚 𑌶𑌿 𑌟𑍍𑌯 𑌚
𑌮𑍃𑌜𑍍𑌯𑌮𑌾𑌨𑌾𑌃।𑌸𑍂𑌹𑌸𑍍𑌤𑌿𑌯𑌾𑌸𑌾𑌮𑍂𑌦𑍍𑌰𑌾𑌇𑌵𑌾𑌚𑌮𑌿
 𑌤𑍀 𑌚 𑌚𑌿 𑌯 𑌟𑌚 𑌯 𑌟𑍍𑌯 𑌕𑍍𑌯
𑌨𑍍𑌵𑌾𑌸𑌾𑌇।𑌰𑌾𑌇𑌂𑌪𑌾𑌇𑌶𑌾𑌂𑌗𑌂𑌬𑌹𑍁𑌲𑌂।𑌪𑍂𑌰𑍂
 𑌚𑍍𑌯 𑌶 𑌯 𑌟𑌚 𑌯 𑌟𑍍𑌯 𑌕 𑌚𑌿
𑌸𑍍𑌖𑍃𑌹𑌾𑌮𑍍।𑌪𑌾𑌵𑌾𑌮𑌾𑌨𑌾𑌭𑌾𑌯𑌾𑌔𑌹𑍋𑌵𑌾।
  𑌚𑍀 𑌯 𑌟𑌚 𑌯 𑌟 𑌟 𑌚 𑌶𑌿
