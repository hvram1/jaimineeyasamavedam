7 ।𑌦𑌾𑌡𑍍𑌯𑌰𑌚𑍍𑌯𑍁𑌤𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿…

𑌆𑌰𑍍𑌷𑌾𑌇𑌹𑌾𑌸𑍋𑌮𑌾𑌦𑍍𑌯𑍁𑌮𑌾𑌕𑍍𑌤3𑌮𑌾𑌇𑌹𑌾।
 𑌚𑍍𑌯 𑌶𑍍𑌯 𑌟𑍀 𑌚𑍍𑌯 𑌶𑍍𑌯
𑌆𑌭𑌾𑌇𑌇𑌹𑌾।𑌦𑍇𑌰𑌾𑌣𑌾𑌨𑌿𑌰𑍋𑌰𑍂𑌵𑌾𑌦𑌿𑌹𑌾।
 𑌚𑍍𑌯 𑌶 𑌚𑍍𑌯 𑌟𑍀𑌚 𑌕 𑌚 𑌶𑍍𑌯
𑌸𑍀𑌦𑌾𑌨𑌿𑌹𑌾।𑌯𑍋𑌨𑍗𑌵𑌨𑌾𑌇।𑌷𑍂𑌵𑌾𑌇𑌹𑌾।।10।।
 𑌕 𑌚 𑌶𑍍𑌯 𑌟𑍀𑌚 𑌶 𑌚𑍍𑌯 𑌶𑍍𑌯
𑌆𑌰𑍍𑌷𑌾𑌹𑌾𑌬𑍁।𑌸𑍋𑌮𑌦𑍍𑌯𑍁𑌮𑌕𑍍𑌤3𑌮𑍋𑌅𑌭𑌿𑌦𑍇𑌰𑌾𑌣𑌾।
 𑌕𑍍𑌯 𑌤 𑌶 𑌟𑍢 𑌤
𑌨𑌿𑌰𑍋𑌔𑌹𑍋।𑌰𑍂𑌵𑌾𑌤𑍍𑌸𑍀𑌦𑌾𑌉𑌵𑌾𑌯𑍋𑌨𑍗𑌵𑌾𑌨𑍇
 𑌭𑌿 𑌤 𑌟𑍍𑌯 𑌤𑍀 𑌪𑌿 𑌶
𑌹𑌿𑌮𑍍𑌮𑌾𑌏।𑌷𑍂𑌵𑌾।।11।।
 𑌭𑌿 𑌚
𑌅𑌰𑍍𑌷𑌾𑌸𑍋𑌮𑌾𑌦𑍍𑌯𑍁𑌮𑌾𑌕𑍍𑌤𑌮𑍋𑌅𑌰𑍍𑌷𑌾𑌸𑍋𑌮𑌾।𑌦𑍍𑌯𑍁𑌮𑌕𑍍𑌤
 𑌫𑍂 𑌚𑍍𑌯 𑌶𑍀
𑌮𑍋𑌅𑌭𑌿𑌦𑍇𑌰𑌾𑌣𑌾𑌹𑌾𑌹𑌾𑌇।𑌨𑍀𑌰𑍋𑌰𑍂𑌵𑌤𑍍𑌸𑌾
