13 ।𑌐𑌷𑌿𑌰𑍇𑌚𑍇 ।

𑌤𑌰𑌣𑌿𑌰𑍀𑌤𑍍।𑌸𑌿𑌷𑌾𑌸𑌾𑌤𑍀।𑌵𑌾𑌜𑌾𑌂𑌪𑍁𑌰𑌾𑌨𑍍𑌧𑌿𑌯𑌾𑌯𑍁
𑌤𑍀𑌚𑌿𑌣𑌕𑌚𑌚𑍍𑌯𑌤𑍍𑌯
𑌜𑌾।𑌆𑌵𑌾𑌇𑌨𑍍𑌦𑌰𑌂।𑌪𑍁𑌰𑍂𑌹𑍂𑌤𑌾𑌨𑍍𑌨𑌮𑍇𑌗𑌾𑌇𑌰𑌾𑌃।
𑌟𑍍𑌯𑌟𑍍𑌯𑌚𑌣𑌚𑍀𑌕𑍍𑌯𑌟𑌿
𑌨𑌾𑌇𑌮𑌿𑌨𑍍𑌤𑌾𑌷𑍍𑌟𑍇।𑌵𑌾𑌸𑍁𑌦𑍍𑌰𑍂𑌵𑌾𑌂।।20।।
𑌟𑌿𑌚𑌣𑌕𑌨𑌿
𑌤𑌾𑌰𑌾𑌹𑌾𑌬𑍁।𑌣𑍀𑌰𑌿𑌤𑍍𑌸𑍀𑌷𑌾𑌸𑌤𑍀𑌹𑌾𑌯𑌾𑌇𑌹𑌾𑌯𑌾𑌇।
𑌤𑍍𑌯𑌤𑌶𑌚𑍀𑌕𑍂𑌤𑌿𑌶
𑌵𑌾𑌜𑌂𑌪𑍁𑌰𑌨𑍍𑌧𑌿𑌯𑌾𑌯𑍁𑌜𑍗।𑌹𑍋𑌵𑌾𑌹𑌾𑌇।𑌪𑍍𑌰।
𑌚𑍍𑌯𑌕𑍍𑌯𑌚𑌚𑍍𑌯𑌯𑌟𑌤𑍍𑌯𑌤𑌶
𑌆𑌵𑌇𑌨𑍍𑌦𑌰𑌂𑌪𑍁𑌰𑍁𑌹𑍂𑌤𑌨𑍍𑌨𑌾𑌮𑌾𑌇𑌗𑌾𑌇𑌰𑍗।𑌪𑍍𑌰।𑌨𑍇
𑌕𑍂𑌥𑍍𑌯𑌕𑍍𑌯𑌯𑍍𑌯𑌟𑍍𑌯
𑌮𑌿𑌨𑍍𑌤𑌷𑍍𑌟𑍇𑌵𑌾𑌸𑌾𑌔𑌹𑍋𑌵𑌾𑌹𑌾𑌔𑌹𑍋𑌵𑌾।𑌊
𑌚𑍍𑌯𑌕𑍍𑌯𑌚𑌚𑍍𑌯𑌕𑌚𑌿𑌶𑌿
𑌦𑍍𑌰𑍂𑌵𑌂।।21।।
𑌚𑍍𑌯𑌶
