।𑌶𑌯𑍍𑌯𑌾𑌤𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

9 ।𑌶𑌯𑍍𑌯𑌾𑌤𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌹𑌾𑌬𑍁𑌤𑍍𑌵𑌮𑌿𑌨𑍍𑌦𑌰𑌾।𑌬𑌲𑌾𑌦𑌧𑌿𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁।
𑌤𑍁𑌕𑍁𑌤𑍍𑌯𑌶
𑌸𑌹𑌾𑌸𑌾𑌜𑌾𑌤𑌾𑌓𑌜𑌾𑌸𑌾𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁।
𑌕𑍍𑌯𑌥𑍍𑌯𑌟𑌭𑌿𑌤𑌿𑌶
𑌤𑍍𑌵𑌾𑌂𑌸𑌾𑌨𑍍𑌨𑍍𑌵𑌾𑌰𑍍𑌷𑌾𑌨𑍍𑌹𑌾𑌬𑍁𑌹𑌾𑌬𑍁।𑌵𑍃𑌷𑌾𑌇𑌦𑌾
𑌚𑌯𑌟𑍍𑌯𑌤𑌿𑌶𑌟𑌿𑌟
𑌸𑌾𑌔𑌹𑍋𑌵𑌾।𑌵𑍃𑌧𑍇।।14।।
𑌚𑌶𑌿𑌤𑍍𑌯𑌵
𑌤𑍍𑌵𑌮𑌿𑌨𑍍𑌦𑌰𑌬𑌲𑌾𑌦𑌧𑍀𑌏।𑌸𑌹𑌾𑌸𑌾𑌜𑌾𑌤𑌾𑌓𑌜𑌸𑌾।
𑌤𑍃𑌤𑌕𑍍𑌯𑌥𑍍𑌯𑌟𑌭𑌿
𑌇𑌯𑌾𑌹𑍋𑌇𑌯𑌾𑌇𑌯𑍇𑌯𑌾𑌵𑌾।𑌪𑍍𑌰।𑌤𑍍𑌵𑌾𑌂𑌸𑌾𑌨𑍍𑌵𑌾𑌰𑍍𑌷𑌾𑌨𑍍
𑌕𑍍𑌯𑌕𑌿𑌚𑍍𑌯𑌶𑌚𑌯𑌟𑍍𑌯
।𑌪𑍍𑌰।𑌵𑍃𑌷𑌾𑌇𑌦𑌾𑌸𑌾𑌔𑌹𑍋𑌵𑌾।𑌮𑌹𑍇।।15।।
𑌟𑍀𑌟𑌚𑌶𑌿𑌤𑍍𑌯𑌚
𑌤𑍍𑌵𑌮𑌿𑌨𑍍𑌦𑌰𑌬𑌲𑌾𑌦𑌧𑍀𑌸𑌹𑌸𑌾।𑌜𑌾𑌤𑌾𑌓𑌜𑌾𑌸𑌾𑌇𑌹𑍋
𑌤𑍢𑌕𑍍𑌯𑌕𑍍𑌯𑌚
𑌇𑌹𑌾।𑌤𑍍𑌵𑌾𑌂𑌸𑌾𑌨𑍍𑌵𑌾𑌰𑍍𑌷𑌾𑌨𑌿𑌹𑍋𑌇𑌹𑌾।𑌵𑍃𑌷𑌾𑌇
𑌶𑍀𑌕𑍍𑌯𑌚𑌕𑌚𑌶𑍀
𑌦𑌾𑌸𑌾𑌔𑌹𑍋𑌵𑌾।𑌈𑌹𑌾।।16।।
𑌟𑍀𑌟𑌚𑌶𑌿𑌚𑌶
