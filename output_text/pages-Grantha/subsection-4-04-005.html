। 𑌮𑌹𑌾𑌵𑍈𑌶𑍍𑌪𑌾𑌮𑌿𑌤𑍇𑌰𑌚𑍇।

5 । 𑌮𑌹𑌾𑌵𑍈𑌶𑍍𑌪𑌾𑌮𑌿𑌤𑍇𑌰𑌚𑍇।

𑌹𑌯𑌾𑌇𑌹𑌯𑌾𑌓𑌹𑌾𑌓𑌹𑌾।𑌤𑍍𑌰𑍀𑌃।𑌇𑌨𑍍𑌦𑌂𑌵𑌿𑌶𑍍𑌪𑌾।𑌅𑌵𑍀𑌵𑌾
 𑌚𑍁 𑌫 𑌫 𑌤𑍀 𑌚𑍍𑌯
𑌰𑍍𑌧𑌾𑌨𑍍।𑌸𑌮𑍁𑌦𑍍𑌰𑌵𑍍𑌯𑌾𑌚𑌸𑌂𑌗𑌾𑌇𑌰𑌾𑌃।𑌰𑌥𑍀𑌤𑌮𑌾𑌂𑌰𑌾𑌥𑌾
 𑌟𑍍𑌯 𑌤𑍀 𑌚𑍍𑌯 𑌟𑌿 𑌤𑍀 𑌕
𑌇𑌨𑌾𑌂।𑌵𑌾𑌜𑌾𑌨𑌾𑌂𑌸𑌾𑌤𑍍𑌪𑌾𑌤𑍀𑌂𑌪𑌾𑌤𑍀𑌂।𑌹𑌯𑌾𑌇𑌹𑌯𑌾
 𑌟𑍍𑌯 𑌤𑍀 𑌚𑍍𑌯 𑌟𑍍𑌯 𑌚𑍁
𑌓𑌹𑌾𑌓𑌹𑌾।𑌤𑍍𑌰𑌿।𑌹𑍋𑌇ழா𑌹𑌹𑍋𑌇ழா𑌹𑌹𑍋𑌇ழா𑌹।।9।।
 𑌫 𑌫 𑌫𑌿 𑌫𑌿𑌪𑍍𑌲 𑌚 𑌶𑍍𑌯
𑌹𑌾𑌯𑌾𑌯𑍇𑌹𑌾𑌯𑌾𑌯𑍇𑌹𑌯𑌾𑌹𑌾𑌆।𑌇𑌨𑍍𑌦𑌰𑌂𑌵𑌿𑌶𑍍𑌪𑌾
 𑌟𑌿𑌚 𑌕 𑌕𑍍𑌯𑌚 𑌤𑍍𑌯 𑌤 𑌪 𑌕𑍀𑌚
𑌅𑌵𑍀𑌵𑍃𑌧𑌾𑌨𑍍।𑌸𑌾𑌮𑍁𑌦𑍍𑌰𑌾𑌵𑍍𑌯𑌾𑌚𑌸𑌂𑌗𑌿𑌰𑌾𑌃।𑌰𑌾
 𑌫 𑌤𑍍𑌯 𑌪 𑌣𑍁𑌫 𑌤𑍍𑌯 𑌪
𑌥𑍀𑌤𑌾𑌮𑌂𑌰𑌥𑍀𑌨𑌾𑌂।𑌵𑌾𑌜𑌾𑌨𑌾𑌂𑌸𑌾𑌤𑍍𑌪𑌾𑌤𑍀𑌂𑌪𑌤𑌾𑌇।
 𑌣𑍁𑌫 𑌤 𑌪 𑌣𑍁𑌫 𑌤𑍍𑌯
𑌹𑌾𑌯𑌾𑌯𑍇𑌹𑌾𑌯𑌾𑌯𑍇𑌹𑌯𑌾𑌹𑌾𑌆।𑌹𑍋𑌇ழா𑌹
 𑌟𑌿𑌚 𑌕 𑌟𑍍𑌯 𑌤𑍍𑌯 𑌤 𑌪 𑌫𑌿
𑌹𑍋𑌇ழா𑌹𑌹𑍋𑌇ழா𑌹
 𑌫𑌿𑌪𑍍𑌲 𑌚 𑌶𑍍𑌯
