2 ।𑌇𑌨𑍍𑌦𑌰𑌸𑍍𑌯𑌵𑍃𑌷𑌕𑌾𑌣𑌿𑌤𑍍𑌰𑌿𑌣𑌿…

𑌪𑌵𑌸𑍍𑌵𑍇𑌨𑍍𑌦𑍋𑌵𑍃𑌷𑌾𑌸𑍁𑌤𑌾:।𑌕𑍃𑌧𑍀𑌨𑍋𑌯𑌶𑌸𑍋𑌜𑌨𑌾
𑌫𑍀𑌶𑍍𑌯𑌕𑍍𑌯𑌷𑌿
𑌏𑌵𑌾𑌇𑌶𑍍𑌪𑌾𑌯𑌾𑌪𑌾।𑌚𑌾𑌇𑌷𑌾𑌔𑌹𑍋𑌵𑌾।𑌜𑌾𑌹𑌿।।3।।
𑌪𑍂𑌶𑍁𑌟𑌚𑍍𑌯𑌶𑌿𑌚𑌶
𑌵𑍃𑌷𑌾𑌹𑌿𑌯𑌾।𑌸𑌿𑌭𑌾𑌨𑍂𑌨𑌾।𑌦𑍍𑌯𑍁𑌮𑌭𑌢3𑌹𑌵𑌾𑌮𑌾𑌹𑌾𑌇
𑌤𑍀𑌚𑌿𑌤𑌯𑍂𑌪𑍍𑌲
𑌪𑌾𑌵𑌾।𑌮𑌾𑌨𑌾𑌸𑍁𑌵𑍋𑌓𑌬𑌾।𑌦𑍃𑌶𑌂।।4।।
𑌤𑌿𑌟𑍀𑌪𑌶𑌚𑌶
𑌵𑍃𑌷𑌾𑌹𑌿𑌯𑌾𑌸𑌿𑌭𑌾𑌨𑍁𑌨𑌾।𑌦𑍍𑌯𑍁𑌮𑌭𑌂𑌢3𑌹𑌵𑌾𑌮𑌹𑍇।
𑌫𑍀𑌕𑍀𑌟𑍀𑌕𑍀
𑌹𑍋𑌵𑌾𑌔𑌹𑍋𑌵𑌾।𑌪𑍍𑌰।𑌪𑌾𑌵𑌮𑌾𑌨𑌾𑌸𑍁𑌵𑌦𑍃𑌶𑌂
𑌕𑍍𑌯𑌟𑌕𑍍𑌯𑌟𑌿𑌚𑌟𑍀
।𑌪𑍍𑌰।𑌹𑍂𑌵𑍋𑌇ழா𑌹।।5।।
