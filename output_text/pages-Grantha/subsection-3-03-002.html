।𑌵𑌾𑌮𑍍𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

2 ।𑌵𑌾𑌮𑍍𑌰𑌾𑌣𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌯𑌾𑌯𑌾𑌹𑍋𑌇𑌨𑍍𑌦𑌰𑌾।𑌭𑍂𑌜𑌾𑌆𑌭𑌾𑌰𑌾।𑌸𑍁𑌵𑌾
𑌤𑍍𑌯𑌚𑍍𑌯𑌶𑌚𑌕𑍍𑌯𑌯𑌟𑌤𑍍𑌯
𑌹𑌾𑌰𑍍𑌵𑌂𑌯𑌾𑌸𑍂𑌰𑌾𑌇𑌭𑌾𑌯𑌾𑌃।𑌸𑍍𑌤𑍋𑌤𑌾𑌹𑌾𑌰𑌾
𑌚𑌿𑌕𑍍𑌯𑌯𑍍𑌯𑌟𑌟𑍍𑌯𑌤𑌚𑌕
𑌮𑌿𑌨𑍍𑌮𑌘𑌵𑌨𑍍𑌨𑌸𑍍𑌯𑌾𑌵𑌾𑌰𑍍𑌧𑌾𑌯𑌾।𑌯𑌾𑌇𑌚𑌾𑌹𑌾𑌇।
𑌚𑍂𑌯𑌟𑍍𑌯𑌟𑌿𑌤𑌶
𑌤𑍍𑌵𑍇𑌵𑍃𑌕𑍍𑌤𑌾𑌬𑌰𑍍𑌹𑌾𑌇𑌷𑌾𑌃।𑌓𑌇ழா𑌹।।4।।
𑌚𑍍𑌯𑌟𑌿𑌚𑍍𑌯𑌣𑌪𑌶𑍍𑌯
𑌯𑌾𑌇𑌨𑍍𑌦𑌰𑌭𑍁𑌜𑌾𑌆𑌭𑌾𑌰𑌾𑌃।𑌸𑍍𑌵𑌰𑍍𑌵𑌂𑌆𑌸𑍂𑌰𑍇𑌭𑌾𑌯𑌾।
𑌟𑍂𑌤𑍍𑌰𑌤𑌥𑍍𑌯𑌚𑌟𑌟𑍍𑌯𑌚𑌶
𑌹𑌾𑌹𑍋𑌹𑌾𑌇।𑌪𑍍𑌰।𑌸𑍍𑌤𑍋𑌤𑌾𑌰𑌾𑌮𑌾𑌇𑌨𑍍𑌮𑌾𑌘𑌵
𑌚𑍍𑌯𑌣𑌶𑌥𑍍𑌯𑌚𑌚𑌿𑌕𑍍𑌯
𑌨𑍍𑌨𑌾𑌸𑍍𑌯𑌵𑌾𑌰𑍍𑌧𑌾𑌯𑌾।𑌪𑍍𑌰।𑌯𑍇𑌚𑌤𑍍𑌵𑌾𑌇𑌵𑍃।𑌪𑍍𑌰।
𑌟𑌿𑌚𑌶𑌕𑌕𑍍𑌯𑌤𑍍𑌯
𑌕𑍍𑌤𑌾𑌬𑌰𑍍𑌹𑌾𑌇𑌷𑌾𑌃।𑌓𑌇ழா𑌹।।5।।
𑌟𑌿𑌚𑍍𑌯𑌣𑌪𑌶𑍍𑌯
𑌯𑌾𑌇𑌨𑍍𑌦𑌰𑌭𑍁𑌜𑌾𑌆𑌭𑌰𑌉𑌹𑍁𑌵𑌾𑌹𑌾𑌇।𑌸𑍍𑌵𑌰𑍍𑌵𑌂𑌆𑌸𑍁
𑌤𑍈𑌤𑌶𑌥𑍍𑌯𑌚𑌟
𑌰𑍇𑌭𑍍𑌯𑌾𑌉𑌹𑍁𑌵𑌾𑌹𑌾𑌇।𑌪𑍍𑌰।𑌸𑍍𑌤𑍋𑌤𑌾𑌰𑌮𑌿𑌨𑍍𑌮𑌘𑌵𑌨𑍍𑌨𑌸𑍍𑌯
𑌟𑍍𑌯𑌚𑌟𑌿𑌤𑌶𑌕
𑌵𑌾𑌰𑍍𑌧𑌾𑌯𑌾।𑌪𑍍𑌰।𑌯𑍇𑌚𑌤𑍍𑌵𑌾𑌇𑌵𑍃।𑌪𑍍𑌰।𑌕𑍍𑌤𑌾𑌬
𑌯𑍢𑌟𑍍𑌯𑌚𑌚𑌟𑍍𑌯𑌤𑍍𑌯𑌚𑌚
𑌰𑍍𑌹𑌾𑌇𑌷𑌾𑌃।𑌓𑌇ழா𑌹।।6।।
𑌟𑍍𑌯𑌚𑍍𑌯𑌣𑌪𑌶𑍍𑌯
