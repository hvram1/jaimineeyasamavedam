।𑌕𑍗𑌨𑍍𑌮𑍁𑌦𑍇𑌚𑍇𑌸𑍍𑌵𑌜𑍋𑌰𑌤𑌿𑌷𑍀𑌵𑌾 ।

7 ।𑌕𑍗𑌨𑍍𑌮𑍁𑌦𑍇𑌚𑍇𑌸𑍍𑌵𑌜𑍋𑌰𑌤𑌿𑌷𑍀𑌵𑌾 ।

𑌕𑌸𑍍𑌤𑌮𑌿𑌨𑍍𑌦𑌰𑌾।𑌤𑍁𑌵𑌾𑌵𑌾𑌸𑌾𑌬𑍁।𑌆𑌮𑌕𑍍𑌤𑍇𑌯𑌾𑌦𑌧𑌰𑍍𑌷𑌤𑌾
𑌤𑍀𑌟𑍍𑌯𑌟𑍍𑌯𑌶𑌚𑍃
𑌇।𑌶𑍍𑌰𑌡𑌾𑌹𑌾𑌇𑌤𑍇।𑌮𑌾𑌘𑌵𑌨𑍍𑌪𑌾𑌰𑍍𑌯𑌾𑌇𑌦𑌾𑌇𑌵𑍀।𑌵𑌾
𑌶𑌥𑍍𑌯𑌟𑌿𑌕𑌿𑌯𑍀𑌟𑍍𑌯
𑌜𑍀𑌵𑌾𑌜𑌾𑌂𑌸𑍀𑌷𑌾𑌸𑌾𑌤𑌾𑌔𑌹𑍋𑌵𑌾।𑌉𑌊𑌪𑌾
𑌟𑍍𑌯𑌟𑍍𑌯𑌚𑌕𑌟𑍍𑌯𑌟𑌚𑌶𑌿𑌚𑍍𑌯𑌶
𑌕𑌸𑍍𑌤𑌮𑌿𑌨𑍍𑌦𑌰𑌾𑌤𑍍𑌵𑌾𑌵𑌸𑌾𑌬𑍂𑌆𑌮𑌕𑍍𑌤𑍇𑌰𑍍𑌯𑌾𑌦𑌧𑌰𑍍𑌷𑌤𑌾𑌇।
𑌫𑍁𑌤𑍍𑌯𑌪𑍍𑌲𑌶𑍃
