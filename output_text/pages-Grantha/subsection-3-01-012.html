11 ।𑌧𑌾𑌨𑌾𑌕𑍇𑌚𑍇।

𑌤𑌰𑍋𑌭𑌿𑌵𑍋𑌰𑌵𑌿𑌦𑌚𑌾𑌸𑍂𑌂।𑌇𑌨𑍍𑌦𑌰𑌾𑌮𑌿𑌨𑍍𑌦𑌰𑌂𑌸𑌬𑌾𑌧𑌾
 𑌤𑍃 𑌤 𑌤𑍍𑌯 𑌚𑍍𑌯
𑌊𑌤𑌾𑌯𑌾𑌇।𑌬𑍍𑌰𑌹𑌾𑌤𑍍𑌬𑍍𑌰𑌹𑌤𑍍𑌗𑌾𑌯𑌨𑍍𑌤𑌸𑍍𑌸𑍂𑌤𑌸𑍋𑌮𑌾
𑌯𑌾𑌧𑍍𑌵𑌾𑌰𑌾𑌇।𑌹𑍂𑌵𑌾𑌇।𑌹𑍂𑌵𑍇𑌭𑌰𑌨𑍍𑌨𑌾𑌕𑌾𑌰𑌿𑌣𑌾𑌮𑌿
 𑌯 𑌟𑍍𑌯 𑌶 𑌚𑍍𑌯 𑌶 𑌚𑍂
ழா𑌹𑌭𑌾।𑌓𑌇ழா𑌹।।15।।
 𑌟𑍀 𑌚𑌣 𑌪 𑌪𑍍𑌲
𑌤𑌰𑍋𑌭𑌿𑌵𑍋𑌰𑌵𑌿𑌦𑌾𑌓𑌚𑌾𑌸𑍂𑌂।𑌇𑌨𑍍𑌦𑌰𑌂𑌸𑌬𑌾𑌧𑌊𑌤𑌾
 𑌤𑍃 𑌤 𑌚𑍀 𑌟𑍍𑌯 𑌚
𑌯𑌾𑌇।𑌬𑍍𑌰𑌹𑌾𑌤𑍍𑌬𑍍𑌰𑌹𑌾𑌆𑌉𑌵𑌾।𑌗𑌾𑌯𑌨𑍍𑌤𑌾𑌸𑍍𑌸𑍂𑌤𑌾
 𑌣 𑌶 𑌚𑍍𑌯 𑌤𑍍𑌯 𑌟𑌿 𑌚𑌿
𑌸𑍋𑌮𑍇𑌆𑌧𑍍𑌵𑌾𑌰𑍇𑌹𑍂𑌵𑍇𑌹𑍋𑌇𑌭𑌾𑌰𑌾𑌂𑌨𑌾𑌕𑌾
 𑌚𑌿 𑌚𑍍𑌯 𑌚𑍍𑌯 𑌕𑍍𑌯 𑌟𑌿 𑌤 𑌚𑍍𑌯
𑌰𑌿𑌣𑌾𑌮𑌿ழா𑌹𑌭𑌾।𑌓𑌇ழா𑌹।।16।।
 𑌟𑍀 𑌚𑌣 𑌪 𑌪𑍍𑌲
