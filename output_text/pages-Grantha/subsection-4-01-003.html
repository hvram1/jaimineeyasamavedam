।𑌵𑌸𑌿𑌷𑍍𑌠𑌸𑍍𑌯𑌚𑌨𑌿𑌨𑍍𑌹𑌵𑌂 ।

2 ।𑌵𑌸𑌿𑌷𑍍𑌠𑌸𑍍𑌯𑌚𑌨𑌿𑌨𑍍𑌹𑌵𑌂 ।

𑌆𑌇𑌹𑌿𑌆𑌇𑌹𑌿𑌏𑌹𑌿𑌯𑌾𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌇।
 𑌟𑌿 𑌟𑌿 𑌚𑍁 𑌪𑍍𑌲 𑌤 𑌶
𑌅𑌸𑌾𑌵𑌿𑌦𑍇𑌵𑌂𑌗𑍋𑌋𑌜𑍀𑌕𑌾𑌮𑌨𑍍𑌧𑍋𑌅𑌨𑍍𑌧𑍋𑌅𑌨𑍍𑌧𑌾
 𑌚𑍂 𑌕𑍍𑌯𑌚 𑌕 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌟𑍍𑌯
𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌇।𑌪𑍍𑌰।𑌨𑍍𑌯𑌸𑍍𑌮𑌿𑌨𑍍𑌨𑌿𑌨𑍍𑌦𑍇𑌰𑌾𑌜𑌨𑍁𑌷𑍇
 𑌚𑍍𑌯 𑌪𑍍𑌲 𑌤 𑌶 𑌚𑍀 𑌕𑌿𑌚
𑌮𑍂𑌵𑍋𑌚𑌾𑌵𑍋𑌚𑌾𑌵𑍋𑌚𑌾𑌉𑌵𑍋𑌵𑌾𑌹𑌾𑌇।
 𑌕 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌚𑍍𑌯 𑌪𑍍𑌲 𑌤 𑌶
𑌬𑍋𑌧𑌾𑌮𑌸𑌿𑌤𑍍𑌵𑌾𑌹𑌰𑌿𑌯𑌾𑌶𑍍𑌪𑌾𑌯𑌾𑌜𑍍𑌤𑍈𑌰𑍍𑌯𑌾𑌜𑍍𑌤𑍈𑌰𑍍𑌯𑌜𑍍𑌤𑌾।
 𑌚𑍁 𑌕𑌿𑌚 𑌕 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌟𑍍𑌯
।𑌪𑍍𑌰।𑌬𑍋𑌧𑌾𑌨𑌸𑍍𑌤𑍋𑌮𑌾𑌮𑌨𑍍𑌧𑌾𑌸𑌾𑌮𑌾𑌦𑌾𑌇𑌷𑍂
 𑌚𑍁 𑌕𑌿𑌚 𑌕 𑌟𑌿
𑌦𑌾𑌇𑌷𑍂𑌦𑍇𑌷𑍍𑌵𑌾।𑌪𑍍𑌰।𑌆𑌇𑌹𑌿𑌆𑌇𑌹𑌿𑌏𑌹𑌿𑌯𑌾𑌉
 𑌟𑌿 𑌟𑍍𑌯 𑌟𑌿 𑌟𑌿
𑌵𑍋𑌵𑌾𑌹𑌾𑌔𑌹𑍋𑌵𑌾।𑌈।।।2।।
 𑌚𑍁 𑌪𑍍𑌲 𑌚 𑌶𑌿 𑌚
