।𑌵𑌷𑌟𑍍𑌕𑌾𑌰𑌨𑌿𑌧𑌨𑌨𑍍𑌜 𑌸𑌾𑌕𑌮𑌶𑍍𑌪𑌂 ।

4 ।𑌵𑌷𑌟𑍍𑌕𑌾𑌰𑌨𑌿𑌧𑌨𑌨𑍍𑌜 𑌸𑌾𑌕𑌮𑌶𑍍𑌪𑌂 ।

𑌅𑌭𑌿𑌤𑍍𑌵𑌾𑌪𑍂𑌰𑍍𑌵𑌪𑍀𑌤𑌯𑍇𑌆𑌭𑍀𑌤𑍍𑌵𑌾𑌪𑍂𑌰𑍍𑌵𑌾𑌪𑍀𑌤𑌯𑌾𑌇।
𑌫𑍇𑌚𑍍𑌯𑌪𑍍𑌲𑌶
𑌇𑌨𑍍𑌦𑌰𑌾𑌸𑍍𑌤𑍋𑌮𑍈𑌭𑍀𑌰𑌾𑌯𑌵𑍋𑌭𑍀𑌰𑌾𑌯𑌾𑌵𑌾𑌓𑌮𑍋
𑌥𑌿𑌚𑌕𑍍𑌯𑌚𑍍𑌯𑌕𑍍𑌯𑌯𑌟
𑌵𑌾।𑌸𑌮𑍀𑌚𑍀𑌨𑌾𑌸𑌾𑍠𑌭𑌾𑌵𑌾𑌸𑌾𑌮𑌾𑌸𑍍𑌵𑌰𑌾𑌨𑍍𑌸𑌾𑌮𑌾𑌸𑍍𑌵𑌾
𑌕𑌿𑌚𑍢𑌕𑍍𑌯𑌟𑍍𑌯𑌚𑌚𑍍𑌯𑌯
𑌰𑌾𑌨𑍋𑌮𑍋𑌵𑌾।𑌰𑍁𑌦𑍍𑌰𑌾𑌗𑍃𑌣𑌨𑍍𑌤𑌾𑌪𑍂𑌰𑍍𑌵𑌾𑌯𑌾𑌨𑍍𑌤𑌾
𑌟𑌕𑌿𑌚𑍍𑌯𑌕𑍍𑌯𑌚𑌕𑍍𑌯𑌟𑍍𑌯𑌚
𑌪𑍂𑌰𑍍𑌵𑌾𑌯𑌾𑌮𑍋𑌮𑍋𑌵𑌾𑌔𑌹𑍋𑌵𑌾।𑌊𑌪𑌾।।10।।
