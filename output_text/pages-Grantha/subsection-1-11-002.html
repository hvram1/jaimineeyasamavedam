।𑌦𑍈𑌰𑍍𑌘𑌤𑌮𑌸𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿।।

2 ।𑌦𑍈𑌰𑍍𑌘𑌤𑌮𑌸𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿।।

𑌪𑍁𑌰𑍁𑌤𑍍𑌵𑌾𑌦𑌾𑌶𑌿𑌵𑌂𑌵𑍋𑌚𑍇𑌤𑍍।𑌆𑌰𑌾𑌇𑌰𑌾𑌗𑍍𑌨𑌾𑌇।
𑌤𑍃𑌤𑌪𑍀𑌶𑍍𑌯
𑌤𑌾𑌵𑌾𑌸𑍍𑌵𑌾𑌇𑌦𑌾।𑌤𑍋𑌦𑌾𑌸𑍍𑌯𑌾𑌇𑌵𑌾𑌶𑌾𑌰𑌾𑌣𑌾
𑌚𑌿𑌣𑍍𑌯𑌚𑌿𑌶𑍍𑌯𑌚𑌿
𑌯𑌾।𑌮𑌾𑌹𑌾𑌸𑍍𑌯𑌾।𑌏।।3।।
𑌣𑌚𑍍𑌯𑌤𑍍𑌰𑌤𑌚
𑌪𑍁𑌰𑍂𑌤𑍍𑌵𑌾𑌦𑌾𑌶𑌿𑌵𑌂𑌵𑍋𑌚𑍇𑌤𑍍।𑌅𑌰𑌿𑌰𑌗𑍍𑌨𑌾𑌇𑌤𑌵𑌾
𑌚𑌿𑌶𑍂
𑌸𑍍𑌵𑌾𑌇𑌦𑌾।𑌤𑍋𑌦𑌾𑌸𑍍𑌯𑌾𑌇𑌵𑌾𑌶𑌾𑌰𑌾𑌣𑌾𑌯𑌾।
𑌟𑍃𑌚𑌣𑌚𑌪𑍍𑌲𑌚𑌶𑍍𑌯𑌚𑌿𑌣
𑌮𑌾𑌹𑌾𑌸𑍍𑌯𑌾।।।4।।
𑌚𑍍𑌯𑌤𑍍𑌰
𑌪𑍁𑌰𑍂𑌤𑍍𑌵𑌾𑌦𑌾𑌶𑌿𑌵𑌂𑌵𑍋𑌚𑍇𑌡𑌾𑌬𑍁।𑌹𑌾𑌬𑍁𑌹𑍋
𑌪𑌿𑌶𑍃𑌕
𑌹𑌾𑌉𑌵𑍋𑌵𑌾।𑌪𑍍𑌰।𑌅𑌰𑌿𑌰𑌗𑍍𑌨𑍇𑌤𑌵𑌾𑌸𑍍𑌵𑍀𑌦𑌾।𑌪𑍍𑌰।
𑌚𑌿𑌣𑌕𑍀𑌕𑌿𑌚
𑌤𑍋𑌦𑌸𑍇𑌯𑌵𑌶𑌰𑌾𑌣𑌾𑌯𑌾।𑌹𑌾𑌬𑍁𑌹𑍋𑌹𑌾
𑌕𑌕𑍍𑌯𑌕𑌿𑌚𑍍𑌯𑌕𑌟𑍍𑌯
𑌉𑌵𑍋𑌵𑌾।𑌏𑌮𑌾𑌹𑌾𑌸𑍍𑌯𑌾।।5।।
