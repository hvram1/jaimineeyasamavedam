।𑌵𑌰𑍁𑌣𑌸𑌾𑌮𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

3 ।𑌵𑌰𑍁𑌣𑌸𑌾𑌮𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌪𑍍𑌰𑌮𑌿𑌤𑍍𑌰𑌾𑌯𑌪𑍍𑌰𑌾𑌹𑌾𑌬𑍁।𑌆𑌰𑍍𑌯𑌮𑍍𑌣𑌾𑌇𑌸𑌚𑌾𑌹𑍋𑌧𑌿
𑌤𑍁𑌤𑌶𑌟𑌚𑍍𑌯𑌟𑌿𑌂
𑌯𑍗𑌹𑍁𑌵𑌾𑌕𑍍𑌤𑌰𑌾𑌵𑌸𑌾𑌬𑍁।𑌵𑌰𑍗𑌹𑍋𑌧𑍀𑌯𑍗𑌹𑍂𑌵𑌾
𑌟𑍍𑌯𑌟𑌿𑌚𑍍𑌯𑌶𑌟𑍍𑌯𑌂𑌟𑍍𑌯
𑌇𑌵𑌰𑍂𑌣𑍇𑌛𑌾𑌨𑍍𑌦𑍀𑌯𑌂𑌵𑌚𑌾𑌃।𑌸𑍍𑌤𑍋𑌤𑍍𑌰𑌾𑌂𑌹𑍋𑌇𑌰𑌾𑌜𑍗
𑌕𑍃𑌚𑌿𑌟𑍍𑌯𑌟𑌿𑌟
𑌹𑍂𑌵𑌾𑌸𑍂𑌗𑌾𑌯𑌾𑌤𑌾𑌆𑌉𑌵𑌾।𑌊𑌪𑌾।।7।।
𑌕𑌿𑌟𑌿𑌟𑌿𑌚𑌶
𑌪𑍍𑌰𑌮𑌿𑌤𑍍𑌰𑌾𑌯𑌪𑍍𑌰𑍗𑌹𑍋𑌵𑌾।𑌅𑌰𑍍𑌯𑌮𑍍𑌣𑌾𑌬𑍁𑌹𑍋𑌔𑌹𑍋
𑌤𑍂𑌤𑌕𑌿𑌚𑌕
𑌵𑌾।𑌪𑍍𑌰।𑌸𑌾𑌚𑌾𑌤𑍍𑌥𑍍𑌯𑌮𑍃𑌤𑌾𑌵𑌾𑌸𑌾।𑌪𑍍𑌰।𑌵𑌾𑌰𑍂𑌥𑍍𑌯
𑌕𑍍𑌯𑌕𑍀𑌚𑍍𑌯
𑌵𑌰𑍂𑌣𑍇𑌛𑌨𑍍𑌦𑍀𑌯𑌂𑌵𑌾𑌚𑌾।𑌪𑍍𑌰।𑌸𑍍𑌤𑍋𑌤𑍍𑌰𑌂𑌰𑌾𑌜𑌾।𑌪𑍍𑌰।
𑌚𑍂𑌕𑌕𑌿𑌕𑌕𑍍𑌯
𑌸𑍂𑌗𑌾𑌯𑌾𑌤𑌾𑌬𑍁𑌹𑍋𑌔𑌹𑍋𑌵𑌾𑌃।𑌓𑌇ழா𑌹।।8।।
𑌕𑌕𑍍𑌯𑌕𑌿𑌚𑌕𑌪𑌣𑌪𑌶𑍍𑌯
𑌪𑍍𑌰𑌮𑌿𑌤𑍍𑌰𑌾𑌯𑌪𑍍𑌰𑌾𑌰𑍍𑌯𑌮𑍋𑌵𑌾𑌓𑌵𑌾𑌸𑌾𑌚𑌾𑌥𑍍𑌯𑌮𑍃𑌤𑌾𑌵𑌾
𑌤𑍃𑌤𑍍𑌯𑌤𑌚𑍍𑌯𑌚𑍍𑌯𑌯𑍍𑌯
𑌸𑌾𑌬𑍁।𑌵𑌾𑌰𑍂𑌥𑌾𑌯𑌾।𑌇𑌵𑌰𑍂𑌣𑍇𑌛𑌨𑍍𑌦𑌾𑌯𑌾𑌂𑌹𑌾
𑌟𑌶𑌟𑌤𑌟𑌤𑌕𑍀𑌚𑌕𑌟𑍍𑌯𑌤
𑌇𑌵𑌚𑍋𑌆।𑌸𑍍𑌤𑍋𑌤𑍍𑌰𑌂𑌰𑌾𑌜𑌸𑍁𑌗𑌾𑌯𑌤𑌸𑍍𑌤𑍋𑌤𑍍𑌰𑌾𑌂।
𑌟𑌿𑌤𑌕𑌕𑍍𑌯𑌚𑌿𑌟𑌿𑌤
𑌰𑌾𑌜𑌸𑍂𑌗𑍗𑌹𑍋𑌓𑌬𑌾𑌯𑌾𑌤𑍋𑌹𑌾𑌇।।9।।
𑌕𑌚𑌕𑌟𑌤𑌪𑌪𑍍𑌲𑌶𑍍𑌯
