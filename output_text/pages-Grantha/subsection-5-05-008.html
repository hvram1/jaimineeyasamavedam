।𑌆𑌯𑌾𑌸𑍇𑌯𑌚𑍇𑌸𑍋𑌮𑌸𑌾𑌮𑌨𑍀𑌵𑌾 ।

8 ।𑌆𑌯𑌾𑌸𑍇𑌯𑌚𑍇𑌸𑍋𑌮𑌸𑌾𑌮𑌨𑍀𑌵𑌾 ।

𑌆𑌯𑌿𑌪𑍁𑌨𑌾।𑌨𑌾𑌸𑍍𑌸𑍋𑌮𑌾𑌧𑌾𑌰𑌯𑌾𑌅𑌪𑍋𑌵𑌾𑌸𑌾𑌓
 𑌤𑍀 𑌕𑌿 𑌕𑍂
𑌨𑍋𑌅𑌰𑍍𑌷𑌸𑍀।𑌆𑌰𑌾𑌤𑍍𑌨𑌾𑌧𑌾𑌓𑌯𑍋𑌨𑌿𑌮𑍃𑌤𑌸𑍍𑌯𑌾𑌸𑍀𑌦𑌸𑍀।
 𑌕𑍍𑌯 𑌚𑌿 𑌕𑍀 𑌕𑍂 𑌚𑌿
𑌉𑌤𑍍𑌸𑍋𑌦𑌾𑌇𑌵𑌾𑌓𑌇।𑌹𑍀𑌰𑌣𑍍𑌯𑌾𑌯𑌾।𑌓𑌇ழா𑌹।।10।।
 𑌕𑍁 𑌚 𑌶 𑌟𑌿 𑌚𑌣 𑌪 𑌶𑍍𑌯
𑌪𑍁𑌨𑌾𑌨𑌸𑍍𑌸𑍋𑌮𑌧𑌾𑌹𑌾𑌬𑍁𑌹𑍋𑌵𑌾।𑌰𑌾𑌯𑌾𑌪𑍋𑌵𑌸𑌾
 𑌤𑍢 𑌤 𑌪𑍍𑌲
𑌨𑌯𑌾।𑌰𑍍𑌷𑌾𑌸𑍀।𑌅𑌰𑌾𑌔𑌹𑍋𑌵𑌾𑌤𑍍𑌨𑌾𑌧𑌾𑌯𑍋
 𑌚𑍂𑌣𑌫𑌪𑍍𑌲 𑌚 𑌶 𑌪𑍍𑌲 𑌕 𑌥𑍍𑌯
𑌨𑌿𑌮𑍃𑌤𑌾𑌸𑍍𑌯𑌸𑌾𑌇।𑌦𑌾𑌸𑍀।𑌉𑌤𑍍𑌸𑍋𑌔𑌹𑍋
 𑌟𑌿 𑌚𑍍𑌯𑌣𑌫𑌪𑍍𑌲 𑌶 𑌚 𑌶 𑌪𑍍𑌲 𑌕
𑌵𑌾।𑌦𑌾𑌇𑌵𑍋𑌹𑌿𑌰𑌾।𑌣𑍍𑌯𑌾𑌯𑌾।।11।।
 𑌥𑍍𑌯 𑌟𑌿 𑌚𑍍𑌯𑌣𑌫𑌪𑍍𑌲 𑌚
