। 𑌋 𑌤𑍁𑌸𑌾𑌮𑌨𑌿𑌚 ।

11 । 𑌋 𑌤𑍁𑌸𑌾𑌮𑌨𑌿𑌚 ।

𑌬𑍍𑌰𑌹𑍍𑌮𑌾𑌬𑍍𑌰𑌹𑍍𑌮𑌾।𑌜𑌜𑍍𑌤𑌾𑌨𑌂𑌪𑍍𑌰𑌾𑌥𑌾𑌮𑌂𑌪𑍁𑌰𑌾𑌸𑍍𑌤𑌾𑌤𑍍।
 𑌟𑌿 𑌤 𑌚𑌿 𑌕𑌿 𑌭𑌿
𑌵𑍀𑌸𑌾𑌇𑌵𑌾𑌇𑌸𑍀।𑌮𑌤𑌸𑍍𑌸𑍁𑌰𑍂𑌚𑍋𑌵𑍇𑌨𑌆𑌵𑌾𑌤𑍍।
 𑌟𑌿 𑌤𑍍𑌯 𑌕𑍀 𑌥𑍍𑌯 𑌭𑌿
𑌸𑌾𑌬𑍂𑌸𑌾𑌬𑍂।𑌧𑌨𑍍𑌯𑌾𑌉𑌪𑌾𑌮𑌾𑌆𑌸𑍍𑌯𑌵𑌾𑌇𑌷𑍍𑌠𑌾𑌃।
 𑌟𑌿 𑌤 𑌚𑍀 𑌕 𑌟 𑌭𑍀
𑌸𑌾𑌤𑌾𑌸𑍍𑌸𑌾𑌤𑌾𑌃।𑌚𑌯𑍋𑌨𑍀𑌮𑌾𑌸𑌾𑌤𑌾𑌶𑍍𑌪𑌵𑌾𑌇𑌵𑌾𑌃।
  𑌟𑌿 𑌤 𑌚𑍁 𑌕𑌿 𑌪𑌣
𑌓𑌇ழா।।17।।
 𑌪 𑌪𑍍𑌲
𑌹𑍁𑌵𑍇𑌹𑌾𑌇𑌹𑍁𑌵𑍇𑌹𑌾𑌇𑌹𑍇𑌷𑌾𑌯𑌾।𑌬𑍍𑌰𑌹𑍍𑌮𑌜𑌜𑍍𑌤𑌾
 𑌟𑍍𑌯 𑌤 𑌟𑌿 𑌤 𑌟𑌿 𑌤 𑌕𑍀
𑌨𑌾𑌂𑌪𑍍𑌰𑌾𑌥𑌾𑌮𑌂𑌪𑍁𑌰𑌾𑌸𑍍𑌤𑌾𑌤𑍍।𑌵𑍀𑌸𑍀𑌮𑌤𑌾𑌸𑍍𑌸𑍁𑌰𑍂𑌚𑍋
 𑌟 𑌕𑌿 𑌚𑍍𑌯 𑌶 𑌕𑍀
𑌵𑍇𑌨𑌆𑌵𑌾𑌤𑍍।𑌸𑌾𑌬𑍁𑌧𑌨𑍍𑌯𑌾𑌉𑌪𑌮𑌾𑌆𑌸𑍍𑌯𑌵𑌾𑌇
 𑌕𑍀 𑌚𑍍𑌯 𑌶 𑌕𑍁 𑌕𑌿 𑌚𑍍𑌯
𑌷𑍍𑌟𑌾𑌃।𑌸𑌾𑌤𑌶𑍍𑌪𑌯𑍋𑌨𑍀𑌮𑌾𑌸𑌾𑌤𑌶𑍍𑌪𑌵𑍀𑌵𑌾𑌃।𑌹𑍁𑌵𑍇𑌹𑌾
 𑌫𑍍𑌲 𑌕𑍁 𑌕𑍍𑌯 𑌚𑌿 𑌪𑍍𑌲 𑌟𑍍𑌯 𑌤
𑌇𑌹𑍁𑌵𑍇𑌹𑌾𑌇𑌹𑍇𑌷𑌾𑌯𑌾𑌔𑌹𑍋𑌵𑌾।𑌏𑌋𑌤𑌮𑌮𑍍𑌰𑌤
 𑌟𑌿 𑌤 𑌟𑌿 𑌚 𑌶𑌿 𑌤
𑌮𑍇𑌋𑌤𑌮𑌮𑍍𑌰𑌤𑌮𑍇𑌋𑌤𑌮𑌮𑍍𑌰𑌤𑌾𑌂।।18।।
 𑌕𑍂 𑌕𑍂 𑌟𑍁𑌚
