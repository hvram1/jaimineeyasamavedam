।𑌸𑍈𑌨𑍍𑌧𑍁𑌕𑍍𑌷𑌿𑌤𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

12 ।𑌸𑍈𑌨𑍍𑌧𑍁𑌕𑍍𑌷𑌿𑌤𑌾𑌨𑌿𑌤𑍍𑌰𑍀𑌣𑌿 ।

𑌵𑍃𑌷𑌾𑌮𑌾𑌤𑍀𑌨𑌾𑌂।𑌪𑌾𑌵𑌤𑌾𑌏।𑌵𑍀𑌚𑌕𑍍𑌷𑌾𑌣𑌾𑌏।
 𑌕𑍍𑌯 𑌟𑍍𑌯 𑌂 𑌟𑌿 𑌤 𑌚𑌿 𑌕 𑌚
𑌸𑍋𑌮𑍋𑌅𑌨𑍍𑌹𑌾𑌂𑌪𑍍𑌰𑌾𑌤𑌾𑌰𑌾𑌇𑌤𑌾𑌏।𑌉𑌷𑌸𑌾𑌨𑍍𑌦𑍀𑌵𑌾
 𑌥𑍍𑌯 𑌟𑍍𑌯 𑌚𑍍𑌯 𑌟𑌿 𑌤 𑌚𑍀 𑌕
𑌏।𑌪𑍍𑌰𑌾𑌣𑌾𑌸𑌾𑌇𑌨𑍍𑌧𑍂𑌨𑌾𑌂।𑌕𑌾𑌲𑌾𑌶𑌂𑌏।𑌅𑌚𑌿𑌕𑍍𑌰𑌾
 𑌚 𑌥𑍍𑌯 𑌟𑌿 𑌂 𑌟𑌿 𑌤
𑌦𑌾𑌦𑍇।𑌇𑌨𑍍𑌦𑌰𑌸𑍍𑌯𑌹𑌾𑌦𑌰𑍍𑌯𑌾𑌵𑌾𑌇𑌶𑌾𑌨𑍍𑌏।𑌮𑌨𑍀𑌷𑍀𑌭𑍀
 𑌕𑍀 𑌚 𑌥 𑌟𑍍𑌯 𑌟 𑌟𑍀 𑌤 𑌚𑌿 𑌕
𑌰𑍇।𑌓𑌇ழா𑌹।।21।।
 𑌚𑌣 𑌪 𑌶
𑌵𑍃𑌷𑌾𑌮𑌤𑌾𑌈𑌨𑌾𑌂𑌪𑌵𑌤𑍇𑌵𑍀𑌚𑌕𑍍𑌷𑌾𑌣𑌾𑌃।𑌸𑍋
 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌕𑍁 𑌟𑌿 𑌤
𑌮𑍋𑌅𑌨𑍍𑌹𑌾𑌂𑌪𑍍𑌰𑌤𑌰𑍀𑌤𑍋𑌷𑌾𑌸𑌾𑌨𑍍𑌦𑌾𑌇𑌵𑌾𑌃।𑌪𑍍𑌰𑌾𑌣𑌾
 𑌟𑍍𑌯 𑌟𑍍𑌯 𑌕𑍀 𑌟𑌿 𑌤𑍍𑌯 𑌟𑍍𑌯
𑌸𑌾𑌇𑌨𑍍𑌧𑍂𑌨𑌾𑌂𑌕𑌲𑌶𑌂।𑌆𑌚𑌿𑌕𑍍𑌰𑌾𑌦𑌾𑌤𑍍।𑌆𑌇𑌦𑍍𑌰𑌾𑌸𑍍𑌯𑌾
  𑌟𑌿 𑌕𑍀 𑌟𑌿 𑌤 𑌟𑌿
𑌹𑌾𑌦𑍍𑌯𑌾𑌵𑌿𑌶𑌨𑍍𑌮𑌾𑌨𑍀𑌷𑌾𑌇𑌭𑌾𑌇𑌃।𑌓𑌇ழா𑌹।।22।।
 𑌟𑍍𑌯 𑌕𑌿 𑌟𑌿 𑌚𑍍𑌯𑌣 𑌪 𑌶𑍍𑌯
𑌵𑍃𑌷𑌾𑌮𑌤𑍀𑌨𑌾𑌂𑌪𑌵𑌤𑌾𑌇𑌵𑌾𑌇𑌚𑌾𑌕𑍍𑌷𑌾𑌣𑌾𑌃।𑌸𑍋
 𑌕𑍁 𑌕𑌿 𑌯𑍍𑌯 𑌪𑍍𑌲 𑌤𑍍𑌯
𑌮𑍋𑌅𑌨𑍍𑌹𑌾𑌂𑌪𑍍𑌰𑌤𑌰𑍀𑌤𑍋𑌷𑌾𑌸𑌾𑌨𑍍𑌦𑌿𑌵𑌾𑌃।𑌪𑍍𑌰𑌾𑌣𑌾𑌸𑌿
 𑌕𑍍𑌯 𑌕𑍍𑌯 𑌕𑌿 𑌯𑍍𑌯 𑌪 𑌕𑍍𑌯 𑌕
𑌨𑍍𑌧𑍂𑌨𑌾𑌂𑌕𑌲𑌶𑌂𑌅𑌚𑌾𑌇𑌕𑍃𑌦𑌾𑌤𑍍।𑌆𑌇𑌨𑍍𑌦𑌰𑌾𑌸𑍍𑌯𑌾
 𑌕𑌿𑌚 𑌕 𑌟𑌿 𑌪𑍍𑌲 𑌕𑌿 𑌕𑍀
𑌹𑌾𑌦𑍍𑌯𑌾𑌵𑌿𑌶𑌨𑍍𑌮𑌾𑌨𑌾𑌇𑌷𑍀𑌹𑌾𑌬𑍁।𑌬𑌾।।23।।
 𑌕𑍀 𑌯 𑌪 𑌶𑍍𑌯 𑌣 𑌶 𑌚
